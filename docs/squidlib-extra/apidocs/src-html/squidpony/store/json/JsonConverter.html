018/**
019 * Augmented version of LibGDX's Json class that knows how to handle various data types common in SquidLib.
020 * This includes OrderedMap, which notably allows non-String keys (LibGDX's default Map serializer requires keys to be
021 * Strings), but does not currently allow the IHasher to be set (which only should affect OrderedMaps with array keys).
022 * It also makes significantly shorter serialized output for 2D char arrays, GreasedRegion and FakeLanguageGen objects,
023 * and various collections (IntDoubleOrderedMap, IntVLA, Arrangement, K2, and K2V1 at least).
024 * Created by Tommy Ettinger on 1/9/2017.
025 */
026public class JsonConverter extends Json {
027    /**
028     * Creates a new JsonConverter using "minimal" output type, so it omits double quotes whenever possible but gives
029     * up compatibility with most other JSON readers. Give the constructor
030     * {@link JsonWriter.OutputType#json} if you need full compatibility.
031     */
032    public JsonConverter() {
033        super();
034        initialize(this);
035    }
036
037    /**
038     * Creates a new JsonConverter with the given OutputType; typically minimal is fine, but compatibility may require
039     * you to use json; the javascript type is a sort of middle ground.
040     * @param outputType a JsonWriter.OutputType enum value that determines what syntax can be omitted from the output
041     */
042    public JsonConverter(JsonWriter.OutputType outputType) {
043        super(outputType);
044        initialize(this);
045    }

249        json.setSerializer(OrderedMap.class, new Serializer<OrderedMap>() {
250            @Override
251            public void write(Json json, OrderedMap object, Class knownType) {
252                if(object == null)
253                {
254                    json.writeValue(null);
255                    return;
256                }
257                json.writeObjectStart();
258                json.writeValue("f", object.f);
259                if(!object.isEmpty()) {
260                    json.writeValue("k", object.firstKey(), null);
261                    json.writeValue("v", object.getAt(0), null);
262                    int sz = object.size();
263                    Object[] r = new Object[(sz - 1) * 2];
264                    for (int i = 1, p = 0; i < sz; i++) {
265                        r[p++] = object.keyAt(i);
266                        r[p++] = object.getAt(i);
267                    }
268                    json.writeValue("r", r, Object[].class, Object.class);
269                }
270                json.writeObjectEnd();
271            }

273            @Override
274            @SuppressWarnings("unchecked")
275            public OrderedMap read(Json json, JsonValue jsonData, Class type) {
276                if(jsonData == null || jsonData.isNull()) return null;
277                float f = json.readValue("f", float.class, jsonData);
278                Object k = json.readValue("k", null, INVALID, jsonData);
279                Object v = json.readValue("v", null, INVALID, jsonData);
280                Object[] r = json.readValue("r", Object[].class, jsonData);
281                if(k == INVALID)
282                    return new OrderedMap(0, f);
283                return Maker.makeOM(f, k, v, r);
284                //return new OrderedMap(json.readValue(OrderedSet.class, jsonData.get("k")),
285                //        json.readValue(ArrayList.class, jsonData.get("v")), jsonData.getFloat("f"));
286            }

288        json.setSerializer(EnumOrderedMap.class, new Serializer<EnumOrderedMap>() {
289            @Override
290            public void write(Json json, EnumOrderedMap object, Class knownType) {
291                if(object == null)
292                {
293                    json.writeValue(null);
294                    return;
295                }
296                json.writeObjectStart();
297                if(!object.isEmpty()) {
298                    json.writeValue("c", object.firstKey().getClass().getName());
299                    json.writeValue("k", object.firstKey(), null);
300                    json.writeValue("v", object.getAt(0), null);
301                    int sz = object.size();
302                    Object[] r = new Object[(sz - 1) * 2];
303                    for (int i = 1, p = 0; i < sz; i++) {
304                        r[p++] = object.keyAt(i);
305                        r[p++] = object.getAt(i);
306                    }
307                    json.writeValue("r", r, Object[].class, Object.class);
308                }
309                else
310                    json.writeValue("c", "default");
311                json.writeObjectEnd();
312            }
313
314            @Override
315            @SuppressWarnings("unchecked")
316            public EnumOrderedMap read(Json json, JsonValue jsonData, Class type) {
317                if(jsonData == null || jsonData.isNull()) return null;
318                String c = json.readValue("c", String.class, "", jsonData);
319                if("default".equals(c))
320                    return new EnumOrderedMap();
321                try {
322                    Class<? extends Enum<?>> cl = ClassReflection.forName(c);
323                    if(!ClassReflection.isEnum(cl))
324                        return null;
325                    Enum<?> k = json.readValue("k", cl, jsonData);
326                    Object v = json.readValue("v", null, INVALID, jsonData);
327                    Object[] r = json.readValue("r", Object[].class, jsonData);
328                    if(v == INVALID)
329                        return new EnumOrderedMap();
330                    return Maker.makeEOM(k, v, r);
331                } catch (ReflectionException e) {
332                    return null;
333                }
334                //return new OrderedMap(json.readValue(OrderedSet.class, jsonData.get("k")),
335                //        json.readValue(ArrayList.class, jsonData.get("v")), jsonData.getFloat("f"));
336            }
337        });

384        json.setSerializer(EnumMap.class, new Serializer<EnumMap>() {
385            @Override
386            public void write(Json json, EnumMap object, Class knownType) {
387                if(object == null)
388                {
389                    json.writeValue(null);
390                    return;
391                }
392                json.writeObjectStart();
393                if(!object.isEmpty()) {
394                    Iterator it = object.entrySet().iterator();
395                    Map.Entry en = (Map.Entry)it.next();
396                    json.writeValue("e", en.getKey(), Enum.class);
397                    json.writeValue("v", en.getValue(), null);
398                    int sz = object.size();
399                    Object[] r = new Object[(sz - 1) * 2];
400                    for (int i = 1, p = 0; i < sz; i++) {
401                        if(!it.hasNext())
402                            break;
403                        en = (Map.Entry)it.next();
404                        r[p++] = en.getKey();
405                        r[p++] = en.getValue();
406                    }
407                    json.writeValue("r", r, Object[].class, Object.class);
408                }
409                json.writeObjectEnd();
410            }
411

413            @SuppressWarnings("unchecked")
414            public EnumMap read(Json json, JsonValue jsonData, Class type) {
415                if(jsonData == null || jsonData.isNull() || jsonData.size == 0) return null;
416                return new EnumMap(Maker.makeOM(0.75f,
417                        json.readValue("e", null, jsonData),
418                        json.readValue("v", null, jsonData),
419                        json.readValue("r", Object[].class, jsonData)));
420                //return new OrderedMap(json.readValue(OrderedSet.class, jsonData.get("k")),
421                //        json.readValue(ArrayList.class, jsonData.get("v")), jsonData.getFloat("f"));
422            }

495            @Override
496            public void write(Json json, char[][] object, Class knownType) {
497                if(object == null)
498                {
499                    json.writeValue(null);
500                    return;
501                }
502                int sz = object.length;
503                json.writeArrayStart();
504                for (int i = 0; i < sz; i++) {

507                json.writeArrayEnd();

510            @Override
511            public char[][] read(Json json, JsonValue jsonData, Class type) {
512                if(jsonData == null || jsonData.isNull())
513                    return null;
514                int sz = jsonData.size;
515                char[][] data = new char[sz][];
516                JsonValue c = jsonData.child();
517                for (int i = 0; i < sz && c != null; i++, c = c.next()) {
518                    data[i] = c.asString().toCharArray();
519                }
520                return data;
521            }
