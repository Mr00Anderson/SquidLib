020    /**
021     * Makes a LinkedHashMap (LHM) with key and value types inferred from the types of k0 and v0, and considers all
022     * parameters key-value pairs, casting the Objects at positions 0, 2, 4... etc. to K and the objects at positions
023     * 1, 3, 5... etc. to V. If rest has an odd-number length, then it discards the last item. If any pair of items in
024     * rest cannot be cast to the correct type of K or V, then this inserts nothing for that pair and logs information
025     * on the problematic pair to the static Maker.issueLog field.
026     * @param k0 the first key; used to infer the types of other keys if generic parameters aren't specified.
027     * @param v0 the first value; used to infer the types of other values if generic parameters aren't specified.
028     * @param rest an array or vararg of keys and values in pairs; should contain alternating K, V, K, V... elements
029     * @param <K> the type of keys in the returned LinkedHashMap; if not specified, will be inferred from k0
030     * @param <V> the type of values in the returned LinkedHashMap; if not specified, will be inferred from v0
031     * @return a freshly-made LinkedHashMap with K keys and V values, using k0, v0, and the contents of rest to fill it
032     */
033    @SuppressWarnings("unchecked")
034    public static <K, V> LinkedHashMap<K, V> makeLHM(K k0, V v0, Object... rest)
035    {
036        if(rest == null || rest.length == 0)
037        {
038            LinkedHashMap<K, V> lhm = new LinkedHashMap<>(2);
039            lhm.put(k0, v0);
040            return lhm;
041        }
042        LinkedHashMap<K, V> lhm = new LinkedHashMap<>(1 + (rest.length / 2));
043        lhm.put(k0, v0);
044
045        for (int i = 0; i < rest.length - 1; i+=2) {
046            try {
047                lhm.put((K) rest[i], (V) rest[i + 1]);
048            }catch (ClassCastException cce) {
049                issueLog.append("makeLHM call had a casting problem with pair at rest[")
050                        .append(i)
051                        .append("] and/or rest[")
052                        .append(i + 1)
053                        .append("], with contents: ")
054                        .append(rest[i])
055                        .append(", ")
056                        .append(rest[i + 1])
057                        .append(".\n\nException messages:\n")
058                        .append(cce);
059                String msg = cce.getMessage();
060                if (msg != null) {
061                    issueLog.append('\n').append(msg);
062                }
063                issueLog.append('\n');
064            }
065        }
066        return lhm;
067    }

362     * Makes a EnumOrderedSet (OS) of the enum type T given at least one T element followed by an array or vararg of any
363     * number of additional T elements. Duplicate items in elements will have all but one item discarded, using the
364     * later item in elements. The order given here will be kept in the result, like in OrderedSet, and
365     * you can use {@link EnumOrderedSet#getAt(int)} to get a T value at a given index, like you would with a List.
366     * @param initial the first item to insert into the EnumOrderedSet; if initial is null, the method returns null
367     * @param elements an array or vararg of T; allowed to be empty
368     * @param <T> an enum type
369     * @return a newly-allocated OrderedSet containing all of the non-duplicate items in elements, in order
370     */
371    @SuppressWarnings("unchecked")
372    public static <T extends Enum<?>> EnumOrderedSet<T> makeEOS(T initial, T... elements)
373    {
374        if(initial == null) return null;
375        EnumOrderedSet<T> eos = new EnumOrderedSet<>(initial);
376        eos.add(initial);
377        if(elements != null)
378            eos.addAll(elements);
379        return eos;
380    }

493     * items in rest cannot be cast to the correct type of K or V, then this inserts nothing for that pair and logs
494     * information on the problematic pair to the static Maker.issueLog field. The order given here will be kept in the
495     * result, unlike in the JDK's EnumMap class, and you can use {@link EnumOrderedMap#keyAt(int)} or
496     * {@link EnumOrderedMap#getAt(int)} to get a key or value at a given index, like you would with a List.
497     *
498     * @param k0 the first key, which must be an Enum; used to infer the types of other keys if generic parameters aren't specified.
499     * @param v0 the first value; used to infer the types of other values if generic parameters aren't specified.
500     * @param rest an array or vararg of keys and values in pairs; should contain alternating K, V, K, V... elements
501     * @param <K> the type of Enum keys in the returned EnumOrderedMap; if not specified, will be inferred from k0
502     * @param <V> the type of values in the returned EnumOrderedMap; if not specified, will be inferred from v0
503     * @return a freshly-made EnumOrderedMap with K keys and V values, using k0, v0, and the contents of rest to fill it
504     */
505    @SuppressWarnings("unchecked")
506    public static <K extends Enum<?>, V> EnumOrderedMap<K, V> makeEOM(K k0, V v0, Object... rest)
507    {
508        if(rest == null || rest.length == 0)
509        {
510            EnumOrderedMap<K, V> eom = new EnumOrderedMap<>();
511            eom.put(k0, v0);
512            return eom;
513        }
514        EnumOrderedMap<K, V> eom = new EnumOrderedMap<>(k0);
515        eom.put(k0, v0);
516
517        for (int i = 0; i < rest.length - 1; i+=2) {
518            try {
519                eom.put((K) rest[i], (V) rest[i + 1]);
520            }catch (ClassCastException cce) {
521                issueLog.append("makeEOM call had a casting problem with pair at rest[")
522                          .append(i)
523                          .append("] and/or rest[")
524                          .append(i + 1)
525                          .append("], with contents: ")
526                          .append(rest[i])
527                          .append(", ")
528                          .append(rest[i+1])
529                          .append(".\n\nException messages:\n")
530                          .append(cce);
531                String msg = cce.getMessage();
532                if (msg != null) {
533                    issueLog.append('\n').append(msg);
534                }
535                issueLog.append('\n');
536            }
537        }
538        return eom;
539    }

541    /**
542     * Makes an empty EnumOrderedMap (EOM); needs key and value types to be specified in order to work. For an empty
543     * EnumOrderedMap with Radius keys and Coord values, you could use {@code Maker.<Radius, Coord>makeEOM()}. Using
544     * the new keyword is probably just as easy in this case; this method is provided for completeness relative to
545     * makeEOM() with 2 or more parameters.
546     * @param <K> the type of Enum keys in the returned EnumOrderedMap; cannot be inferred and must be specified
547     * @param <V> the type of values in the returned EnumOrderedMap; cannot be inferred and must be specified
548     * @return an empty EnumOrderedMap with the given key and value types.
549     */
550    public static <K extends Enum<K>, V> EnumOrderedMap<K, V> makeEOM() { return new EnumOrderedMap<>(); }
