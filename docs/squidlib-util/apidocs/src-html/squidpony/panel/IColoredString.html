269                 * @param one
270                 * @param two 
271                 * @return Whether {@code one} represents the same content as
272                 *         {@code two}.
273                 */
274                /*
275                 * Method could be smarter, i.e. return true more often, by doing some
276                 * normalization. It is unnecessary if you only create instances of
277                 * IColoredString.Impl.
278                 */
279                public static <T> boolean equals(IColoredString<T> one, IColoredString<T> two) {
280                        if (one == two)
281                                return true;
282
283                        final Iterator<IColoredString.Bucket<T>> oneIt = one.iterator();
284                        final Iterator<IColoredString.Bucket<T>> twoIt = two.iterator();
285                        while (true) {
286                                if (oneIt.hasNext()) {
287                                        if (twoIt.hasNext()) {
288                                                final Bucket<T> oneb = oneIt.next();
289                                                final Bucket<T> twob = twoIt.next();
290                                                if (!equals(oneb.getText(), twob.getText()))
291                                                        return false;
292                                                if (!equals(oneb.getColor(), twob.getColor()))
293                                                        return false;
294                                                continue;
295                                        } else
296                                                /* 'this' not terminated, but 'other' is. */
297                                                return false;
298                                } else {
299                                        if (twoIt.hasNext())
300                                                /* 'this' terminated, but not 'other'. */
301                                                return false;
302                                        else
303                                                /* Both terminated */
304                                                break;
305                                }
306
307                        }
308                        return true;
309                }

326                @Override
327                public void append(String text, T color) {
328                        if (text == null || text.isEmpty())
329                                return;
330
331                        if (fragments.isEmpty())
332                                fragments.add(new Bucket<>(text, color));
333                        else {
334                                final Bucket<T> last = fragments.get(fragments.size() - 1);
335                                if (equals(last.color, color)) {
336                                        /* Append to the last bucket, to avoid extending the list */
337                                        final Bucket<T> novel = last.append(text);
338                                        fragments.remove(fragments.size() - 1);
339                                        fragments.add(novel);
340                                } else
341                                        fragments.add(new Bucket<>(text, color));
342                        }
343                }

363                @Override
364                public void replaceColor(/* @Nullable */ T old, /* @Nullable */ T new_) {
365                        if (equals(old, new_))
366                                /* Nothing to do */
367                                return;
368
369                        final ListIterator<Bucket<T>> it = fragments.listIterator();
370                        while (it.hasNext()) {
371                                final Bucket<T> bucket = it.next();
372                                if (equals(bucket.color, old)) {
373                                        /* Replace */
374                                        it.remove();
375                                        it.add(new Bucket<T>(bucket.getText(), new_));
376                                }
377                                /* else leave untouched */
378                        }
379                }

397                @Override
398                public void setLength(int len) {
399                        int l = 0;
400                        final ListIterator<IColoredString.Bucket<T>> it = fragments.listIterator();
401                        while (it.hasNext()) {
402                                final IColoredString.Bucket<T> next = it.next();
403                                final String ftext = next.text;
404                                final int flen = ftext.length();
405                                final int nextl = l + flen;
406                                if (nextl < len) {
407                                        /* Nothing to do */
408                                        l += flen;
409                                        continue;
410                                } else if (nextl == len) {
411                                        /* Delete all next fragments */
412                                        while (it.hasNext()) {
413                                                it.next();
414                                                it.remove();
415                                        }
416                                        /* We'll exit the outer loop right away */
417                                } else {
418                                        assert len < nextl;
419                                        /* Trim this fragment */
420                                        final IColoredString.Bucket<T> trimmed = next.setLength(len - l);
421                                        /* Replace this fragment */
422                                        it.remove();
423                                        it.add(trimmed);
424                                        /* Delete all next fragments */
425                                        while (it.hasNext()) {
426                                                it.next();
427                                                it.remove();
428                                        }
429                                        /* We'll exit the outer loop right away */
430                                }
431                        }
432                }

434                @Override
435                public List<IColoredString<T>> wrap(int width)
436                {
437                        // the one special case that wouldn't involve buf
438                        if (width <= 0) {
439                                /* Really, you should not rely on this behavior */
440                                System.err.println("Cannot wrap string in empty display");
441                                final List<IColoredString<T>> result = new ArrayList<>();
442                                result.add(this);
443                                return result;
444                        }
445                        // delegate to the buf-appending overload
446                        return wrap(width, new ArrayList<IColoredString<T>>());
447                }

449                @Override
450                public List<IColoredString<T>> wrap(int width, List<IColoredString<T>> buf) {
451                        if (width <= 0) {
452                                /* Really, you should not rely on this behavior */
453                                System.err.println("Cannot wrap string in empty display");
454                                final List<IColoredString<T>> result = new ArrayList<>();
455                                result.add(this);
456                                return result;
457                        }
458                        if (isEmpty()) {
459                                /*
460                                 * Catch this case early on, as empty lines are eaten below (see
461                                 * code after the while). Checking emptiness is cheap anyway.
462                                 */
463                                buf.add(this);
464                                return buf;
465                        }
466
467                        IColoredString<T> current = create();
468                        int curlen = 0;
469                        final Iterator<Bucket<T>> it = iterator();
470                        while (it.hasNext()) {
471                                final Bucket<T> next = it.next();
472                                final String bucket = next.getText();
473                                final String[] split = StringKit.split(bucket," ");
474                                final T color = next.color;
475                                for (int i = 0; i < split.length; i++) {
476                                        // This section was needed when using String.split() above, but not for
477                                        // StringKit.split(), which keeps leading and trailing delimiters.
478//                                      if (i == split.length - 1 && bucket.endsWith(" "))
479//                                              /*
480//                                               * Do not lose trailing space that got eaten by
481//                                               * 'bucket.split'.
482//                                               */
483//                                              split[i] = split[i] + " ";
484                                        final String chunk = split[i];
485                                        final int chunklen = chunk.length();
486                                        final boolean addLeadingSpace = 0 < curlen && 0 < i;
487                                        if (curlen + chunklen + (addLeadingSpace ? 1 : 0) <= width) {
488                                                if (addLeadingSpace) {
489                                                        /*
490                                                         * Do not forget space on which chunk got split. If
491                                                         * the space is offscreen, it's harmless, hence not
492                                                         * checking it.
493                                                         */
494                                                        current.append(' ', null);
495                                                        curlen++;
496                                                }
497
498                                                /* Can add it */
499                                                current.append(chunk, color);
500                                                /* Extend size */
501                                                curlen += chunklen;
502                                        } else {
503                                                /* Need to wrap */
504                                                /* Flush content so far */
505                                                if (!current.isEmpty())
506                                                        buf.add(current);
507                                                /*
508                                                 * else: line was prepared, but did not contain anything
509                                                 */
510                                                if (chunklen <= width) {
511                                                        current = create();
512                                                        current.append(chunk, color);
513                                                        /* Reinit size */
514                                                        curlen = chunklen;
515                                                } else {
516                                                        /*
517                                                         * This word is too long. Adding it and preparing a
518                                                         * new line immediately.
519                                                         */
520                                                        /* Add */
521                                                        buf.add(new Impl<>(chunk, color));
522                                                        /* Prepare for next rolls */
523                                                        current = create();
524                                                        /* Reinit size */
525                                                        curlen = 0;
526                                                }
527                                        }
528                                }
529                        }
530
531                        if (!current.isEmpty()) {
532                                /* Flush rest */
533                                buf.add(current);
534                        }
535
536                        return buf;
537                }

539                @Override
540                /*
541                 * smelC: not the cutest result (we should add spaces both from the left
542                 * and the right, instead of just from the left), but better than
543                 * nothing.
544                 */
545                public IColoredString<T> justify(int width) {
546                        int length = length();
547
548                        if (width <= length)
549                                /*
550                                 * If width==length, we're good. If width<length, we cannot
551                                 * adjust
552                                 */
553                                return this;
554
555                        int totalDiff = width - length;
556                        assert 0 < totalDiff;
557
558                        if (width <= totalDiff * 3)
559                                /* Too much of a difference, it would look very weird. */
560                                return this;
561
562                        final IColoredString.Impl<T> result = create();
563
564                        ListIterator<IColoredString.Bucket<T>> it = fragments.listIterator();
565                        final int nbb = fragments.size();
566                        final int[] bucketToNbSpaces = new int[nbb];
567                        /* The number of buckets that can contribute to justifying */
568                        int totalNbSpaces = 0;
569                        /* The index of the last bucket that has spaces */
570                        int lastHopeIndex = -1;
571                        {
572                                int i = 0;
573                                while (it.hasNext()) {
574                                        final Bucket<T> next = it.next();
575                                        final int nbs = nbSpaces(next.getText());
576                                        totalNbSpaces += nbs;
577                                        bucketToNbSpaces[i] = nbs;
578                                        i++;
579                                }
580
581                                if (totalNbSpaces == 0)
582                                        /* Cannot do anything */
583                                        return this;
584
585                                for (int j = bucketToNbSpaces.length - 1; 0 <= j; j--) {
586                                        if (0 < bucketToNbSpaces[j]) {
587                                                lastHopeIndex = j;
588                                                break;
589                                        }
590                                }
591                                /* Holds because we ruled out 'totalNbSpaces == 0' before */
592                                assert 0 <= lastHopeIndex;
593                        }
594
595                        // we know totalNbSpaces cannot be 0 from prior checks, so division is OK
596                        int toAddPerSpace = (totalDiff / totalNbSpaces);
597                        int totalRest = totalDiff - (toAddPerSpace * totalNbSpaces);
598                        assert 0 <= totalRest;
599
600                        int bidx = -1;
601
602                        it = fragments.listIterator();
603
604                        while (it.hasNext() && 0 < totalDiff) {
605                                bidx++;
606                                final Bucket<T> next = it.next();
607                                final String bucket = next.getText();
608                                final int blength = bucket.length();
609                                final int localNbSpaces = bucketToNbSpaces[bidx];
610                                if (localNbSpaces == 0) {
611                                        /* Cannot change it */
612                                        result.append(next);
613                                        continue;
614                                }
615                                int localDiff = localNbSpaces * toAddPerSpace;
616                                assert localDiff <= totalDiff;
617                                int nb = localDiff / localNbSpaces;
618                                int localRest = localDiff - (nb * localNbSpaces);
619                                if (localRest == 0 && 0 < totalRest) {
620                                        /*
621                                         * Take one for the group. This avoids flushing all spaces
622                                         * needed in the 'last hope' cases below.
623                                         */
624                                        localRest = 1;
625                                }
626                                assert 0 <= localRest;
627                                assert localRest <= totalRest;
628                                String novel = "";
629                                int eatenSpaces = 1;
630                                for (int i = 0; i < blength; i++) {
631                                        final char c = bucket.charAt(i);
632                                        novel += c;
633                                        if (c == ' ' && (0 < localDiff || 0 < totalDiff || 0 < localRest || 0 < totalRest)) {
634                                                /* Can (and should) add an extra space */
635                                                for (int j = 0; j < nb && 0 < localDiff; j++) {
636                                                        novel += " ";
637                                                        localDiff--;
638                                                        totalDiff--;
639                                                }
640                                                if (0 < localRest || 0 < totalRest) {
641                                                        if (eatenSpaces == localNbSpaces) {
642                                                                /* I'm the last hope for this bucket */
643                                                                for (int j = 0; j < localRest; j++) {
644                                                                        novel += " ";
645                                                                        localRest--;
646                                                                        totalRest--;
647                                                                }
648                                                                if (bidx == lastHopeIndex) {
649                                                                        /* I'm the last hope globally */
650                                                                        while (0 < totalRest) {
651                                                                                novel += " ";
652                                                                                totalRest--;
653                                                                        }
654                                                                }
655                                                        } else {
656                                                                if (0 < localRest && 0 < totalRest) {
657                                                                        /* Not the last hope: take one only */
658                                                                        novel += " ";
659                                                                        localRest--;
660                                                                        totalRest--;
661                                                                }
662                                                        }
663                                                }
664                                                eatenSpaces++;
665                                        }
666                                }
667                                /* I did my job */
668                                assert localRest == 0;
669                                /* If I was the hope, I did my job */
670                                assert bidx != lastHopeIndex || totalRest == 0;
671                                result.append(novel, next.getColor());
672                        }
673
674                        while (it.hasNext())
675                                result.append(it.next());
676
677                        return result;
678                }

709                @Override
710                public T colorAt(int index) {
711                        final ListIterator<IColoredString.Bucket<T>> it = fragments.listIterator();
712                        int now = 0;
713                        while (it.hasNext()) {
714                                final IColoredString.Bucket<T> next = it.next();
715                                final String ftext = next.text;
716                                final int flen = ftext.length();
717                                final int nextl = now + flen;
718                                if (index < nextl)
719                                        return next.color;
720                                now += flen;
721                        }
722                        throw new NoSuchElementException("Color at index " + index + " in " + this);
723                }

833         * A piece of a {@link IColoredString}: a text and its color.
834         * 
835         * @author smelC
836         * 
837         * @param <T>
838         *            The type of colors;
839         */
840        class Bucket<T> {
841
842                protected final String text;
843                protected final/* @Nullable */T color;
844
845                public Bucket(String text, /* @Nullable */T color) {
846                        this.text = text == null ? "" : text;
847                        this.color = color;
848                }
849
850                /**
851                 * @param text
852                 * @return An instance whose text is {@code this.text + text}. Color is
853                 *         unchanged.
854                 */
855                public Bucket<T> append(String text) {
856                        if (text == null || text.isEmpty())
857                                /* Let's save an allocation */
858                                return this;
859                        else
860                                return new Bucket<>(this.text + text, color);
861                }

864                {

866                }

868                public Bucket<T> setLength(int l) {
869                        final int here = text.length();
870                        if (here <= l)
871                                return this;
872                        else
873                                return new Bucket<>(text.substring(0, l), color);
874                }

878                }
