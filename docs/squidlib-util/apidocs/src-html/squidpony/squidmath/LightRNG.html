015/**
016 * This is a SplittableRandom-style generator, meant to have a tiny state
017 * that permits storing many different generators with low overhead.
018 * It should be rather fast, though no guarantees can be made on all hardware.
019 * It should be faster than using SplittableRandom from Java 8 because it has a
020 * single "gamma" value that prevents it from being split but also makes runtime
021 * a bit quicker. It is based on using a unary hash on a large-increment counter,
022 * which makes even the first item obtained from similarly-seeded LightRNGs very
023 * likely to be very different. This is an advantage over non-hash-based RNGs.
024 * <br>
025 * This generator is especially fast on OpenJ9, version 0.17.0 or later (from
026 * late 2019), and can be several times faster there than on HotSpot.
027 * <br>
028 * Written in 2015 by Sebastiano Vigna (vigna@acm.org)
029 * @author Sebastiano Vigna
030 * @author Tommy Ettinger
031 */
032public final class LightRNG implements RandomnessSource, StatefulRandomness, SkippingRandomness, Serializable
033{
034        /** 2 raised to the 53, - 1. */
035    private static final long DOUBLE_MASK = ( 1L << 53 ) - 1;
036    /** 2 raised to the -53. */
037    private static final double NORM_53 = 1. / ( 1L << 53 );
038    /** 2 raised to the 24, -1. */
039    private static final long FLOAT_MASK = ( 1L << 24 ) - 1;
040    /** 2 raised to the -24. */
041    private static final float NORM_24 = 1f / (1 << 24);

047    /** Creates a new generator seeded using Math.random. */
048    public LightRNG() {
049        this((long) ((Math.random() - 0.5) * 0x10000000000000L)
050                ^ (long) (((Math.random() - 0.5) * 2.0) * 0x8000000000000000L));
051    }

057    /**
058     * Gets a pseudo-random int with at most the specified count of bits; for example, if bits is 3 this can return any
059     * int between 0 and 2 to the 3 (that is, 8), exclusive on the upper end. That would mean 7 could be returned, but
060     * no higher ints, and 0 could be returned, but no lower ints.
061     * 
062     * The algorithm used here changed on March 8, 2018 when LightRNG was remade the default generator in SquidLib.
063     * The older method is available as {@link #compatibleNext(int)}, but its use is discouraged; it's slightly slower 
064     * for no good reason.
065     * @param bits the number of bits to be returned; if 0 or less, or if 32 or greater, can return any 32-bit int
066     * @return a pseudo-random int that uses at most the specified amount of bits
067     */
068    @Override
069    public final int next( int bits ) {
070        long z = state += 0x9E3779B97F4A7C15L;
071        z = (z ^ (z >>> 30)) * 0xBF58476D1CE4E5B9L;
072        z = (z ^ (z >>> 27)) * 0x94D049BB133111EBL;
073        return (int)(z ^ (z >>> 31)) >>> (32 - bits);
074    }
075    /**
076     * Gets a pseudo-random int with at most the specified count of bits; for example, if bits is 3 this can return any
077     * int between 0 and 2 to the 3 (that is, 8), exclusive on the upper end. That would mean 7 could be returned, but
078     * no higher ints, and 0 could be returned, but no lower ints.
079     *
080     * The algorithm used here is the older version of {@link #next(int)} before some things changed on March 8 2018.
081     * Using this method is discouraged unless you need to reproduce values exactly; it's slightly slower for no good
082     * reason. Calling {@code next(32)} and {@code compatibleNext(32)} should have identical results, but other values
083     * for bits will probably be different.
084     * @param bits the number of bits to be returned; if 0 or less, or if 32 or greater, can return any 32-bit int
085     * @return a pseudo-random int that uses at most the specified amount of bits
086     */
087    public final int compatibleNext( int bits ) {
088        long z = state += 0x9E3779B97F4A7C15L;
089        z = (z ^ (z >>> 30)) * 0xBF58476D1CE4E5B9L;
090        z = (z ^ (z >>> 27)) * 0x94D049BB133111EBL;
091        return (int)( (z ^ (z >>> 31)) & ( 1L << bits ) - 1 );
092    }

128    /**
129     * Returns a random non-negative integer between 0 (inclusive) and the given bound (exclusive),
130     * or 0 if the bound is 0. The bound can be negative, which will produce 0 or a negative result.
131     * Uses an aggressively optimized technique that has some bias, but mostly for values of
132     * bound over 1 billion. This method uses the same technique as {@link RNG#nextIntHasty(int)},
133     * and like that method will always advance state exactly once (equivalent to one call to
134     * {@link #nextLong()}).
135     * <br>
136     * Credit goes to Daniel Lemire, http://lemire.me/blog/2016/06/27/a-fast-alternative-to-the-modulo-reduction/
137     *
138     * @param bound the outer bound (exclusive), can be negative or positive
139     * @return the found number
140     */
141    public int nextInt( final int bound ) {
142        long z = state += 0x9E3779B97F4A7C15L;
143        z = (z ^ (z >>> 30)) * 0xBF58476D1CE4E5B9L;
144        z = (z ^ (z >>> 27)) * 0x94D049BB133111EBL;
145        return (int) ((bound * ((z ^ (z >>> 31)) & 0x7FFFFFFFL)) >> 31);
146    }

148    /**
149     * Like {@link #compatibleNext(int)}, but for compatibility with {@link #nextInt(int)}.
150     * Exclusive on the upper bound.  The lower bound is 0.
151     * @param bound the upper bound; should be positive
152     * @return a random int less than n and at least equal to 0
153     */
154    public int compatibleNextInt( final int bound ) {
155        if (bound <= 0) return 0;
156        int threshold = (0x7fffffff - bound + 1) % bound;
157        for (; ; ) {
158            int bits = (int) (nextLong() & 0x7fffffff);
159            if (bits >= threshold)
160                return bits % bound;
161        }
162    }
163    /**
164     * Inclusive lower, exclusive upper. Although you should usually use a higher value for upper than for lower, you
165     * can use a lower "upper" than "lower" and this will still work, producing an int between the two bounds.
166     * @param lower the lower bound, inclusive, can be positive or negative
167     * @param upper the upper bound, exclusive, can be positive or negative, usually should be greater than lower
168     * @return a random int between lower (inclusive) and upper (exclusive)
169     */
170    public int nextInt( final int lower, final int upper ) {
171        return lower + nextInt(upper - lower);
172    }

174    /**
175     * Inclusive lower, exclusive upper.
176     * @param lower the lower bound, inclusive, can be positive or negative
177     * @param upper the upper bound, exclusive, should be positive, must be greater than lower
178     * @return a random int between lower (inclusive) and upper (exclusive)
179     */
180    public int compatibleNextInt( final int lower, final int upper ) {
181        if ( upper - lower <= 0 ) throw new IllegalArgumentException("Upper bound must be greater than lower bound");
182        return lower + compatibleNextInt(upper - lower);
183    }

185    /**
186     * Exclusive on the outer bound; the inner bound is 0. The bound may be negative, which will produce a non-positive
187     * result.
188     * @param bound the outer exclusive bound; may be positive or negative
189     * @return a random long between 0 (inclusive) and bound (exclusive)
190     */
191    public long nextLong(long bound) {
192        long rand = nextLong();
193        final long randLow = rand & 0xFFFFFFFFL;
194        final long boundLow = bound & 0xFFFFFFFFL;
195        rand >>>= 32;
196        bound >>= 32;
197        final long z = (randLow * boundLow >> 32);
198        long t = rand * boundLow + z;
199        final long tLow = t & 0xFFFFFFFFL;
200        t >>>= 32;
201        return rand * bound + t + (tLow + randLow * bound >> 32) - (z >> 63) - (bound >> 63);
202    }
203    
204    /**
205     * Inclusive inner, exclusive outer; both inner and outer can be positive or negative.
206     * @param inner the inner bound, inclusive, can be positive or negative
207     * @param outer the outer bound, exclusive, can be positive or negative and may be greater than or less than inner
208     * @return a random long that may be equal to inner and will otherwise be between inner and outer
209     */
210    public long nextLong(final long inner, final long outer) {
211        return inner + nextLong(outer - inner);
212    }

215    /**
216     * Exclusive on the upper bound. The lower bound is 0. Unlike {@link #nextInt(int)} or {@link #nextLong(long)}, this
217     * may sometimes advance the state more than once, depending on what numbers are produced internally and the bound.
218     * {@link #nextLong(long)} is preferred because it is much faster and reliably advances the state only once. Because
219     * this method uses rejection sampling, getting multiple random longs to "smooth the odds" when the bound is such
220     * that it can't fairly distribute one random long across all possible outcomes, it may be more "fair" than
221     * {@link #nextLong(long)}, though it could potentially consume more of the period faster if pathologically bad
222     * bounds were used very often, and if enough of the period is gone then statistical flaws may become detectable.
223     * @param bound the upper bound; if this isn't positive, this method always returns 0
224     * @return a random long less than n and at least equal to 0
225     */
226    public long compatibleNextLong(final long bound) {
227        if ( bound <= 0 ) return 0;
228        long threshold = (0x7fffffffffffffffL - bound + 1) % bound;
229        for (;;) {
230            long bits = nextLong() & 0x7fffffffffffffffL;
231            if (bits >= threshold)
232                return bits % bound;
233        }
234    }

237     * Inclusive lower, exclusive upper.
238     * @param lower the lower bound, inclusive, can be positive or negative
239     * @param upper the upper bound, exclusive, should be positive, must be greater than lower
240     * @return a random long at least equal to lower and less than upper
241     */
242    public long compatibleNextLong( final long lower, final long upper ) {
243        if ( upper - lower <= 0 )  throw new IllegalArgumentException("Upper bound must be greater than lower bound");
244        return lower + compatibleNextLong(upper - lower);

259     * is negative, it will be the (exclusive) lower bound and 0.0 will be the (inclusive) upper bound.

279    /**
280     * Gets a random value, true or false.
281     * Calls nextLong() once.
282     * @return a random true or false value.
283     */
284    public boolean nextBoolean() {
285        return nextLong() < 0L;
286    }

288    /**
289     * Given a byte array as a parameter, this will fill the array with random bytes (modifying it
290     * in-place). Calls nextLong() {@code Math.ceil(bytes.length / 8.0)} times.
291     * @param bytes a byte array that will have its contents overwritten with random bytes.
292     */
293    public void nextBytes( final byte[] bytes ) {
294        int i = bytes.length, n;
295        while( i != 0 ) {
296            n = Math.min( i, 8 );
297            for ( long bits = nextLong(); n-- != 0; bits >>>= 8 ) bytes[ --i ] = (byte)bits;
298        }
299    }

327    /**
328     * Advances or rolls back the LightRNG's state without actually generating each number. Skips forward
329     * or backward a number of steps specified by advance, where a step is equal to one call to nextLong(),
330     * and returns the random number produced at that step (you can get the state with {@link #getState()}).
331     *
332     * @param advance Number of future generations to skip over; can be negative to backtrack, 0 gets the most recent generated number
333     * @return the random long generated after skipping advance numbers
334     */
335    @Override
336    public long skip(long advance) {
337        long z = (state += 0x9E3779B97F4A7C15L * advance);
338        z = (z ^ (z >>> 30)) * 0xBF58476D1CE4E5B9L;
339        z = (z ^ (z >>> 27)) * 0x94D049BB133111EBL;
340        return z ^ (z >>> 31);
341    }

365    {
366        return ((state = ((state = ((state *= 0x9E3779B97F4A7C15L) ^ state >>> 30) * 0xBF58476D1CE4E5B9L) ^ state >>> 27) * 0x94D049BB133111EBL) ^ state >>> 31);

368    public static long determine(final int a, final int b)
369    {
370        long state = 0x9E3779B97F4A7C15L + (a & 0xFFFFFFFFL) + ((long)b << 32);
371        state = ((state >>> 30) ^ state) * 0xBF58476D1CE4E5B9L;
372        state = (state ^ (state >>> 27)) * 0x94D049BB133111EBL;
373        return state ^ (state >>> 31);
374    }

382     * Returns a random float that is deterministic based on state; if state is the same on two calls to this, this will
383     * return the same float. This is expected to be called with a changing variable, e.g. {@code determine(++state)},
384     * where the increment for state should be odd but otherwise doesn't really matter. This multiplies state by
385     * {@code 0x9E3779B97F4A7C15L} within this method, so using a small increment won't be much different from using a
386     * very large one, as long as it is odd. The period is 2 to the 64 if you increment or decrement by 1, but there are
387     * only 2 to the 30 possible floats between 0 and 1.
388     * @param state a variable that should be different every time you want a different random result;
389     *              using {@code determine(++state)} is recommended to go forwards or {@code determine(--state)} to
390     *              generate numbers in reverse order
391     * @return a pseudo-random float between 0f (inclusive) and 1f (exclusive), determined by {@code state}
392     */
