033/**
034 * A Java implementation of an incremental 2D Delaunay triangulation algorithm.
035 * This is a port of <a href="https://github.com/jdiemke/delaunay-triangulator">Johannes Diemke's code</a>, with
036 * some substantial non-algorithmic changes to better work in SquidLib and to reduce allocations. You should consider
037 * using the {@code com.badlogic.gdx.math.DelaunayTriangulator} class if you use libGDX, which allocates fewer objects.
038 * @author Johannes Diemke
039 * @author Tommy Ettinger
040 */
041@Beta
042public class DelaunayTriangulator implements Serializable {
043    private static final long serialVersionUID = 1L;
044
045    private CoordDouble[] points;
046    private ArrayList<Triangle> triangleSoup;
047
048    /**
049     * Constructs a triangulator instance but does not insert any points; you should add points to
050     * {@link #getPoints()}, which is an array that can hold 256 points, before running {@link #triangulate()}.
051     */
052    public DelaunayTriangulator() {
053        this.points = new CoordDouble[256];
054        this.triangleSoup = new ArrayList<>(256);
055    }
056
057    /**
058     * Constructs a new triangulator instance using the specified point set.
059     *
060     * @param points The point set to be triangulated
061     */
062    public DelaunayTriangulator(Collection<CoordDouble> points) {
063        this.points = points.toArray(new CoordDouble[points.size()]);
064        this.triangleSoup = new ArrayList<>(points.size());
065    }
066    /**
067     * Returns the triangle from this triangle soup that contains the specified
068     * point or null if no triangle from the triangle soup contains the point.
069     *
070     * @param point
071     *            The point
072     * @return Returns the triangle from this triangle soup that contains the
073     *         specified point or null
074     */
075    public Triangle findContainingTriangle(CoordDouble point) {
076        for (Triangle triangle : triangleSoup) {
077            if (triangle.contains(point)) {
078                return triangle;
079            }
080        }
081        return null;
082    }
083    /**
084     * Returns the neighbor triangle of the specified triangle sharing the same
085     * edge as specified. If no neighbor sharing the same edge exists null is
086     * returned.
087     *
088     * @param triangle
089     *            The triangle
090     * @param edge
091     *            The edge
092     * @return The triangles neighbor triangle sharing the same edge or null if
093     *         no triangle exists
094     */
095    public Triangle findNeighbor(Triangle triangle, Edge edge) {
096        for (Triangle triangleFromSoup : triangleSoup) {
097            if (triangleFromSoup.isNeighbor(edge) && triangleFromSoup != triangle) {
098                return triangleFromSoup;
099            }
100        }
101        return null;
102    }
103    public Triangle findNeighbor(Triangle triangle, CoordDouble ea, CoordDouble eb) {
104        for (Triangle triangleFromSoup : triangleSoup) {
105            if (triangleFromSoup.isNeighbor(ea, eb) && triangleFromSoup != triangle) {
106                return triangleFromSoup;
107            }
108        }
109        return null;
110    }
111
112    /**
113     * Returns one of the possible triangles sharing the specified edge. Based
114     * on the ordering of the triangles in this triangle soup the returned
115     * triangle may differ. To find the other triangle that shares this edge use
116     * the {@link #findNeighbor(Triangle, Edge)} method.
117     *
118     * @param edge
119     *            The edge
120     * @return Returns one triangle that shares the specified edge
121     */
122    public Triangle findOneTriangleSharing(Edge edge) {
123        for (Triangle triangle : triangleSoup) {
124            if (triangle.isNeighbor(edge)) {
125                return triangle;
126            }
127        }
128        return null;
129    }
130
131    /**
132     * Returns the edge from the triangle soup nearest to the specified point.
133     *
134     * @param point
135     *            The point
136     * @return The edge from the triangle soup nearest to the specified point
137     */
138    public Edge findNearestEdge(CoordDouble point) {
139//            List<EdgeDistancePack> edgeList = new ArrayList<EdgeDistancePack>();
140
141        OrderedMap<Edge, Double> edges = new OrderedMap<>(triangleSoup.size());
142        for (Triangle triangle : triangleSoup) {
143            Edge ab = new Edge(triangle.a, triangle.b);
144            Edge bc = new Edge(triangle.b, triangle.c);
145            Edge ca = new Edge(triangle.c, triangle.a);
146            double abd = triangle.computeClosestPoint(ab, point).subtract(point).lengthSq();
147            double bcd = triangle.computeClosestPoint(bc, point).subtract(point).lengthSq();
148            double cad = triangle.computeClosestPoint(ca, point).subtract(point).lengthSq();
149
150            if(abd <= bcd && abd <= cad)
151                edges.put(ab, abd);
152            else if(bcd <= abd && bcd <= cad)
153                edges.put(bc, bcd);
154            else
155                edges.put(ca, cad);
156        }
157        edges.sortByValue(doubleComparator);
158        return edges.keyAt(0);
159    }
160
161    /**
162     * Removes all triangles from this triangle soup that contain the specified
163     * vertex.
164     *
165     * @param vertex
166     *            The vertex
167     */
168    public void removeTrianglesUsing(CoordDouble vertex) {
169        ListIterator<Triangle> li = triangleSoup.listIterator();
170        while (li.hasNext())
171        {
172            Triangle triangle = li.next();
173            if(triangle.hasVertex(vertex))
174                li.remove();
175        }
176    }
177
178    /**
179     * This method generates a Delaunay triangulation from the specified point
180     * set.
181     */
182    public ArrayList<Triangle> triangulate() {
183        int numPoints;
184        if (points == null || (numPoints = points.length) < 3) {
185            throw new IllegalArgumentException("Less than three points in point set.");
186        }
187        triangleSoup.clear();
188
189        /*
190         * In order for the in circumcircle test to not consider the vertices of
191         * the super triangle we have to start out with a big triangle
192         * containing the whole point set. We have to scale the super triangle
193         * to be very large. Otherwise the triangulation is not convex.
194         */
195        double maxOfAnyCoordinate = 0.0;
196
197        for (int i = 0; i < numPoints; i++) {
198            final CoordDouble pt = points[i];
199            maxOfAnyCoordinate = Math.max(Math.max(pt.x, pt.y), maxOfAnyCoordinate);
200        }
201
202        maxOfAnyCoordinate *= 48.0;
203
204        CoordDouble p1 = new CoordDouble(0.0, maxOfAnyCoordinate);
205        CoordDouble p2 = new CoordDouble(maxOfAnyCoordinate, 0.0);
206        CoordDouble p3 = new CoordDouble(-maxOfAnyCoordinate, -maxOfAnyCoordinate);
207
208        Triangle superTriangle = new Triangle(p1, p2, p3);
209
210        triangleSoup.add(superTriangle);
211
212        for (int i = 0; i < points.length; i++) {
213            Triangle triangle = findContainingTriangle(points[i]);
214
215            if (triangle == null) {
216                /*
217                  If no containing triangle exists, then the vertex is not
218                  inside a triangle (this can also happen due to numerical
219                  errors) and lies on an edge. In order to find this edge we
220                  search all edges of the triangle soup and select the one
221                  which is nearest to the point we try to add. This edge is
222                  removed and four new edges are added.
223                 */
224                Edge edge = findNearestEdge(points[i]);
225
226                Triangle first = findOneTriangleSharing(edge);
227                Triangle second = findNeighbor(first, edge);
228
229                CoordDouble firstNonEdgeVertex = first.getNonEdgeVertex(edge);
230                CoordDouble secondNonEdgeVertex = second.getNonEdgeVertex(edge);
231                CoordDouble p = points[i];
232                
233                triangleSoup.remove(first);
234                triangleSoup.remove(second);
235
236                Triangle triangle1 = new Triangle(edge.a, firstNonEdgeVertex,  p);
237                Triangle triangle2 = new Triangle(edge.b, firstNonEdgeVertex,  p);
238                Triangle triangle3 = new Triangle(edge.a, secondNonEdgeVertex, p);
239                Triangle triangle4 = new Triangle(edge.b, secondNonEdgeVertex, p);
240
241                triangleSoup.add(triangle1);
242                triangleSoup.add(triangle2);
243                triangleSoup.add(triangle3);
244                triangleSoup.add(triangle4);
245
246                legalizeEdge(triangle1, edge.a, firstNonEdgeVertex,  p);
247                legalizeEdge(triangle2, edge.b, firstNonEdgeVertex,  p);
248                legalizeEdge(triangle3, edge.a, secondNonEdgeVertex, p);
249                legalizeEdge(triangle4, edge.b, secondNonEdgeVertex, p);
250            } else {
251                /*
252                 * The vertex is inside a triangle.
253                 */
254                CoordDouble a = triangle.a;
255                CoordDouble b = triangle.b;
256                CoordDouble c = triangle.c;
257                CoordDouble p = points[i];
258                
259                triangleSoup.remove(triangle);
260
261                Triangle first = new Triangle(a, b, p);
262                Triangle second = new Triangle(b, c, p);
263                Triangle third = new Triangle(c, a, p);
264
265                triangleSoup.add(first);
266                triangleSoup.add(second);
267                triangleSoup.add(third);
268
269                legalizeEdge(first,  a, b, p);
270                legalizeEdge(second, b, c, p);
271                legalizeEdge(third,  c, a, p);
272            }
273        }
274
275        /*
276         * Remove all triangles that contain vertices of the super triangle.
277         */
278        removeTrianglesUsing(superTriangle.a);
279        removeTrianglesUsing(superTriangle.b);
280        removeTrianglesUsing(superTriangle.c);
281        return triangleSoup;
282    }
283
284    /**
285     * This method legalizes edges by recursively flipping all illegal edges.
286     * 
287     * @param triangle
288     *            The triangle
289     * @param ea
290     *            The "a" point on the edge to be legalized
291     * @param eb
292     *            The "b" point on the edge to be legalized
293     * @param newVertex
294     *            The new vertex
295     */
296    private void legalizeEdge(Triangle triangle, CoordDouble ea, CoordDouble eb, CoordDouble newVertex) {
297        Triangle neighborTriangle = findNeighbor(triangle, ea, eb);
298
299        /*
300          If the triangle has a neighbor, then legalize the edge
301         */
302        if (neighborTriangle != null) {
303            if (neighborTriangle.isPointInCircumcircle(newVertex)) {
304                triangleSoup.remove(triangle);
305                triangleSoup.remove(neighborTriangle);
306
307                CoordDouble noneEdgeVertex = neighborTriangle.getNonEdgeVertex(ea, eb);
308
309                Triangle firstTriangle = new Triangle(noneEdgeVertex, ea, newVertex);
310                Triangle secondTriangle = new Triangle(noneEdgeVertex, eb, newVertex);
311
312                triangleSoup.add(firstTriangle);
313                triangleSoup.add(secondTriangle);
314
315                legalizeEdge(firstTriangle, noneEdgeVertex, ea, newVertex);
316                legalizeEdge(secondTriangle, noneEdgeVertex, eb, newVertex);
317            }
318        }
319    }

322     * Creates a random permutation of the specified point set. Based on the
323     * implementation of the Delaunay algorithm this can speed up the
324     * computation.
325     */

327        rng.shuffleInPlace(points);

349    public static class Edge implements Serializable
350    {
351        private static final long serialVersionUID = 1L;
352        
353        public CoordDouble a;
354        public CoordDouble b;
355        
356        public Edge()
357        {
358            a = new CoordDouble(0.0, 0.0);
359            b = new CoordDouble(0.0, 1.0);
360        }
361        public Edge(CoordDouble a, CoordDouble b)
362        {
363            this.a = a;
364            this.b = b;
365        }
366        public Edge(double ax, double ay, double bx, double by)
367        {
368            a = new CoordDouble(ax, ay);
369            b = new CoordDouble(bx, by);
370        }
371    }

373    public static class Triangle implements Serializable
374    {
375        private static final long serialVersionUID = 1L;
376
377        public CoordDouble a;
378        public CoordDouble b;
379        public CoordDouble c;
380
381        /**
382         * Constructor of the 2D triangle class used to create a new triangle
383         * instance from three 2D vectors describing the triangle's vertices.
384         *
385         * @param a
386         *            The first vertex of the triangle
387         * @param b
388         *            The second vertex of the triangle
389         * @param c
390         *            The third vertex of the triangle
391         */
392        public Triangle(CoordDouble a, CoordDouble b, CoordDouble c) {
393            this.a = a;
394            this.b = b;
395            this.c = c;
396        }
397
398        /**
399         * Tests if a 2D point lies inside this 2D triangle. See Real-Time Collision
400         * Detection, chap. 5, p. 206.
401         *
402         * @param point
403         *            The point to be tested
404         * @return Returns true iff the point lies inside this 2D triangle
405         */
406        public boolean contains(CoordDouble point) {
407            double px = point.x - a.x;
408            double py = point.y - a.y;
409            double sx = b.x - a.x;
410            double sy = b.y - a.y;
411            double pab = py * sx - px * sy;//point.sub(a).cross(b.sub(a));
412            px = point.x - b.x;
413            py = point.y - b.y;
414            sx = c.x - b.x;
415            sy = c.y - b.y;
416            double pbc = py * sx - px * sy;//point.sub(b).cross(c.sub(b));
417
418            if (!hasSameSign(pab, pbc)) {
419                return false;
420            }
421
422            px = point.x - c.x;
423            py = point.y - c.y;
424            sx = a.x - c.x;
425            sy = a.y - c.y;
426            double pca = py * sx - px * sy;//point.sub(c).cross(a.sub(c));
427
428            return hasSameSign(pab, pca);
429
430        }
431
432        /**
433         * Tests if a given point lies in the circumcircle of this triangle. Let the
434         * triangle ABC appear in counterclockwise (CCW) order. Then when det &gt;
435         * 0, the point lies inside the circumcircle through the three points a, b
436         * and c. If instead det &lt; 0, the point lies outside the circumcircle.
437         * When det = 0, the four points are cocircular. If the triangle is oriented
438         * clockwise (CW) the result is reversed. See Real-Time Collision Detection,
439         * chap. 3, p. 34.
440         *
441         * @param point
442         *            The point to be tested
443         * @return Returns true iff the point lies inside the circumcircle through
444         *         the three points a, b, and c of the triangle
445         */
446        public boolean isPointInCircumcircle(CoordDouble point) {
447            double a11 = a.x - point.x;
448            double a21 = b.x - point.x;
449            double a31 = c.x - point.x;
450
451            double a12 = a.y - point.y;
452            double a22 = b.y - point.y;
453            double a32 = c.y - point.y;
454
455            double a13 = (a.x - point.x) * (a.x - point.x) + (a.y - point.y) * (a.y - point.y);
456            double a23 = (b.x - point.x) * (b.x - point.x) + (b.y - point.y) * (b.y - point.y);
457            double a33 = (c.x - point.x) * (c.x - point.x) + (c.y - point.y) * (c.y - point.y);
458
459            double det = a11 * a22 * a33 + a12 * a23 * a31 + a13 * a21 * a32 - a13 * a22 * a31 - a12 * a21 * a33
460                    - a11 * a23 * a32;
461
462            if (isOrientedCCW()) {
463                return det > 0.0;
464            }
465
466            return det < 0.0;
467        }
468
469        /**
470         * Test if this triangle is oriented counterclockwise (CCW). Let A, B and C
471         * be three 2D points. If det &gt; 0, C lies to the left of the directed
472         * line AB. Equivalently the triangle ABC is oriented counterclockwise. When
473         * det &lt; 0, C lies to the right of the directed line AB, and the triangle
474         * ABC is oriented clockwise. When det = 0, the three points are colinear.
475         * See Real-Time Collision Detection, chap. 3, p. 32
476         *
477         * @return Returns true iff the triangle ABC is oriented counterclockwise
478         *         (CCW)
479         */
480        public boolean isOrientedCCW() {
481            double a11 = a.x - c.x;
482            double a21 = b.x - c.x;
483
484            double a12 = a.y - c.y;
485            double a22 = b.y - c.y;
486
487            double det = a11 * a22 - a12 * a21;
488
489            return det > 0.0;
490        }
491
492        /**
493         * Returns true if this triangle contains the given edge.
494         *
495         * @param edge
496         *            The edge to be tested
497         * @return Returns true if this triangle contains the edge
498         */
499        public boolean isNeighbor(Edge edge) {
500            return (a == edge.a || b == edge.a || c == edge.a) && (a == edge.b || b == edge.b || c == edge.b);
501        }
502        public boolean isNeighbor(CoordDouble ea, CoordDouble eb) {
503            return (a == ea || b == ea || c == ea) && (a == eb || b == eb || c == eb);
504        }
505
506        /**
507         * Returns the vertex of this triangle that is not part of the given edge.
508         *
509         * @param edge
510         *            The edge
511         * @return The vertex of this triangle that is not part of the edge
512         */
513        public CoordDouble getNonEdgeVertex(Edge edge) {
514            if (a != edge.a && a != edge.b) {
515                return a;
516            } else if (b != edge.a && b != edge.b) {
517                return b;
518            } else if (c != edge.a && c != edge.b) {
519                return c;
520            }
521
522            return null;
523        }
524        public CoordDouble getNonEdgeVertex(CoordDouble ea, CoordDouble eb) {
525            if (a != ea && a != eb) {
526                return a;
527            } else if (b != ea && b != eb) {
528                return b;
529            } else if (c != ea && c != eb) {
530                return c;
531            }
532
533            return null;
534        }
535
536        /**
537         * Returns true if the given vertex is one of the vertices describing this
538         * triangle.
539         *
540         * @param vertex
541         *            The vertex to be tested
542         * @return Returns true if the Vertex is one of the vertices describing this
543         *         triangle
544         */
545        public boolean hasVertex(CoordDouble vertex) {
546            if (a == vertex || b == vertex || c == vertex) {
547                return true;
548            }
549
550            return false;
551        }
552
553        /**
554         * Computes the closest point on the given edge to the specified point.
555         *
556         * @param edge
557         *            The edge on which we search the closest point to the specified
558         *            point
559         * @param point
560         *            The point to which we search the closest point on the edge
561         * @return The closest point on the given edge to the specified point
562         */
563        CoordDouble computeClosestPoint(Edge edge, CoordDouble point) {
564            //CoordDouble ab = edge.b.sub(edge.a);
565            double abx = edge.b.x = edge.a.x;
566            double aby = edge.b.y = edge.a.y;
567            double t = ((point.x - edge.a.x) * abx + (point.y - edge.a.y) * aby) / (abx * abx + aby * aby);
568            //double t = point.sub(edge.a).dot(ab) / ab.dot(ab);
569
570            if (t < 0.0d) {
571                t = 0.0d;
572            } else if (t > 1.0d) {
573                t = 1.0d;
574            }
575            return new CoordDouble(edge.a.x + abx * t, edge.a.y + aby * t);
576//            return edge.a.add(ab.mult(t));
577        }
578
579        /**
580         * Tests if the two arguments have the same sign.
581         *
582         * @param a
583         *            The first floating point argument
584         * @param b
585         *            The second floating point argument
586         * @return Returns true iff both arguments have the same sign
587         */
588        private boolean hasSameSign(double a, double b) {
589            return (a == 0.0 && b == 0.0) || ((a > 0.0) == (b > 0.0));
590        }
591
592        @Override
593        public String toString() {
594            return "Triangle[" + a + ", " + b + ", " + c + "]";
595        }
596    }
597    private static final Comparator<Double> doubleComparator = new Comparator<Double>() {
598        @Override
599        public int compare(Double o1, Double o2) {
600            return o1.compareTo(o2);
601        }
602    };
