005/**
006 * One of Mark Overton's subcycle generators from <a href="http://www.drdobbs.com/tools/229625477">this article</a>,
007 * specifically a cmr^cmr with two 32-bit states; this is the fastest 32-bit generator that still passes statistical
008 * tests, plus it's optimized for GWT (sometimes). It has a period of just under 2 to the 64, 0xFFF1F6F18B2A1330, which
009 * is roughly 2 to the 63.999691, and allows 2 to the 32 initial seeds.
010 * <br>
011 * This seems to do well in PractRand testing (32 TB passed), but this is not a generator Overton tested. "Chaotic"
012 * generators like this one tend to score well in PractRand, but it isn't clear if they will fail other tests (in
013 * particular, they can't generate all possible long values, and also can't generate 0 or possibly some other ints). As
014 * for desktop/server speed, this is faster than {@link Lathe32RNG} (which is also high-quality) and is also faster than
015 * {@link Starfish32RNG} (which is very fast but has quality issues) and {@link ThrustAlt32RNG} (which has a very small
016 * period and probably isn't very useful). However, this is slower than Lathe32RNG when using GWT and viewing in
017 * Firefox; for some reason {@link Starfish32RNG} optimizes well on Firefox and less well on Chrome, but Mover does very
018 * well in older Chrome (faster than Lathe) and rather poorly on Firefox. It doesn't do amazingly well in current Chrome
019 * versions, however, and Lathe beats it most of the time there. On 64-bit desktop or server Java, you may want to
020 * prefer {@link Mover64RNG}, which is the same algorithm using larger words and constants. While each of the two parts
021 * of a Mover32RNG can have their full period evaluated, making the total period possible to calculate, the same cannot
022 * be said for Mover64RNG (its period is high enough for most usage, but the actual total is still unknown).
023 * <br>
024 * Its period is 0xFFF1F6F18B2A1330 for the largest cycle, which it always initializes into if {@link #setState(int)} is
025 * used. setState() only allows 2 to the 32 starting states, but less than 2 to the 64 states are in the largest cycle,
026 * so using a long or two ints to set the state seems ill-advised. The generator has two similar parts, each updated
027 * without needing to read from the other part. Each is a 32-bit CMR generator, which multiplies a state by a constant,
028 * rotates by another constant, and stores that as the next state. The particular constants used here were found by
029 * randomly picking 16-bit odd numbers as multipliers, checking the period for every non-zero rotation, and reporting
030 * the multiplier and rotation amount when a period was found that was greater than 0xFF000000. Better multipliers are
031 * almost guaranteed to exist, but finding them would be a challenge.
032 * <br>
033 * This is a RandomnessSource but not a StatefulRandomness because it needs to take care and avoid seeds that would put
034 * it in a short-period subcycle. It uses two generators with different cycle lengths, and skips at most 65536 times
035 * into each generator's cycle independently when seeding. It uses constants to store 128 known midpoints for each
036 * generator, which ensures it calculates an advance for each generator at most 511 times. 
037 * <br>
038 * The name comes from M. Overton, who discovered this category of subcycle generators, and also how this generator can
039 * really move when it comes to speed.
040 * <br>
041 * Created by Tommy Ettinger on 8/6/2018.
042 * @author Mark Overton
043 * @author Tommy Ettinger
044 */
045public final class Mover32RNG implements RandomnessSource {
046    private int stateA, stateB;
047    public Mover32RNG()
048    {
049        setState((int)((Math.random() * 2.0 - 1.0) * 0x80000000));
050    }
051    public Mover32RNG(final int state)
052    {
053        setState(state);
054    }

060     * @param stateB
061     */

063    {
064        this.stateA = stateA == 0 ? 1 : stateA;
065        this.stateB = stateB == 0 ? 1 : stateB;
066    }

104    public final void setState(final int s) {
105        stateA = startingA[s >>> 9 & 0x7F];
106        for (int i = s & 0x1FF; i > 0; i--) {
107            stateA *= 0x89A7;
108            stateA = (stateA << 13 | stateA >>> 19);
109        }
110        stateB = startingB[s >>> 25];
111        for (int i = s >>> 16 & 0x1FF; i > 0; i--) {
112            stateB *= 0xBCFD;
113            stateB = (stateB << 17 | stateB >>> 15);
114        }
115    }
116
117    public final int nextInt()
118    {
119        int y = stateA * 0x89A7;
120        stateA = (y = (y << 13 | y >>> 19));
121        final int x = stateB * 0xBCFD;
122        return (y ^ (stateB = (x << 17 | x >>> 15)));
123    }
124    @Override
125    public final int next(final int bits)
126    {
127        int y = stateA * 0x89A7;
128        stateA = (y = (y << 13 | y >>> 19));
129        final int x = stateB * 0xBCFD;
130        return (y ^ (stateB = (x << 17 | x >>> 15))) >>> (32 - bits);
131    }
132    @Override
133    public final long nextLong()
134    {
135        int y = stateA * 0x89A7;
136        y = (y << 13 | y >>> 19);
137        int x = stateB * 0xBCFD;
138        final long t = (y ^ (x = (x << 17 | x >>> 15))) & 0xFFFFFFFFL;
139        y *= 0x89A7;
140        stateA = (y = (y << 13 | y >>> 19));
141        x *= 0xBCFD;
142        return t << 32 | ((y ^ (stateB = (x << 17 | x >>> 15))) & 0xFFFFFFFFL);
143    }

145    /**
146     * Produces a copy of this Mover32RNG that, if next() and/or nextLong() are called on this object and the
147     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just need to
148     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
149     *
150     * @return a copy of this Mover32RNG
151     */
152    @Override
153    public Mover32RNG copy() {
154        return new Mover32RNG(stateA, stateB);
155    }

157    /**
158     * Gets the "A" part of the state; if this generator was set with {@link #Mover32RNG()}, {@link #Mover32RNG(int)},

160     * @return the "A" part of the state, an int
161     */
162    public int getStateA()
163    {
164        return stateA;
165    }

174        return stateB;

196    @Override
197    public String toString() {
198        return "Mover32RNG with stateA 0x" + StringKit.hex(stateA) + " and stateB 0x" + StringKit.hex(stateB);
199    }

201    @Override
202    public boolean equals(Object o) {
203        if (this == o) return true;
204        if (o == null || getClass() != o.getClass()) return false;
205
206        Mover32RNG mover32RNG = (Mover32RNG) o;
207
208        return stateA == mover32RNG.stateA && stateB == mover32RNG.stateB;
209    }
210
211    @Override
212    public int hashCode() {
213        return 31 * stateA + stateB | 0;
214    }
