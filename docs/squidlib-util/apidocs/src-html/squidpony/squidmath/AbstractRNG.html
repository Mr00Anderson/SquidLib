011/**
012 * A helper class for implementing {@link IRNG} without so much busy-work.
013 * You need to provide implementations for the abstract methods {@link #nextInt()}, {@link #next(int)},
014 * {@link #nextLong()}, {@link #nextBoolean()}, {@link #nextDouble()}, {@link #nextFloat()}, {@link #copy()}, and
015 * {@link #toSerializable()}, many of which may be built off of each other and some of which have sample code in their
016 * documentation strings in this class.
017 * <br>
018 * Big thanks to smelc for the concept in his hgameshrogue library.
019 * @author Tommy Ettinger
020 * @author smelC
021 */
022public abstract class AbstractRNG implements IRNG {
023    /**
024     * Get up to 32 bits (inclusive) of random output; the int this produces
025     * will not require more than {@code bits} bits to represent.
026     *
027     * @param bits an int between 1 and 32, both inclusive
028     * @return a random number that fits in the specified number of bits
029     */
030    @Override
031    public abstract int next(int bits);
032    /**
033     * Get a random integer between Integer.MIN_VALUE to Integer.MAX_VALUE (both inclusive).
034     *
035     * @return a 32-bit random int.
036     */
037    @Override
038    public abstract int nextInt();
039
040    /**
041     * Returns a random non-negative integer below the given bound, or 0 if the bound is 0 or
042     * negative.
043     *
044     * @param bound the upper bound (exclusive)
045     * @return the found number
046     */
047    @Override
048    public int nextInt(int bound) {
049        return (int) ((bound * ((long)next(31))) >>> 31) & ~(bound >> 31);
050    }
051
052    /**
053     * Get a random long between Long.MIN_VALUE to Long.MAX_VALUE (both inclusive).
054     *
055     * @return a 64-bit random long.
056     */
057    @Override
058    public abstract long nextLong();
059
060    /**
061     * Exclusive on bound (which must be positive), with an inner bound of 0.
062     * If bound is negative or 0 this always returns 0.
063     * <br>
064     * Credit for this method goes to <a href="https://oroboro.com/large-random-in-range/">Rafael Baptista's blog</a>,
065     * with some adaptation for signed long values and a 64-bit generator. This method is drastically faster than the
066     * previous implementation when the bound varies often (roughly 4x faster, possibly more). It also always gets
067     * exactly one random number, so it advances the state as much as {@link #nextInt(int)}.
068     * @param bound the outer exclusive bound; should be positive, otherwise this always returns 0L
069     * @return a random long between 0 (inclusive) and bound (exclusive)
070     */
071    @Override
072    public long nextLong(long bound) {
073        long rand = nextLong();
074        if (bound <= 0) return 0;
075        final long randLow = rand & 0xFFFFFFFFL;
076        final long boundLow = bound & 0xFFFFFFFFL;
077        rand >>>= 32;
078        bound >>= 32;
079        final long z = (randLow * boundLow >> 32);
080        final long t = rand * boundLow + z;
081        return rand * bound + (t >> 32) + ((t & 0xFFFFFFFFL) + randLow * bound >> 32) - (z >> 63);
082    }

084    /**
085     * Get a random bit of state, interpreted as true or false with approximately equal likelihood.
086     * <br>
087     * Note: This is abstract because some implementations may be best served by using {@link #next(int)} to get 1 bit,
088     * returning {@code next(1) == 1}, but others will get much better results with a sign check by calling their choice
089     * of {@link #nextInt()} or {@link #nextLong()} and returning {@code nextInt() < 0} or {@code nextLong < 0L}. For
090     * example, an implementation that uses a linear congruential generator without truncating some lower bits will have
091     * very-low-period results for the bottom bit (alternating true and false), but perfectly fine results from a sign
092     * check. There are tested issues on the bottom (at least 2) bits of {@link XoRoRNG}, but again not on a sign check.
093     * @return a random boolean.
094     */
095    @Override
096    public abstract boolean nextBoolean();
097    /**
098     * Gets a random double between 0.0 inclusive and 1.0 exclusive.
099     * This returns a maximum of 0.9999999999999999 because that is the largest double value that is less than 1.0 .
100     * <br>
101     * This is abstract because some generators may natively work with double or float values, but others may need to
102     * convert a long to a double as with {@code (nextLong() & 0x1fffffffffffffL) * 0x1p-53}, which is recommended if
103     * longs are fast to produce.
104     * @return a double between 0.0 (inclusive) and 0.9999999999999999 (inclusive)
105     */
106    @Override
107    public abstract double nextDouble();
108    /**
109     * This returns a random double between 0.0 (inclusive) and outer (exclusive). The value for outer can be positive
110     * or negative. Because of how math on doubles works, there are at most 2 to the 53 values this can return for any
111     * given outer bound, and very large values for outer will not necessarily produce all numbers you might expect.
112     *
113     * @param outer the outer exclusive bound as a double; can be negative or positive
114     * @return a double between 0.0 (inclusive) and outer (exclusive)
115     */
116    @Override
117    public double nextDouble(double outer) {
118        return nextDouble() * outer;
119    }
120
121    /**
122     * Gets a random float between 0.0f inclusive and 1.0f exclusive.
123     * This returns a maximum of 0.99999994 because that is the largest float value that is less than 1.0f .
124     * <br>
125     * This is abstract because some generators may natively work with double or float values, but others may need to
126     * convert an int or long to a float as with {@code (nextInt() & 0xffffff) * 0x1p-24f},
127     * {@code (nextLong() & 0xffffffL) * 0x1p-24f}, or {@code next(24) * 0x1p-24f}, any of which can work when the
128     * method they call is high-quality and fast. You probably would want to use nextInt() or next() if your
129     * implementation is natively 32-bit and is slower at producing longs, for example.
130     * @return a float between 0f (inclusive) and 0.99999994f (inclusive)
131     */
132    @Override
133    public abstract float nextFloat();
134    /**
135     * This returns a random float between 0.0f (inclusive) and outer (exclusive). The value for outer can be positive
136     * or negative. Because of how math on floats works, there are at most 2 to the 24 values this can return for any
137     * given outer bound, and very large values for outer will not necessarily produce all numbers you might expect.
138     *
139     * @param outer the outer exclusive bound as a float; can be negative or positive
140     * @return a float between 0f (inclusive) and outer (exclusive)
141     */
142    @Override
143    public float nextFloat(float outer) {
144        return nextFloat() * outer;
145    }
146    /**
147     * Exclusive on bound (which may be positive or negative), with an inner bound of 0.
148     * If bound is negative this returns a negative long; if bound is positive this returns a positive long. The bound
149     * can even be 0, which will cause this to return 0L every time. This uses a biased technique to get numbers from
150     * large ranges, but the amount of bias is incredibly small (expected to be under 1/1000 if enough random ranged
151     * numbers are requested, which is about the same as an unbiased method that was also considered). It may have
152     * noticeable bias if the generator's period is exhausted by only calls to this method. Unlike all unbiased methods,
153     * this advances the state by an equivalent to exactly one call to {@link #nextLong()}, where rejection sampling
154     * would sometimes advance by one call, but other times by arbitrarily many more.
155     * <br>
156     * Credit for this method goes to <a href="https://oroboro.com/large-random-in-range/">Rafael Baptista's blog</a>,
157     * with some adaptation for signed long values and a 64-bit generator. This method is drastically faster than the
158     * previous implementation when the bound varies often (roughly 4x faster, possibly more). It also always gets at
159     * most one random number, so it advances the state as much as {@link #nextInt(int)} or {@link #nextLong()}.
160     * @param bound the outer exclusive bound; can be positive or negative
161     * @return a random long between 0 (inclusive) and bound (exclusive)
162     */
163    @Override
164    public long nextSignedLong(long bound) {
165        long rand = nextLong();
166        final long randLow = rand & 0xFFFFFFFFL;
167        final long boundLow = bound & 0xFFFFFFFFL;
168        rand >>>= 32;
169        bound >>= 32;
170        final long z = (randLow * boundLow >> 32);
171        final long t = rand * boundLow + z;
172        return rand * bound + (t >> 32) + ((t & 0xFFFFFFFFL) + randLow * bound >> 32) - (z >> 63);
173    }

175    /**
176     * Returns a random non-negative integer between 0 (inclusive) and the given bound (exclusive),
177     * or 0 if the bound is 0. The bound can be negative, which will produce 0 or a negative result.
178     * <br>
179     * Credit goes to Daniel Lemire, http://lemire.me/blog/2016/06/27/a-fast-alternative-to-the-modulo-reduction/
180     *
181     * @param bound the outer bound (exclusive), can be negative or positive
182     * @return the found number
183     */
184    @Override
185    public int nextSignedInt(final int bound) {
186        return (int) ((bound * (nextInt() & 0xFFFFFFFFL)) >> 32);
187    }

189    /**
190     * Returns a value between min (inclusive) and max (exclusive) as ints.
191     * <br>
192     * The inclusive and exclusive behavior is to match the behavior of the similar
193     * method that deals with floating point values.
194     * <br>
195     * If {@code min} and {@code max} happen to be the same, {@code min} is returned
196     * (breaking the exclusive behavior, but it's convenient to do so).
197     *
198     * @param min the minimum bound on the return value (inclusive)
199     * @param max the maximum bound on the return value (exclusive)
200     * @return the found value
201     */
202    @Override
203    public int between(int min, int max) {
204        return nextInt(max - min) + min;
205    }

207    /**
208     * Returns a value between min (inclusive) and max (exclusive) as longs.
209     * <br>
210     * The inclusive and exclusive behavior is to match the behavior of the similar
211     * method that deals with floating point values.
212     * <br>
213     * If {@code min} and {@code max} happen to be the same, {@code min} is returned
214     * (breaking the exclusive behavior, but it's convenient to do so).
215     *
216     * @param min the minimum bound on the return value (inclusive)
217     * @param max the maximum bound on the return value (exclusive)
218     * @return the found value
219     */
220    @Override
221    public long between(long min, long max) {
222        return nextLong(max - min) + min;
223    }

226     * Returns a value from a uniform distribution from min (inclusive) to max
227     * (exclusive).
228     *
229     * @param min the minimum bound on the return value (inclusive)

238    /**
239     * Returns a random element from the provided array and maintains object
240     * type.
241     *
242     * @param array the array to get an element from
243     * @return the randomly selected element
244     */
245    @Override
246    public <T> T getRandomElement(T[] array) {
247        if (array.length < 1) {
248            return null;
249        }
250        return array[nextInt(array.length)];
251    }

253    /**
254     * Returns a random element from the provided list. If the list is empty
255     * then null is returned. This will perform well on Lists that allow random access,
256     * but will not perform as well on {@link java.util.LinkedList} or similar classes
257     * that need to iterate one-by-one in their {@link List#get(int)} method.
258     *
259     * @param list the list to get an element from
260     * @return the randomly selected element
261     */
262    @Override
263    public <T> T getRandomElement(List<T> list) {
264        if (list.isEmpty()) {
265            return null;
266        }
267        return list.get(nextInt(list.size()));
268    }

270    /**
271     * Returns a random element from the provided Collection, which should have predictable iteration order if you want
272     * predictable behavior for identical RNG seeds, though it will get a random element just fine for any Collection
273     * (just not predictably in all cases). If you give this a Set, it should be a LinkedHashSet or some form of sorted
274     * Set like TreeSet if you want predictable results. Any List or Queue should be fine. Map does not implement
275     * Collection, thank you very much Java library designers, so you can't actually pass a Map to this, though you can
276     * pass the keys or values. If coll is empty, returns null.
277     * <br>
278     * Requires iterating through a random amount of coll's elements, so performance depends on the size of coll but is
279     * likely to be decent, as long as iteration isn't unusually slow. This replaces {@code getRandomElement(Queue)},
280     * since Queue implements Collection and the older Queue-using implementation was probably less efficient.
281     * <br>
282     * You should generally prefer {@link #getRandomElement(List)} whenever possible, or in some cases you can use
283     * methods that get a random value on the Collection (or Map, in the case of OrderedMap) itself.
284     *
285     * @param coll the Collection to get an element from; remember, Map does not implement Collection
286     * @return the randomly selected element
287     */
288    @Override
289    public <T> T getRandomElement(Collection<T> coll) {
290        int n;
291        if ((n = coll.size()) <= 0) {
292            return null;
293        }
294        n = nextInt(n);
295        T t = null;
296        Iterator<T> it = coll.iterator();
297        while (n-- >= 0 && it.hasNext())
298            t = it.next();
299        return t;
300    }

302    /**
303     * Mutates the array arr by switching the contents at pos1 and pos2.
304     * @param arr an array of T; must not be null
305     * @param pos1 an index into arr; must be at least 0 and no greater than arr.length
306     * @param pos2 an index into arr; must be at least 0 and no greater than arr.length
307     */
308    protected static <T> void swap(T[] arr, int pos1, int pos2) {
309        final T tmp = arr[pos1];
310        arr[pos1] = arr[pos2];
311        arr[pos2] = tmp;
312    }

314    /**
315     * Shuffle an array using the Fisher-Yates algorithm and returns a shuffled copy, freshly-allocated, without
316     * modifying elements.
317     * <br>
318     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
319     *
320     * @param elements an array of T; will not be modified
321     * @return a shuffled copy of elements
322     */
323    @Override
324    public <T> T[] shuffle(T[] elements) {
325        final int size = elements.length;
326        final T[] array = Arrays.copyOf(elements, size);
327        for (int i = size; i > 1; i--) {
328            swap(array, i - 1, nextInt(i));
329        }
330        return array;
331    }

334     * Shuffles an array in-place using the Fisher-Yates algorithm.
335     * If you don't want the array modified, use {@link #shuffle(Object[], Object[])}.
336     * <br>
337     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
338     *
339     * @param elements an array of T; <b>will</b> be modified
340     * @return elements after shuffling it in-place
341     */
342    @Override
343    public <T> T[] shuffleInPlace(T[] elements) {
344        final int size = elements.length;
345        for (int i = size; i > 1; i--) {
346            swap(elements, i - 1, nextInt(i));
347        }
348        return elements;
349    }

351    /**
352     * Shuffle an array using the Fisher-Yates algorithm. If possible, create a new array or reuse an existing array
353     * with the same length as elements and pass it in as dest; the dest array will contain the shuffled contents of
354     * elements and will also be returned as-is. If dest does not have the same length as elements, this will throw an
355     * IllegalArgumentException
356     * <br>
357     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
358     *
359     * @param elements an array of T; will not be modified
360     * @param <T>      can be any non-primitive type.
361     * @param dest     Where to put the shuffle. If it does not have the same length as {@code elements}, this will
362     *                 throw an IllegalArgumentException.
363     * @return {@code dest} after modifications
364     */
365    @Override
366    public <T> T[] shuffle(T[] elements, T[] dest) {
367        if (dest.length != elements.length)
368            throw new IllegalArgumentException("Not allowed! In AbstractRNG.shuffle(), elements and dest must not have the same length.");
369        System.arraycopy(elements, 0, dest, 0, elements.length);
370        shuffleInPlace(dest);
371        return dest;
372    }

374    /**
375     * Shuffles a {@link Collection} of T using the Fisher-Yates algorithm and returns an ArrayList of T.
376     * <br>
377     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
378     *
379     * @param elements a Collection of T; will not be modified
380     * @return a shuffled ArrayList containing the whole of elements in pseudo-random order.
381     */
382    @Override
383    public <T> ArrayList<T> shuffle(Collection<T> elements) {
384        return shuffle(elements, null);
385    }

387    /**
388     * Shuffles a {@link Collection} of T using the Fisher-Yates algorithm and puts it in a buffer.
389     * The result is allocated if {@code buf} is null or if {@code buf} isn't empty,
390     * otherwise {@code elements} is poured into {@code buf}.
391     * <br>
392     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
393     *
394     * @param elements a Collection of T; will not be modified
395     * @param buf      a buffer as an ArrayList that will be filled with the shuffled contents of elements;
396     *                 if null or non-empty, a new ArrayList will be allocated and returned
397     * @return a shuffled ArrayList containing the whole of elements in pseudo-random order, which may be {@code buf}
398     */
399    @Override
400    public <T> ArrayList<T> shuffle(Collection<T> elements, ArrayList<T> buf) {
401        final ArrayList<T> al;
402        if (buf == null || !buf.isEmpty())
403            al = new ArrayList<>(elements);
404        else {
405            al = buf;
406            al.addAll(elements);
407        }
408        final int n = al.size();
409        for (int i = n; i > 1; i--) {
410            Collections.swap(al, nextInt(i), i - 1);
411        }
412        return al;
413    }

435     * Generates a random permutation of the range from 0 (inclusive) to length (exclusive).
436     * Useful for passing to OrderedMap or OrderedSet's reorder() methods.
437     *
438     * @param length the size of the ordering to produce
439     * @return a random ordering containing all ints from 0 to length (exclusive)
440     */

442    public int[] randomOrdering(int length) {
443        if (length <= 0)
444            return new int[0];
445        return randomOrdering(length, new int[length]);
446    }

448    /**
449     * Generates a random permutation of the range from 0 (inclusive) to length (exclusive) and stores it in
450     * the dest parameter, avoiding allocations.
451     * Useful for passing to OrderedMap or OrderedSet's reorder() methods.
452     *
453     * @param length the size of the ordering to produce; the smaller of {@code dest.length} and length will be used
454     * @param dest   the destination array; will be modified
455     * @return dest, filled with a random ordering containing all ints from 0 to length (exclusive)
456     */
457    @Override
458    public int[] randomOrdering(int length, int[] dest) {
459        if (dest == null) return null;
460
461        final int n = Math.min(length, dest.length);
462        for (int i = 0; i < n; i++) {
463            dest[i] = i;
464        }
465        for (int i = n - 1; i > 0; i--) {
466            final int r = nextInt(i+1),
467                    t = dest[r];
468            dest[r] = dest[i];
469            dest[i] = t;
470        }
471        return dest;
472    }
473    /**
474     * Gets a random portion of data (an array), assigns that portion to output (an array) so that it fills as much as
475     * it can, and then returns output. Will only use a given position in the given data at most once; uses the
476     * Swap-Or-Not Shuffle to accomplish this without allocations. Internally, makes 1 call to {@link #nextInt()} and
477     * 6 calls to {@link #nextSignedInt(int)}, regardless of the data being randomized. It will get progressively less
478     * random as output gets larger.
479     * <br>
480     * Uses approximately the same code as {@link SwapOrNotShuffler}, but without any object or array allocations.
481     *
482     * @param data   an array of T; will not be modified.
483     * @param output an array of T that will be overwritten; should always be instantiated with the portion length
484     * @param <T>    can be any non-primitive type.
485     * @return output, after {@code Math.min(output.length, data.length)} unique items have been put into it from data
486     */
487    @Override
488    public <T> T[] randomPortion(T[] data, T[] output) {
489        final int length = data.length, n = Math.min(length, output.length),
490                func = nextInt(),
491                a = nextSignedInt(length), b = nextSignedInt(length), c = nextSignedInt(length),
492                d = nextSignedInt(length), e = nextSignedInt(length), f = nextSignedInt(length);
493        int key, index;
494        for (int i = 0; i < n; i++) {
495            index = i;
496            key = a - index;
497            key += (key >> 31 & length);
498            if(((func) + Math.max(index, key) & 1) == 0) index = key;
499            key = b - index;
500            key += (key >> 31 & length);
501            if(((func >>> 1) + Math.max(index, key) & 1) == 0) index = key;
502            key = c - index;
503            key += (key >> 31 & length);
504            if(((func >>> 2) + Math.max(index, key) & 1) == 0) index = key;
505            key = d - index;
506            key += (key >> 31 & length);
507            if(((func >>> 3) + Math.max(index, key) & 1) == 0) index = key;
508            key = e - index;
509            key += (key >> 31 & length);
510            if(((func >>> 4) + Math.max(index, key) & 1) == 0) index = key;
511            key = f - index;
512            key += (key >> 31 & length);
513            if(((func >>> 5) + Math.max(index, key) & 1) == 0) index = key;
514
515            output[i] = data[index];
516        }
517        return output;
518    }
