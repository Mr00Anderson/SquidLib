001package squidpony.squidmath;
002
003import squidpony.annotation.Beta;
004
005import java.io.Serializable;
006
007/**
008 * Highly experimental RNG that can be configured to smoothly transition between producing mostly values in the center
009 * of its range, to producing more values at or near the extremes. The probability distribution is... unusual, with
010 * lumps that rise or fall based on centrality.
011 * <br>
012 * <a href="https://i.imgur.com/VCvtlSc.gifv">Here's an animation of the distribution graph changing.</a>
013 * <br>
014 * Created by Tommy Ettinger on 10/6/2019.
015 */
016@Beta
017public class TweakRNG extends AbstractRNG implements Serializable {
018
019    private static final long serialVersionUID = 1L;
020
021    private long stateA, stateB, centrality;
022    public TweakRNG() {
023        this((long) ((Math.random() - 0.5) * 0x10000000000000L)
024                        ^ (long) (((Math.random() - 0.5) * 2.0) * 0x8000000000000000L),
025                (long) ((Math.random() - 0.5) * 0x10000000000000L)
026                        ^ (long) (((Math.random() - 0.5) * 2.0) * 0x8000000000000000L),
027                0L);
028    }
029
030    public TweakRNG(long seed) {
031        this((seed = ((seed = (((seed * 0x632BE59BD9B4E019L) ^ 0x9E3779B97F4A7C15L) * 0xC6BC279692B5CC83L)) ^ seed >>> 27) * 0xAEF17502108EF2D9L) ^ seed >>> 25, 
032                ((seed = ((seed = (((seed * 0x632BE59BD9B4E019L) ^ 0x9E3779B97F4A7C15L) * 0xC6BC279692B5CC83L)) ^ seed >>> 27) * 0xAEF17502108EF2D9L) ^ seed >>> 25),
033                0L);
034    }
035
036    public TweakRNG(final long seedA, final long seedB, final long centrality) {
037        stateA = seedA;
038        stateB = seedB | 1L;
039        this.centrality = centrality % 0x10000L;
040    }
041
042    @Override
043    public int next(int bits) {
044        return (int)(nextLong() >>> 64 - bits);
045    }
046
047    @Override
048    public int nextInt() {
049        return (int) (nextLong() >> 32);
050    }
051
052    @Override
053    public long nextLong() {
054        final long r = internalLong(), s = internalLong();
055        return (long) (NumberTools.atan2_(((r & 0xFF) - (r >>> 8 & 0xFF)) * ((r >>> 16 & 0xFF) - (r >>> 24 & 0xFF)),
056                ((r >>> 32 & 0xFF) - (r >>> 40 & 0xFF)) * ((r >>> 48 & 0xFF) - (r >>> 56 & 0xFF)) + centrality) * 0x6000000000000000L)
057                + (s & 0x1FFFFFFFFFFFFFFFL) << 1 ^ (s >>> 63);
058    }
059
060    @Override
061    public boolean nextBoolean() {
062        return nextLong() < 0;
063    }
064    
065    @Override
066    public double nextDouble() {
067        final long r = internalLong();
068        return NumberTools.atan2_(((r & 0xFF) - (r >>> 8 & 0xFF)) * ((r >>> 16 & 0xFF) - (r >>> 24 & 0xFF)),
069                ((r >>> 32 & 0xFF) - (r >>> 40 & 0xFF)) * ((r >>> 48 & 0xFF) - (r >>> 56 & 0xFF)) + centrality) * 0.75
070                + (internalLong() & 0xfffffffffffffL) * 0x1p-54;
071    }
072
073    @Override
074    public float nextFloat() {
075        return (float) nextDouble();
076    }
077    
078    @Override
079    public TweakRNG copy() {
080        return this;
081    }
082
083    @Override
084    public Serializable toSerializable() {
085        return this;
086    }
087
088    public long getStateA() {
089        return stateA;
090    }
091
092    public void setStateA(long stateA) {
093        this.stateA = stateA;
094    }
095
096    public long getStateB() {
097        return stateB;
098    }
099
100    public void setStateB(long stateB) {
101        this.stateB = stateB | 1L;
102    }
103
104    public long getCentrality() {
105        return centrality;
106    }
107
108    /**
109     * Adjusts the central bias of this TweakRNG, often to positive numbers (which bias toward the center of the range),
110     * but also often to negative numbers (which bias toward extreme values, though still within the range).
111     * @param centrality should be between -65535 and 65535; positive values bias toward the center of the range
112     */
113    public void setCentrality(long centrality) {
114        this.centrality = centrality % 65536L;
115    }
116
117    private long internalLong()
118    {
119        final long s = (stateA += 0xC6BC279692B5C323L);
120        final long z = (s ^ s >>> 31) * (stateB += 0x9E3779B97F4A7C16L);
121        return z ^ z >>> 26;
122    }
123}
