001package squidpony.squidmath;

003import squidpony.annotation.Beta;
004
005/**
006 * A low-quality continuous noise generator with strong grid artifacts, this is nonetheless useful as a building block.
007 * This implements Noise2D, Noise3D, and Noise4D, and could have more dimensionality support added later. It has much
008 * lower quality than {@link ClassicNoise}, but is structured similarly in many ways, and should be a little faster.
009 * @see FoamNoise FoamNoise produces high-quality noise by combining a few rotated results of ValueNoise with domain warping.
010 */
011@Beta
012public class ValueNoise implements Noise.Noise1D, Noise.Noise2D, Noise.Noise3D, Noise.Noise4D {
013    public static final ValueNoise instance = new ValueNoise();
014    
015    public int seed = 0xD1CEBEEF;
016    public ValueNoise() {
017    }
018
019    public ValueNoise(int seed) {
020        this.seed = seed;
021    }
022
023    public ValueNoise(long seed) {
024        this.seed = (int) (seed ^ seed >>> 32);
025    }
026
027    public static double valueNoise(int seed, double x) {
028        int xFloor = x >= 0.0 ? (int) x : (int) x - 1;
029        x -= xFloor;
030        x *= x * (3.0 - 2.0 * x);
031        xFloor *= 0x9E377;
032        seed ^= seed >>> 17;
033        return ((1.0 - x) * hashPart1024(xFloor, seed) + x * hashPart1024(xFloor + 0x9E377, seed))
034                * (0x1.0040100401004p-10);
035    }
036    private static int hashPart1024(final int x, int s) {
037        s *= ((x ^ x >>> 12) | 1);
038        s = (s ^ s >>> 16) * 0xAC451;
039        return (s >>> 3 ^ s >>> 10) & 0x3FF;
040    }
041
042    public static double valueNoise(int seed, double x, double y)
043    {
044
045        int xFloor = x >= 0.0 ? (int) x : (int) x - 1;
046        x -= xFloor;
047        x *= x * (3.0 - 2.0 * x);
048        int yFloor = y >= 0.0 ? (int) y : (int) y - 1;
049        y -= yFloor;
050        y *= y * (3.0 - 2.0 * y);
051        xFloor *= 0xD1B55;
052        yFloor *= 0xABC99;
053        return ((1.0 - y) * ((1.0 - x) * hashPart1024(xFloor, yFloor, seed) + x * hashPart1024(xFloor + 0xD1B55, yFloor, seed))
054                + y * ((1.0 - x) * hashPart1024(xFloor, yFloor + 0xABC99, seed) + x * hashPart1024(xFloor + 0xD1B55, yFloor + 0xABC99, seed)))
055                * (0x1.0040100401004p-10);
056    }
057
058    //// constants are the most significant 20 bits of constants from MummyNoise, incremented if even
059    //// they should normally be used for the 3D version of R2, but we only use 2 of the 3 constants
060    //x should be premultiplied by 0xD1B55
061    //y should be premultiplied by 0xABC99
062    private static int hashPart1024(final int x, final int y, int s) {
063        s += x ^ y;
064        return (s >>> 3 ^ s >>> 10) & 0x3FF;
065    }
066
067    public static double valueNoise(int seed, double x, double y, double z)
068    {
069        int xFloor = x >= 0.0 ? (int) x : (int) x - 1;
070        x -= xFloor;
071        x *= x * (3.0 - 2.0 * x);
072        int yFloor = y >= 0.0 ? (int) y : (int) y - 1;
073        y -= yFloor;
074        y *= y * (3.0 - 2.0 * y);
075        int zFloor = z >= 0.0 ? (int) z : (int) z - 1;
076        z -= zFloor;
077        z *= z * (3.0 - 2.0 * z);
078        //0xDB4F1, 0xBBE05, 0xA0F2F
079        xFloor *= 0xDB4F1;
080        yFloor *= 0xBBE05;
081        zFloor *= 0xA0F2F;
082        return ((1.0 - z) *
083                ((1.0 - y) * ((1.0 - x) * hashPart1024(xFloor, yFloor, zFloor, seed) + x * hashPart1024(xFloor + 0xDB4F1, yFloor, zFloor, seed))
084                        + y * ((1.0 - x) * hashPart1024(xFloor, yFloor + 0xBBE05, zFloor, seed) + x * hashPart1024(xFloor + 0xDB4F1, yFloor + 0xBBE05, zFloor, seed)))
085                + z * 
086                ((1.0 - y) * ((1.0 - x) * hashPart1024(xFloor, yFloor, zFloor + 0xA0F2F, seed) + x * hashPart1024(xFloor + 0xDB4F1, yFloor, zFloor + 0xA0F2F, seed)) 
087                        + y * ((1.0 - x) * hashPart1024(xFloor, yFloor + 0xBBE05, zFloor + 0xA0F2F, seed) + x * hashPart1024(xFloor + 0xDB4F1, yFloor + 0xBBE05, zFloor + 0xA0F2F, seed)))
088                ) * (0x1.0040100401004p-10);
089    }
090
091    //// constants are the most significant 20 bits of constants from MummyNoise, incremented if even
092    //// they should normally be used for the 4D version of R2, but we only use 3 of the 4 constants
093    //x should be premultiplied by 0xDB4F1
094    //y should be premultiplied by 0xBBE05
095    //z should be premultiplied by 0xA0F2F
096    private static int hashPart1024(final int x, final int y, final int z, int s) {
097        s += x ^ y ^ z;
098        return (s >>> 3 ^ s >>> 10) & 0x3FF;
099    }
100
101    public static double valueNoise(int seed, double x, double y, double z, double w)
102    {
103        int xFloor = x >= 0.0 ? (int) x : (int) x - 1;
104        x -= xFloor;
105        x *= x * (3.0 - 2.0 * x);
106        int yFloor = y >= 0.0 ? (int) y : (int) y - 1;
107        y -= yFloor;
108        y *= y * (3.0 - 2.0 * y);
109        int zFloor = z >= 0.0 ? (int) z : (int) z - 1;
110        z -= zFloor;
111        z *= z * (3.0 - 2.0 * z);
112        int wFloor = w >= 0.0 ? (int) w : (int) w - 1;
113        w -= wFloor;
114        w *= w * (3.0 - 2.0 * w);
115        //0xE19B1, 0xC6D1D, 0xAF36D, 0x9A695
116        xFloor *= 0xE19B1;
117        yFloor *= 0xC6D1D;
118        zFloor *= 0xAF36D;
119        wFloor *= 0x9A695;
120        return ((1.0 - w) *
121                ((1.0 - z) *
122                ((1.0 - y) * ((1.0 - x) * hashPart1024(xFloor, yFloor, zFloor, wFloor, seed) + x * hashPart1024(xFloor + 0xE19B1, yFloor, zFloor, wFloor, seed))
123                        + y * ((1.0 - x) * hashPart1024(xFloor, yFloor + 0xC6D1D, zFloor, wFloor, seed) + x * hashPart1024(xFloor + 0xE19B1, yFloor + 0xC6D1D, zFloor, wFloor, seed)))
124                + z * 
125                ((1.0 - y) * ((1.0 - x) * hashPart1024(xFloor, yFloor, zFloor + 0xAF36D, wFloor, seed) + x * hashPart1024(xFloor + 0xE19B1, yFloor, zFloor + 0xAF36D, wFloor, seed)) 
126                        + y * ((1.0 - x) * hashPart1024(xFloor, yFloor + 0xC6D1D, zFloor + 0xAF36D, wFloor, seed) + x * hashPart1024(xFloor + 0xE19B1, yFloor + 0xC6D1D, zFloor + 0xAF36D, wFloor, seed))))
127              + (w * 
128                ((1.0 - z) *
129                ((1.0 - y) * ((1.0 - x) * hashPart1024(xFloor, yFloor, zFloor, wFloor + 0x9A695, seed) + x * hashPart1024(xFloor + 0xE19B1, yFloor, zFloor, wFloor + 0x9A695, seed))
130                        + y * ((1.0 - x) * hashPart1024(xFloor, yFloor + 0xC6D1D, zFloor, wFloor + 0x9A695, seed) + x * hashPart1024(xFloor + 0xE19B1, yFloor + 0xC6D1D, zFloor, wFloor + 0x9A695, seed)))
131                + z *
132                ((1.0 - y) * ((1.0 - x) * hashPart1024(xFloor, yFloor, zFloor + 0xAF36D, wFloor + 0x9A695, seed) + x * hashPart1024(xFloor + 0xE19B1, yFloor, zFloor + 0xAF36D, wFloor + 0x9A695, seed))
133                        + y * ((1.0 - x) * hashPart1024(xFloor, yFloor + 0xC6D1D, zFloor + 0xAF36D, wFloor + 0x9A695, seed) + x * hashPart1024(xFloor + 0xE19B1, yFloor + 0xC6D1D, zFloor + 0xAF36D, wFloor + 0x9A695, seed)))
134        ))) * (0x1.0040100401004p-10);
135    }
136
137    //// constants are the most significant 20 bits of constants from MummyNoise, incremented if even
138    //// they should normally be used for the 5D version of R2, but we only use 4 of the 5 constants
139    //x should be premultiplied by 0xE19B1
140    //y should be premultiplied by 0xC6D1D
141    //z should be premultiplied by 0xAF36D
142    //w should be premultiplied by 0x9A695
143    private static int hashPart1024(final int x, final int y, final int z, final int w, int s) {
144        s += x ^ y ^ z ^ w;
145        return (s >>> 3 ^ s >>> 10) & 0x3FF;
146    }
147    
148    @Override
149    public double getNoise(double x) {
150        return valueNoise(seed, x) * 2.0 - 1.0;
151    }
152
153    @Override
154    public double getNoiseWithSeed(double x, long seed) {
155        return valueNoise((int) (seed ^ seed >>> 32), x) * 2.0 - 1.0;
156    }
157
158    @Override
159    public double getNoise(double x, double y) {
160        return valueNoise(seed, x, y) * 2.0 - 1.0;
161    }
162    @Override
163    public double getNoiseWithSeed(double x, double y, long seed) {
164        return valueNoise((int) (seed ^ seed >>> 32), x, y) * 2.0 - 1.0;
165    }
166    @Override
167    public double getNoise(double x, double y, double z) {
168        return valueNoise(seed, x, y, z) * 2.0 - 1.0;
169    }
170    @Override
171    public double getNoiseWithSeed(double x, double y, double z, long seed) {
172        return valueNoise((int) (seed ^ seed >>> 32), x, y, z) * 2.0 - 1.0;
173    }
174    @Override
175    public double getNoise(double x, double y, double z, double w) {
176        return valueNoise(seed, x, y, z, w) * 2.0 - 1.0;
177    }
178    @Override
179    public double getNoiseWithSeed(double x, double y, double z, double w, long seed) {
180        return valueNoise((int) (seed ^ seed >>> 32), x, y, z, w) * 2.0 - 1.0;
181    }
