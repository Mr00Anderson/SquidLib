025/**
026 * A bi-directional mapping of objects to positions in an ordering (which this generates), and vice versa.
027 * Useful for various indirection techniques where you want to be able to retrieve a value in several ways, such as with
028 * multiple Arrangements all sharing the same ordering but with different types for keys.
029 * <br>
030 * Originally, this started as a generic linked hash map with with a fast implementation, originally from fastutil as
031 * Object2IntLinkedOpenHashMap but modified to support indexed access. The code is extremely close to {@link OrderedMap},
032 * and like OrderedMap you can look up keys by index, but this is specialized to int values which it can automatically assign.
033 * <br>
034 * <p>Instances of this class use a hash table to represent a map. The table is filled up to a specified <em>load factor</em>, and then doubled in size to accommodate new entries. If the table is
035 * emptied below <em>one fourth</em> of the load factor, it is halved in size. However, halving is not performed when deleting entries from an iterator, as it would interfere with the iteration
036 * process.
037 * </p>
038 * <p>Note that {@link #clear()} does not modify the hash table size. Rather, a family of {@linkplain #trim() trimming methods} lets you control the size of the table; this is particularly useful if
039 * you reuse instances of this class.
040 * </p>
041 * <p>Iterators generated by this map will enumerate pairs in the same order in which they have been added to the map (addition of pairs whose key is already present in the set does not change the
042 * iteration order). Note that this order has nothing in common with the natural order of the keys. The order is kept by means of a int-specialized list, {@link IntVLA}, and is modifiable with this
043 * class' {@link #reorder(int...)} and {@link #shuffle(IRNG)} methods, among other tools.
044 * </p>
045 * <p>This class implements the interface of a sorted map, so to allow easy access of the iteration order: for instance, you can get the first key in iteration order with {@code firstKey()} without
046 * having to create an iterator; however, this class partially violates the {@link SortedMap} contract because all submap methods throw an exception and {@link #comparator()} returns always
047 * <code>null</code>.
048 * </p>
049 * <p>
050 * Additional methods, such as <code>getAndMoveToFirst()</code>, make it easy to use instances of this class as a cache (e.g., with LRU policy).
051 * </p>
052 * <br>
053 * Thank you, Sebastiano Vigna, for making FastUtil available to the public with such high quality.
054 * <br>
055 * See https://github.com/vigna/fastutil for the original library.
056 *
057 * @author Sebastiano Vigna (responsible for all the hard parts)
058 * @author Tommy Ettinger (mostly responsible for squashing several layers of parent classes into one monster class)
059 */

352        final int needed = (int) Math.min(
353                1 << 30,
354                Math.max(2, HashCommon.nextPowerOfTwo((long) Math.ceil(capacity

359    private int removeEntry(final int pos) {
360        size--;
361        fixOrder(pos);
362        shiftKeys(pos);
363        final int oldValue = value[pos];
364        fixValues(oldValue);
365        if (size < (maxFill >>> 2) && n > DEFAULT_INITIAL_SIZE)
366            rehash(n / 2);
367        return oldValue;
368    }
369    private int removeNullEntry() {
370        containsNullKey = false;
371        key[n] = null;
372        final int oldValue = value[n];
373        size--;
374        fixOrder(n);
375        fixValues(oldValue);
376        if (size < (maxFill >>> 2) && n > DEFAULT_INITIAL_SIZE)
377            rehash(n / 2);
378        return oldValue;
379    }

421    private int insert(final K k) {
422        int pos;
423        if (k == null) {
424            if (containsNullKey)
425                return n;
426            containsNullKey = true;
427            pos = n;
428        } else {
429            K curr;
430            final K[] key = this.key;
431            // The starting point.
432            if ((curr = key[pos = (hasher.hash(k)) & mask]) != null) {
433                if (hasher.areEqual(curr, k)) {
434                    return pos;
435                }
436                while ((curr = key[pos = (pos + 1) & mask]) != null)
437                    if (hasher.areEqual(curr, k)) {
438                        return pos;
439                    }
440            }
441        }
442        key[pos] = k;
443        order.add(pos);
444        value[pos] = size;
445        if (size++ >= maxFill)
446            rehash(arraySize(size + 1, f));
447        return -1;
448    }
449    private int insertAt(final K k, final int at) {
450        int pos;
451        if (k == null) {
452            if (containsNullKey)
453                return n;
454            containsNullKey = true;
455            pos = n;
456        } else {
457            K curr;
458            final K[] key = this.key;
459            // The starting point.
460            if ((curr = key[pos = (hasher.hash(k)) & mask]) != null) {
461                if (hasher.areEqual(curr, k)) {
462                    order.removeIndex(value[pos]);
463                    order.insert(at, pos);
464                    return pos;
465                }
466                while ((curr = key[pos = (pos + 1) & mask]) != null)
467                    if (hasher.areEqual(curr, k)) {
468                        order.removeIndex(value[pos]);
469                        order.insert(at, pos);
470                        return pos;
471                    }
472            }
473        }
474        key[pos] = k;
475        order.insert(at, pos);
476        value[pos] = at;
477        if (size++ >= maxFill)
478            rehash(arraySize(size + 1, f));
479        return pos;
480    }

549    /**
550     * Shifts left entries with the specified hash code, starting at the
551     * specified position, and empties the resulting free entry.
552     *
553     * @param pos
554     *            a starting position.
555     */
556    protected final void shiftKeys(int pos) {
557        // Shift entries with the same hash.
558        int last, slot;
559        K curr;
560        final K[] key = this.key;
561        for (;;) {
562            pos = ((last = pos) + 1) & mask;
563            for (;;) {
564                if ((curr = key[pos]) == null) {
565                    key[last] = null;
566                    return;
567                }
568                slot = (hasher.hash(curr))
569                        & mask;
570                if (last <= pos ? last >= slot || slot > pos : last >= slot
571                        && slot > pos)
572                    break;
573                pos = (pos + 1) & mask;
574            }
575            key[last] = curr;
576            fixOrder(pos, last);
577        }
578    }

581     * Shifts left entries with the specified hash code, starting at the
582     * specified position, and empties the resulting free entry.
583     *
584     * @param pos
585     *            a starting position.
586     */
587    protected final void shiftKeysValues(int pos) {
588        // Shift entries with the same hash.
589        int last, slot;
590        K curr;
591        int v;
592        final K[] key = this.key;
593        final int[] val = this.value;
594        for (;;) {
595            pos = ((last = pos) + 1) & mask;
596            for (;;) {
597                if ((curr = key[pos]) == null) {
598                    key[last] = null;
599                    val[last] = -1;
600                    return;
601                }
602                v = val[pos];
603                slot = (hasher.hash(curr))
604                        & mask;
605                if (last <= pos ? last >= slot || slot > pos : last >= slot
606                        && slot > pos)
607                    break;
608                pos = (pos + 1) & mask;
609            }
610            key[last] = curr;
611            val[last] = v;
612            fixOrder(pos, last);
613        }

615    @SuppressWarnings("unchecked")
616    protected Integer rem(final Object k) {
617        if (k == null) {
618            if (containsNullKey)
619                return removeNullEntry();
620            return null;
621        }
622        K curr;
623        final K[] key = this.key;
624        int pos;
625        // The starting point.
626        if ((curr = key[pos = (hasher.hash(k)) & mask]) == null)
627            return null;
628        if (hasher.areEqual(k, curr))
629            return removeEntry(pos);
630        while (true) {
631            if ((curr = key[pos = (pos + 1) & mask]) == null)
632                return null;
633            if (hasher.areEqual(k, curr))
634                return removeEntry(pos);
635        }
636    }

643        if (k == null) {
644            if (containsNullKey)
645                return removeNullEntry();
646            return defRetValue;
647        }
648        K curr;
649        final K[] key = this.key;
650        int pos;
651        // The starting point.
652        if ((curr = key[pos = (hasher.hash(k)) & mask]) == null)
653            return defRetValue;
654        if (hasher.areEqual(k, curr))
655            return removeEntry(pos);
656        while (true) {
657            if ((curr = key[pos = (pos + 1) & mask]) == null)
658                return defRetValue;
659            if (hasher.areEqual(k, curr))
660                return removeEntry(pos);
661        }
662    }

1963     * Rehashes the map.
1964     *
1965     * <P>
1966     * This method implements the basic rehashing strategy, and may be overriden
1967     * by subclasses implementing different rehashing strategies (e.g.,
1968     * disk-based rehashing). However, you should not override this method
1969     * unless you understand the internal workings of this class.
1970     *
1971     * @param newN
1972     *            the new size
1973     */

1976    protected void rehash(final int newN) {
1977        final K[] key = this.key;
1978        final int[] value = this.value;
1979        final int mask = newN - 1; // Note that this is used by the hashing macro
1980        final K[] newKey = (K[]) new Object[newN + 1];
1981        final int[] newValue = new int[newN + 1];
1982        final int sz = order.size;
1983        K k;
1984        int i, pos;
1985        final int[] oi = order.items;
1986        for (int q = 0; q < sz; q++) {
1987            i = oi[q];
1988            if ((k = key[i]) == null)
1989                pos = newN;
1990            else {
1991                pos = (hasher.hash(k)) & mask;
1992                while (newKey[pos] != null)
1993                    pos = (pos + 1) & mask;
1994            }
1995            newKey[pos] = k;
1996            newValue[pos] = value[i];
1997            oi[q] = pos;
1998        }
1999
2000        n = newN;
2001        this.mask = mask;
2002        maxFill = maxFill(n, f);
2003        this.key = newKey;
2004        this.value = newValue;
