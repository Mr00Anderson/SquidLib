007@Beta
008public class PhantomNoise {
009    public static final PhantomNoise instance = new PhantomNoise();
010    
011    private final CrossHash.Yolk yolk;
012    public final int dim;
013    private final double inverse;
014    private final double[] working, points;
015    private final double[][] vertices;
016    private final int[] floors, hashFloors;
017    public PhantomNoise() {
018        this(0xFEEDBEEF1337CAFEL, 3);
019    }
020
021    public PhantomNoise(long seed, int dimension) {
022        dim = Math.max(2, dimension);
023        working = new double[dim+1];
024        points = new double[dim+1];
025        vertices = new double[dim+1][dim];
026        double id = -1.0 / dim;
027        vertices[0][0] = 1.0;
028        for (int v = 1; v <= dim; v++) {
029            vertices[v][0] = id;
030        }
031        for (int d = 1; d < dim; d++) {
032            double t = 0.0;
033            for (int i = 0; i < d; i++) {
034                t += vertices[d][i] * vertices[d][i];
035            }
036            vertices[d][d] = Math.sqrt(1.0 - t);
037            t = (id - t) / vertices[d][d];
038            for (int v = d + 1; v <= dim; v++) {
039                vertices[v][d] = t;
040            }
041        }
042        floors = new int[dim+1];
043        hashFloors = new int[dim+1];
044        yolk = new CrossHash.Yolk(seed);
045        inverse = 1.0 / (dim + 1.0);
046    }
047
048    public double valueNoise()
049    {
050        hashFloors[dim] = NumberTools.doubleToMixedIntBits(working[dim]);
051        for (int i = 0; i < dim; i++) {
052            floors[i] = working[i] >= 0.0 ? (int)working[i] : (int)working[i] - 1;
053            working[i] -= floors[i];
054            working[i] *= working[i] * (3.0 - 2.0 * working[i]);
055        }
056        double sum = 0.0, temp;
057        final int limit = 1 << dim;
058        int bit;
059        for (int i = 0; i < limit; i++) {
060            temp = 1.0;
061            for (int j = 0; j < dim; j++) {
062                bit = (i >>> j & 1);
063                temp *= bit + (1|-bit) * working[j];
064                hashFloors[j] = floors[j] - bit;
065            }
066            sum += temp * yolk.hash(hashFloors);
067        }
068        return (sum * 0x1p-32 + 0.5);
069    }
070
071    public double getNoise(double... args) {
072        for (int v = 0; v <= dim; v++) {
073            points[v] = 0.0;
074            for (int d = 0; d < dim; d++) {
075                points[v] += args[d] * vertices[v][d];
076            }
077        }
078        working[dim] = Math.PI;
079        double result = 0.0, warp = 0.0;
080        for (int i = 0; i <= dim; i++) {
081            for (int j = 0, d = 0; j < dim; j++, d++) {
082                if(d == i) d++;
083                working[j] = points[d];
084            }
085            working[0] += warp;
086            warp = valueNoise();
087            result += warp;
088            working[dim] += Math.E;
089        }
090        result *= inverse;
091        return (result <= 0.5)
092                ? Math.pow(result * 2, dim) - 1.0
093                : Math.pow((result - 1) * 2, dim) * (((dim & 1) << 1) - 1) + 1.0;
094//        for (int i = 1; i < dim; i++) {
095//            result *= result * (3.0 - 2.0 * result);
096//        }
097//        return  (result * result * (6.0 - 4.0 * result) - 1.0);
098    }
099
100    public double getNoise(double x, double y) {
101        points[0] = -0.4161468365471422 * x + 0.9092974268256818 * y;
102        points[1] = -0.5794012529532914 * x + -0.8150424455671962 * y;
103        points[2] = 0.9955480895004332 * x + -0.09425498125848553 * y;
104        working[dim] = Math.PI;
105        double result = 0.0, warp = 0.0;
106        for (int i = 0; i <= dim; i++) {
107            for (int j = 0, d = 0; j < dim; j++, d++) {
108                if(d == i) d++;
109                working[j] = points[d] + warp;
110            }
111            warp = valueNoise();
112            result += warp;
113            working[dim] += Math.E;
114        }
115        result *= inverse;
116        return (result <= 0.5)
117                ? (result * result * 4) - 1.0
118                : ((result - 1) * (result - 1) * -4) + 1.0;
119
120//        result *= result * (3.0 - 2.0 * result);
121//        return  (result * result * (6.0 - 4.0 * result) - 1.0);
122    }

125        System.out.println("PhantomNoise with Dimension " + dim + ":");
126        final String dimNames = "xyzwuvabcdefghijklmnopqrst";
127        for (int v = 0; v <= dim; v++) {
128            System.out.print("points[" + v + "] = ");
129            for (int i = 0; i < dim; i++) {
130                if(vertices[v][i] != 0.0) 
131                {
132                    if(i > 0)
133                        System.out.print(" + ");
134                    if(vertices[v][i] == 1.0)
135                        System.out.print(dimNames.charAt(i % dimNames.length()));
136                    else 
137                        System.out.print(dimNames.charAt(i % dimNames.length()) + " * " + vertices[v][i]);
138                }
139            }

141        }
