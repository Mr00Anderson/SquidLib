020import java.io.Serializable;
021import java.util.NoSuchElementException;

023/** An unordered set that uses short keys. This implementation uses cuckoo hashing using 3 hashes, random walking, and a
024 * small stash for problematic keys. No allocation is done except when growing the table size. Used internally by
025 * CoordPacker, and unlikely to be used outside of it.
026 * <br>
027 * This set performs very fast contains and remove (typically O(1), worst case O(log(n))). Add may be a bit slower,
028 * depending on hash collisions. Load factors greater than 0.91 greatly increase the chances the set will have to rehash
029 * to the next higher POT size.
030 * @author Nathan Sweet
031 * Ported from libGDX by Tommy Ettinger on 10/19/2015.
032 */
033public class ShortSet implements Serializable{
034    private static final long serialVersionUID = -4390851800502156007L;
035
036    private static final int PRIME2 = 0xb4b82e39;
037    private static final int PRIME3 = 0xced1c241;
038    private static final short EMPTY = 0;
039
040    public int size;
041
042    short[] keyTable;
043    int capacity, stashSize;
044    boolean hasZeroValue;
045
046    private float loadFactor;
047    private int hashShift, threshold;
048    private int stashCapacity;
049    private int pushIterations;
050    private int mask;
051    private static long rngState;
052
053    private ShortSetIterator iterator1, iterator2;

055    /** Creates a new sets with an initial capacity of 32 and a load factor of 0.8. This set will hold 25 items before growing the
056     * backing table. */
057    public ShortSet() {
058        this(32, 0.8f);
059    }

061    /** Creates a new set with a load factor of 0.8. This set will hold initialCapacity * 0.8 items before growing the backing
062     * table. */
063    public ShortSet(int initialCapacity) {
064        this(initialCapacity, 0.8f);
065    }

067    /** Creates a new set with the specified initial capacity and load factor. This set will hold initialCapacity * loadFactor items
068     * before growing the backing table. */
069    public ShortSet(int initialCapacity, float loadFactor) {
070        if (initialCapacity < 0) throw new IllegalArgumentException("initialCapacity must be >= 0: " + initialCapacity);
071        if (initialCapacity > 1 << 30) throw new IllegalArgumentException("initialCapacity is too large: " + initialCapacity);
072        capacity = nextPowerOfTwo(initialCapacity);
073
074        rngState = (long) ((Math.random() - 0.5) * 0x10000000000000L)
075                ^ (long) (((Math.random() - 0.5) * 2.0) * 0x8000000000000000L);
076
077        if (loadFactor <= 0) throw new IllegalArgumentException("loadFactor must be > 0: " + loadFactor);
078        this.loadFactor = loadFactor;
079
080        threshold = (int)(capacity * loadFactor);
081        mask = capacity - 1;
082        hashShift = 31 - Integer.numberOfTrailingZeros(capacity);
083        stashCapacity = Math.max(3, (int)Math.ceil(Math.log(capacity)) * 2);
084        pushIterations = Math.max(Math.min(capacity, 8), (int)Math.sqrt(capacity) / 8);
085
086        keyTable = new short[capacity + stashCapacity];
087    }
088
089    /** Creates a new map identical to the specified map. */
090    public ShortSet(ShortSet map) {
091        this(map.capacity, map.loadFactor);
092        stashSize = map.stashSize;
093        System.arraycopy(map.keyTable, 0, keyTable, 0, map.keyTable.length);
094        size = map.size;
095        hasZeroValue = map.hasZeroValue;
096    }

098    /** Returns true if the key was not already in the set. */
099    public boolean add (short key) {
100        if (key == 0) {
101            if (hasZeroValue) return false;
102            hasZeroValue = true;
103            size++;
104            return true;
105        }
106
107        short[] keyTable = this.keyTable;
108
109        // Check for existing keys.
110        int index1 = key & mask;
111        short key1 = keyTable[index1];
112        if (key1 == key) return false;
113
114        int index2 = hash2(key);
115        short key2 = keyTable[index2];
116        if (key2 == key) return false;
117
118        int index3 = hash3(key);
119        short key3 = keyTable[index3];
120        if (key3 == key) return false;
121
122        // Find key in the stash.
123        for (int i = capacity, n = i + stashSize; i < n; i++)
124            if (keyTable[i] == key) return false;
125
126        // Check for empty buckets.
127        if (key1 == EMPTY) {
128            keyTable[index1] = key;
129            if (size++ >= threshold) resize(capacity << 1);
130            return true;
131        }
132
133        if (key2 == EMPTY) {
134            keyTable[index2] = key;
135            if (size++ >= threshold) resize(capacity << 1);
136            return true;
137        }
138
139        if (key3 == EMPTY) {
140            keyTable[index3] = key;
141            if (size++ >= threshold) resize(capacity << 1);
142            return true;
143        }
144
145        push(key, index1, key1, index2, key2, index3, key3);
146        return true;
147    }

154        if (offset + length > array.size)

156        addAll(array.items, offset, length);

159    public void addAll (short... array) {
160        addAll(array, 0, array.length);
161    }

163    public void addAll (short[] array, int offset, int length) {
164        ensureCapacity(length);
165        for (int i = offset, n = i + length; i < n; i++)
166            add(array[i]);
167    }
168
169    public void addAll (ShortSet set) {
170        ensureCapacity(set.size);
171        ShortSetIterator iterator = set.iterator();
172        while (iterator.hasNext)
173            add(iterator.next());
174    }

176    /** Skips checks for existing keys. */
177    private void addResize (short key) {
178        if (key == 0) {
179            hasZeroValue = true;
180            return;
181        }
182
183        // Check for empty buckets.
184        int index1 = key & mask;
185        short key1 = keyTable[index1];
186        if (key1 == EMPTY) {
187            keyTable[index1] = key;
188            if (size++ >= threshold) resize(capacity << 1);
189            return;
190        }
191
192        int index2 = hash2(key);
193        short key2 = keyTable[index2];
194        if (key2 == EMPTY) {
195            keyTable[index2] = key;
196            if (size++ >= threshold) resize(capacity << 1);
197            return;
198        }
199
200        int index3 = hash3(key);
201        short key3 = keyTable[index3];
202        if (key3 == EMPTY) {
203            keyTable[index3] = key;
204            if (size++ >= threshold) resize(capacity << 1);
205            return;
206        }
207
208        push(key, index1, key1, index2, key2, index3, key3);
209    }

211    private void push (short insertKey, int index1, short key1, int index2, short key2, int index3, short key3) {
212        short[] keyTable = this.keyTable;
213
214        int mask = this.mask;
215
216        // Push keys until an empty bucket is found.
217        short evictedKey;
218        int i = 0, pushIterations = this.pushIterations;
219        do {
220            // Replace the key and value for one of the hashes.
221            switch (DiverRNG.determineBounded(++rngState, 3)) {
222                case 0:
223                    evictedKey = key1;
224                    keyTable[index1] = insertKey;
225                    break;
226                case 1:
227                    evictedKey = key2;
228                    keyTable[index2] = insertKey;
229                    break;
230                default:
231                    evictedKey = key3;
232                    keyTable[index3] = insertKey;
233                    break;
234            }
235
236            // If the evicted key hashes to an empty bucket, put it there and stop.
237            index1 = evictedKey & mask;
238            key1 = keyTable[index1];
239            if (key1 == EMPTY) {
240                keyTable[index1] = evictedKey;
241                if (size++ >= threshold) resize(capacity << 1);
242                return;
243            }
244
245            index2 = hash2(evictedKey);
246            key2 = keyTable[index2];
247            if (key2 == EMPTY) {
248                keyTable[index2] = evictedKey;
249                if (size++ >= threshold) resize(capacity << 1);
250                return;
251            }
252
253            index3 = hash3(evictedKey);
254            key3 = keyTable[index3];
255            if (key3 == EMPTY) {
256                keyTable[index3] = evictedKey;
257                if (size++ >= threshold) resize(capacity << 1);
258                return;
259            }
260
261            if (++i == pushIterations) break;
262
263            insertKey = evictedKey;
264        } while (true);
265
266        addStash(evictedKey);
267    }
268
269    private void addStash (short key) {
270        if (stashSize == stashCapacity) {
271            // Too many pushes occurred and the stash is full, increase the table size.
272            resize(capacity << 1);
273            add(key);
274            return;
275        }
276        // Store key in the stash.
277        int index = capacity + stashSize;
278        keyTable[index] = key;
279        stashSize++;
280        size++;
281    }

283    /** Returns true if the key was removed. */
284    public boolean remove (short key) {
285        if (key == 0) {
286            if (!hasZeroValue) return false;
287            hasZeroValue = false;
288            size--;
289            return true;
290        }
291
292        int index = key & mask;
293        if (keyTable[index] == key) {
294            keyTable[index] = EMPTY;
295            size--;
296            return true;
297        }
298
299        index = hash2(key);
300        if (keyTable[index] == key) {
301            keyTable[index] = EMPTY;
302            size--;
303            return true;
304        }
305
306        index = hash3(key);
307        if (keyTable[index] == key) {
308            keyTable[index] = EMPTY;
309            size--;
310            return true;
311        }
312
313        return removeStash(key);
314    }
315
316    boolean removeStash (short key) {
317        short[] keyTable = this.keyTable;
318        for (int i = capacity, n = i + stashSize; i < n; i++) {
319            if (keyTable[i] == key) {
320                removeStashIndex(i);
321                size--;
322                return true;
323            }
324        }
325        return false;
326    }
327
328    void removeStashIndex (int index) {
329        // If the removed location was not last, move the last tuple to the removed location.
330        stashSize--;
331        int lastIndex = capacity + stashSize;
332        if (index < lastIndex) keyTable[index] = keyTable[lastIndex];
333    }

335    /** Reduces the size of the backing arrays to be the specified capacity or less. If the capacity is already less, nothing is
336     * done. If the set contains more items than the specified capacity, the next highest power of two capacity is used instead. */
337    public void shrink (int maximumCapacity) {
338        if (maximumCapacity < 0) throw new IllegalArgumentException("maximumCapacity must be >= 0: " + maximumCapacity);
339        if (size > maximumCapacity) maximumCapacity = size;
340        if (capacity <= maximumCapacity) return;
341        maximumCapacity = nextPowerOfTwo(maximumCapacity);
342
343        resize(maximumCapacity);
344    }
345
346    /** Clears the map and reduces the size of the backing arrays to be the specified capacity if they are larger. */
347    public void clear (int maximumCapacity) {
348        if (capacity <= maximumCapacity) {
349            clear();
350            return;
351        }
352        hasZeroValue = false;
353        size = 0;
354        resize(maximumCapacity);
355    }
356
357    public void clear () {
358        if (size == 0) return;
359        short[] keyTable = this.keyTable;
360        for (int i = capacity + stashSize; i-- > 0;)
361            keyTable[i] = EMPTY;
362        size = 0;
363        stashSize = 0;
364        hasZeroValue = false;
365    }
366
367    public boolean contains (short key) {
368        if (key == 0) return hasZeroValue;
369        int index = key & mask;
370        if (keyTable[index] != key) {
371            index = hash2(key);
372            if (keyTable[index] != key) {
373                index = hash3(key);
374                if (keyTable[index] != key) return containsKeyStash(key);
375            }
376        }
377        return true;
378    }
379
380    private boolean containsKeyStash (short key) {
381        short[] keyTable = this.keyTable;
382        for (int i = capacity, n = i + stashSize; i < n; i++)
383            if (keyTable[i] == key) return true;
384        return false;
385    }
386
387    public int first () {
388        if (hasZeroValue) return 0;
389        short[] keyTable = this.keyTable;
390        for (int i = 0, n = capacity + stashSize; i < n; i++)
391            if (keyTable[i] != EMPTY) return keyTable[i];
392        throw new IllegalStateException("ShortSet is empty.");
393    }
394
395    /** Increases the size of the backing array to accommodate the specified number of additional items. Useful before adding many
396     * items to avoid multiple backing array resizes. */
397    public void ensureCapacity (int additionalCapacity) {
398        int sizeNeeded = size + additionalCapacity;
399        if (sizeNeeded >= threshold) resize(nextPowerOfTwo((int)(sizeNeeded / loadFactor)));
400    }
401
402    private void resize (int newSize) {
403        int oldEndIndex = capacity + stashSize;
404
405        capacity = newSize;
406        threshold = (int)(newSize * loadFactor);
407        mask = newSize - 1;
408        hashShift = 31 - Integer.numberOfTrailingZeros(newSize);
409        stashCapacity = Math.max(3, (int)Math.ceil(Math.log(newSize)) * 2);
410        pushIterations = Math.max(Math.min(newSize, 8), (int)Math.sqrt(newSize) / 8);
411
412        short[] oldKeyTable = keyTable;
413
414        keyTable = new short[newSize + stashCapacity];
415
416        int oldSize = size;
417        size = hasZeroValue ? 1 : 0;
418        stashSize = 0;
419        if (oldSize > 0) {
420            for (int i = 0; i < oldEndIndex; i++) {
421                short key = oldKeyTable[i];
422                if (key != EMPTY) addResize(key);
423            }
424        }
425    }

427    private int hash2 (int h) {
428        h *= PRIME2;
429        return (h ^ h >>> hashShift) & mask;
430    }
431
432    private int hash3 (int h) {
433        h *= PRIME3;
434        return (h ^ h >>> hashShift) & mask;
435    }

437    @Override
438        public int hashCode () {
439        int h = 0;
440        for (int i = 0, n = capacity + stashSize; i < n; i++)
441            if (keyTable[i] != EMPTY) h += keyTable[i];
442        return h;
443    }

445    @Override
446        public boolean equals (Object obj) {
447        if (!(obj instanceof ShortSet)) return false;
448        ShortSet other = (ShortSet)obj;
449        if (other.size != size) return false;
450        if (other.hasZeroValue != hasZeroValue) return false;
451        for (int i = 0, n = capacity + stashSize; i < n; i++)
452            if (keyTable[i] != EMPTY && !other.contains(keyTable[i])) return false;
453        return true;
454    }

456    @Override
457        public String toString () {
458        if (size == 0) return "[]";
459        StringBuilder buffer = new StringBuilder(32);
460        buffer.append('[');
461        short[] keyTable = this.keyTable;
462        int i = keyTable.length;
463        if (hasZeroValue)
464            buffer.append("0");
465        else {
466            while (i-- > 0) {
467                int key = keyTable[i];
468                if (key == EMPTY) continue;
469                buffer.append(key);
470                break;
471            }
472        }
473        while (i-- > 0) {
474            int key = keyTable[i];
475            if (key == EMPTY) continue;
476            buffer.append(", ").append(key);
477        }
478        return buffer.append(']').toString();
479    }

481    /**
482     * Gets a random short from this ShortSet, using the given {@link IRNG} to generate random values.
483     * If this ShortSet is empty, throws an UnsupportedOperationException. This method operates in linear time, unlike
484     * the random item retrieval methods in {@link OrderedSet} and {@link OrderedMap}, which take constant time.
485     * @param rng an {@link IRNG}, such as {@link RNG} or {@link GWTRNG}
486     * @return a random short from this ShortSet
487     */
488    public short random(IRNG rng)
489    {
490        if (size <= 0) {
491            throw new UnsupportedOperationException("ShortSet cannot be empty when getting a random element");
492        }
493        int n = rng.nextInt(size);
494        short s = 0;
495        ShortSet.ShortSetIterator ssi = iterator();
496        while (n-- >= 0 && ssi.hasNext)
497            s = ssi.next();
498        ssi.reset();
499        return s;
500
501    }

503    private static int nextPowerOfTwo(int n)
504    {
505        int highest = Integer.highestOneBit(n);
506        return  (highest == NumberTools.lowestOneBit(n)) ? highest : highest << 1;
507    }

509    /** Returns an iterator for the keys in the set. Remove is supported. Note that the same iterator instance is returned each time
510     * this method is called. Use the {@link ShortSetIterator} constructor for nested or multithreaded iteration. */
511    public ShortSetIterator iterator () {
512        if (iterator1 == null) {
513            iterator1 = new ShortSetIterator(this);
514            iterator2 = new ShortSetIterator(this);
515        }
516        if (!iterator1.valid) {
517            iterator1.reset();
518            iterator1.valid = true;
519            iterator2.valid = false;
520            return iterator1;
521        }
522        iterator2.reset();
523        iterator2.valid = true;
524        iterator1.valid = false;
525        return iterator2;
526    }

533
534    public static class ShortSetIterator {
535        static final int INDEX_ILLEGAL = -2;
536        static final int INDEX_ZERO = -1;
537
538        public boolean hasNext;
539
540        final ShortSet set;
541        int nextIndex, currentIndex;
542        boolean valid = true;
543
544        public ShortSetIterator(ShortSet set) {
545            this.set = set;
546            reset();
547        }
548
549        public void reset () {
550            currentIndex = INDEX_ILLEGAL;
551            nextIndex = INDEX_ZERO;
552            if (set.hasZeroValue)
553                hasNext = true;
554            else
555                findNextIndex();
556        }
557
558        void findNextIndex () {
559            hasNext = false;
560            short[] keyTable = set.keyTable;
561            for (int n = set.capacity + set.stashSize; ++nextIndex < n;) {
562                if (keyTable[nextIndex] != EMPTY) {
563                    hasNext = true;
564                    break;
565                }
566            }
567        }
568
569        public void remove () {
570            if (currentIndex == INDEX_ZERO && set.hasZeroValue) {
571                set.hasZeroValue = false;
572            } else if (currentIndex < 0) {
573                throw new IllegalStateException("next must be called before remove.");
574            } else if (currentIndex >= set.capacity) {
575                set.removeStashIndex(currentIndex);
576                nextIndex = currentIndex - 1;
577                findNextIndex();
578            } else {
579                set.keyTable[currentIndex] = EMPTY;
580            }
581            currentIndex = INDEX_ILLEGAL;
582            set.size--;
583        }
584
585        public short next () {
586            if (!hasNext) throw new NoSuchElementException();
587            if (!valid) throw new RuntimeException("ShortSetIterator cannot be used nested.");
588            short key = nextIndex == INDEX_ZERO ? 0 : set.keyTable[nextIndex];
589            currentIndex = nextIndex;
590            findNextIndex();
591            return key;
592        }
593
594        /** Returns a new array containing the remaining keys. */
595        public ShortVLA toArray () {
596            ShortVLA array = new ShortVLA(true, set.size);
597            while (hasNext)
598                array.add(next());
599            return array;
600        }
601    }
