007/**
008 * An unusual continuous noise generator that tends to produce organic-looking forms, currently supporting 2D, 3D, and
009 * 4D. Produces noise values from -1.0 inclusive to 1.0 exclusive. Typically needs about a third as many octaves as the
010 * Simplex option in FastNoise to produce roughly comparable quality, but it also has about a third the speed. A useful
011 * property of FoamNoise is how its visual "character" doesn't change much as dimensions are added; whereas 4D simplex
012 * noise tends to separate into "surflets" separated by spans of 0, and higher dimensions of simplex only have larger
013 * such spans, FoamNoise seems to stay approximately as coherent in 2D, 3D, and 4D. Verifying this claim about FoamNoise
014 * is not easy, but it makes sense intuitively because of how this generator works. Simplex noise in N dimensions relies
015 * on a lattice of N-simplices (such as triangles in 2D or tetrahedra in 3D) and evaluates the noise at a point by
016 * hashing each of the N+1 vertices, looking up a gradient vector for each vertex, and combining the gradient vectors
017 * based on proximity of the evaluated point to each vertex. FoamNoise in N dimensions is not nearly as complex; it
018 * relies on making N+1 averaged calls to N-dimensional value noise, each call using a rotated (and potentially skewed)
019 * set of axes, with each call's result also affecting the inputs to the next call (domain warping). Value noise uses a
020 * cubic lattice or its hypercube equivalent in higher dimensions, which seems to be more "stable" as dimensionality
021 * increases, and the number of value noise calls increases at the same rate as simplex noise adds gradient vectors.
022 * Averaging more calls causes the distribution of the noise to gradually approach Gaussian, but the effect of this
023 * approach gets less pronounced past 3 calls.
024 * <br>
025 * It's strongly encouraged to experiment with the lacunarity parameter in {@link Noise.Layered3D} and similar classes
026 * if you use one of those variants, which also probably needs adjustments to frequency. Changing lacunarity with
027 * multiple octaves can be useful to edit how tightly the noise clumps together.
028 * <br>
029 * <a href="https://i.imgur.com/WpUz1xP.png">2D FoamNoise, one octave</a>,
030 * <a href="https://i.imgur.com/39Brm9Y.png">2D FoamNoise, one octave colorized</a>,
031 * <a href="https://i.imgur.com/ytKe4MW.png">2D FoamNoise, two octaves colorized</a> (note fewer peaks and valleys),
032 * <a href="https://i.imgur.com/5FTjEIR.gifv">3D FoamNoise animated over time, one octave</a>,
033 * <a href="https://i.imgur.com/su9naGx.gifv">3D FoamNoise animated over time, one octave colorized</a>,
034 * <a href="https://i.imgur.com/nr4TROD.gifv">3D FoamNoise animated over time, two octaves colorized</a>,
035 * <a href="https://i.imgur.com/r5cx6im.gifv">4D FoamNoise animated over time, one octave colorized</a>,
036 * <a href="https://i.imgur.com/jEbUdun.gifv">4D FoamNoise animated over time, two octaves colorized</a>,
037 * <a href="https://i.imgur.com/ktCTiIK.jpg">World map made using FoamNoise</a>.
038 */

055    public static double foamNoise(final double x, final double y, int seed) {
056        final double p0 = x;
057        final double p1 = x * -0.5 + y * 0.8660254037844386;
058        final double p2 = x * -0.5 + y * -0.8660254037844387;
059        
060        double xin = p1;
061        double yin = p2;        
062        //double xin = x * 0.540302 + y * 0.841471; // sin and cos of 1
063        //double yin = x * -0.841471 + y * 0.540302;
064        final double a = valueNoise(seed, xin, yin);
065        seed += 0x9E3779BD;
066        seed = (seed ^ seed >>> 12) * 0xDAB;
067        seed ^= seed >>> 14;
068        xin = p2;
069        yin = p0;
070        //xin = x * -0.989992 + y * 0.141120; // sin and cos of 3
071        //yin = x * -0.141120 + y * -0.989992;
072        final double b = valueNoise(seed, xin + a, yin);
073        seed += 0x9E3779BD;
074        seed = (seed ^ seed >>> 12) * 0xDAB;
075        seed ^= seed >>> 14;
076        xin = p0;
077        yin = p1;
078        //xin = x * 0.283662 + y * -0.958924; // sin and cos of 5
079        //yin = x * 0.958924 + y * 0.283662;
080        final double c = valueNoise(seed, xin + b, yin);
081        final double result = a * 0.3125 + (b + c) * 0.34375;
082        return result * result * (6.0 - 4.0 * result) - 1.0;
083    }

104    public static double foamNoise(final double x, final double y, final double z, int seed) {
105        final double p0 = x;
106        final double p1 = x * -0.3333333333333333 + y * 0.9428090415820634;
107        final double p2 = x * -0.3333333333333333 + y * -0.4714045207910317 + z * 0.816496580927726;
108        final double p3 = x * -0.3333333333333333 + y * -0.4714045207910317 + z * -0.816496580927726;
109
110        //final double p0 = (x + y + z);// * 0.5;
111        //final double p1 = x - (y + z);// * 0.25;
112        //final double p2 = y - (x + z);// * 0.25;
113        //final double p3 = z - (x + y);// * 0.25;
114////rotated version of above points on a tetrahedron; the ones below are "more correct" but more complex (and slower?)       
115//        final double p0 = x * 0.139640 + y * -0.304485 + z * 0.942226;
116//        final double p1 = x * -0.185127 + y * -0.791704 + z * -0.582180;
117//        final double p2 = x * -0.776796 + y * 0.628752 + z * -0.035464;
118//        final double p3 = x * 0.822283 + y * 0.467437 + z * -0.324582;
119        double xin = p1;
120        double yin = p2;
121        double zin = p3;
122        final double a = valueNoise(seed, xin, yin, zin);
123        //seed = (seed ^ 0x9E3779BD) * 0xDAB;
124        seed += 0x9E3779BD;
125        seed = (seed ^ seed >>> 12) * 0xDAB;
126        seed ^= seed >>> 14;
127        xin = p0;
128        yin = p2;
129        zin = p3;
130        final double b = valueNoise(seed, xin + a, yin, zin);
131        //seed = (seed ^ 0x9E3779BD) * 0xDAB;
132        seed += 0x9E3779BD;
133        seed = (seed ^ seed >>> 12) * 0xDAB;
134        seed ^= seed >>> 14;
135        xin = p0;
136        yin = p1;
137        zin = p3;
138        final double c = valueNoise(seed, xin + b, yin, zin);
139        //seed = (seed ^ 0x9E3779BD) * 0xDAB;
140        seed += 0x9E3779BD;
141        seed = (seed ^ seed >>> 12) * 0xDAB;
142        seed ^= seed >>> 14;
143        xin = p0;
144        yin = p1;
145        zin = p2;
146        final double d = valueNoise(seed, xin + c, yin, zin);
147
148        final double result = (a + b + c + d) * 0.25;
149        return  (result * result * (6.0 - 4.0 * result) - 1.0);
150    }

152    public static double foamNoise(final double x, final double y, final double z, final double w, int seed) {
153        final double p0 = x;
154        final double p1 = x * -0.25 + y * 0.9682458365518543;
155        final double p2 = x * -0.25 + y * -0.3227486121839514 + z * 0.9128709291752769;
156        final double p3 = x * -0.25 + y * -0.3227486121839514 + z * -0.45643546458763834 + w * 0.7905694150420949;
157        final double p4 = x * -0.25 + y * -0.3227486121839514 + z * -0.45643546458763834 + w * -0.7905694150420947;
158        //final double p0 = (x + y + z + w);
159        //final double p1 = x - (y + z + w);
160        //final double p2 = y - (x + z + w);
161        //final double p3 = z - (x + y + w);
162        //final double p4 = w - (x + y + z);
163////rotated version of above points on a 5-cell; the ones below are "more correct" but more complex (and slower?)       
164//        final double p0 = x * 0.139640 + y * -0.304485 + z * 0.942226;
165//        final double p1 = x * -0.185127 + y * -0.791704 + z * -0.582180;
166//        final double p2 = x * -0.776796 + y * 0.628752 + z * -0.035464;
167//        final double p3 = x * 0.822283 + y * 0.467437 + z * -0.324582;
168        double xin = p1;
169        double yin = p2;
170        double zin = p3;
171        double win = p4;
172        final double a = valueNoise(seed, xin, yin, zin, win);
173        //seed = (seed ^ 0x9E3779BD) * 0xDAB;
174        seed += 0x9E3779BD;
175        seed = (seed ^ seed >>> 12) * 0xDAB;
176        seed ^= seed >>> 14;
177        xin = p0;
178        yin = p2;
179        zin = p3;
180        win = p4;
181        final double b = valueNoise(seed, xin + a, yin, zin, win);
182        //seed = (seed ^ 0x9E3779BD) * 0xDAB;
183        seed += 0x9E3779BD;
184        seed = (seed ^ seed >>> 12) * 0xDAB;
185        seed ^= seed >>> 14;
186        xin = p0;
187        yin = p1;
188        zin = p3;
189        win = p4;
190        final double c = valueNoise(seed, xin + b, yin, zin, win);
191        //seed = (seed ^ 0x9E3779BD) * 0xDAB;
192        seed += 0x9E3779BD;
193        seed = (seed ^ seed >>> 12) * 0xDAB;
194        seed ^= seed >>> 14;
195        xin = p0;
196        yin = p1;
197        zin = p2;
198        win = p4;
199        final double d = valueNoise(seed, xin + c, yin, zin, win);
200        seed += 0x9E3779BD;
201        seed = (seed ^ seed >>> 12) * 0xDAB;
202        seed ^= seed >>> 14;
203        xin = p0;
204        yin = p1;
205        zin = p2;
206        win = p3;
207        final double e = valueNoise(seed, xin + d, yin, zin, win);
208
209        final double result = (a + b + c + d + e) * 0.2;
210        return  (result * result * (6.0 - 4.0 * result) - 1.0);
211    }
