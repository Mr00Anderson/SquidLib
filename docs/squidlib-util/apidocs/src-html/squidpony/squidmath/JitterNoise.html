012/**
013 * "Classic Perlin" noise with jitter applied to the grid it operates on. Not to be confused with the Simplex Noise also
014 * created by Ken Perlin (which is in {@link PerlinNoise}), or with the non-jittered but similar {@link ClassicNoise}.
015 * This noise can in theory be scaled up to arbitrary dimensions, but in practice uses unreasonably hefty amounts of
016 * memory when dimensionality exceeds 10 or so, since it needs to hash {@code Math.pow(2, dimensionality)} points per
017 * sample of noise, which involves over a thousand points in 10 dimensions and over a million points in 20 dimensions.
018 * For that reason, it's limited to 6D noise here, and also implements 2D, 3D, and 4D. Its performance is similar to
019 * {@link WhirlingNoise} in 2D, and slows down further on 3D, 4D, and especially 6D. However, due to the grid jitter,
020 * its quality is improved substantially over vanilla Simplex Noise, and it appears more detailed with one octave than
021 * Simplex does. This uses different gradient vectors than what was recommended in the "Improved
022 * Perlin Noise" paper, since the ones this uses avoid 45-degree angular artifacts in all dimensions implemented.
023 * <br>
024 * ClassicNoise is recommended for most usage in {@link squidpony.squidgrid.mapping.WorldMapGenerator} that needs a
025 * Noise3D implementation, and it tends to about as fast as {@link WhirlingNoise} in 3D with less artifacts. JitterNoise
026 * is a fair amount slower than ClassicNoise, but it has much better quality in 2D and may have somewhat-better quality
027 * in higher dimensions as well. There's probably no reason to prefer JitterNoise over ClassicNoise for 6D output,
028 * because they're practically indistinguishable.
029 */
030@Beta
031public class JitterNoise implements Noise.Noise2D, Noise.Noise3D, Noise.Noise4D, Noise.Noise6D {

042    protected static double gradCoord2D(long seed, int x, int y,
043                                              double xd, double yd) {
044        final long hash = Noise.HastyPointHash.hashAll(x, y, seed);
045        final double[] grad = phiGrad2[(int)hash & 0xFF];
046        return (xd + ((hash >>> 8 & 0xFFL) * 0x1p-8 - 0.5f)) * grad[0]
047                + (yd + ((hash >>> 16 & 0xFFL) * 0x1p-8 - 0.5f)) * grad[1];
048    }

055    protected static double gradCoord3D(long seed, int x, int y, int z, double xd, double yd, double zd) {
056        final long hash = Noise.HastyPointHash.hashAll(x, y, z, seed);
057        final int idx = ((int) hash & 31) * 3;
058        return ((xd + ((hash >>> 8 & 0xFFL) * 0x1p-8 - 0.5f)) * grad3d[idx]
059                + (yd + ((hash >>> 16 & 0xFFL) * 0x1p-8 - 0.5f)) * grad3d[idx + 1]
060                + (zd + ((hash >>> 24 & 0xFFL) * 0x1p-8 - 0.5f)) * grad3d[idx + 2]);
061    }

069    protected static double gradCoord4D(long seed, int x, int y, int z, int w,
070                                        double xd, double yd, double zd, double wd) {
071        final int hash =
072                (int)(((seed ^= 0xB4C4D * x ^ 0xEE2C1 * y ^ 0xA7E07 * z ^ 0xCD5E9 * w) ^ seed >>> 13) * (seed)),
073                        idx = (hash >>> 24) & 0xFC, jitter = (hash >>> 18) & 0xFC;
074        return ((xd+ grad4d[jitter]*0.25) * grad4d[idx]
075                + (yd+ grad4d[jitter + 1]*0.25) * grad4d[idx + 1]
076                + (zd+ grad4d[jitter + 2]*0.25) * grad4d[idx + 2]
077                + (wd+ grad4d[jitter + 3]*0.25) * grad4d[idx + 3]);
078    }
079    protected static double gradCoord6D(long seed, int x, int y, int z, int w, int u, int v,
080                                        double xd, double yd, double zd, double wd, double ud, double vd) {
081        final int hash = (int)(((seed ^= 0xB4C4D * x ^ 0xEE2C1 * y ^ 0xA7E07 * z ^ 0xCD5E9 * w ^ 0x94B5B * u ^ 0xD2385 * v)
082                        ^ seed >>> 13) * (seed)), idx = (hash >>> 24) * 6, jitter = (hash >>> 16 & 0xFF) * 6;
083        return (  (xd+gradient6DLUT[jitter]*0.3) * gradient6DLUT[idx]
084                + (yd+gradient6DLUT[jitter+1]*0.3) * gradient6DLUT[idx+1]
085                + (zd+gradient6DLUT[jitter+2]*0.3) * gradient6DLUT[idx+2]
086                + (wd+gradient6DLUT[jitter+3]*0.3) * gradient6DLUT[idx+3]
087                + (ud+gradient6DLUT[jitter+4]*0.3) * gradient6DLUT[idx+4]
088                + (vd+gradient6DLUT[jitter+5]*0.3) * gradient6DLUT[idx+5]);
089    }

097    public double getNoiseWithSeed(double x, double y, final long seed) {
098        x *= 2.0;
099        y *= 2.0;
100        final int
101                x0 = fastFloor(x),
102                y0 = fastFloor(y);         
103        return Noise.emphasizeSigned(
104                cerp(cerp(gradCoord2D(seed, x0, y0, x - x0, y - y0), gradCoord2D(seed, x0+1, y0, x - x0 - 1, y - y0), x - x0),
105                                cerp(gradCoord2D(seed, x0, y0+1, x - x0, y - y0-1), gradCoord2D(seed, x0+1, y0+1, x - x0 - 1, y - y0 - 1), x - x0),
106                                y - y0) * 0.875);// * 1.4142;
107    }
