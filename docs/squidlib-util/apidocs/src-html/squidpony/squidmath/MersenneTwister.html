041package squidpony.squidmath;
042
043import squidpony.StringKit;
044
045import java.io.Serializable;
046import java.util.Arrays;
047
048/**
049 * Mersenne Twister, 64-bit version as a RandomnessSource.
050 * <br>
051 * Similar to the regular 32-bit Mersenne Twister but implemented with 64-bit
052 * values (Java <code>long</code>), and with different output. This generator is probably
053 * not the best to use because of known statistical problems and low speed, but its period
054 * is absurdly high, {@code pow(2, 19937) - 1}. {@link LongPeriodRNG} has significantly
055 * better speed and statistical quality, and also has a large period, {@code pow(2, 1024) - 1}.
056 * {@link IsaacRNG} is slower, but offers impeccable quality, and from its webpage, "Cycles
057 * are guaranteed to be at least {@code pow(2, 40)} values long, and they are
058 * {@code pow(2, 8295)} values long on average." IsaacRNG should be your choice if security is a
059 * concern, LongPeriodRNG if quality and speed are important, and MersenneTwister should be used
060 * if period is the only criterion to judge an RNG on. There may be a CMWC generator added at
061 * some point, which would have potentially a greater period than the Mersenne Twister.
062 * <br>
063 * This is mostly a straight port of the
064 * <a href="http://www.math.sci.hiroshima-u.ac.jp/~m-mat/MT/VERSIONS/C-LANG/mt19937-64.c">
065 * C-code (mt19937-64.c)</a>, but made more "java-like". This version was originally found
066 * at <a href="https://github.com/javabeanz/javarng/blob/master/com/modp/random/MersenneTwister64.java">
067 * an archived version of a Google Code repo</a>, and it is licensed under the 3-clause BSD license, like
068 * the Mersenne Twister.
069 * <br>
070 * References:
071 * <br>
072 * <ul>
073 * <li>
074 * <a href="http://www.math.sci.hiroshima-u.ac.jp/~m-mat/MT/emt.html">
075 * The Mersenne Twister Home Page </a>
076 * </li>
077 * <li>  Makato Matsumoto and Takuji Nishimura,
078 * <a href="http://www.math.sci.hiroshima-u.ac.jp/~m-mat/MT/ARTICLES/mt.pdf">"Mersenne Twister:A 623-Dimensionally Equidistributed Uniform Pseudo-Random Number Generator"</a>,
079 * <i>ACM Transactions on Modeling and Computer Simulation, </i> Vol. 8, No. 1,
080 * January 1998, pp 3--30.</li>
081 * </ul>
082 *
083 * @author Nick Galbreath -- nickg [at] modp [dot] com
084 * @author Tommy Ettinger
085 * @version 1.1 -- 07-Oct-2017
086 */
087public class MersenneTwister implements Serializable, RandomnessSource {
088
089    private static final long serialVersionUID = 1001000L;
090
091    private static final int NN = 312;
092
093    private static final int MM = 156;
094
095    private static final long MATRIX_A = 0xB5026F5AA96619E9L;
096
097    /**
098     * Mask: Most significant 33 bits
099     */
100    private static final long UM = 0xFFFFFFFF80000000L;
101
102    /**
103     * Mask: Least significant 31 bits
104     */
105    private static final long LM = 0x7FFFFFFFL;
106
107    /**
108     * Mersenne Twister data.
109     */
110    private long[] mt = new long[NN];
111
112    /**
113     * Mersenne Twister Index.
114     */
115    private int mti = NN + 1;
116
117    /**
118     * Internal to hold 64 bits, that might
119     * used to generate two 32 bit values.
120     */
121    private long extra;
122
123    /**
124     * Set to true if we need to generate another 64 bits, false if we have enough bits available for an int.
125     */
126    private boolean bitState = true;
127
128    /**
129     * Seeds this using two calls to Math.random().
130     */
131    public MersenneTwister() {
132        setSeed((long) ((Math.random() * 2.0 - 1.0) * 0x8000000000000L)
133                ^ (long) ((Math.random() * 2.0 - 1.0) * 0x8000000000000000L));
134    }
135
136    /**
137     * Seeds this with the given long, which will be used to affect the large state.
138     *
139     * @param seed any long
140     */
141    public MersenneTwister(final long seed) {
142        setSeed(seed);
143    }
144
145    /**
146     * Seeds this with the given long array, which will be used to affect the large state, and not used directly.
147     *
148     * @param seed a long array; generally should be non-null
149     */
150    public MersenneTwister(final long[] seed) {
151        setSeed(seed);
152    }
153
154    /**
155     * Initalize the pseudo random number generator with 64 bits.
156     * Not the same as setState() in StatefulRandomness; this changes the seed quite a bit.
157     *
158     * @param seed any long
159     */
160    public void setSeed(final long seed) {
161        mt[0] = seed;
162        for (mti = 1; mti < NN; mti++) {
163            mt[mti] = (6364136223846793005L * (mt[mti - 1] ^ (mt[mti - 1] >>> 62)) + mti);
164        }
165    }
166
167    /**
168     * Initalize the pseudo random number generator with a long array of any size, which should not be null but can be.
169     * Not the same as setState() in StatefulRandomness; this changes the seed quite a bit.
170     *
171     * @param array any long array
172     */
173    public void setSeed(final long[] array) {
174        setSeed(19650218L);
175        if (array == null)
176            return;
177        int i = 1;
178        int j = 0;
179        int k = (NN > array.length ? NN : array.length);
180        for (; k != 0; k--) {
181            mt[i] = (mt[i] ^ ((mt[i - 1] ^ (mt[i - 1] >>> 62)) * 3935559000370003845L))
182                    + array[j] + j;
183            i++;
184            j++;
185            if (i >= NN) {
186                mt[0] = mt[NN - 1];
187                i = 1;
188            }
189            if (j >= array.length)
190                j = 0;
191        }
192        for (k = NN - 1; k != 0; k--) {
193            mt[i] = (mt[i] ^ ((mt[i - 1] ^ (mt[i - 1] >>> 62)) * 2862933555777941757L))
194                    - i;
195            i++;
196            if (i >= NN) {
197                mt[0] = mt[NN - 1];
198                i = 1;
199            }
200        }
201
202        mt[0] = 0x8000000000000000L; /* MSB is 1; assuring non-zero initial array */
203    }
204
205    /**
206     * Returns up to 32 random bits.
207     * <br>
208     * The implementation splits a 64-bit long into two 32-bit chunks.
209     * @param bits the number of bits to output, between 1 and 32 (both inclusive)
210     * @return an int with the specified number of pseudo-random bits
211     */
212    @Override
213    public int next(final int bits) {
214        //return ((int)nextLong()) >>> (32 - numbits);
215
216        if (bitState) {
217            extra = nextLong();
218            bitState = false;
219            return (int) (extra >>> (64 - bits));
220        } else {
221            bitState = true;
222            return ((int) extra) >>> (32 - bits);
223        }
224    }
225
226    /**
227     * Returns 64 random bits.
228     * @return a pseudo-random long, which can have any 64-bit value, positive or negative
229     */
230    @Override
231    public long nextLong() {
232        int i;
233        long x;
234        if (mti >= NN) { /* generate NN words at one time */
235
236            for (i = 0; i < NN - MM; i++) {
237                x = (mt[i] & UM) | (mt[i + 1] & LM);
238                mt[i] = mt[i + MM] ^ (x >>> 1) ^ (x & 1L) * MATRIX_A;
239            }
240            for (; i < NN - 1; i++) {
241                x = (mt[i] & UM) | (mt[i + 1] & LM);
242                mt[i] = mt[i + (MM - NN)] ^ (x >>> 1) ^ (x & 1L) * MATRIX_A;
243            }
244            x = (mt[NN - 1] & UM) | (mt[0] & LM);
245            mt[NN - 1] = mt[MM - 1] ^ (x >>> 1) ^ (x & 1L) * MATRIX_A;
246
247            mti = 0;
248        }
249
250        x = mt[mti++];
251
252        x ^= (x >>> 29) & 0x5555555555555555L;
253        x ^= (x << 17) & 0x71D67FFFEDA60000L;
254        x ^= (x << 37) & 0xFFF7EEE000000000L;
255        x ^= (x >>> 43);
256
257        return x;
258    }
259
260    @Override
261    public final MersenneTwister copy() {
262        MersenneTwister f = new MersenneTwister(MATRIX_A);
263        f.mti = mti;
264        f.extra = extra;
265        f.bitState = bitState;
266        System.arraycopy(mt, 0, f.mt, 0, mt.length);
267        return f;
268    }
269
270    @Override
271    public String toString() {
272        return "MersenneTwister with state hashed as " + StringKit.hexHash(mt) +
273                " and index " + mti;
274    }
275
276    @Override
277    public boolean equals(Object o) {
278        if (this == o) return true;
279        if (o == null || getClass() != o.getClass()) return false;
280
281        MersenneTwister mt64RNG = (MersenneTwister) o;
282
283        return mti == mt64RNG.mti && extra == mt64RNG.extra && bitState == mt64RNG.bitState && Arrays.equals(mt, mt64RNG.mt);
284    }
285
286    @Override
287    public int hashCode() {
288        int result = CrossHash.hash(mt);
289        result = 31 * result + mti;
290        result = 31 * result + (int) (extra ^ (extra >>> 32));
291        result = 31 * result + (bitState ? 1 : 0);
292        return result;
293    }
