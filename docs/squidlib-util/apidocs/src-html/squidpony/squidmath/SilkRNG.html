007/**
008 * An IStatefulRNG implementation that is meant to provide random numbers very quickly when targeting GWT but also to
009 * produce the same numbers when used on desktop, Android, or other platforms, and that can have its state read as a
010 * StatefulRandomness; it is thus like {@link GWTRNG} but should perform better on recent desktop JVMs. This uses a
011 * related algorithm to {@link OrbitRNG}, modified to use 32-bit math and more stringently randomize output. It has two
012 * 32-bit ints for state and a period of 0x10000000000000000 (2 to the 64), while passing 32TB of PractRand tests
013 * without any failures or anomalies (so its quality is very good). It is extremely fast when run on Java 13, at least
014 * using OpenJ9 as the compiler; it can produce a billion ints a second on moderately-recent laptop hardware. A nice
015 * quality of the implementation is that it allows any int for both of its states, so you don't need to check and avoid
016 * setting both states to 0 (which GWTRNG has to do in its internals).
017 * <br>
018 * Internally, this uses two Weyl sequences (counters with different large increments), and rarely but at specific
019 * intervals (when stateA is 0) introduces a lag into one sequence (making stateB keep its value instead of incrementing
020 * for one generated number). A multiplier is taken from the upper bits of stateB and multiplied with a xorshifted
021 * stateA, then part of a unary hash in the style of SplitMix64 is used to improve quality. A particularly devious piece
022 * of bitwise hackery allows this to avoid branching as it decides whether to add the lag or not, and is responsible for
023 * this implementation's high speed.
024 * <br>
025 * The name comes from spider silk, used in a web, and how this is optimized for Google Web Toolkit.
026 * <br>
027 * Written in 2019 by Tommy Ettinger.
028 * @author Tommy Ettinger
029 */
030public final class SilkRNG extends AbstractRNG implements IStatefulRNG, Serializable {
031    private static final long serialVersionUID = 5L;
032
033    public int stateA, stateB;
034
035    /**
036     * Creates a new generator seeded using two calls to Math.random().
037     */
038    public SilkRNG()
039    {
040        setState((int)((Math.random() - 0.5) * 0x1p32), (int)((Math.random() - 0.5) * 0x1p32));
041    }
042    /**
043     * Constructs this SilkRNG by dispersing the bits of seed using {@link #setSeed(int)} across the two parts of state
044     * this has.
045     * @param seed an int that won't be used exactly, but will affect both components of state
046     */
047    public SilkRNG(final int seed) {
048        setSeed(seed);
049    }
050    /**
051     * Constructs this SilkRNG by splitting the given seed across the two parts of state this has with
052     * {@link #setState(long)}.
053     * @param seed a long that will be split across both components of state
054     */
055    public SilkRNG(final long seed) {
056        setState(seed);
057    }
058    /**
059     * Constructs this SilkRNG by calling {@link #setState(int, int)} on stateA and stateB as given; see that method
060     * for the specific details (stateA and stateB are kept as-is unless they are both 0).
061     * @param stateA the number to use as the first part of the state
062     * @param stateB the number to use as the second part of the state
063     */
064    public SilkRNG(final int stateA, final int stateB) {
065        setState(stateA, stateB);
066    }
067
068    /**
069     * Hashes {@code seed} using both {@link CrossHash#hash(CharSequence)} and {@link String#hashCode()} and uses those
070     * two results as the two states with {@link #setState(int, int)}. If seed is null, this won't call
071     * String.hashCode() on it and will instead use 0 as that state.
072     * @param seed any String; may be null
073     */
074    public SilkRNG(final String seed) {
075        setState(CrossHash.hash(seed), seed == null ? 0 : seed.hashCode());
076    }

078    /**
079     * Get up to 32 bits (inclusive) of random output; the int this produces
080     * will not require more than {@code bits} bits to represent.
081     *
082     * @param bits an int between 1 and 32, both inclusive
083     * @return a random number that fits in the specified number of bits
084     */
085    @Override
086    public final int next(int bits) {
087        final int s = (stateA += 0xC1C64E6D);
088        int x = (s ^ s >>> 17) * ((stateB += (s | -s) >> 31 & 0x9E3779BB) >>> 12 | 1);
089        x = (x ^ x >>> 16) * 0xAC451;
090        return (x ^ x >>> 15) >>> (32 - bits);
091    }
092
093    /**
094     * Get a random integer between Integer.MIN_VALUE to Integer.MAX_VALUE (both inclusive).
095     *
096     * @return a 32-bit random int.
097     */
098    @Override
099    public final int nextInt() {
100        final int s = (stateA += 0xC1C64E6D);
101        int x = (s ^ s >>> 17) * ((stateB += (s | -s) >> 31 & 0x9E3779BB) >>> 12 | 1);
102        x = (x ^ x >>> 16) * 0xAC451;
103        return (x ^ x >>> 15);
104    }
105    /**
106     * Returns a random non-negative integer below the given bound, or 0 if the bound is 0 or
107     * negative.
108     *
109     * @param bound the upper bound (exclusive)
110     * @return the found number
111     */
112    @Override
113    public final int nextInt(final int bound) {
114        final int s = (stateA += 0xC1C64E6D);
115        int x = (s ^ s >>> 17) * ((stateB += (s | -s) >> 31 & 0x9E3779BB) >>> 12 | 1);
116        x = (x ^ x >>> 16) * 0xAC451;
117        return (int) ((bound * ((x ^ x >>> 15) & 0xFFFFFFFFL)) >>> 32) & ~(bound >> 31);
118    }
119
120    /**
121     * Get a random long between Long.MIN_VALUE to Long.MAX_VALUE (both inclusive).
122     *
123     * @return a 64-bit random long.
124     */
125    @Override
126    public final long nextLong() {
127        int s = (stateA + 0xC1C64E6D);
128        int x = (s ^ s >>> 17) * ((stateB += (s | -s) >> 31 & 0x9E3779BB) >>> 12 | 1);
129        x = (x ^ x >>> 16) * 0xAC451;
130        final long high = (x ^ x >>> 15);
131        s = (stateA += 0x838C9CDA);
132        x = (s ^ s >>> 17) * ((stateB += (s | -s) >> 31 & 0x9E3779BB) >>> 12 | 1);
133        x = (x ^ x >>> 16) * 0xAC451;
134        return (high << 32) | ((x ^ x >>> 15) & 0xFFFFFFFFL);
135    }
136
137    /**
138     * Get a random bit of state, interpreted as true or false with approximately equal likelihood.
139     * This implementation uses a sign check as a safeguard, since its algorithm is based on (but is not equivalent to)
140     * xoroshiro, which recommends a sign check instead of using the least significant bit.
141     *
142     * @return a random boolean.
143     */
144    @Override
145    public final boolean nextBoolean() {
146        final int s = (stateA += 0xC1C64E6D);
147        final int x = (s ^ s >>> 17) * ((stateB += (s | -s) >> 31 & 0x9E3779BB) >>> 12 | 1);
148        return ((x ^ x >>> 16) & 1) == 0;
149    }
150
151    /**
152     * Gets a random double between 0.0 inclusive and 1.0 exclusive.
153     * This returns a maximum of 0.9999999999999999 because that is the largest double value that is less than 1.0 .
154     *
155     * @return a double between 0.0 (inclusive) and 0.9999999999999999 (inclusive)
156     */
157    @Override
158    public final double nextDouble() {
159        int s = (stateA + 0xC1C64E6D);
160        int x = (s ^ s >>> 17) * ((stateB += (s | -s) >> 31 & 0x9E3779BB) >>> 12 | 1);
161        x = (x ^ x >>> 16) * 0xAC451;
162        final long high = (x ^ x >>> 15);
163        s = (stateA += 0x838C9CDA);
164        x = (s ^ s >>> 17) * ((stateB += (s | -s) >> 31 & 0x9E3779BB) >>> 12 | 1);
165        x = (x ^ x >>> 16) * 0xAC451;
166        return  (((high << 32) | ((x ^ x >>> 15) & 0xFFFFFFFFL))
167                & 0x1FFFFFFFFFFFFFL) * 0x1p-53;
168    }
169
170    /**
171     * Gets a random float between 0.0f inclusive and 1.0f exclusive.
172     * This returns a maximum of 0.99999994 because that is the largest float value that is less than 1.0f .
173     *
174     * @return a float between 0f (inclusive) and 0.99999994f (inclusive)
175     */
176    @Override
177    public final float nextFloat() {
178        final int s = (stateA += 0xC1C64E6D);
179        int x = (s ^ s >>> 17) * ((stateB += (s | -s) >> 31 & 0x9E3779BB) >>> 12 | 1);
180        x = (x ^ x >>> 16) * 0xAC451;
181        return ((x ^ x >>> 15) & 0xffffff) * 0x1p-24f;
182    }

184    /**
185     * Creates a copy of this SilkRNG; it will generate the same random numbers, given the same calls in order, as this
186     * SilkRNG at the point copy() is called. The copy will not share references with this SilkRNG.
187     * 
188     * @return a copy of this SilkRNG
189     */
190    @Override
191    public SilkRNG copy() {
192        return new SilkRNG(stateA, stateB);
193    }

195    /**
196     * Gets a view of this IRNG in a way that implements {@link Serializable}, which is simply this IRNG.
197     * @return a {@link Serializable} view of this IRNG or a similar one; always {@code this}
198     */
199    @Override
200    public Serializable toSerializable() {
201        return this;
202    }
203    /**
204     * Sets the state of this generator using one int, running it through Zog32RNG's algorithm two times to get 
205     * two ints. If the states would both be 0, state A is assigned 1 instead.
206     * @param seed the int to use to produce this generator's state
207     */
208    public void setSeed(final int seed) {
209        int z = seed + 0xC74EAD55, a = seed ^ z;
210        a ^= a >>> 14;
211        z = (z ^ z >>> 10) * 0xA5CB3;
212        a ^= a >>> 15;
213        stateA = (z ^ z >>> 20) + (a ^= a << 13);
214        z = seed + 0x8E9D5AAA;
215        a ^= a >>> 14;
216        z = (z ^ z >>> 10) * 0xA5CB3;
217        a ^= a >>> 15;
218        stateB = (z ^ z >>> 20) + (a ^ a << 13);
219    }

259    /**
260     * Get the current internal state of the StatefulRandomness as a long.
261     *
262     * @return the current internal state of this object.
263     */
264    @Override
265    public long getState() {
266        return stateA & 0xFFFFFFFFL | ((long)stateB) << 32;
267    }
268
269    /**
270     * Set the current internal state of this StatefulRandomness with a long.
271     *
272     * @param state a 64-bit long. You should avoid passing 0; this implementation will treat it as 1.
273     */
274    @Override
275    public void setState(final long state) {
276        stateA = (int)(state & 0xFFFFFFFFL);
277        stateB = (int)(state >>> 32);
278    }

280    @Override
281    public boolean equals(Object o) {
282        if (this == o) return true;
283        if (o == null || getClass() != o.getClass()) return false;
284
285        SilkRNG silkRNG = (SilkRNG) o;
286
287        if (stateA != silkRNG.stateA) return false;
288        return stateB == silkRNG.stateB;
289    }

291    @Override
292    public int hashCode() {
293        return 31 * stateA + stateB;
294    }

298        return "SilkRNG with stateA 0x" + StringKit.hex(stateA) + " and stateB 0x" + StringKit.hex(stateB);

301    /**
302     * A deterministic random int generator that, given one int {@code state} as input, irreversibly returns an 
303     * almost-always-different int as a result. Unlike the rest of SilkRNG, this will not produce all possible ints given
304     * all ints as inputs, and probably a third of all possible ints cannot be returned. You should call this with
305     * {@code SilkRNG.determineInt(state = state + 1 | 0)} (you can subtract 1 to go backwards instead of forwards),
306     * which will allow overflow in the incremented state to be handled the same on GWT as on desktop.
307     * @param state an int that should go up or down by 1 each call, as with {@code SilkRNG.determineInt(state = state + 1 | 0)} to handle overflow 
308     * @return a not-necessarily-unique int that is usually very different from {@code state}
309     */
310    public static int determineInt(int state) {
311        return (state = ((state = (state ^ 0xD1B54A35) * 0x102473) ^ state >>> 11 ^ state >>> 21) * (state | 0xFFE00001)) ^ state >>> 13 ^ state >>> 19;
312    }

314    /**
315     * A deterministic random int generator that, given one int {@code state} and an outer int {@code bound} as input,
316     * returns an int between 0 (inclusive) and {@code bound} (exclusive) as a result, which should have no noticeable
317     * correlation between {@code state} and the result. You should call this with
318     * {@code SilkRNG.determineBound(state = state + 1 | 0, bound)} (you can subtract 1 to go backwards instead of
319     * forwards), which will allow overflow in the incremented state to be handled the same on GWT as on desktop.
320     * Like most bounded int generation in SquidLib, this uses some long math, but most of the function uses ints.
321     * @param state an int that should go up or down by 1 each call, as with {@code SilkRNG.determineBounded(state = state + 1 | 0, bound)} to handle overflow
322     * @param bound the outer exclusive bound, as an int; may be positive or negative
323     * @return an int between 0 (inclusive) and {@code bound} (exclusive)
324     */
325    public static int determineBounded(int state, final int bound)
326    {
327        return (int) ((((state = ((state = (state ^ 0xD1B54A35) * 0x102473) ^ state >>> 11 ^ state >>> 21) * (state | 0xFFE00001)) ^ state >>> 13 ^ state >>> 19) & 0xFFFFFFFFL) * bound >> 32);
328    }
329    /**
330     * A deterministic random long generator that, given one int {@code state} as input, returns an 
331     * almost-always-different long as a result. This can only return a tiny fraction of all possible longs, since there
332     * are at most 2 to the 32 possible ints and this doesn't even return different values for each of those. You should
333     * call this with {@code SilkRNG.determine(state = state + 1 | 0)} (you can subtract 1 to go backwards instead of
334     * forwards), which will allow overflow in the incremented state to be handled the same on GWT as on desktop.
335     * @param state an int that should go up or down by 1 each call, as with {@code SilkRNG.determine(state = state + 1 | 0)} to handle overflow 
336     * @return a not-necessarily-unique long that is usually very different from {@code state}
337     */
338    public static long determine(int state)
339    {
340        int r = (state ^ 0xD1B54A35) * 0x102473;
341        r = (r = (r ^ r >>> 11 ^ r >>> 21) * (r | 0xFFE00001)) ^ r >>> 13 ^ r >>> 19;
342        return ((long) r << 32) | (((state = ((state = (state ^ 0xD1B54A35) * 0x102473) ^ state >>> 11 ^ state >>> 21) * (state | 0xFFE00001)) ^ state >>> 13 ^ state >>> 19) & 0xFFFFFFFFL);
343    }
344    /**
345     * A deterministic random float generator that, given one int {@code state} as input, returns an 
346     * almost-always-different float between 0.0f and 1.0f as a result. Unlike the rest of SilkRNG, this might not
347     * produce all possible floats given all ints as inputs, and some fraction of possible floats cannot be returned.
348     * You should call this with {@code SilkRNG.determineFloat(state = state + 1 | 0)} (you can subtract 1 to go
349     * backwards instead of forwards), which will allow overflow in the incremented state to be handled the same on GWT
350     * as on desktop.
351     * @param state an int that should go up or down by 1 each call, as with {@code SilkRNG.determineFloat(state = state + 1 | 0)} to handle overflow 
352     * @return a not-necessarily-unique float from 0.0f to 1.0f that is usually very different from {@code state}
353     */
354    public static float determineFloat(int state) {
355        return (((state = ((state = (state ^ 0xD1B54A35) * 0x102473) ^ state >>> 11 ^ state >>> 21) * (state | 0xFFE00001)) ^ state >>> 13 ^ state >>> 19) & 0xFFFFFF) * 0x1p-24f;
356    }
357    /**
358     * A deterministic random double generator that, given one int {@code state} as input, returns an 
359     * almost-always-different double between 0.0 and 1.0 as a result. This cannot produce more than a tiny fraction of
360     * all possible doubles because the input is 32 bits and at least 53 bits are needed to represent most doubles from
361     * 0.0 to 1.0. You should call this with {@code SilkRNG.determineDouble(state = state + 1 | 0)} (you can subtract 1
362     * to go backwards instead of forwards), which will allow overflow in the incremented state to be handled the same
363     * on GWT as on desktop.
364     * @param state an int that should go up or down by 1 each call, as with {@code SilkRNG.determineDouble(state = state + 1 | 0)} to handle overflow 
365     * @return a not-necessarily-unique double from 0.0 to 1.0 that is usually very different from {@code state}
366     */
367    public static double determineDouble(int state)
368    {
369        return ((state = ((state = (state ^ 0xD1B54A35) * 0x102473) ^ state >>> 11 ^ state >>> 21) * (state | 0xFFE00001)) ^ state >>> 13 ^ state >>> 19) * 0x1p-32 + 0.5;
370    }
