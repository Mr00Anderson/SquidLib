010/**
011 * Class for emulating various traditional RPG-style dice rolls.
012 * Supports rolling multiple virtual dice of arbitrary size, summing all, the highest <i>n</i>, or the lowest <i>n</i>
013 * dice, treating dice as "exploding" as in some tabletop games (where the max result is rolled again and added),
014 * getting value from inside a range, and applying simple arithmetic modifiers to the result (like adding a number).
015 * <br>
016 * Based on code from the Blacken library.
017 *
018 * @author yam655
019 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
020 * @author Tommy Ettinger
021 */
022@Beta
023public class Dice implements Serializable {

027    private static final Matcher mat = Pattern.compile("\\s*(?:(?:(-?\\d+)?\\s*(?:([:><])\\s*(\\d+))?\\s*(?:([d:!])\\s*(\\d+))?)|([+/*-]))\\s*").matcher();
028    private IRNG rng;
029    private transient IntVLA temp = new IntVLA(20);
030    /**

038     * Creates a new dice roller that uses the given IRNG, which can be seeded before it's given here. The IRNG will be

042     */

076    /**
077     * Rolls the given number of dice with the given number of sides and returns
078     * the total of the best n dice.
079     *
080     * @param n number of best dice to total
081     * @param dice total number of dice to roll
082     * @param sides number of sides on the dice
083     * @return sum of best n out of <em>dice</em><b>d</b><em>sides</em>
084     */
085    private int bestOf(int n, int dice, int sides) {
086        int rolls = Math.min(n, dice);
087        temp.clear();
088        for (int i = 0; i < dice; i++) {
089            temp.add(rollDice(1, sides));
090        }
091
092        return bestOf(rolls, temp);
093
094    }

097     * Rolls the given number of exploding dice with the given number of sides and returns
098     * the total of the best n dice (counting a die that explodes as one die).

103     * @return sum of best n out of <em>dice</em><b>d</b><em>sides</em>

150    /**
151     * Totals the highest n numbers in the pool.
152     *
153     * @param n the number of dice to be totaled
154     * @param pool the dice to pick from
155     * @return the sum
156     */
157    private int bestOf(int n, IntVLA pool) {
158        int rolls = Math.min(n, pool.size);
159        pool.sort();
160
161        int ret = 0;
162        for (int i = pool.size - 1, r = 0;  r < rolls && i >= 0; i--, r++) {
163            ret += pool.get(i);
164        }
165        return ret;
166    }

169     * Totals the lowest n numbers in the pool.
170     *
171     * @param n the number of dice to be totaled
172     * @param pool the dice to pick from

175    private int worstOf(int n, IntVLA pool) {
176        int rolls = Math.min(n, pool.size);
177        pool.sort();
178
179        int ret = 0;
180        for (int r = 0;  r < rolls; r++) {
181            ret += pool.get(r);
182        }
183        return ret;
184    }

226    /**
227     * Emulate a dice roll and return the sum.
228     *
229     * @param n number of dice to sum
230     * @param sides positive integer; number of sides on the rolled dice
231     * @return sum of rolled dice
232     */
233    public int rollDice(int n, int sides) {
234        int ret = 0;
235        for (int i = 0; i < n; i++) {
236            ret += rng.nextInt(sides) + 1;
237        }
238        return ret;
239    }
240
241    /**
242     * Emulate an exploding dice roll and return the sum.
243     *
244     * @param n number of dice to sum
245     * @param sides number of sides on the rollDice; should be greater than 1
246     * @return sum of rollDice
247     */
248    public int rollExplodingDice(int n, int sides) {
249        int ret = 0, curr;
250        if(sides <= 1) return n; // avoid infinite loop, act like they can't explode
251        for (int i = 0; i < n;) {
252            ret += (curr = rng.nextInt(sides) + 1);
253            if(curr != sides) i++;
254        }
255        return ret;
256    }

258    /**
259     * Get a list of the independent results of n rolls of dice with the given
260     * number of sides.
261     *
262     * @param n number of dice used
263     * @param sides positive integer; number of sides on each die
264     * @return list of results
265     */
266    public IntVLA independentRolls(int n, int sides) {
267        IntVLA ret = new IntVLA(n);
268        for (int i = 0; i < n; i++) {
269            ret.add(rng.nextInt(sides) + 1);
270        }
271        return ret;
272    }

274    /**
275     * Evaluate the String {@code rollCode} as dice roll notation and roll to get a random result of that dice roll.
276     * This is the main way of using the Dice class. This effectively allows rolling one or more dice and performing
277     * certain operations on the dice and their result. One of the more frequent uses is rolling some amount of dice and
278     * summing their values, which can be done with e.g. "4d10" to roll four ten-sided dice and add up their results.
279     * You can choose to sum only some of the dice, either the "n highest" or "n lowest" values in a group, with "3&gt;4d6"
280     * to sum the three greatest-value dice in four rolls of six-sided dice, or "2&lt;3d8" to sum the two lowest-value dice
281     * in three rolls of eight-sided dice. You can apply modifiers to these results, such as "1d20+7" to roll one
282     * twenty-sided die and add 7 to its result. These modifiers can be other dice, such as "1d10-1d6", and while
283     * multiplication and division are supported, order of operations isn't, so it just rolls dice from left to right
284     * and applies operators it find along the way. You can get a random value in an inclusive range with "50:100",
285     * which is equivalent to "1d51+49" but is easier to read and understand. You can treat dice as "exploding,"
286     * where any dice that get the maximum result are rolled again and added to the total along with the previous
287     * maximum result. As an example, if two exploding six-sided dice are rolled, and their results are 3 and 6, then
288     * because 6 is the maximum value it is rolled again and added to the earlier rolls; if the additional roll is a 5,
289     * then the sum is 3 + 6 + 5 (for a total of 14), but if the additional roll was a 6, then it would be rolled again
290     * and added again, potentially many times if 6 is rolled continually. Some players may be familiar with this game
291     * mechanic from various tabletop games, but many potential players might not be, so it should be explained if you
292     * show the kinds of dice being rolled to players. The syntax used for exploding dice replaced the "d" in "3d6" for
293     * normal dice with "!", making "3!6" for exploding dice. Inclusive ranges are not supported with best-of and
294     * worst-of notation, but exploding dice are. If using a range, the upper bound can be random, decided by dice rolls
295     * such as with "1:6d6" (which rolls six 6-sided dice and uses that as the upper bound of the range) or by other
296     * ranges such as with "10:100:200", which gets a random number between 100 and 200, then returns a random number
297     * between 10 and that. While it is technically allowed to end a dice string with an operator, the partial
298     * operator will be ignored. If you start a dice string with an operator, its left-hand-side will always be 0. If
299     * you have two operators in a row, only the last will be used, unless one is '-' and can be treated as part of a
300     * negative number (this allows "1d20 * -3" to work). Whitespace is allowed between most parts of a dice string.
301     * <br>
302     * The following notation is supported:
303     * <ul>
304     *     <li>{@code 42} : simple absolute string; can start with {@code -} to make it negative</li>
305     *     <li>{@code 3d6} : sum of 3 6-sided dice</li>
306     *     <li>{@code d6} : synonym for {@code 1d6}</li>
307     *     <li>{@code 3>4d6} : best 3 of 4 6-sided dice</li>
308     *     <li>{@code 3:4d6} : gets a random value between 3 and a roll of {@code 4d6}; this syntax has changed</li>
309     *     <li>{@code 2<5d6} : worst 2 of 5 6-sided dice</li>
310     *     <li>{@code 10:20} : simple random range (inclusive between 10 and 20)</li>
311     *     <li>{@code :20} : synonym for {@code 0:20}</li>
312     *     <li>{@code 3!6} : sum of 3 "exploding" 6-sided dice; see above for the semantics of "exploding" dice</li>
313     *     <li>{@code !6} : synonym for {@code 1!6}</li>
314     * </ul>
315     * The following types of operators are supported:
316     * <ul>
317     *     <li>{@code +4} : add 4 to the value</li>
318     *     <li>{@code -3} : subtract 3 from the value</li>
319     *     <li>{@code *100} : multiply value by 100</li>
320     *     <li>{@code /8} : divide value by 8</li>
321     * </ul>
322     * @param rollCode dice string using the above notation
323     * @return a random number that is possible with the given dice string
324     */
325    public int roll(String rollCode) {
326        mat.setTarget(rollCode);
327        int ret, prev = 0;
328        char currentMode = '+';
329        while (mat.find()) {
330            ret = 0;
331            String num1 = mat.group(1); // number constant
332            String wmode = mat.group(2); // between notation
333            String wnum = mat.group(3); // number constant
334            String mode = mat.group(4); // dice, range, or explode
335            String num2 = mat.group(5); // number constant
336            String pmode = mat.group(6); // math operation
337
338            if(pmode != null)
339            {
340                currentMode = pmode.charAt(0);
341                continue;
342            }
343            int a = num1 == null ? 0 : StringKit.intFromDec(num1);
344            int b = num2 == null ? 0 : StringKit.intFromDec(num2);
345            int w = wnum == null ? 0 : StringKit.intFromDec(wnum);
346
347            if (num1 != null && num2 != null) {
348                if (wnum != null) {
349                    if (">".equals(wmode)) {
350                        if ("d".equals(mode)) {
351                            ret = bestOf(a, w, b);
352                        }
353                        else if("!".equals(mode))
354                        {
355                            ret = bestOfExploding(a, w, b);
356                        }
357                    }
358                    else if("<".equals(wmode))
359                    {
360                        if ("d".equals(mode)) {
361                            ret = worstOf(a, w, b);
362                        }
363                        else if("!".equals(mode))
364                        {
365                            ret = worstOfExploding(a, w, b);
366                        }
367                    }
368                    else
369                    // Here, wmode is ":", there is a constant lower bound for the range, and the upper bound is some
370                    // dice roll or other range. This can be negative, easily, if the random upper bound is negative
371                    {
372                        if ("d".equals(mode)) {
373                            ret = a + rng.nextSignedInt(rollDice(w, b) + 1 - a);
374                        } else if ("!".equals(mode)) {
375                            ret = a + rng.nextSignedInt(rollExplodingDice(w, b) + 1 - a);
376                        } else if (":".equals(mode)) {
377                            ret = a + rng.nextSignedInt(w + rng.nextSignedInt(b + 1 - w) + 1 - a);
378                        }
379                    }
380                } else if ("d".equals(mode)) {
381                    ret = rollDice(a, b);
382                } else if ("!".equals(mode)) {
383                    ret = rollExplodingDice(a, b);
384                } else if (":".equals(mode)) {
385                    ret = a + rng.nextSignedInt(b + 1 - a);
386                }
387            } else if (num1 != null) {
388                if (":".equals(wmode)) {
389                    ret = a + rng.nextSignedInt(w + 1 - a);
390                } else {
391                    ret = a;
392                }
393            } else if (num2 != null) {
394                if (mode != null) {
395                    switch (mode) {
396                        case "d":
397                            ret = rollDice(1, b);
398                            break;
399                        case "!":
400                            ret = rollExplodingDice(1, b);
401                            break;
402                        case ":":
403                            ret = rng.nextSignedInt(b + 1);
404                            break;
405                    }
406                }
407            }
408            switch (currentMode)
409            {
410                case '-':
411                    prev -= ret;
412                    break;
413                case '*':
414                    prev *= ret;
415                    break;
416                case '/':
417                    prev /= ret;
418                    break;
419                default:
420                    prev += ret;
421                    break;
422            }
423            currentMode = '+';
424        }
425        return prev;
426    }
