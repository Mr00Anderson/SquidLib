001package squidpony.squidmath;
002
003import java.io.Serializable;
004import java.util.Arrays;
005import java.util.Random;
006
007/**
008 * A low-quality but very fast RNG that has no apparent visual artifacts here; uses Mark Overton's CMR subcycle
009 * generator type, but modified to be especially GWT-friendly. Even though it has no visual issues when rendered as
010 * pixels, it still fails PractRand testing almost immediately. This is meant to be an answer to when people ask for
011 * a bare-minimum generator that's still "good enough" for games. It has a period of 0xFFF43787 or 4294195079, which
012 * can be exhausted in seconds if you only generate numbers in that time, but some seeds will be in a different
013 * cycle with a much lower period. The likelihood of choosing one of these seeds is low, less than a fiftieth of one
014 * percent, but it can happen. It cannot produce all possible ints in its longest cycle, and it can't produce even a
015 * fraction of all possible ints in its smallest cycle. It implements RandomnessSource, but if you just want to copy
016 * this class with no dependencies, then the class declaration can easily be changed to
017 * {@code public class BasicRandom32 extends Random implements Serializable} without any other changes. Note, it does
018 * extend java.util.Random for additional ease of integration, but doesn't use the slow {@code synchronized} keyword
019 * that Random's implementations do.
020 * <br>
021 * <a href="http://www.drdobbs.com/tools/fast-high-quality-parallel-random-number/231000484">This Dr. Dobb's article has
022 * more on this type of generator</a>.
023 * @author Mark Overton
024 * @author Tommy Ettinger
025 */
026public class BasicRandom32 extends Random implements RandomnessSource, Serializable {
027    private static final long serialVersionUID = 1L;
028    public int state;
029
030    public BasicRandom32()
031    {
032        state = 1;
033    }
034
035    public BasicRandom32(final int seed) {
036        setState(seed);
037    }
038
039    public void setState(final int seed)
040    {
041        state = (seed ^ 0x41C64E6D) * 0x9E373 ^ (seed >>> 16);
042    }
043
044    public final long nextLong() {
045        int y = state * 0xBCFD;
046        y = (y << 17 | y >>> 15);
047        final int x = y * 0xBCFD;
048        return ((long) y << 32 ^ (state = (x << 17 | x >>> 15)));
049
050    }
051
052    /**
053     * Gets an int with at most the specified amount of bits; don't confuse this with {@link #nextInt(int)}, which gets
054     * a number between 0 and its int argument, where this draws from a different (larger) range of random results. For
055     * example, {@code next(2)} can return any 2-bit int,
056     * which is limited to 0, 1, 2, or 3. Note that if you request 0 bits, this can give you any int (32 bits).
057     * @param bits the number of bits to get, from 1 to 32
058     * @return an int with at most the specified bits
059     */
060    public final int next(final int bits) {
061        final int y = state * 0xBCFD;
062        return (state = (y << 17 | y >>> 15)) >>> (32 - bits);
063    }
064    public final int nextInt() {
065        final int y = state * 0xBCFD;
066        return (state = (y << 17 | y >>> 15));
067    }
068
069    /**
070     * Returns a random non-negative integer between 0 (inclusive) and the given bound (exclusive),
071     * or 0 if the bound is 0. The bound can be negative, which will produce 0 or a negative result.
072     * <br>
073     * Credit goes to Daniel Lemire, http://lemire.me/blog/2016/06/27/a-fast-alternative-to-the-modulo-reduction/
074     *
075     * @param bound the outer bound (exclusive), can be negative or positive
076     * @return the found number
077     */
078    public int nextInt(final int bound) {
079        return (int) ((bound * (nextInt() & 0xFFFFFFFFL)) >> 32);
080    }
081    /**
082     * Exclusive on bound (which must be positive), with an inner bound of 0.
083     * If bound is negative or 0 this always returns 0.
084     * <br>
085     * Credit for this method goes to <a href="https://oroboro.com/large-random-in-range/">Rafael Baptista's blog</a>,
086     * with some adaptation for signed long values and a 64-bit generator. It also always gets
087     * exactly two random numbers, so it advances the state as much as {@link #nextLong()}.
088     * @param bound the outer exclusive bound; should be positive, otherwise this always returns 0L
089     * @return a random long between 0 (inclusive) and bound (exclusive)
090     */
091    public long nextLong(long bound) {
092        if (bound <= 0) return 0;
093        final long rand = nextInt() & 0xFFFFFFFFL;
094        final long randLow = nextInt() & 0xFFFFFFFFL;
095        final long boundLow = bound & 0xFFFFFFFFL;
096        bound >>= 32;
097        final long z = (randLow * boundLow >> 32);
098        final long t = rand * boundLow + z;
099        return rand * bound + (t >> 32) + ((t & 0xFFFFFFFFL) + randLow * bound >> 32) - (z >> 63);
100    }
101
102    /**
103     * Sets the seed using a long, by XORing the upper and lower halves of {@code seed} and passing that to
104     * {@link #setState(int)}.
105     * @param seed the initial seed
106     */
107    @Override
108    public void setSeed(long seed) {
109        setState((int)(seed ^ seed >>> 32));
110    }
111    /**
112     * Mutates the array arr by switching the contents at pos1 and pos2.
113     * @param arr an array of T; must not be null
114     * @param pos1 an index into arr; must be at least 0 and no greater than arr.length
115     * @param pos2 an index into arr; must be at least 0 and no greater than arr.length
116     */
117    protected static <T> void swap(T[] arr, int pos1, int pos2) {
118        final T tmp = arr[pos1];
119        arr[pos1] = arr[pos2];
120        arr[pos2] = tmp;
121    }
122
123    /**
124     * Shuffle an array using the Fisher-Yates algorithm and returns a shuffled copy, freshly-allocated, without
125     * modifying elements.
126     * <br>
127     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
128     *
129     * @param elements an array of T; will not be modified
130     * @return a shuffled copy of elements
131     */
132    public <T> T[] shuffle(T[] elements) {
133        final int size = elements.length;
134        final T[] array = Arrays.copyOf(elements, size);
135        for (int i = size; i > 1; i--) {
136            swap(array, i - 1, nextInt(i));
137        }
138        return array;
139    }
140
141    /**
142     * Shuffles an array in-place using the Fisher-Yates algorithm, affecting indices from 0 (inclusive) to length
143     * (exclusive). May be useful with libGDX Array instances, which can be shuffled with
144     * {@code random.shuffleInPlace(arr.items, arr.size)}. If you don't want the array modified, use
145     * {@link #shuffle(Object[])}.
146     * <br>
147     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
148     *
149     * @param elements an array of T; <b>will</b> be modified
150     * @return elements after shuffling it in-place
151     */
152    public <T> T[] shuffleInPlace(T[] elements, int length) {
153        final int size = Math.min(elements.length, length);
154        for (int i = size; i > 1; i--) {
155            swap(elements, i - 1, nextInt(i));
156        }
157        return elements;
158    }
159
160    /**
161     * Shuffles an array in-place using the Fisher-Yates algorithm.
162     * If you don't want the array modified, use {@link #shuffle(Object[])}.
163     * <br>
164     * <a href="https://en.wikipedia.org/wiki/Fisher%E2%80%93Yates_shuffle">Wikipedia has more on this algorithm</a>.
165     *
166     * @param elements an array of T; <b>will</b> be modified
167     * @return elements after shuffling it in-place
168     */
169    public <T> T[] shuffleInPlace(T[] elements) {
170        final int size = elements.length;
171        for (int i = size; i > 1; i--) {
172            swap(elements, i - 1, nextInt(i));
173        }
174        return elements;
175    }
176
177    public BasicRandom32 copy() {
178        BasicRandom32 sr = new BasicRandom32();
179        sr.state = state;
180        return sr;
181    }
182}
