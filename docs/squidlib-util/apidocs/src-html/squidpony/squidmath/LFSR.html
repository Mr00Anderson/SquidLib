008package squidpony.squidmath;
009
010import squidpony.StringKit;
011import squidpony.annotation.Beta;
012
013import java.io.Serializable;
014
015/**
016 * A Linear Feedback Shift Register that may be used like a StatefulRandomness but is not truly random. This has a
017 * period of (2 to the 64) minus 1, and is based on Wikipedia's code for a Galois LFSR but uses data from
018 * http://web.archive.org/web/20161007061934/http://courses.cse.tamu.edu/csce680/walker/lfsr_table.pdf .
019 * It is important to note that an LFSR will produce each number from 1 until its maximum exactly once before repeating,
020 * so this may be useful as a way of generating test data in an unpredictable order.
021 * @author Tommy Ettinger
022 */
023@Beta
024public class LFSR implements StatefulRandomness, Serializable {
025
026        private static final long DOUBLE_MASK = (1L << 53) - 1;
027    private static final double NORM_53 = 1. / (1L << 53);
028    private static final long FLOAT_MASK = (1L << 24) - 1;
029    private static final double NORM_24 = 1. / (1L << 24);
030
031        private static final long serialVersionUID = -2373549048478690398L;
032
033    public long state;
034
035    /**
036     * Creates a new generator seeded using Math.random.
037     */
038    public LFSR() {
039        this((long) (Math.random() * Long.MAX_VALUE));
040    }
041
042    public LFSR(final long seed) {
043        setState(seed);
044    }
045
046    public LFSR(final CharSequence seed)
047    {
048        this(CrossHash.hash64(seed));
049    }
050
051
052    @Override
053    public final int next(int bits) {
054        return (int)nextLong() >>> (32 - bits);
055    }
056
057    @Override
058    public final long nextLong() {
059        return state = (state >>> 1 ^ (-(state & 1L) & 0xD800000000000000L));
060    }
061
062    /**
063     * Produces a copy of this RandomnessSource that, if next() and/or nextLong() are called on this object and the
064     * copy, both will generate the same sequence of random numbers from the point copy() was called. This just needs to
065     * copy the state so it isn't shared, usually, and produce a new value with the same exact state.
066     *
067     * @return a copy of this RandomnessSource
068     */
069    @Override
070    public LFSR copy() {
071        return new LFSR(state);
072    }
073
074
075    /**
076     * Can return any int, positive or negative, of any size permissible in a 32-bit signed integer.
077     * @return any int, all 32 bits are random
078     */
079    public int nextInt() {
080        return (int)nextLong();
081    }
082
083    /**
084     * Exclusive on the outer bound; the inner bound is 0. The bound may be negative, which will produce a non-positive
085     * result.
086     * @param bound the outer exclusive bound; may be positive or negative
087     * @return a random long between 0 (inclusive) and bound (exclusive)
088     */
089    public int nextInt( final int bound ) {
090        return (int)((bound * (nextLong() & 0x7FFFFFFFL)) >> 31);
091    }
092    /**
093     * Inclusive lower, exclusive upper.
094     * @param inner the inner bound, inclusive, can be positive or negative
095     * @param outer the outer bound, exclusive, should be positive, should usually be greater than inner
096     * @return a random int that may be equal to inner and will otherwise be between inner and outer
097     */
098    public int nextInt(final int inner, final int outer) {
099        return inner + nextInt(outer - inner);
100    }
101
102    /**
103     * Exclusive on the outer bound; the inner bound is 0. The bound may be negative, which will produce a non-positive
104     * result.
105     * @param bound the outer exclusive bound; may be positive or negative
106     * @return a random long between 0 (inclusive) and bound (exclusive)
107     */
108    public long nextLong(long bound) {
109        long rand = nextLong();
110        final long randLow = rand & 0xFFFFFFFFL;
111        final long boundLow = bound & 0xFFFFFFFFL;
112        rand >>>= 32;
113        bound >>= 32;
114        final long z = (randLow * boundLow >> 32);
115        long t = rand * boundLow + z;
116        final long tLow = t & 0xFFFFFFFFL;
117        t >>>= 32;
118        return rand * bound + t + (tLow + randLow * bound >> 32) - (z >> 63) - (bound >> 63);
119    }
120
121    /**
122     * Inclusive inner, exclusive outer; both inner and outer can be positive or negative.
123     * @param inner the inner bound, inclusive, can be positive or negative
124     * @param outer the outer bound, exclusive, can be positive or negative and may be greater than or less than inner
125     * @return a random long that may be equal to inner and will otherwise be between inner and outer
126     */
127    public long nextLong(final long inner, final long outer) {
128        return inner + nextLong(outer - inner);
129    }
130
131    public double nextDouble() {
132        return (nextLong() & DOUBLE_MASK) * NORM_53;
133    }
134
135    public float nextFloat() {
136        return (float) ((nextLong() & FLOAT_MASK) * NORM_24);
137    }
138
139    public boolean nextBoolean() {
140        return nextLong() < 0L;
141    }
142
143    public void nextBytes(final byte[] bytes) {
144        int i = bytes.length, n = 0;
145        while (i != 0) {
146            n = Math.min(i, 8);
147            for (long bits = nextLong(); n-- != 0; bits >>>= 8) {
148                bytes[--i] = (byte) bits;
149            }
150        }
151    }
152
153    /**
154     * Get the current internal state of the StatefulRandomness as a long.
155     *
156     * @return the current internal state of this object.
157     */
158    @Override
159    public long getState() {
160        return state;
161    }
162
163    /**
164     * Sets the seed of this generator using one long, running that through LightRNG's algorithm twice to get the state.
165     * @param seed the number to use as the seed
166     */
167    public void setState(final long seed) {
168        if(seed == 0)
169            state = -1L;
170        else
171            state = seed;
172    }
173
174    @Override
175    public String toString() {
176        return "LFSR with state 0x" + StringKit.hex(state) + 'L';
177    }
178
179    @Override
180    public boolean equals(Object o) {
181        if (this == o) return true;
182        if (o == null || getClass() != o.getClass()) return false;
183
184        LFSR lfsr = (LFSR) o;
185
186        return (state == lfsr.state);
187    }
188
189    @Override
190    public int hashCode() {
191        return (int) (state ^ (state >>> 32));
192    }
193
194    /**
195     * Gets the next number that an LFSR would produce using {@link #nextLong()} if its state was {@code state}.
196     * Does not allow state to be 0. Strongly consider using the result of this and assigning it to state if you expect
197     * to call this again, such as with {@code (state = LFSR.determine(state))}, which will ensure the long-term
198     * properties of an LFSR hold up (such as having a period of ((2 to the 64) minus 1), or the guarantee that every
199     * number from 1 to ((2 to the 64) minus 1), inclusive on both, will be generated once per period).
200     * @param state any long other than 0
201     * @return the next long that a 64-bit LFSR would produce with the given state
202     */
203    public static long determine(final long state)
204    {
205        return state >>> 1 ^ (-(state & 1L) & 0xD800000000000000L);
206    }
207
208    /**
209     * Gets the next number from 1 to 255 that a different kind of LFSR would produce if its state was {@code state}.
210     * Does not allow state to be 0. If given all byte values except 0 as arguments, will produce all ints 1-255.
211     * Strongly consider using the result of this and assigning it to state if you expect to call this again, such as
212     * with {@code (state = LFSR.determineByte(state))}, which will ensure the long-term properties of an LFSR hold up
213     * (such as having a period of 255, or the guarantee that every number from 1 to 255, inclusive on both, will be
214     * generated once per period).
215     * @param state any byte other than 0
216     * @return the next int between 1 and 255 that an 8-bit LFSR would produce with the given state
217     */
218    public static int determineByte(final byte state)
219    {
220        return state >>> 1 ^ (-(state & 1) & 0xB8);
221    }
222
223    /**
224     * Gets the next number that a different kind of 32-bit LFSR would produce if its state was {@code state}.
225     * Does not allow state to be 0. If given all int values except 0 as arguments, will produce all ints except 0.
226     * Strongly consider using the result of this and assigning it to state if you expect to call this again, such as
227     * with {@code (state = LFSR.determineInt(state))}, which will ensure the long-term properties of an LFSR hold up
228     * (such as having a period of ((2 to the 32) minus 1), or the guarantee that every number from 1 to ((2 to the 32)
229     * minus 1), inclusive on both, will be generated once per period).
230     * @param state any long other than 0
231     * @return the next int that a 32-bit LFSR would produce with the given state
232     */
233    public static int determineInt(final int state)
234    {
235        return state >>> 1 ^ (-(state & 1) & 0xA3000000);
236    }
237
238    /**
239     * Gets the next positive long that a different kind of 63-bit LFSR would produce if its state was {@code state}.
240     * Does not allow state to be 0 or negative. If given all positive long values (not including 0) as arguments, will
241     * produce all longs greater than 0. Strongly consider using the result of this and assigning it to state if you
242     * expect to call this again, such as with {@code (state = LFSR.determinePositiveLong(state))}, which will ensure
243     * the long-term properties of an LFSR hold up (such as having a period of ((2 to the 63) minus 1), or the guarantee
244     * that every number from 1 to ((2 to the 63) minus 1), inclusive on both, will be generated once per period).
245     * @param state any positive long, not including 0
246     * @return the next int that a 63-bit LFSR would produce with the given state
247     */
248    public static long determinePositiveLong(final long state)
249    {
250        return state >>> 1 ^ (-(state & 1L) & 0x6000000000000000L);
251    }
252
253    /**
254     * Gets the next positive int that a different kind of 31-bit LFSR would produce if its state was {@code state}.
255     * Does not allow state to be 0 or negative. If given all positive int values (not including 0) as arguments, will
256     * produce all ints greater than 0. Strongly consider using the result of this and assigning it to state if you
257     * expect to call this again, such as with {@code (state = LFSR.determinePositiveInt(state))}, which will ensure the
258     * long-term properties of an LFSR hold up (such as having a period of ((2 to the 31) minus 1), or the guarantee
259     * that every number from 1 to ((2 to the 31) minus 1), inclusive on both, will be generated once per period).
260     * <br>
261     * A potential benefit of using this particular LFSR type is that the period is a prime number, 2147483647; this can
262     * sometimes be relevant if you simultaneously get pseudo-random numbers from sources of randomness with different
263     * periods that are "relatively co-prime" (that is, they share no common factors other than 1). This case lengthens
264     * the total period of the combined generators significantly, generally multiplying the periods together to get the
265     * combined period, as opposed to other cases that may simply add them together.
266     * @param state any positive int, not including 0
267     * @return the next int that a 31-bit LFSR would produce with the given state
268     */
269    public static int determinePositiveInt(final int state)
270    {
271        return state >>> 1 ^ (-(state & 1) & 0x48000000);
272    }
273
274    /**
275     * Gets the next int that a different kind of LFSR would produce if its state was {@code state}.
276     * Does not allow state to be {@link Integer#MIN_VALUE}, nor will this produce a result of {@link Integer#MIN_VALUE}
277     * (as long as {@link Integer#MIN_VALUE} was not the input). If given all int values except
278     * {@link Integer#MIN_VALUE} as arguments, will produce all ints in the range {@code [-2147483647,2147483647]},
279     * including 0 but not -2147483648 (the minimum int). Strongly consider using the result of this and assigning it to
280     * state if you expect to call this again, such as with {@code (state = LFSR.determineIntSymmetrical(state))}, which
281     * will ensure the long-term properties of an LFSR hold up (such as having a period of ((2 to the 32) minus 1), or
282     * the guarantee that every int except {@link Integer#MIN_VALUE} will be generated once per period).
283     * <br>
284     * This is called Symmetrical because it produces the same amount of positive and negative numbers, instead of the
285     * normal generation of more negative ones (due to how ints are represented, the min value is always further from 0
286     * than the max value for any signed integer type).
287     * @param state any int other than -2147483648 (0x80000000), which is {@link Integer#MIN_VALUE}; can produce 0
288     * @return the next int other than -2147483648 that an LFSR would produce with the given state
289     */
290    public static int determineIntSymmetrical(final int state)
291    {
292        return ((state ^ 0x80000000) >>> 1 ^ (-(state & 1) & 0xA3000000));
293    }
294
295    /**
296     * Gets the next number that an LFSR would produce using {@link #nextInt(int)} if its state was {@code state} and
297     * {@code bound} was passed to nextInt(). Does not allow state to be 0, but bound can be negative, which causes this
298     * not to produce positive numbers. This method is very predictable and its use is not encouraged; prefer using
299     * {@link #determineBounded(int, int)}.
300     * @param state any long other than 0
301     * @param bound the exclusive bound on the result as an int; does better if the bound is not too high (below 10000?)
302     * @return the next value that {@link LFSR#determine(long)} would produce with the given state, but limited to bound; can return 0
303     */
304    public static int determineBounded(final long state, final int bound)
305    {
306        return (int)((bound * (state >>> 1 & 0xFFFFFFFFL)) >> 32);
307    }
308    /**
309     * Gets an int using {@link #determineInt(int)} and bounds it to fit between 0 (inclusive) and bound (exclusive).
310     * Does not allow state to be 0, but bound can be negative, which causes this not to produce positive numbers.
311     * @param state any int other than 0
312     * @param bound the exclusive bound on the result as an int; does better if the bound is not too high (below 10000?)
313     * @return the next int that {@link LFSR#determineInt(int)} would produce with the given state, but limited to bound; can return 0
314     */
315    public static int determineBounded(final int state, final int bound)
316    {
317        return (int)((bound * ((state >>> 1 ^ (-(state & 1) & 0xA3000000)) & 0xFFFFFFFFL)) >> 32);
318    }
