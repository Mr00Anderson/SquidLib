101    /**
102     * Allows this Thesaurus to find the exact words in synonyms and, when requested, replace each occurrence with a
103     * different word from the same Collection. Each word in synonyms should have the same part of speech, so "demon"
104     * and "devils" should not be in the same list of synonyms (singular noun and plural noun), but "demon" and "devil"
105     * could be (each is a singular noun). The Strings in synonyms should all be lower-case, since case is picked up
106     * from the text as it is being replaced and not from the words themselves. Proper nouns should normally not be used
107     * as synonyms, since this could get very confusing if it changed occurrences of "Germany" to "France" at random and
108     * a character's name, like "Dorothy", to "Anne", "Emily", "Cynthia", etc. in the middle of a section about Dorothy.
109     * The word matching pattern this uses only matches all-letter words, not words that contain hyphens, apostrophes,
110     * or other punctuation.
111     * @param synonyms a Collection of lower-case Strings with similar meaning and the same part of speech
112     * @return this for chaining
113     */
114    public Thesaurus addSynonyms(Collection<String> synonyms)
115    {
116        if(synonyms.isEmpty())
117            return this;
118        //long prevState = rng.getState();
119        //rng.setState(prevState ^ CrossHash.hash64(synonyms));
120        GapShuffler<String> shuffler = new GapShuffler<>(synonyms, rng, true);
121        for(String syn : synonyms)
122        {
123            mappings.put(syn, shuffler);
124        }
125        //rng.setState(prevState);
126        return this;
127    }

311    private StringBuilder modify(CharSequence text)
312    {
313        Matcher m;
314        StringBuilder sb = new StringBuilder(text);
315        Replacer.StringBuilderBuffer tb, working = Replacer.wrap(sb);
316        StringBuilder tmp;
317        boolean found;
318        FakeLanguageGen.Alteration alt;
319        for (int a = 0; a < alterations.size(); a++) {
320            alt = alterations.get(a);
321            tmp = working.sb;
322            tb = Replacer.wrap(new StringBuilder(tmp.length()));
323            m = alt.replacer.getPattern().matcher(tmp);
324
325            found = false;
326            while (true) {
327                if (rng.nextDouble() < alt.chance) {
328                    if (!Replacer.replaceStep(m, alt.replacer.getSubstitution(), tb))
329                        break;
330                    found = true;
331                } else {
332                    if (!m.find())
333                        break;
334                    found = true;
335                    m.getGroup(MatchResult.PREFIX, tb);
336                    m.getGroup(MatchResult.MATCH, tb);
337                    m.setTarget(m, MatchResult.SUFFIX);
338                }
339            }
340            if (found) {
341                m.getGroup(MatchResult.TARGET, tb);
342                working = tb;
343            }
344        }
345        return working.sb;
346
347    }
348    /**
349     * Given a String, StringBuilder, or other CharSequence that should contain words this knows synonyms for, this
350     * replaces each occurrence of such a known word with one of its synonyms, leaving unknown words untouched. Words
351     * that were learned together as synonyms with addSynonyms() will be replaced in such a way that an individual
352     * replacement word should not occur too close to a previous occurrence of the same word; that is, replacing the
353     * text "You fiend! You demon! You despoiler of creation; devil made flesh!", where "fiend", "demon", and "devil"
354     * are all synonyms, would never produce a string that contained "fiend" as the replacement for all three of those.
355     * @param text a CharSequence, such as a String, that contains words in the source language
356     * @return a String of the translated text.
357     */
358    public String process(CharSequence text)
359    {
360        Replacer rep = wordMatch.replacer(new SynonymSubstitution());
361        if(alterations.isEmpty())
362            return rep.replace(text);
363        else
364            return modify(rep.replace(text)).toString();
365    }

367    public String lookup(String word)
368    {
369        if(word.isEmpty())
370            return word;
371        String word2 = word.toLowerCase();
372        if(mappings.containsKey(word2))
373        {
374            String nx = mappings.get(word2).next();
375            if(nx.isEmpty())
376                return nx;
377            if(word.length() > 1 && Category.Lu.contains(word.charAt(1)))
378                return nx.toUpperCase();
379            if(Category.Lu.contains(word.charAt(0)))
380            {
381                return Character.toUpperCase(nx.charAt(0)) + nx.substring(1);
382            }
383            return nx;
384        }
385        else if(languages.containsKey(word2))
386        {
387            if(word.length() > 1 && Category.Lu.contains(word.charAt(1)))
388                return languages.get(word2).word(rng, false, rng.between(2, 4)).toUpperCase();
389            if(Category.Lu.contains(word.charAt(0)))
390            {
391                return languages.get(word2).word(rng, true, rng.between(2, 4));
392            }
393            return languages.get(word2).word(rng, false, rng.between(2, 4));
394        }
395        return word;
396    }
