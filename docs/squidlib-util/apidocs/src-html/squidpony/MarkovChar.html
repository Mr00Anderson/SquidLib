014/**
015 * A simple Markov chain text generator; call {@link #analyze(CharSequence)} once on a large sample text, then you can
016 * call {@link #chain(long)} many times to get odd-sounding "remixes" of the sample text. This is meant to allow easy
017 * serialization of the necessary data to call chain(); if you can store the {@link #chars} and {@link #processed}
018 * arrays in some serialized form, then you can reassign them to the same fields to avoid calling analyze(). One way to
019 * do this conveniently is to use {@link #serializeToString()} after calling analyze() once and to save the resulting
020 * String; then, rather than calling analyze() again on future runs, you would call
021 * {@link #deserializeFromString(String)} to create the MarkovText without needing any repeated analysis.
022 * <br>
023 * Created by Tommy Ettinger on 1/30/2018.
024 */
025@Beta
026public class MarkovChar implements Serializable {
027    private static final long serialVersionUID = 1L;
028
029    /**
030     * All chars (case-sensitive and only counting chars that are letters in Unicode) that this encountered during the 
031     * latest call to {@link #analyze(CharSequence)}. Will be null if {@link #analyze(CharSequence)} was never called.
032     */
033    public char[] chars;
034
035    /**
036     * Map of all pairs of chars encountered to the position in the order they were encountered. Pairs are stored using
037     * their 16-bit {@link #chars} indices placed into the most-significant bits for the first word and the
038     * least-significant bits for the second word. The size of this IntIntOrderedMap is likely to be larger than the
039     * char array {@link #chars}, but should be equal to {@code processed.length}. Will be null if
040     * {@link #analyze(CharSequence)} was never called.
041     */
042    public IntIntOrderedMap pairs;
043    /**
044     * Complicated data that mixes probabilities of chars using their indices in {@link #chars} and the indices of char
045     * pairs in {@link #pairs}, generated during the latest call to {@link #analyze(CharSequence)}. This is a jagged 2D
046     * array. Will be null if {@link #analyze(CharSequence)} was never called.
047     */
048    public int[][] processed;
049
050    private static final Character INITIAL = '^', END = ' ';
051    private static final Matcher matcher = Pattern.compile("[\\p{L}']").matcher();
052    public MarkovChar()
053    {
054    }
055
056    /**
057     * This is the main necessary step before using a MarkovText; you must call this method at some point before you can
058     * call any other methods. You can serialize this MarkovText after calling to avoid needing to call this again on later
059     * runs, or even include serialized MarkovText objects with a game to only need to call this during pre-processing.
060     * This method analyzes the pairings of words in a (typically large) corpus text, including some punctuation as part
061     * of words and some kinds as their own "words." It only uses one preceding word to determine the subsequent word.
062     * When it finishes processing, it stores the results in {@link #chars} and {@link #processed}, which allows other
063     * methods to be called (they will throw a {@link NullPointerException} if analyze() hasn't been called).
064     * @param corpus a typically-large sample text in the style that should be mimicked
065     */
066    public void analyze(CharSequence corpus)
067    {
068        final int clen = corpus.length();
069        Arrangement<Character> body = new Arrangement<>((clen >> 4) + 5);
070        pairs = new IntIntOrderedMap(clen / 5 + 5);
071        ArrayList<IntVLA> working = new ArrayList<>(clen / 5 + 5);
072        body.add(INITIAL);
073        working.add(new IntVLA(128));
074        pairs.put(0, 0);
075        body.add(END);
076//        working.add(new IntVLA(links));
077
078        //matcher.setTarget(corpus);
079        int current = 0, pair = 0, pre = 0, post;
080        for (int i = 0; i < clen; i++) {
081            char c = corpus.charAt(i);
082            if('\'' != c && !Category.L.contains(c))
083                c = END;
084            current = body.addOrIndex(c);
085            pair = pair << 16 | (current & 0xFFFF);
086            if(pair == 1)
087                continue;
088            post = pairs.putIfAbsent(pair, pairs.size());
089            if(working.size() != pairs.size())
090            {
091                working.add(new IntVLA(16));
092            }
093            working.get(pre).add(current);
094            if(current == 1)
095            {
096                working.get(post).add(0);
097                pair = 0;
098                pre = 0;
099            }
100            else
101            {
102                pre = post;
103            }
104        }
105        IntVLA w = working.get(pre), v;
106        if(w.size == 0) w.add(0);
107        final int len = body.size(), pairLen = working.size();
108        chars = new char[len];
109        for (int i = 0; i < len; i++) {
110            chars[i] = body.keyAt(i);
111        }
112
113        processed = new int[pairLen][];
114        w = new IntVLA(128);
115        IntVLA small = new IntVLA(128);
116        IntVLA large = new IntVLA(128);
117        IntVLA probabilities = new IntVLA(128);
118        for(int iv = 0; iv < pairLen; iv++ )
119        {
120            v = working.get(iv);
121            w.clear();
122            probabilities.clear();
123            if(v.size <= 0)
124            {
125                v.add(1);
126            }
127            int vv, sum = 0;
128            final int vs = v.size;
129            OUTER:
130            for (int i = 0; i < vs; ++i) {
131                vv = v.get(i);
132                for (int j = 0; j < w.size; j++) {
133                    if (w.get(j) == vv) {
134                        probabilities.incr(j, 0x10000);
135                        sum += 0x10000;
136                        continue OUTER;
137                    }
138                }
139                w.add(vv);
140                probabilities.add(0x10000);
141                sum += 0x10000;
142            }
143            int iAverage = (sum / w.size);
144
145            small.clear();
146            large.clear();
147            /* Populate the stacks with the input probabilities. */
148            for (int i = 0; i < probabilities.size; i++) {
149                /* If the probability is below the average probability, then we add
150                 * it to the small list; otherwise we add it to the large list.
151                 */
152                if (probabilities.get(i) >= iAverage)
153                    large.add(i);
154                else
155                    small.add(i);
156            }
157
158            processed[iv] = new int[w.size * 3];
159
160            while (!small.isEmpty() && !large.isEmpty()) {
161                /* Get the index of the small and the large probabilities. */
162                int less = small.pop(), less2 = less * 3;
163                int more = large.pop();
164
165                /* These probabilities have not yet been scaled up to be such that
166                 * sum/n is given weight 1.0.  We do this here instead.
167                 */
168                processed[iv][less2] = (probabilities.size * probabilities.get(less)) / (sum >> 16);
169                processed[iv][less2+1] = w.get(less);
170                processed[iv][less2+2] = w.get(more);
171                vv = probabilities.get(less) - iAverage;
172                probabilities.incr(more, vv);
173                if (probabilities.get(more) >= iAverage)
174                    large.add(more);
175                else
176                    small.add(more);
177            }
178            int t;
179            while (!small.isEmpty())
180            {
181                processed[iv][(t = small.pop()) * 3] = 0xFFFF;
182                processed[iv][t * 3 + 1] = processed[iv][t * 3 + 2] = w.get(t);
183            }
184            while (!large.isEmpty())
185            {
186                processed[iv][(t = large.pop()) * 3] = 0xFFFF;
187                processed[iv][t * 3 + 1] = processed[iv][t * 3 + 2] = w.get(t);
188            }
189        }
190    }

192    /**
193     * Generate a roughly-sentence-sized piece of text based on the previously analyzed corpus text (using
194     * {@link #analyze(CharSequence)}) that terminates when stop punctuation is used (".", "!", "?", or "..."), or once
195     * the length would be greater than 200 characters without encountering stop punctuation(it terminates such a
196     * sentence with "." or "...").
197     * @param seed the seed for the random decisions this makes, as a long; any long can be used
198     * @return a String generated from the analyzed corpus text's word placement, usually a small sentence
199     */
200    public String chain(long seed) {
201        return chain(seed, 200);
202    }
203
204    /**
205     * Generate a roughly-sentence-sized piece of text based on the previously analyzed corpus text (using
206     * {@link #analyze(CharSequence)}) that terminates when stop punctuation is used (".", "!", "?", or "...") or once
207     * the maxLength would be exceeded by any other words (it terminates such a sentence with "." or "...").
208     * @param seed the seed for the random decisions this makes, as a long; any long can be used
209     * @param maxLength the maximum length for the generated String, in number of characters
210     * @return a String generated from the analyzed corpus text's word placement, usually a small sentence
211     */
212    public String chain(long seed, int maxLength) {
213        int before = 0, pair = 0;
214        boolean later;
215        long state;
216        StringBuilder sb = new StringBuilder(1000);
217        int[] rf;
218        while (sb.length() < maxLength) {
219            later = (pair != 0);
220            rf = processed[pairs.get(pair)];
221            // This is LightRNG's algorithm to generate a random long given sequential states
222            state = ((state = ((state = ((seed += 0x9E3779B97F4A7C15L) ^ seed >>> 30) * 0xBF58476D1CE4E5B9L) ^ state >>> 27) * 0x94D049BB133111EBL) ^ state >>> 31);
223            // get a random int (using half the bits of our previously-calculated state) that is less than size
224            int column = (int) ((rf.length * (state & 0xFFFFFFFFL)) / 0x300000000L) * 3; // divide by 2^32, round down to multiple of 3
225            // use the other half of the bits of state to get a double, compare to probability and choose either the
226            // current column or the alias for that column based on that probability
227            //before = ((state >>> 33) > rf[column]) ? rf[column + 1] : rf[column + 2];
228            if((state >>> 48) > rf[column])
229                before = rf[column + 1];
230            else
231                before = rf[column + 2];
232            if(before > 1)
233            {
234                if(sb.length() + 1 < maxLength)
235                {
236                    sb.append(chars[before]);
237                    pair = pair << 16 | (before & 0xFFFF);
238                }
239                else
240                {
241                    break;
242                }
243            }
244            else
245            {
246                break;
247            }
248            
249        }
250        return sb.toString();
251    }

253    /**
254     * Returns a representation of this MarkovText as a String; use {@link #deserializeFromString(String)} to get a
255     * MarkovText back from this String. The {@link #chars} and {@link #processed} fields must have been given values by
256     * either direct assignment, calling {@link #analyze(CharSequence)}, or building this MarkovTest with the
257     * aforementioned deserializeToString method. Uses spaces to separate words and a tab to separate the two fields.
258     * @return a String that can be used to store the analyzed words and frequencies in this MarkovText
259     */
260    public String serializeToString()
261    {
262        return String.valueOf(chars) + "\t" + StringKit.join(",", pairs.keysAsArray()) + "\t" + Converters.convertArrayInt2D.stringify(processed);
263    }
264
265    /**
266     * Recreates an already-analyzed MarkovText given a String produced by {@link #serializeToString()}.
267     * @param data a String returned by {@link #serializeToString()}
268     * @return a MarkovText that is ready to generate text with {@link #chain(long)}
269     */
270    public static MarkovChar deserializeFromString(String data)
271    {
272        int split = data.indexOf('\t');
273        MarkovChar markov = new MarkovChar();
274        markov.chars = data.substring(0, split).toCharArray();
275        int[] arr = Converters.convertArrayInt.restore(data.substring(split+1, split = data.indexOf('\t', split + 1)));
276        markov.pairs = new IntIntOrderedMap(arr, ArrayTools.range(arr.length));
277        markov.processed = Converters.convertArrayInt2D.restore(data.substring(split + 1));
278        return markov;
279    }

281    /**
282     * Copies the String array {@link #chars} and the 2D jagged int array {@link #processed} into a new MarkovText.
283     * None of the arrays will be equivalent references, but the Strings (being immutable) will be the same objects in
284     * both MarkovText instances.
285     * @return a copy of this MarkovText
286     */
287    public MarkovChar copy()
288    {
289        MarkovChar other = new MarkovChar();
290        other.chars = new char[chars.length];
291        System.arraycopy(chars, 0, other.chars, 0, chars.length);
292        other.processed = new int[processed.length][];
293        int len;
294        for (int i = 0; i < processed.length; i++) {
295            other.processed[i] = new int[len = processed[i].length];
296            System.arraycopy(processed[i], 0, other.processed[i], 0, len);
297        }
298        return other;
299    }
