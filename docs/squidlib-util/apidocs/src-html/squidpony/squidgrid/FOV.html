011/**
012 * This class provides methods for calculating Field of View in grids. Field of
013 * View (FOV) algorithms determine how much area surrounding a point can be
014 * seen. They return a two dimensional array of doubles, representing the amount
015 * of view (typically sight, but perhaps sound, smell, etc.) which the origin
016 * has of each cell.
017 * <br>
018 * The input resistanceMap is considered the percent of opacity. This resistance
019 * is on top of the resistance applied from the light spreading out. You can
020 * obtain a resistance map easily with the DungeonUtility.generateResistances()
021 * method, which uses defaults for common chars used in SquidLib, but you may
022 * also want to create a resistance map manually if a given char means something
023 * very different in your game. This is easy enough to do by looping over all the
024 * x,y positions in your char[][] map and running a switch statement on each char,
025 * assigning a double to the same x,y position in a double[][]. The value should
026 * be between 0.0 (unblocked) for things light passes through, 1.0 (blocked) for
027 * things light can't pass at all, and possibly other values if you have
028 * translucent obstacles.
029 * <br>
030 * The returned light map is considered the percent of light in the cells.
031 * <br>
032 * Not all implementations are required to provide percentage levels of light.
033 * In such cases the returned values will be 0 for no light and 1.0 for fully
034 * lit. Implementations that return this way note so in their documentation.
035 * Currently, all implementations do provide percentage levels.
036 * <br>
037 * All solvers perform bounds checking so solid borders in the map are not
038 * required.
039 * <br>
040 * Static methods are provided to add together FOV maps in the simple way
041 * (disregarding visibility of distant FOV from a given cell), or the more
042 * practical way for roguelikes (where a cell needs to be within line-of-sight
043 * in the first place for a distant light to illuminate it). The second method
044 * relies on an LOS map, which is essentially the same as a very-high-radius
045 * FOV map and can be easily obtained with calculateLOSMap().
046 * <br>
047 * If you want to iterate through cells that are visible in a double[][] returned
048 * by FOV, you can pass that double[][] to the constructor for Region, and you
049 * can use the Region as a reliably-ordered List of Coord (among other things).
050 * The order Region iterates in is somewhat strange, and doesn't, for example,
051 * start at the center of an FOV map, but it will be the same every time you
052 * create a Region with the same FOV map (or the same visible Coords).
053 *
054 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
055 */

157    /**
158     * Calculates the Field Of View for the provided map from the given x, y
159     * coordinates. Returns a light map where the values represent a percentage
160     * of fully lit.
161     *
162     * The starting point for the calculation is considered to be at the center
163     * of the origin cell. Radius determinations are determined by the provided
164     * RadiusStrategy.
165     *
166     * @param resistanceMap the grid of cells to calculate on; the kind made by {@link squidpony.squidgrid.mapping.DungeonUtility#generateResistances(char[][])}
167     * @param startX the horizontal component of the starting location
168     * @param startY the vertical component of the starting location
169     * @param radius the distance the light will extend to
170     * @param radiusTechnique provides a means to calculate the radius as desired
171     * @return the computed light grid
172     */
173    public double[][] calculateFOV(double[][] resistanceMap, int startX, int startY, double radius, Radius radiusTechnique) {
174        double decay = 1.0 / radius;
175
176        int width = resistanceMap.length;
177        int height = resistanceMap[0].length;
178
179        initializeLightMap(width, height);
180        light[startX][startY] = Math.min(1.0, radius);//make the starting space full power unless radius is tiny
181
182        switch (type) {
183            case RIPPLE:
184            case RIPPLE_LOOSE:
185            case RIPPLE_TIGHT:
186            case RIPPLE_VERY_LOOSE:
187                initializeNearLight(width, height);
188                doRippleFOV(light, rippleValue(type), startX, startY, startX, startY, decay, radius, resistanceMap, nearLight, radiusTechnique);
189                break;
190            case SHADOW:
191                for (Direction d : Direction.DIAGONALS) {
192                    shadowCast(0, d.deltaX, d.deltaY, 0, radius, startX, startY, decay, light, resistanceMap, radiusTechnique);
193                    shadowCast(d.deltaX, 0, 0, d.deltaY, radius, startX, startY, decay, light, resistanceMap, radiusTechnique);
194                }
195                break;
196        }
197        return light;
198    }

200        /**
201     * Calculates the Field Of View for the provided map from the given x, y
202     * coordinates. Returns a light map where the values represent a percentage
203     * of fully lit.
204     *
205     * The starting point for the calculation is considered to be at the center
206     * of the origin cell. Radius determinations are determined by the provided
207     * RadiusStrategy. A conical section of FOV is lit by this method if
208     * span is greater than 0.
209     *
210     * @param resistanceMap the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()
211     * @param startX the horizontal component of the starting location
212     * @param startY the vertical component of the starting location
213     * @param radius the distance the light will extend to
214     * @param radiusTechnique provides a means to calculate the radius as desired
215     * @param angle the angle in degrees that will be the center of the FOV cone, 0 points right
216     * @param span the angle in degrees that measures the full arc contained in the FOV cone
217     * @return the computed light grid
218     */
219    public double[][] calculateFOV(double[][] resistanceMap, int startX, int startY, double radius,
220                                   Radius radiusTechnique, double angle, double span) {
221        double decay = 1.0 / radius;
222        angle = ((angle >= 360.0 || angle < 0.0)
223                ? MathExtras.remainder(angle, 360.0) : angle) * 0.002777777777777778;
224        span = span * 0.002777777777777778;
225        int width = resistanceMap.length;
226        int height = resistanceMap[0].length;
227
228        initializeLightMap(width, height);
229        light[startX][startY] = Math.min(1.0, radius);//make the starting space full power unless radius is tiny
230
231        switch (type) {
232            case RIPPLE:
233            case RIPPLE_LOOSE:
234            case RIPPLE_TIGHT:
235            case RIPPLE_VERY_LOOSE:
236                initializeNearLight(width, height);
237                doRippleFOV(light, rippleValue(type), startX, startY, startX, startY, decay, radius, resistanceMap, nearLight, radiusTechnique, angle, span);
238                break;
239            case SHADOW:
240                light = shadowCastLimited(1, 1.0, 0.0, 0, 1, 1, 0, radius, startX, startY, decay, light, resistanceMap, radiusTechnique, angle, span);
241                light = shadowCastLimited(1, 1.0, 0.0, 1, 0, 0, 1, radius, startX, startY, decay, light, resistanceMap, radiusTechnique, angle, span);
242
243                light = shadowCastLimited(1, 1.0, 0.0, 0, -1, 1, 0, radius, startX, startY, decay, light, resistanceMap, radiusTechnique, angle, span);
244                light = shadowCastLimited(1, 1.0, 0.0, -1, 0, 0, 1, radius, startX, startY, decay, light, resistanceMap, radiusTechnique, angle, span);
245
246                light = shadowCastLimited(1, 1.0, 0.0, 0, -1, -1, 0, radius, startX, startY, decay, light, resistanceMap, radiusTechnique, angle, span);
247                light = shadowCastLimited(1, 1.0, 0.0, -1, 0, 0, -1, radius, startX, startY, decay, light, resistanceMap, radiusTechnique, angle, span);
248
249                light = shadowCastLimited(1, 1.0, 0.0, 0, 1, -1, 0, radius, startX, startY, decay, light, resistanceMap, radiusTechnique, angle, span);
250                light = shadowCastLimited(1, 1.0, 0.0, 1, 0, 0, -1, radius, startX, startY, decay, light, resistanceMap, radiusTechnique, angle, span);
251                break;
252        }
253
254        return light;
255    }

334        shadowCast(1, 0, 0, 1, radius, startX, startY, decay, light, resistanceMap, radiusTechnique);

340        shadowCast(-1, 0, 0, -1, radius, startX, startY, decay, light, resistanceMap, radiusTechnique);

344        return light;

358     * @param resistanceMap the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()
359     * @param light the grid of cells to assign to; may have existing values, and 0.0 is used to mean "unlit"

361     * @param startY the vertical component of the starting location

367    {
368        double decay = 1.0 / radius;
369        ArrayTools.fill(light, 0);
370        light[startX][startY] = Math.min(1.0, radius);//make the starting space full power unless radius is tiny

372
373        shadowCast(0, 1, 1, 0, radius, startX, startY, decay, light, resistanceMap, radiusTechnique);
374        for (int row = 0; row <= radius + 1.0; row++) {
375            for (int col = Math.max(1,row); col <= radius + 1.0; col++) {
376                if(startX - col >= 0 && startY - row >= 0 && resistanceMap[startX - col][startY - row] < 1.0 &&
377                        !shadowCastCheck(1, 1.0, 0.0, 0, -1, -1, 0, radius, startX - col, startY - row, decay, light, resistanceMap, radiusTechnique, 0, 0, light.length, light[0].length, startX, startY))

379            }
380        }
381        shadowCast(1, 0, 0, 1, radius, startX, startY, decay, light, resistanceMap, radiusTechnique);
382        for (int col = 0; col <= radius + 1.0; col++) {
383            for (int row = Math.max(1,col); row <= radius + 1.0; row++) {
384                if(startX - col >= 0 && startY - row >= 0 && resistanceMap[startX - col][startY - row] < 1.0 &&
385                        !shadowCastCheck(1, 1.0, 0.0, -1, 0, 0, -1, radius, startX - col, startY - row, decay, light, resistanceMap, radiusTechnique, 0, 0, light.length, light[0].length, startX, startY))
386                    light[startX - col][startY - row] = 0.0;
387            }
388        }
389
390        shadowCast(0, 1, -1, 0, radius, startX, startY, decay, light, resistanceMap, radiusTechnique);
391        for (int row = 0; row <= radius + 1.0; row++) {
392            for (int col = Math.max(1,row); col <= radius + 1.0; col++) {
393                if(startX - col >= 0 && startY + row < light[0].length &&  resistanceMap[startX - col][startY + row] < 1.0 &&
394                        !shadowCastCheck(1, 1.0, 0.0, 0, -1, 1, 0, radius, startX - col, startY + row, decay, light, resistanceMap, radiusTechnique, 0, 0, light.length, light[0].length, startX, startY))
395                    light[startX - col][startY + row] = 0.0;
396            }
397        }
398        shadowCast(1, 0, 0, -1, radius, startX, startY, decay, light, resistanceMap, radiusTechnique);
399        for (int col = 0; col <= radius + 1.0; col++) {
400            for (int row = Math.max(1,col); row <= radius + 1.0; row++) {
401                if(startX - col >= 0 && startY + row < light[0].length && resistanceMap[startX - col][startY + row] < 1.0 &&
402                        !shadowCastCheck(1, 1.0, 0.0, -1, 0, 0, 1, radius, startX - col, startY + row, decay, light, resistanceMap, radiusTechnique, 0, 0, light.length, light[0].length, startX, startY))
403                    light[startX - col][startY + row] = 0.0;
404            }
405        }
406
407        shadowCast(0, -1, -1, 0, radius, startX, startY, decay, light, resistanceMap, radiusTechnique);
408        for (int row = 0; row <= radius + 1.0; row++) {
409            for (int col = Math.max(1,row); col <= radius + 1.0; col++) {
410                if(startX + col < light.length && startY + row < light[0].length && resistanceMap[startX + col][startY + row] < 1.0 &&
411                        !shadowCastCheck(1, 1.0, 0.0, 0, 1, 1, 0, radius, startX + col, startY + row, decay, light, resistanceMap, radiusTechnique, 0, 0, light.length, light[0].length, startX, startY))
412                    light[startX + col][startY + row] = 0.0;
413            }
414        }
415        shadowCast(-1, 0, 0, -1, radius, startX, startY, decay, light, resistanceMap, radiusTechnique);
416        for (int col = 0; col <= radius + 1.0; col++) {
417            for (int row = Math.max(1,col); row <= radius + 1.0; row++) {
418                if(startX + col < light.length && startY + row < light[0].length && resistanceMap[startX + col][startY + row] < 1.0 &&
419                        !shadowCastCheck(1, 1.0, 0.0, 1, 0, 0, 1, radius, startX + col, startY + row, decay, light, resistanceMap, radiusTechnique, 0, 0, light.length, light[0].length, startX, startY))
420                    light[startX + col][startY + row] = 0.0;
421            }
422        }
423
424        shadowCast(0, -1, 1, 0, radius, startX, startY, decay, light, resistanceMap, radiusTechnique);
425        for (int row = 0; row <= radius + 1.0 && startY + row < light[0].length; row++) {
426            for (int col = Math.max(1,row); col <= radius + 1.0; col++) {
427                if(startX + col < light.length && startY - row >= 0 && resistanceMap[startX + col][startY - row] < 1.0 &&
428                        !shadowCastCheck(1, 1.0, 0.0, 0, 1, -1, 0, radius, startX + col, startY - row, decay, light, resistanceMap, radiusTechnique, 0, 0, light.length, light[0].length, startX, startY))
429                    light[startX + col][startY - row] = 0.0;
430            }
431        }
432        shadowCast(-1, 0, 0, 1, radius, startX, startY, decay, light, resistanceMap, radiusTechnique);
433        for (int col = 0; col <= radius + 1.0; col++) {
434            for (int row = Math.max(1,col); row <= radius + 1.0; row++) {
435                if(startX + col < light.length && startY - row >= 0 && resistanceMap[startX + col][startY - row] < 1.0 &&
436                        !shadowCastCheck(1, 1.0, 0.0, 1, 0, 0, -1, radius, startX + col, startY - row, decay, light, resistanceMap, radiusTechnique, 0, 0, light.length, light[0].length, startX, startY))
437                    light[startX + col][startY - row] = 0.0;
438            }
439        }
440        return light;
441    }
442    /**
443     * Calculates which cells have line of sight from the given x, y coordinates.
444     * Assigns to, and returns, a light map where the values
445     * are always either 0.0 for "not in line of sight" or 1.0 for "in line of
446     * sight," which doesn't mean a cell is actually visible if there's no light
447     * in that cell. Always uses shadowcasting FOV, which allows this method to
448     * be static since it doesn't need to keep any state around, and can reuse the
449     * state the user gives it via the {@code light} parameter. The values in light
450     * are always cleared before this is run, because prior state can make this give
451     * incorrect results.
452     * <br>
453     * The starting point for the calculation is considered to be at the center
454     * of the origin cell. Radius determinations are pretty much irrelevant because
455     * the distance doesn't matter, only the presence of a clear line, but this uses
456     * {@link Radius#SQUARE} if it matters.
457     * @param resistanceMap the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()
458     * @param light the grid of cells to assign to; may have existing values, and 0.0 is used to mean "no line"
459     * @param startX the horizontal component of the starting location
460     * @param startY the vertical component of the starting location

556    /**
557     * Reuses the existing light 2D array and fills it with a straight-line bouncing path of light that reflects its way
558     * through the given resistanceMap from startX, startY until it uses up the given distance. The angle the path
559     * takes is given in degrees, and the angle used can change as obstacles are hit (reflecting backwards if it hits a
560     * corner pointing directly into or away from its path). This can be used something like an LOS method, but because
561     * the path can be traveled back over, an array or Queue becomes somewhat more complex, and the decreasing numbers
562     * for a straight line that stack may make more sense for how this could be used (especially with visual effects).
563     * This currently allows the path to pass through single-cell wall-like obstacles without changing direction, e.g.
564     * it passes through pillars, but will bounce if it hits a bigger wall.
565     * @param resistanceMap the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()
566     * @param light the grid of cells to assign to; may have existing values, and 0.0 is used to mean "unlit"
567     * @param startX the horizontal component of the starting location
568     * @param startY the vertical component of the starting location
569     * @param distance the distance the light will extend to
570     * @param angle in degrees, the angle to start the path traveling in
571     * @return the given light parameter, after modifications
572     */
573    public static double[][] bouncingLine(double[][] resistanceMap, double[][] light, int startX, int startY, double distance, double angle)
574    {
575        double rad = Math.max(1, distance);
576        double decay = 1.0 / rad;
577        ArrayTools.fill(light, 0);
578        light[startX][startY] = 1;//make the starting space full power
579        angle = Math.toRadians((angle > 360.0 || angle < 0.0)
580                ? MathExtras.remainder(angle, 360.0) : angle);
581        float s = (float) NumberTools.sin(angle),
582                c = (float) NumberTools.cos(angle);
583        double deteriorate = 1.0;
584        int dx, dy, width = resistanceMap.length, height = resistanceMap[0].length;
585        for (int d = 1; d <= rad; ) {
586            dx = startX + Math.round(c * d);
587            if(dx < 0 || dx > width)
588                break;
589            dy = startY + Math.round(s * d);
590            if(dy < 0 || dy > height)
591                break;
592            deteriorate -= decay;
593            //check if it's within the lightable area and light if needed
594            if (deteriorate > 0.0) {
595                light[dx][dy] = Math.min(light[dx][dy] + deteriorate, 1.0);
596                if (resistanceMap[dx][dy] >= 1 && deteriorate > decay)
597                {
598                    startX = dx;
599                    startY = dy;
600                    d = 1;
601                    double flipX = resistanceMap[startX + Math.round(-c * d)][dy],
602                            flipY = resistanceMap[dx][startY + Math.round(-s * d)];
603                    if(flipX >= 1.0)
604                        s = -s;
605                    if(flipY >= 1.0)
606                        c = -c;
607                }
608                else ++d;
609
610            }
611            else break;
612        }
613
614        return light;
615    }

703                }

971                double rightSlope = (deltaX + 0.5f) / (deltaY - 0.5f);
972

974                    continue;
975                } else if (end > leftSlope) {
976                    break;
977                }
978                double deltaRadius = radiusStrategy.radius(deltaX, deltaY),
979                        at2 = Math.abs(angle - NumberTools.atan2_(currentY - starty, currentX - startx));// + 1.0) % 1.0;
980                //check if it's within the lightable area and light if needed
981                if (deltaRadius <= radius
982                        && (at2 <= span * 0.5
983                        || at2 >= 1.0 - span * 0.5)) {
984                    double bright = 1 - decay * deltaRadius;
985                    lightMap[currentX][currentY] = bright;
986                }
987
988                if (blocked) { //previous cell was a blocking one
989                    if (map[currentX][currentY] >= 1) {//hit a wall

994                    }
995                } else {

1001                }
1002            }

1008    /**
1009     * Calculates the Field Of View for the provided map from the given x, y
1010     * coordinates, lighting with the view "pointed at" the given {@code angle} in degrees,
1011     * extending to different ranges based on the direction the light is traveling.
1012     * The direction ranges are {@code forward}, {@code sideForward}, {@code side},
1013     * {@code sideBack}, and {@code back}; all are multiplied by {@code radius}.
1014     * Assigns to, and returns, a light map where the values represent a percentage of fully
1015     * lit. The values in light are cleared before this is run, because prior state can make
1016     * this give incorrect results. You can use {@link #addFOVsInto(double[][], double[][]...)}
1017     * if you want to mix FOV results, which works as an alternative to using the prior light state.
1018     * <br>
1019     * The starting point for the calculation is considered to be at the center
1020     * of the origin cell. Radius determinations are determined by the provided
1021     * RadiusStrategy. If all direction ranges are the same, this acts like
1022     * {@link #reuseFOV(double[][], double[][], int, int, double, Radius)}; otherwise
1023     * may produce conical shapes (potentially more than one, or overlapping ones).
1024     *
1025     * @param resistanceMap the grid of cells to calculate on; the kind made by DungeonUtility.generateResistances()
1026     * @param light the grid of cells to assign to; may have existing values, and 0.0 is used to mean "unlit"
1027     * @param startX the horizontal component of the starting location
1028     * @param startY the vertical component of the starting location
1029     * @param radius the distance the light will extend to (roughly); direction ranges will be multiplied by this
1030     * @param radiusTechnique provides a means to shape the FOV by changing distance calculation (circle, square, etc.)
1031     * @param angle the angle in degrees that will be the center of the FOV cone, 0 points right
1032     * @param forward the range to extend when the light is within 22.5 degrees of angle; will be interpolated with sideForward
1033     * @param sideForward the range to extend when the light is between 22.5 and 67.5 degrees of angle; will be interpolated with forward or side
1034     * @param side the range to extend when the light is between 67.5 and 112.5 degrees of angle; will be interpolated with sideForward or sideBack
1035     * @param sideBack the range to extend when the light is between 112.5 and 157.5 degrees of angle; will be interpolated with side or back
1036     * @param back the range to extend when the light is more than 157.5 degrees away from angle; will be interpolated with sideBack
1037     * @return the computed light grid (the same as {@code light})
1038     */
1039    public static double[][] reuseFOV(double[][] resistanceMap, double[][] light, int startX, int startY,
1040                                      double radius, Radius radiusTechnique, double angle,
1041                                      double forward, double sideForward, double side, double sideBack, double back) {
1042        directionRanges[0] = forward * radius;
1043        directionRanges[7] = directionRanges[1] = sideForward * radius;
1044        directionRanges[6] = directionRanges[2] = side * radius;
1045        directionRanges[5] = directionRanges[3] = sideBack * radius;
1046        directionRanges[4] = back * radius;
1047
1048        radius = Math.max(1, radius);
1049        ArrayTools.fill(light, 0);
1050        light[startX][startY] = 1;//make the starting space full power
1051        angle = ((angle >= 360.0 || angle < 0.0)
1052                ? MathExtras.remainder(angle, 360.0) : angle) * 0.002777777777777778;
1053
1054
1055        light = shadowCastPersonalized(1, 1.0, 0.0, 0, 1, 1, 0,   radius, startX, startY, light, resistanceMap, radiusTechnique, angle, directionRanges);
1056        light = shadowCastPersonalized(1, 1.0, 0.0, 1, 0, 0, 1,   radius, startX, startY, light, resistanceMap, radiusTechnique, angle, directionRanges);
1057        light = shadowCastPersonalized(1, 1.0, 0.0, 0, -1, 1, 0,  radius, startX, startY, light, resistanceMap, radiusTechnique, angle, directionRanges);
1058        light = shadowCastPersonalized(1, 1.0, 0.0, -1, 0, 0, 1,  radius, startX, startY, light, resistanceMap, radiusTechnique, angle, directionRanges);
1059        light = shadowCastPersonalized(1, 1.0, 0.0, 0, -1, -1, 0, radius, startX, startY, light, resistanceMap, radiusTechnique, angle, directionRanges);
1060        light = shadowCastPersonalized(1, 1.0, 0.0, -1, 0, 0, -1, radius, startX, startY, light, resistanceMap, radiusTechnique, angle, directionRanges);
1061        light = shadowCastPersonalized(1, 1.0, 0.0, 0, 1, -1, 0,  radius, startX, startY, light, resistanceMap, radiusTechnique, angle, directionRanges);
1062        light = shadowCastPersonalized(1, 1.0, 0.0, 1, 0, 0, -1,  radius, startX, startY, light, resistanceMap, radiusTechnique, angle, directionRanges);
1063        return light;
1064    }

1067                                                     double radius, int startx, int starty, double[][] lightMap,
1068                                                     double[][] map, Radius radiusStrategy, double angle, final double[] directionRanges) {
1069        double newStart = 0;
1070        if (start < end) {
1071            return lightMap;
1072        }
1073        int width = lightMap.length;
1074        int height = lightMap[0].length;
1075
1076        boolean blocked = false;
1077        for (int distance = row; distance <= radius && distance < width + height && !blocked; distance++) {
1078            int deltaY = -distance;
1079            for (int deltaX = -distance; deltaX <= 0; deltaX++) {
1080                int currentX = startx + deltaX * xx + deltaY * xy;
1081                int currentY = starty + deltaX * yx + deltaY * yy;
1082                double leftSlope = (deltaX - 0.5f) / (deltaY + 0.5f);
1083                double rightSlope = (deltaX + 0.5f) / (deltaY - 0.5f);
1084
1085                if (!(currentX >= 0 && currentY >= 0 && currentX < width && currentY < height) || start < rightSlope) {
1086                    continue;
1087                } else if (end > leftSlope) {
1088                    break;
1089                }
1090                double at2 = Math.abs(angle - NumberTools.atan2_(currentY - starty, currentX - startx)) * 8.0,
1091                        deltaRadius = radiusStrategy.radius(deltaX, deltaY);
1092                int ia = (int)(at2), low = ia & 7, high = ia + 1 & 7;
1093                double a = at2 - ia, adjRadius = (1.0 - a) * directionRanges[low] + a * directionRanges[high];
1094                //check if it's within the lightable area and light if needed
1095                if (deltaRadius <= adjRadius) {
1096                    lightMap[currentX][currentY] = 1.0 - (deltaRadius / (adjRadius + 1.0)); // how bright the tile is
1097                }
1098
1099                if (blocked) { //previous cell was a blocking one

1101                        newStart = rightSlope;
1102                    } else {

1105                    }
1106                } else {

1108                        blocked = true;
1109                        lightMap = shadowCastPersonalized(distance + 1, start, leftSlope, xx, xy, yx, yy, radius, startx, starty, lightMap, map, radiusStrategy, angle, directionRanges);
1110                        newStart = rightSlope;
1111                    }
1112                }
1113            }
1114        }
1115        return lightMap;
