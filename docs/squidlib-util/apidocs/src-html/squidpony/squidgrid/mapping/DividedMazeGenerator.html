001package squidpony.squidgrid.mapping;
002
003import squidpony.squidgrid.Direction;
004import squidpony.squidmath.GWTRNG;
005import squidpony.squidmath.IRNG;
006import squidpony.squidmath.IntVLA;
007
008import java.util.ArrayDeque;
009
010/**
011 * Recursively divided maze. Creates only walls and passages.
012 * <p>
013 * This dungeon generator is based on a port of the rot.js version.
014 *
015 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
016 */
017public class DividedMazeGenerator {
018
019    private class DividedMazeRoom {
020
021        private int left, top, right, bottom;
022
023        public DividedMazeRoom(int left, int top, int right, int bottom) {
024            this.left = left;
025            this.top = top;
026            this.right = right;
027            this.bottom = bottom;
028        }
029    }
030
031    private int width, height;
032    private boolean[][] map;
033    private IRNG rng;
034
035    /**
036     * Sets up the generator to make mazes the given width and height. The mazes
037     * have a solid wall border.
038     *
039     * @param width
040     * @param height
041     */
042    public DividedMazeGenerator(int width, int height) {
043        this.width = width;
044        this.height = height;
045        rng = new GWTRNG();
046    }
047
048    /**
049     * Sets up the generator to make mazes the given width and height. The mazes
050     * have a solid wall border.
051     *
052     * @param width  in cells
053     * @param height in cells
054     * @param rng    the random number generator to use
055     */
056    public DividedMazeGenerator(int width, int height, IRNG rng) {
057        this.width = width;
058        this.height = height;
059        this.rng = rng;
060    }
061
062    /**
063     * Builds a maze. True values represent walls.
064     *
065     * @return
066     */
067    public boolean[][] create() {
068        map = new boolean[width][height];
069
070        for (int x = 0; x < width; x++) {
071            for (int y = 0; y < height; y++) {
072                map[x][y] = x == 0 || y == 0 || x + 1 == width || y + 1 == height;
073            }
074        }
075
076        process();
077
078        return map;
079    }
080
081    private void process() {
082        ArrayDeque<DividedMazeRoom> stack = new ArrayDeque<>();
083        stack.offer(new DividedMazeRoom(1, 1, width - 2, height - 2));
084        while (!stack.isEmpty()) {
085            DividedMazeRoom room = stack.removeFirst();
086            IntVLA availX = new IntVLA(), availY = new IntVLA();
087
088            for (int x = room.left + 1; x < room.right; x++) {
089                boolean top = map[x][room.top - 1];
090                boolean bottom = map[x][room.bottom + 1];
091                if (top && bottom && (x & 1) == 0) {
092                    availX.add(x);
093                }
094            }
095
096            for (int y = room.top + 1; y < room.bottom; y++) {
097                boolean left = map[room.left - 1][y];
098                boolean right = map[room.right + 1][y];
099                if (left && right && (y & 1) == 0) {
100                    availY.add(y);
101                }
102            }
103
104            if (availX.isEmpty() || availY.isEmpty()) {
105                continue;
106            }
107
108            int x2 = availX.getRandomElement(rng);
109            int y2 = availY.getRandomElement(rng);
110
111            map[x2][y2] = true;
112
113            for (Direction dir : Direction.CARDINALS) {
114                switch (dir) {
115                    case LEFT:
116                        for (int x = room.left; x < x2; x++) {
117                            map[x][y2] = true;
118                        }
119                        break;
120                    case RIGHT:
121                        for (int x = x2 + 1; x <= room.right; x++) {
122                            map[x][y2] = true;
123                        }
124                        break;
125                    case UP:
126                        for (int y = room.top; y < y2; y++) {
127                            map[x2][y] = true;
128                        }
129                        break;
130                    case DOWN:
131                        for (int y = y2 + 1; y <= room.bottom; y++) {
132                            map[x2][y] = true;
133                        }
134                        break;
135                    case NONE:
136                        break;
137                    case DOWN_LEFT:
138                    case DOWN_RIGHT:
139                    case UP_LEFT:
140                    case UP_RIGHT:
141                        throw new IllegalStateException("There should only be cardinal directions here");
142                }
143            }
144
145            //// the commented code below actually works fine, but it's a little wasteful on RAM
146//            ArrayList<Direction> dirs = Maker.makeList(Direction.CARDINALS);
147//            dirs.remove(rng.getRandomElement(dirs));
148
149            //// not sure if the order being random matters, but this will remove a random item
150            Direction[] dirs = rng.randomPortion(Direction.CARDINALS, new Direction[3]);
151
152            for (Direction dir : dirs) {
153                switch (dir) {
154                    case LEFT:
155                        map[rng.between(room.left, x2)][y2] = false;
156                        break;
157                    case RIGHT:
158                        map[rng.between(x2 + 1, room.right + 1)][y2] = false;
159                        break;
160                    case UP:
161                        map[x2][rng.between(room.top, y2)] = false;
162                        break;
163                    case DOWN:
164                        map[x2][rng.between(y2 + 1, room.bottom + 1)] = false;
165                        break;
166                    case NONE:
167                        break;
168                    case DOWN_LEFT:
169                    case DOWN_RIGHT:
170                    case UP_LEFT:
171                    case UP_RIGHT:
172                        throw new IllegalStateException("There should only be cardinal directions here");
173                }
174            }
175
176            stack.offer(new DividedMazeRoom(room.left, room.top, x2 - 1, y2 - 1));
177            stack.offer(new DividedMazeRoom(x2 + 1, room.top, room.right, y2 - 1));
178            stack.offer(new DividedMazeRoom(room.left, y2 + 1, x2 - 1, room.bottom));
179            stack.offer(new DividedMazeRoom(x2 + 1, y2 + 1, room.right, room.bottom));
180        }
181    }
182
183}
