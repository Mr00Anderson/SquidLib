014/**
015 * When you have a world map as produced by {@link WorldMapGenerator}, you may want to fill it with claims by various
016 * factions, where each faction may be hand-made and may consist of humans or some fantasy species, such as goblins,
017 * elves, or demons. This can assign contiguous areas of land to various factions, while acknowledging any preferences
018 * some species may have for specific types of land (elves may strongly prefer forest terrain, or flying demons may be
019 * the ideal residents for difficult mountainous terrain). This needs both a {@link WorldMapGenerator}, preferably
020 * {@link squidpony.squidgrid.mapping.WorldMapGenerator.TilingMap} or
021 * {@link squidpony.squidgrid.mapping.WorldMapGenerator.EllipticalMap} because they distribute area equally across the
022 * map, and a {@link squidpony.squidgrid.mapping.WorldMapGenerator.BiomeMapper} to allocate biomes and height/moisture
023 * info.
024 */
025@Beta
026public class FantasyPoliticalMapper implements Serializable {
027    private static final long serialVersionUID = 0L;

029    /**
030     * Represents a group that claims territory on a world-map, such as a nation. Each Faction has a name, a short name
031     * that may be the same as the regular name, a FakeLanguageGen that would be used to generate place names in that
032     * Faction's territory, and a number of (possibly null) arrays and Sets that represent what terrains this Faction
033     * wants to claim and which it will avoid (e.g. elves may prefer claiming forests, while frost giants won't claim
034     * anywhere that's even remotely warm).
035     */
036    public static class Faction implements Serializable

040        public String name, shortName;
041        public FakeLanguageGen language;
042        /**
043         * An UnorderedSet of String keys, where each key is the name of a biome this Faction wants to occupy.
044         * May be null if no biomes are specifically preferred.
045         */
046        public HashSet<String> preferredBiomes;
047        /**
048         * An UnorderedSet of String keys, where each key is the name of a biome this Faction will never occupy.
049         * May be null if all biomes are available, but unless this is specified in a constructor, the default will be
050         * to consider "Ocean" blocked.
051         */
052        public HashSet<String> blockedBiomes;
053
054        /**
055         * An int array of height codes that this Faction prefers; 0, 1, 2, and 3 are all oceans, while 4 is shoreline
056         * or low-lying land and higher numbers (up to 8, inclusive) are used for increasing elevations.
057         */
058        public int[] preferredHeight;
059        /**
060         * An int array of heat codes that this Faction prefers; typically a 6-code scale is used where 0, 1, and 2 are
061         * cold and getting progressively warmer, while 3, 4, and 5 are warm to warmest.
062         */
063        public int[] preferredHeat;
064        /**
065         * An int array of moisture codes that this Faction prefers; typically a 6-code scale is used where 0, 1, and 2
066         * are dry and getting progressively more precipitation, while 3, 4, and 5 are wet to wettest.
067         */
068        public int[] preferredMoisture;

292    /**
293     * For when you really don't care what arguments you give this, you can use this zero-parameter overload of
294     * generate() to produce a 128x128 {@link squidpony.squidgrid.mapping.WorldMapGenerator.TilingMap} world map with a
295     * {@link squidpony.squidgrid.mapping.WorldMapGenerator.SimpleBiomeMapper} biome mapper, filling it with 30 random
296     * Factions and trying to avoid unclaimed land. You may need to use {@link #atlas} to make sense of the randomly
297     * generated Factions. The seed will be random here.
298     * @return a 2D char array where each char can be used as a key into {@link #atlas} to find the Faction that claims it
299     */
300    public char[][] generate() {
301        wmg = new WorldMapGenerator.TilingMap(rng.nextLong(),128, 128);
302        wmg.generate();
303        biomeMapper = new WorldMapGenerator.SimpleBiomeMapper();
304        biomeMapper.makeBiomes(wmg);
305        return generate(rng.nextLong(), wmg, biomeMapper, null, 30, 1.0);
306    }

308    /**
309     * Generates a 2D char array that represents the claims to the land described by the WorldMapGenerator {@code wmg}
310     * and the BiomeMapper {@code biomeMapper} by various Factions, where {@link Faction} is an inner class.
311     * This starts with two default Factions for "Ocean" and "Wilderness" (unclaimed land) and adds randomly generated
312     * Factions to fill factionCount (the two default factions aren't counted against this limit). These Factions
313     * typically claim contiguous spans of land stretching out from a starting point that matches the Faction's
314     * preferences for biome, land height, heat, and moisture. If a Faction requires a biome (like "TropicalRainforest") 
315     * and the world has none of that type, then that Faction won't claim any land. If the WorldMapGenerator zooms in or
316     * out, you should call {@link #adjustZoom()} to get a different 2D char array that represents the zoomed-in area.
317     * This overload tries to claim all land that can be reached by an existing Faction, though islands will often be
318     * unclaimed.

321     * @param biomeMapper a WorldMapGenerator.BiomeMapper, which must have been initialized with wmg and refer to the same world
322     * @param factionCount the number of factions to have claiming land; cannot be negative or more than 253
323     * @return a 2D char array where each char can be used as a key into {@link #atlas} to find the Faction that claims it
324     */

351    /**
352     * Generates a 2D char array that represents the claims to the land described by the WorldMapGenerator {@code wmg}
353     * and the BiomeMapper {@code biomeMapper} by various Factions, where {@link Faction} is an inner class.
354     * This starts with two default Factions for "Ocean" and "Wilderness" (unclaimed land) and adds all of
355     * {@code factions} until {@code factionCount} is reached; if it isn't reached, random Factions will be generated to
356     * fill factionCount (the two default factions aren't counted against this limit). These Factions typically claim
357     * contiguous spans of land stretching out from a starting point that matches the Faction's preferences for biome,
358     * land height, heat, and moisture. If a Faction requires a biome (like "TropicalRainforest") and the world has none
359     * of that type, then that Faction won't claim any land. If the WorldMapGenerator zooms in or out, you should call
360     * {@link #adjustZoom()} to get a different 2D char array that represents the zoomed-in area. This overload tries to
361     * claim the given {@code controlledFraction} of land in total, though 1.0 can rarely be reached unless there are
362     * many factions and few islands.
363     *
364     * @param seed the seed that determines how Factions will randomly spread around the world
365     * @param wmg a WorldMapGenerator, which must have produced a map by calling its generate() method
366     * @param biomeMapper a WorldMapGenerator.BiomeMapper, which must have been initialized with wmg and refer to the same world
367     * @param factions a Collection of {@link Faction} that will be copied, shuffled and used before adding any random Factions
368     * @param factionCount the number of factions to have claiming land; cannot be negative or more than 253
369     * @param controlledFraction between 0.0 and 1.0 inclusive; higher means more land has a letter, lower has more '%'
370     * @return a 2D char array where each char can be used as a key into {@link #atlas} to find the Faction that claims it
371     */
372    public char[][] generate(long seed, WorldMapGenerator wmg, WorldMapGenerator.BiomeMapper biomeMapper,
373                                  Collection<Faction> factions, int factionCount, double controlledFraction) {
374        rng.setState(seed);
375        factionCount = Math.abs(factionCount % 254);
376        Thesaurus th = new Thesaurus(rng.nextLong());
377        th.addKnownCategories();
378        ArrayList<Faction> fact = factions == null ? new ArrayList<Faction>() : rng.shuffle(factions);
379        for (int i = fact.size(); i < factionCount; i++) {
380            String name = th.makeNationName(), shortName = th.latestGenerated;
381            FakeLanguageGen lang;
382            if(th.randomLanguages == null || th.randomLanguages.isEmpty())
383                lang = FakeLanguageGen.randomLanguage(rng);
384            else
385                lang = th.randomLanguages.get(0);
386            fact.add(new Faction(lang, name, shortName));
387        }
388        if(factionCount > 0)
389            rng.shuffleInPlace(fact);
390        fact.add(0, new Faction(FakeLanguageGen.DEMONIC, "The Lost Wilderness", "Wilderness"));
391        fact.add(0, new Faction(FakeLanguageGen.ALIEN_O, "The Vast Domain of the Seafolk", "Seafolk", new String[]{"Ocean"}));
392        atlas = new OrderedMap<>(letters.subList(0, factionCount + 2), fact.subList(0, factionCount + 2), 0.5f);
393        this.wmg = wmg;
394        this.biomeMapper = biomeMapper;
395        width = wmg.width;
396        height = wmg.height;
397        GreasedRegion land = new GreasedRegion(wmg.heightCodeData, 4, 999);
398        politicalMap = land.toChars('%', '~');
399        int controlled = (int) (land.size() * Math.max(0.0, Math.min(1.0, controlledFraction)));
400
401        int[] centers = land.copy().randomScatter(rng, (int) (Math.sqrt(width * height) * 0.1 + 0.999), factionCount).asTightEncoded();
402        int cen, cx, cy, cx2, cy2, biome, high, hot, moist, count = centers.length, re;
403        String biomeName;
404        String[] biomeTable = biomeMapper.getBiomeNameTable();
405        int[] reorder = new int[count];
406        Faction current;
407        int[] factionIndices = new int[count];
408        for (int c = 0; c < count; c++) {
409            cen = centers[c];
410            cx = cen % width;
411            cy = cen / width;
412            biome = biomeMapper.getBiomeCode(cx, cy);
413            biomeName = biomeTable[biome];
414            high = wmg.heightCodeData[cx][cy];
415            hot = biomeMapper.getHeatCode(cx, cy);
416            moist = biomeMapper.getMoistureCode(cx, cy);
417            rng.randomOrdering(count, reorder);
418            PER_FACTION:
419            for (int i = 0; i < count; i++) {
420                current = fact.get(re = reorder[i]);
421                if(current.preferredBiomes == null || current.preferredBiomes.contains(biomeName))
422                {
423                    factionIndices[c] = re;
424                    break;
425                }
426                if(current.blockedBiomes != null && current.blockedBiomes.contains(biomeName))
427                    continue;
428                if(current.preferredHeight != null)
429                {
430                    for (int j = 0; j < current.preferredHeight.length; j++) {
431                        if(high == current.preferredHeight[j])
432                        {
433                            factionIndices[c] = re;
434                            break PER_FACTION;
435                        }
436                    }
437                }
438                if(current.preferredHeat != null)
439                {
440                    for (int j = 0; j < current.preferredHeat.length; j++) {
441                        if(hot == current.preferredHeat[j])
442                        {
443                            factionIndices[c] = re;
444                            break PER_FACTION;
445                        }
446                    }
447                }
448                if(current.preferredMoisture != null)
449                {
450                    for (int j = 0; j < current.preferredMoisture.length; j++) {
451                        if(moist == current.preferredMoisture[j])
452                        {
453                            factionIndices[c] = re;
454                            break PER_FACTION;
455                        }
456                    }
457                }
458            }
459        }
460        IntVLA[] fresh = new IntVLA[count];
461        int filled = 0;
462        boolean hasFresh = false;
463        int approximateArea = (controlled * 4) / (count * 3);
464        char[] keys = new char[count];
465        double[] biases = new double[count];
466        for (int i = 0; i < count; i++) {
467            fresh[i] = new IntVLA(approximateArea);
468            cen = centers[i];
469            fresh[i].add(cen);
470            cx = cen % width;
471            cy = cen / width;
472            politicalMap[cx][cy] = keys[i] = atlas.keyAt(factionIndices[i] + 2);
473            biases[i] = rng.nextDouble() * rng.nextDouble() + rng.nextDouble() + 0.03125;
474            hasFresh = true;
475        }
476        Direction[] dirs = Direction.CARDINALS;
477        IntVLA currentFresh;
478        GreasedRegion anySpillMap = new GreasedRegion(width, height),
479                anyFreshMap = new GreasedRegion(width, height);
480
481        while (hasFresh && filled < controlled) {
482            hasFresh = false;
483            for (int i = 0; i < count && filled < controlled; i++) {
484                currentFresh = fresh[i];
485                if (currentFresh.isEmpty())
486                    continue;
487                else
488                    hasFresh = true;
489                if (rng.nextDouble() < biases[i]) {
490                    int index = rng.nextIntHasty(currentFresh.size), cell = currentFresh.get(index);
491                    cx = cell % width;
492                    cy = cell / width;
493
494
495                    politicalMap[cx][cy] = keys[i];
496                    filled++;
497                    anySpillMap.insert(cx, cy);
498
499                    for (int d = 0; d < dirs.length; d++) {
500                        cx2 = wmg.wrapX(cx + dirs[d].deltaX, cy);
501                        cy2 = wmg.wrapY(cx, cy + dirs[d].deltaY);
502                        if (cx == cx2 && cy == cy2)
503                            continue;
504                        if (land.contains(cx2, cy2) && !anySpillMap.contains(cx2, cy2)) {
505                            if(!anyFreshMap.contains(cx2, cy2)) {
506                                currentFresh.add(cx2 + cy2 * width);
507                                anyFreshMap.insert(cx2, cy2);
508                            }
509                        }
510                    }
511                    currentFresh.removeIndex(index);
512                    anyFreshMap.remove(cx, cy);
513                }
514            }
515        }
516        zoomedMap = ArrayTools.copy(politicalMap);
517        return politicalMap;
518    }

520    /**
521     * If the WorldMapGenerator used by
522     * {@link #generate(long, WorldMapGenerator, WorldMapGenerator.BiomeMapper, Collection, int, double)} zooms in or
523     * out, you can call this method to make the {@link #zoomedMap} 2D char array match its zoom. The world-scale map,
524     * {@link #politicalMap}, will remain unchanged unless generate() is called again, but zoomedMap will change each
525     * time either generate() or adjustZoom() is called. This method isn't 100% precise on how it places borders; for
526     * aesthetic reasons, the borders are tattered with {@link GreasedRegion#fray(double)} so they don't look like a
527     * wall of angular bubbles. Using fray() at each level of zoom is quasi-random, so if you zoom in on the same
528     * sequence of points on two different occasions, the change from fray() will be the same, but it may be slightly
529     * different if any point of zoom is different.
530     * @return a direct reference to {@link #zoomedMap}, which will hold the correctly-zoomed version of {@link #politicalMap}
531     */
532    public char[][] adjustZoom() {
533        if(wmg.zoom <= 0)
534        {
535            return ArrayTools.insert(politicalMap, zoomedMap, 0, 0);
536        }
537        ArrayTools.fill(zoomedMap, ' ');
538        char c;
539        int stx = Math.min(Math.max((wmg.zoomStartX - (width  >> 1)) / ((2 << wmg.zoom) - 2), 0), width ),
540                sty = Math.min(Math.max((wmg.zoomStartY - (height >> 1)) / ((2 << wmg.zoom) - 2), 0), height);
541        GreasedRegion nation = new GreasedRegion(wmg.landData);
542        GreasedRegion fillable = new GreasedRegion(politicalMap, '~').not();
543        for (int i = 0; i < wmg.zoom; i++) {
544            fillable.zoom(stx, sty);
545        }
546        fillable.flood(nation, width + height);
547        for (int i = 1; i < atlas.size(); i++) {
548            nation.refill(politicalMap, c = atlas.keyAt(i));
549            if(nation.isEmpty()) continue;
550            for (int z = 0; z < wmg.zoom; z++) {
551                nation.zoom(stx, sty).expand8way().expand().fray(0.5);
552            }
553            fillable.andNot(nation);
554            nation.intoChars(zoomedMap, c);
555        }
556        for (int i = 1; i < atlas.size(); i++) {
557            nation.refill(zoomedMap, c = atlas.keyAt(i));
558            if(nation.isEmpty()) continue;
559            nation.flood(fillable, 4 << wmg.zoom).intoChars(zoomedMap, c);
560        }
561        nation.refill(wmg.heightCodeData, 4, 999).and(new GreasedRegion(zoomedMap, ' ')).intoChars(zoomedMap, '%');
562        nation.refill(wmg.heightCodeData, -999, 4).intoChars(zoomedMap, '~');
563        return zoomedMap;
564    }
