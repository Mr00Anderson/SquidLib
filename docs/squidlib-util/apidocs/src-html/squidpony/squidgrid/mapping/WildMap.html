018/**
019 * A finite 2D area map for some kind of wilderness, with specifics handled by subclasses.
020 * Regional maps for wilderness areas have very different requirements from mostly-artificial dungeons. This is intended
021 * to work alongside {@link WorldMapGenerator} and {@link WorldMapGenerator.DetailedBiomeMapper} to produce blends
022 * between biomes where a strict line can't be drawn between, say, woodland and grassland.
023 * <br>
024 * Using this code mostly involves constructing a subclass, adding kinds of game objects that you want to appear
025 * in this type of map to {@link #contentTypes}, and only then calling {@link #generate()}, which is implemented by
026 * subclasses only. You can use just about any type for {@code T}, including {@link Object}; all this code uses is the
027 * number of items in {@link #contentTypes} to put correct indices into {@link #content}. Because contentTypes is an
028 * ArrayList, you can have and are encouraged to have duplicates when an object should appear more often. An index of -1
029 * in content indicates nothing of note is present there. There is also a String array of {@link #floorTypes} that is
030 * not typically user-set unless you subclass WildMap yourself; it is used to look up the indices in {@link #floors}.
031 * The floors are set to reasonable values for the particular subclass, so a forest might have "dirt" and "leaves" among
032 * others, while a desert might only have "sand". Again, only the indices matter, so you could change the values in
033 * {@link #floorTypes} to match names of textures in a graphical game and make lookup easier, or to a char followed by
034 * the name of a color (as in SColor in the display module) for a text-based game.
035 * <br>
036 * Created by Tommy Ettinger on 10/16/2019.
037 */

048    /**
049     * Meant for generating large ArrayLists of Strings where an individual String may occur quite a few times.
050     * The rest parameter is a vararg (it may also be an Object array) of alternating String and Integer values, where
051     * an Integer is how many times to repeat the preceding String in the returned ArrayList.
052     * @param rest a vararg (or Object array) of alternating String and Integer values
053     * @return an ArrayList of Strings, probably with some or most of them repeated; you may want to shuffle this result
054     */
055    public static ArrayList<String> makeRepeats(Object... rest)
056    {
057        if(rest == null || rest.length < 2)
058        {
059            return new ArrayList<>(0);
060        }
061        ArrayList<String> al = new ArrayList<>(rest.length);
062
063        for (int i = 0; i < rest.length - 1; i+=2) {
064            try {
065                int count = (int)rest[i+1];
066                String v = (String) rest[i];
067                for (int j = 0; j < count; j++) {
068                    al.add(v);
069                }
070            }catch (ClassCastException ignored) {
071            }
072        }
073        return al;
074    }
075    public static ArrayList<String> makeShuffledRepeats(IRNG rng, Object... rest) 
076    {
077        ArrayList<String> al = makeRepeats(rest);
078        rng.shuffleInPlace(al);
079        return al;
080    }
081    public static ArrayList<String> makeVegetation(IRNG rng, int size, double monoculture, FakeLanguageGen naming)
082    {
083        Thesaurus t = new Thesaurus(rng);
084        ArrayList<String> al = new ArrayList<>(size);
085        String latest;
086        for (int i = size; i > 0; i--) {
087            al.add(latest = t.makePlantName(naming));
088            for (double j = rng.nextDouble(monoculture * 2.0 * size); j >= 1 && i > 0; j--, i--) {
089                al.add(latest);
090            }
091        }
092        rng.shuffleInPlace(al);
093        return al;
094    }

379    /**
380     * Produces a map by filling the {@link #floors} 2D array with indices into {@link #floorTypes}, and similarly
381     * filling the {@link #content} 2D array with indices into {@link #contentTypes}. You only need to call this method
382     * when you first generate a map with the specific parameters you want, such as biome, and later if you want another
383     * map with the same parameters.
384     * <br>
385     * Virtually all of this method is a wrapper around functionality provided by {@link BlueNoise}, adjusted to fit
386     * wilderness maps slightly.
387     */
388    public void generate() {
389        ArrayTools.fill(content, -1);
390        final int seed = rng.nextInt();//, otherSeed = rng.nextInt(), choice = seed + otherSeed & 15;
391        final int limit = contentTypes.size(), floorLimit = floorTypes.size();
392        int b;
393        BlueNoise.blueSpill(floors, floorLimit, rng);
394        for (int x = 0; x < width; x++) {
395            for (int y = 0; y < height; y++) {
396                if((b = BlueNoise.getSeeded(x, y, seed) + 128) < limit)
397                    content[x][y] = b;
398                //floors[x][y] = (int)((FastNoise.instance.layered2D(x,  y, otherSeed, 2, 0x1p-5f) * 0.4999f + 0.5f) * (floorLimit - 1) + 0.25f + rng.nextFloat(0.5f));
399            }
400        }
401    }

403    public static class MixedWildMap extends WildMap implements Serializable
404    {
405        private static final long serialVersionUID = 1L;
406        public final int[][] pieceMap;
407        public final WildMap[] pieces;
408        protected final int[] minFloors, maxFloors, minContents, maxContents;
409        
410        public MixedWildMap()
411        {
412            this(new WildMap(), new WildMap(), new WildMap(), new WildMap(), new SilkRNG());
413        }
414        
415        public MixedWildMap(WildMap northeast, WildMap southeast, WildMap southwest, WildMap northwest, IStatefulRNG rng)
416        {
417            super(northeast.width, northeast.height, northeast.biome, rng, new ArrayList<>(northeast.floorTypes), new ArrayList<>(northeast.contentTypes));
418            minFloors = new int[4];
419            maxFloors = new int[4];
420            minContents = new int[4];
421            maxContents = new int[4];
422            floorTypes.addAll(southeast.floorTypes);
423            floorTypes.addAll(southwest.floorTypes);
424            floorTypes.addAll(northwest.floorTypes);
425            contentTypes.addAll(southeast.contentTypes);
426            contentTypes.addAll(southwest.contentTypes);
427            contentTypes.addAll(northwest.contentTypes);
428            minFloors[1]   = maxFloors[0]   = northeast.floorTypes.size();
429            minContents[1] = maxContents[0] = northeast.contentTypes.size();
430            minFloors[2]   = maxFloors[1]   = maxFloors[0]    + southeast.floorTypes.size();
431            minContents[2] = maxContents[1] = maxContents[0]  + southeast.contentTypes.size();
432            minFloors[3]   = maxFloors[2]   = maxFloors[1]    + southwest.floorTypes.size();
433            minContents[3] = maxContents[2] = maxContents[1]  + southwest.contentTypes.size();
434            maxFloors[3]   = maxFloors[2]    + northwest.floorTypes.size();
435            maxContents[3] = maxContents[2]  + northwest.contentTypes.size();
436            pieces = new WildMap[]{northeast, southeast, southwest, northwest};
437            pieceMap = new int[width][height];
438        }
439        
440        protected void preparePieceMap()
441        {
442            ArrayTools.fill(pieceMap, 255);
443            pieceMap[width - 1][0] = 0; // northeast
444            pieceMap[width - 1][height - 1] = 1; // southeast
445            pieceMap[0][height - 1] = 2; // southwest
446            pieceMap[0][0] = 3; //northwest
447            final int spillerLimit = 4;
448            final Direction[] dirs = Direction.CARDINALS;
449            Direction d;
450            int t, rx, ry, ctr;
451            int[] ox = new int[width], oy = new int[height];
452            boolean anySuccesses = false;
453            do {
454                ctr = 0;
455                rng.randomOrdering(width, ox);
456                rng.randomOrdering(height, oy);
457                for (int x = 0; x < width; x++) {
458                    rx = ox[x];
459                    for (int y = 0; y < height; y++) {
460                        ry = oy[y];
461                        if ((t = pieceMap[rx][ry]) < spillerLimit) {
462                            d = dirs[rng.next(2)];
463                            if (rx + d.deltaX >= 0 && rx + d.deltaX < width && ry + d.deltaY >= 0 && ry + d.deltaY < height &&
464                                    pieceMap[rx + d.deltaX][ry + d.deltaY] >= spillerLimit) {
465                                pieceMap[rx + d.deltaX][ry + d.deltaY] = t;
466                                ctr++;
467                            }
468                            d = dirs[rng.next(2)];
469                            if (rx + d.deltaX >= 0 && rx + d.deltaX < width && ry + d.deltaY >= 0 && ry + d.deltaY < height &&
470                                    pieceMap[rx + d.deltaX][ry + d.deltaY] >= spillerLimit) {
471                                pieceMap[rx + d.deltaX][ry + d.deltaY] = t;
472                                ctr++;
473                            }
474                        }
475
476                    }
477                }
478                if(!anySuccesses && ctr == 0)
479                {
480                    ArrayTools.fill(pieceMap, 0);
481                    return;
482                }
483                else
484                    anySuccesses = true;
485            } while (ctr > 0);
486            do {
487                ctr = 0;
488                rng.randomOrdering(width, ox);
489                rng.randomOrdering(height, oy);
490                for (int x = 0; x < width; x++) {
491                    rx = ox[x];
492                    for (int y = 0; y < height; y++) {
493                        ry = oy[y];
494                        if ((t = pieceMap[rx][ry]) < spillerLimit) {
495                            for (int i = 0; i < 4; i++) {
496                                d = dirs[i];
497                                if (rx + d.deltaX >= 0 && rx + d.deltaX < width && ry + d.deltaY >= 0 && ry + d.deltaY < height &&
498                                        pieceMap[rx + d.deltaX][ry + d.deltaY] >= spillerLimit) {
499                                    pieceMap[rx + d.deltaX][ry + d.deltaY] = t;
500                                    ctr++;
501                                }
502                            }
503                        }
504
505                    }
506                }
507            } while (ctr > 0);
508
509        }
510
511        @Override
512        public void generate() {
513            ArrayTools.fill(content, -1);
514            for (int i = 0; i < pieces.length; i++) {
515                pieces[i].generate();
516            }
517            preparePieceMap();
518            int p, c;
519            WildMap piece;
520            for (int x = 0; x < width; x++) {
521                for (int y = 0; y < height; y++) {
522                    p = pieceMap[x][y];
523                    piece = pieces[p];
524                    floors[x][y] = piece.floors[x][y] + minFloors[p];
525                    if((c = piece.content[x][y]) >= 0)
526                    {
527                        content[x][y] = c + minContents[p];
528                    }
529                }
530            }
531        }
532    }
