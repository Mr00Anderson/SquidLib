013/**
014 * Creates a dungeon in the style of the original Rogue game. It will always
015 * make a grid style of rooms where there are a certain number horizontally and
016 * vertically and it will link them only next to each other.
017 *
018 * This dungeon generator is based on a port of the rot.js version.
019 *
020 * @author hyakugei
021 * @author Eben Howard - http://squidpony.com - howard@squidpony.com
022 */
023public class ClassicRogueMapGenerator implements IDungeonGenerator{
024
025    /**
026     * Holds the information needed to track rooms in the classic rogue
027     * generation algorithm.
028     */
029    private static class ClassicRogueRoom {
030
031        private int x, y, width, height, cellx, celly;
032        private final OrderedSet<ClassicRogueRoom> connections = new OrderedSet<>(4);
033        ClassicRogueRoom(int x, int y, int width, int height, int cellx, int celly) {
034            this.x = x;
035            this.y = y;
036            this.width = width;
037            this.height = height;
038            this.cellx = cellx;
039            this.celly = celly;
040        }
041
042        @Override
043        public int hashCode() {
044            // same as Coord.hashCode(); has extremely good resistance to collisions on common points
045            int r = cellx ^ celly;
046            r ^= (cellx << 13 | cellx >>> 19) ^ (r << 5) ^ (r << 28 | r >>> 4);
047            r = cellx ^ (r << 11 | r >>> 21);
048            return r ^ (r << 25 | r >>> 7);
049        }
050
051        @Override
052        public boolean equals(Object obj) {
053            if (obj == null) {
054                return false;
055            }
056            if (getClass() != obj.getClass()) {
057                return false;
058            }
059            final ClassicRogueRoom other = (ClassicRogueRoom) obj;
060            if (cellx != other.cellx) {
061                return false;
062            }
063            return celly == other.celly;
064        }
065    }

069    private int horizontalRooms, verticalRooms, dungeonWidth, dungeonHeight,
070            minRoomWidth, maxRoomWidth, minRoomHeight, maxRoomHeight;
071    private ClassicRogueRoom[][] rooms;
072    private char[][] dungeon;
073    private Direction[] dirToCheck = new Direction[4];

095    /**
096     * Initializes the generator to turn out random dungeons within the specific
097     * parameters.
098     *
099     * Will size down the room width and height parameters if needed to ensure
100     * the desired number of rooms will fit both horizontally and vertically.
101     *
102     * @param horizontalRooms How many rooms will be made horizontally
103     * @param verticalRooms How many rooms will be made vertically
104     * @param dungeonWidth How wide the total dungeon will be
105     * @param dungeonHeight How high the total dungeon will be
106     * @param minRoomWidth The minimum width a room can be
107     * @param maxRoomWidth The maximum width a room can be
108     * @param minRoomHeight The minimum height a room can be
109     * @param maxRoomHeight The maximum height a room can be
110     */
111    public ClassicRogueMapGenerator(int horizontalRooms, int verticalRooms, int dungeonWidth, int dungeonHeight,
112                                    int minRoomWidth, int maxRoomWidth, int minRoomHeight, int maxRoomHeight, IRNG rng)
113    {
114        this.rng = rng;
115        this.horizontalRooms = horizontalRooms;
116        this.verticalRooms = verticalRooms;
117        this.dungeonWidth = dungeonWidth;
118        this.dungeonHeight = dungeonHeight;
119        this.minRoomWidth = minRoomWidth;
120        this.maxRoomWidth = maxRoomWidth;
121        this.minRoomHeight = minRoomHeight;
122        this.maxRoomHeight = maxRoomHeight;
123
124        sanitizeRoomDimensions();
125    }

127    private void sanitizeRoomDimensions() {
128        int test = (dungeonWidth - 3 * horizontalRooms) / horizontalRooms;//have to leave space for hallways
129        maxRoomWidth = Math.min(test, maxRoomWidth);
130        minRoomWidth = Math.max(minRoomWidth, 2);
131        minRoomWidth = Math.min(minRoomWidth, maxRoomWidth);
132
133        test = (dungeonHeight - 3 * verticalRooms) / verticalRooms;//have to leave space for hallways
134        maxRoomHeight = Math.min(test, maxRoomHeight);
135        minRoomHeight = Math.max(minRoomHeight, 2);
136        minRoomHeight = Math.min(minRoomHeight, maxRoomHeight);
137    }

139    /**
140     * Builds and returns a map in the Classic Rogue style, returned as a 2D char array.
141     *
142     * Only includes rooms ('.' for floor and '#' for walls), corridors (using the same chars as rooms) and doors ('+'
143     * for closed doors, does not generate open doors).
144     * @return a 2D char array version of the map
145     */
146    public char[][] generate()
147    {
148        initRooms();
149        connectRooms();
150        connectUnconnectedRooms();
151        fullyConnect();
152        createRooms();
153        createCorridors();
154        return dungeon;
155    }
156
157    public char[][] getDungeon() {
158        return dungeon;
159    }

161    private void initRooms() {
162        rooms = new ClassicRogueRoom[horizontalRooms][verticalRooms];
163        dungeon = new char[dungeonWidth][dungeonHeight];
164        for (int x = 0; x < horizontalRooms; x++) {
165            for (int y = 0; y < verticalRooms; y++) {
166                rooms[x][y] = new ClassicRogueRoom(0, 0, 0, 0, x, y);
167            }
168        }
169        for (int x = 0; x < dungeonWidth; x++) {
170            for (int y = 0; y < dungeonHeight; y++) {
171                dungeon[x][y] = '#';
172            }
173        }
174    }
175
176    private void connectRooms() {
177        ArrayList<ClassicRogueRoom> unconnected = new ArrayList<>();
178        for (int x = 0; x < horizontalRooms; x++) {
179            for (int y = 0; y < verticalRooms; y++) {
180                unconnected.add(rooms[x][y]);
181            }
182        }
183        rng.shuffleInPlace(unconnected);
184
185        for (int i = 0; i < unconnected.size(); i++) {
186            ClassicRogueRoom room = unconnected.get(i);
187            rng.shuffle(Direction.CARDINALS, dirToCheck);
188            for (Direction dir : dirToCheck) {
189                int nextX = room.x + dir.deltaX;
190                int nextY = room.y + dir.deltaY;
191                if (nextX < 0 || nextX >= horizontalRooms || nextY < 0 || nextY >= verticalRooms) {
192                    continue;
193                }
194                ClassicRogueRoom otherRoom = rooms[nextX][nextY];
195
196                if (room.connections.contains(otherRoom)) {
197                    break;//already connected to this room
198                }
199
200                if (!otherRoom.connections.isEmpty()) {
201                    room.connections.add(otherRoom);
202                    break;
203                }
204            }
205        }
206    }
207
208    private void connectUnconnectedRooms() {
209        for (int x = 0; x < horizontalRooms; x++) {
210            for (int y = 0; y < verticalRooms; y++) {
211                ClassicRogueRoom room = rooms[x][y];
212
213                if (room.connections.isEmpty()) {
214                    rng.shuffle(Direction.CARDINALS, dirToCheck);
215
216                    boolean validRoom = false;
217                    ClassicRogueRoom otherRoom = null;
218                    int idx = 0;
219                    do {
220                        Direction dir = dirToCheck[idx++];
221
222                        int nextX = x + dir.deltaX;
223                        if (nextX < 0 || nextX >= horizontalRooms) {
224                            continue;
225                        }
226                        int nextY = y + dir.deltaY;
227                        if (nextY < 0 || nextY >= verticalRooms) {
228                            continue;
229                        }
230
231                        otherRoom = rooms[nextX][nextY];
232                        validRoom = true;
233
234                        if (otherRoom.connections.contains(room)) {
235                            validRoom = false;
236                        }
237                        else {
238                            break;
239                        }
240                    } while (idx < 4);
241
242                    if (validRoom) {
243                        room.connections.add(otherRoom);
244                    }
245                }
246            }
247        }
248    }

250    private void fullyConnect() {
251        boolean allGood;
252        do {
253            ArrayDeque<ClassicRogueRoom> deq = new ArrayDeque<>(horizontalRooms * verticalRooms);
254            for (int x = 0; x < horizontalRooms; x++) {
255                for (int y = 0; y < verticalRooms; y++) {
256                    deq.offer(rooms[x][y]);
257                }
258            }
259            ArrayList<ClassicRogueRoom> connected = new ArrayList<>();
260            connected.add(deq.removeFirst());
261            boolean changed = true;
262            TESTING:
263            while (changed) {
264                changed = false;
265                for (ClassicRogueRoom test : deq) {
266                    for (int i = 0, connectedSize = connected.size(); i < connectedSize; i++) {
267                        ClassicRogueRoom r = connected.get(i);
268                        if (test.connections.contains(r) || r.connections.contains(test)) {
269                            connected.add(test);
270                            deq.remove(test);
271                            changed = true;
272                            continue TESTING;
273                        }
274                    }
275                }
276            }
277
278            allGood = true;
279            if (!deq.isEmpty()) {
280                TESTING:
281                for (ClassicRogueRoom room : deq) {
282                    for (Direction dir : Direction.CARDINALS) {
283                        int x = room.cellx + dir.deltaX;
284                        int y = room.celly + dir.deltaY;
285                        if (x >= 0 && y >= 0 && x < horizontalRooms && y < verticalRooms) {
286                            ClassicRogueRoom otherRoom = rooms[x][y];
287                            if (connected.contains(otherRoom)) {
288                                room.connections.add(otherRoom);
289                                allGood = false;
290                                break TESTING;
291                            }
292                        }
293                    }
294                }
295            }
296
297        } while (!allGood);
298    }

300    private void createRooms() {
301        int cwp = dungeonWidth / horizontalRooms;
302        int chp = dungeonHeight / verticalRooms;
303
304        ClassicRogueRoom otherRoom;
305
306        for (int x = 0; x < horizontalRooms; x++) {
307            for (int y = 0; y < verticalRooms; y++) {
308                int sx = cwp * x;
309                int sy = chp * y;
310
311                sx = Math.max(sx, 2);
312                sy = Math.max(sy, 2);
313
314                int roomw = rng.between(minRoomWidth, maxRoomWidth + 1);
315                int roomh = rng.between(minRoomHeight, maxRoomHeight + 1);
316
317                if (y > 0) {
318                    otherRoom = rooms[x][y - 1];
319                    while (sy - (otherRoom.y + otherRoom.height) < 3) {
320                        sy++;
321                    }
322                }
323
324                if (x > 0) {
325                    otherRoom = rooms[x - 1][y];
326                    while (sx - (otherRoom.x + otherRoom.width) < 3) {
327                        sx++;
328                    }
329                }
330
331                int sxOffset = Math.round(rng.nextInt(cwp - roomw) * 0.5f);
332                int syOffset = Math.round(rng.nextInt(chp - roomh) * 0.5f);
333
334                while (sx + sxOffset + roomw >= dungeonWidth) {
335                    if (sxOffset > 0) {
336                        sxOffset--;
337                    } else {
338                        roomw--;
339                    }
340                }
341                while (sy + syOffset + roomh >= dungeonHeight) {
342                    if (syOffset > 0) {
343                        syOffset--;
344                    } else {
345                        roomh--;
346                    }
347                }
348
349                sx += sxOffset;
350                sy += syOffset;
351
352                ClassicRogueRoom r = rooms[x][y];
353                r.x = sx;
354                r.y = sy;
355                r.width = roomw;
356                r.height = roomh;
357
358                for (int xx = sx; xx < sx + roomw; xx++) {
359                    for (int yy = sy; yy < sy + roomh; yy++) {
360                        dungeon[xx][yy] = '.';
361                    }
362                }
363            }
364        }
365    }

367    private Coord randomWallPosition(ClassicRogueRoom room, Direction dir) {
368        int x, y;
369        Coord p = null;
370
371        switch (dir) {
372            case LEFT:
373                y = rng.between(room.y + 1, room.y + room.height);
374                x = room.x - 1;
375                dungeon[x][y] = '+';
376                p = Coord.get(x - 1, y);
377                break;
378            case RIGHT:
379                y = rng.between(room.y + 1, room.y + room.height);
380                x = room.x + room.width;
381                dungeon[x][y] = '+';
382                p = Coord.get(x + 1, y);
383                break;
384            case DOWN:
385                x = rng.between(room.x + 1, room.x + room.width);
386                y = room.y - 1;
387                dungeon[x][y] = '+';
388                p = Coord.get(x, y - 1);
389                break;
390            case UP:
391                x = rng.between(room.x + 1, room.x + room.width);
392                y = room.y + room.height;
393                dungeon[x][y] = '+';
394                p = Coord.get(x, y + 1);
395                break;
396        case NONE:
397                break;
398                case DOWN_LEFT:
399                case DOWN_RIGHT:
400                case UP_LEFT:
401                case UP_RIGHT:
402                        throw new IllegalStateException("There should only be cardinal positions here");
403        }
404
405        return p;
406    }

408    /**
409     * Draws a corridor between the two points with a zig-zag in between.
410     *
411     * @param start
412     * @param end
413     */
414    private void digPath(Coord start, Coord end) {
415        int xOffset = end.x - start.x;
416        int yOffset = end.y - start.y;
417        int xpos = start.x;
418        int ypos = start.y;
419
420        ArrayDeque<Magnitude> moves = new ArrayDeque<>();
421
422        int xAbs = Math.abs(xOffset);
423        int yAbs = Math.abs(yOffset);
424
425        double firstHalf = rng.nextDouble();
426        double secondHalf = 1 - firstHalf;
427
428        Direction xDir = xOffset < 0 ? Direction.LEFT : Direction.RIGHT;
429        Direction yDir = yOffset > 0 ? Direction.DOWN : Direction.UP;
430
431        if (xAbs < yAbs) {
432            int tempDist = (int) Math.ceil(yAbs * firstHalf);
433            moves.add(new Magnitude(yDir, tempDist));
434            moves.add(new Magnitude(xDir, xAbs));
435            tempDist = (int) Math.floor(yAbs * secondHalf);
436            moves.add(new Magnitude(yDir, tempDist));
437        } else {
438            int tempDist = (int) Math.ceil(xAbs * firstHalf);
439            moves.add(new Magnitude(xDir, tempDist));
440            moves.add(new Magnitude(yDir, yAbs));
441            tempDist = (int) Math.floor(xAbs * secondHalf);
442            moves.add(new Magnitude(xDir, tempDist));
443        }
444
445        dungeon[xpos][ypos] = '.';
446
447        while (!moves.isEmpty()) {
448            Magnitude move = moves.removeFirst();
449            Direction dir = move.dir;
450            int dist = move.distance;
451            while (dist > 0) {
452                xpos += dir.deltaX;
453                ypos += dir.deltaY;
454                dungeon[xpos][ypos] = '.';
455                dist--;
456            }
457        }
458    }

460    private void createCorridors() {
461        for (int x = 0; x < horizontalRooms; x++) {
462            for (int y = 0; y < verticalRooms; y++) {
463                ClassicRogueRoom room = rooms[x][y];
464                for (int r = 0; r < room.connections.size(); r++) {
465                    ClassicRogueRoom otherRoom = room.connections.getAt(r);
466                    Direction dir = Direction.getCardinalDirection(otherRoom.cellx - room.cellx, otherRoom.celly - room.celly);
467                    digPath(randomWallPosition(room, dir), randomWallPosition(otherRoom, dir.opposite()));
468                }
469            }
470        }
471    }

473    private static class Magnitude {
474
475        public Direction dir;
476        public int distance;
477
478        public Magnitude(Direction dir, int distance) {
479            this.dir = dir;
480            this.distance = distance;
481        }
