150    /**
151     * Produces a 4x4 2D char array by interpreting the bits of the given long as line information. Uses the box drawing
152     * chars from {@link #light}, which are compatible with most fonts.
153     * @param encoded a long, which can be random, that encodes some pattern of (typically box drawing) characters
154     * @return a 4x4 2D char array containing elements from symbols assigned based on encoded
155     */
156    public static char[][] decode4x4(long encoded)
157    {
158        return decode4x4(encoded, light);
159    }
160    /**
161     * Produces a 4x4 2D char array by interpreting the bits of the given long as line information. Uses the given char
162     * array, which must have at least 16 elements and is usually one of {@link #light}, {@link #heavy},
163     * {@link #lightAlt}, or {@link #heavyAlt}, with the last two usable only if using a font that supports the chars
164     * {@code ╴╵╶╷} (this is true for Iosevka and Source Code Pro, for instance, but not Inconsolata or GoMono).
165     * @param encoded a long, which can be random, that encodes some pattern of (typically box drawing) characters
166     * @param symbols a 16-element-or-larger char array; usually a constant in this class like {@link #light}
167     * @return a 4x4 2D char array containing elements from symbols assigned based on encoded
168     */
169    public static char[][] decode4x4(long encoded, char[] symbols)
170    {
171        char[][] v = new char[4][4];
172        for (int i = 0; i < 16; i++) {
173            v[i & 3][i >> 2] = symbols[(int) (encoded >>> (i << 2) & 15L)];
174        }
175        return v;
176    }
177    /**
178     * Fills a 4x4 area of the given 2D char array {@code into} by interpreting the bits of the given long as line
179     * information. Uses the given char array {@code symbols}, which must have at least 16 elements and is usually one
180     * of {@link #light}, {@link #heavy}, {@link #lightAlt}, or {@link #heavyAlt}, with the last two usable only if
181     * using a font that supports the chars {@code ╴╵╶╷} (this is true for Iosevka and Source Code Pro, for instance,
182     * but not Inconsolata or GoMono).
183     * @param encoded a long, which can be random, that encodes some pattern of (typically box drawing) characters
184     * @param symbols a 16-element-or-larger char array; usually a constant in this class like {@link #light}
185     * @param into a 2D char array that will be modified in a 4x4 area
186     * @param startX the first x position to modify in into
187     * @param startY the first y position to modify in into
188     * @return into, after modification
189     */
190    public static char[][] decodeInto4x4(long encoded, char[] symbols, char[][] into, int startX, int startY)
191    {
192        for (int i = 0; i < 16; i++) {
193            into[(i & 3) + startX][(i >> 2) + startY] = symbols[(int) (encoded >>> (i << 2) & 15L)];
194        }
195        return into;
196    }

198    /**
199     * Reads a 2D char array {@code decoded}, which must be at least 4x4 in size, and returns a long that encodes the cells from 0,0 to
200     * 3,3 in a way that this class can interpret and manipulate. The 2D array {@code decoded} must contain box drawing
201     * symbols, which can be any of those from {@link #light}, {@link #heavy}, {@link #lightAlt}, or {@link #heavyAlt}.
202     * Valid chars are {@code ╴╵┘╶─└┴╷┐│┤┌┬├┼╸╹┛╺━┗┻╻┓┃┫┏┳┣╋}; any other chars will be treated as empty space.
203     * @param decoded a 2D char array that must be at least 4x4 and should usually contain box drawing characters
204     * @return a long that encodes the box drawing information in decoded so this class can manipulate it
205     */
206    public static long encode4x4(char[][] decoded)
207    {
208        long v = 0L;
209        for (int i = 0; i < 16; i++) {
210            switch (decoded[i & 3][i >> 2])
211            {
212                // ╴╵┘╶─└┴╷┐│┤┌┬├┼
213                // ╸╹┛╺━┗┻╻┓┃┫┏┳┣╋
214                //0123456789ABCDEF
215                case '─':
216                case '━':
217                    v |= 5L << (i << 2);
218                    break;
219                case '│':
220                case '┃':
221                    v |= 10L << (i << 2);
222                    break;
223                case '┘':
224                case '┛':
225                    v |= 3L << (i << 2);
226                    break;
227                case '└':
228                case '┗':
229                    v |= 6L << (i << 2);
230                    break;
231                case '┐':
232                case '┓':
233                    v |= 9L << (i << 2);
234                    break;
235                case '┌':
236                case '┏':
237                    v |= 12L << (i << 2);
238                    break;
239                case '┴':
240                case '┻':
241                    v |= 7L << (i << 2);
242                    break;
243                case '┤':
244                case '┫':
245                    v |= 11L << (i << 2);
246                    break;
247                case '┬':
248                case '┳':
249                    v |= 13L << (i << 2);
250                    break;
251                case '├':
252                case '┣':
253                    v |= 14L << (i << 2);
254                    break;
255                case '┼':
256                case '╋':
257                    v |= 15L << (i << 2);
258                    break;
259                case '╴':
260                case '╸':
261                    v |= 1L << (i << 2);

280    /**
281     * Makes a variant on the given encoded 4x4 pattern so the left side is flipped to the right side and vice versa.
282     * @param encoded an encoded pattern long that represents a 4x4 area
283     * @return a different encoded pattern long that represents the argument flipped left-to-right
284     */
285    public static long flipHorizontal4x4(long encoded)
286    {
287        long v = 0L;
288        for (int i = 0, i4 = 0; i < 16; i++, i4 += 4) {
289            v |= ((encoded >>> i4 & 10L) | ((encoded >>> i4 & 1) << 2L) | ((encoded >>> i4 & 4L) >>> 2)) << (i + 3 - ((i & 3) << 1) << 2);
290        }
291        return v;
292    }
293    /**
294     * Makes a variant on the given encoded 4x4 pattern so the top side is flipped to the bottom side and vice versa.
295     * @param encoded an encoded pattern long that represents a 4x4 area
296     * @return a different encoded pattern long that represents the argument flipped top-to-bottom
297     */
298    public static long flipVertical4x4(long encoded)
299    {
300        long v = 0L;
301        for (int i = 0, i4 = 0; i < 16; i++, i4 += 4) {
302            v |= ((encoded >>> i4 & 5L) | ((encoded >>> i4 & 2L) << 2) | ((encoded >>> i4 & 8L) >>> 2)) << (i + 12 - ((i >> 2) << 3) << 2);
303        }
304        return v;
305    }

307    /**
308     * Makes a variant on the given encoded 4x4 pattern so the x and y axes are interchanged, making the top side become
309     * the left side and vice versa, while the bottom side becomes the right side and vice versa.
310     * @param encoded an encoded pattern long that represents a 4x4 area
311     * @return a different encoded pattern long that represents the argument transposed top-to-left and bottom-to-right
312     */
313    public static long transpose4x4(long encoded)
314    {
315        long v = 0L;
316        for (int i4 = 0; i4 < 64; i4 += 4) {
317            v |= (((encoded >>> i4 & 5L) << 1) | ((encoded >>> i4 & 10L) >>> 1)) << ((i4 >>> 2 & 12L) | ((i4 & 12L) << 2));
318        }
319        return v;
320    }
321    /**
322     * Makes a variant on the given encoded 4x4 pattern so the lines are rotated 90 degrees clockwise, changing their
323     * positions as well as what chars they will decode to. This can be called twice to get a 180 degree rotation, but
324     * {@link #rotateCounterclockwise(long)} should be used for a 270 degree rotation.
325     * @param encoded an encoded pattern long that represents a 4x4 area
326     * @return a different encoded pattern long that represents the argument rotated 90 degrees clockwise
327     */
328    public static long rotateClockwise(long encoded)
329    {
330        // this is functionally equivalent to, but faster than, the following:
331        // return flipHorizontal4x4(transpose4x4(encoded));
332        long v = 0L;
333        for (int i4 = 0; i4 < 64; i4 += 4) {
334            v |= (((encoded >>> i4 & 7L) << 1) | ((encoded >>> i4 & 8L) >>> 3)) << ((~i4 >>> 2 & 12L) | ((i4 & 12L) << 2));
335        }
336        return v;
337    }
338    /**
339     * Makes a variant on the given encoded 4x4 pattern so the lines are rotated 90 degrees counterclockwise, changing
340     * their positions as well as what chars they will decode to. This can be called twice to get a 180 degree rotation,
341     * but {@link #rotateClockwise(long)} should be used for a 270 degree rotation.
342     * @param encoded an encoded pattern long that represents a 4x4 area
343     * @return a different encoded pattern long that represents the argument rotated 90 degrees counterclockwise
344     */
345    public static long rotateCounterclockwise(long encoded)
346    {
347        // this is functionally equivalent to, but faster than, the following:
348        // return flipVertical4x4(transpose4x4(encoded));
349        long v = 0L;
350        for (int i4 = 0; i4 < 64; i4 += 4) {
351            v |= ((encoded >>> (i4 + 1) & 7L) | ((encoded >>> i4 & 1L) << 3)) << ((i4 >>> 2 & 12L) | ((~i4 & 12L) << 2));
352        }
353        return v;
354    }

365     * @param seen a GreasedRegion where "on" cells are visible now or were visible in the past
366     * @param writeInto a 2D char array that must have at least the dimensions of map; will be modified
367     * @return writeInto, after modifications
368     */

375     * Takes a map that was produced using {@link DungeonUtility#hashesToLines(char[][])} a GreasedRegion that stores
376     * already-seen cells, an optional char array that refers to a line drawing style constant in this class (defaults
377     * to {@link #light}, and writes an altered version of the 2D char array {@code map} to {@code writeInto}, leaving
378     * non-box-drawing chars unchanged. This method modifies writeInto in-place, and also returns it after those changes
379     * are made. The way this works is explained well with an example: if the player is north of a T-junction wall, '┬',
380     * then unless he has already explored the area south of his position, the bottom segment of the wall isn't visible
381     * to him, and so '─' should be rendered instead of '┬'. If a cell has already been seen, it is considered still
382     * visible for the purpose of calculating shown segments (it won't change once you leave an area).
383     * @param map a 2D char array that should have been produced by {@link DungeonUtility#hashesToLines(char[][])}
384     * @param seen a GreasedRegion where "on" cells are visible now or were visible in the past
385     * @param symbols a char array that should be {@link #light} or {@link #heavy} unless you know your font supports
386     *                the chars "╴╵╶╷", in which case you can use {@link #lightAlt}, or the heavy-weight versions of 
387     *                those chars, in which case you can use {@link #heavyAlt}
388     * @param writeInto a 2D char array that must have at least the dimensions of map; will be modified
389     * @return writeInto, after modifications
390     */
391    public static char[][] pruneLines(char[][] map, GreasedRegion seen, char[] symbols, char[][] writeInto)
392    {
393        final int width = map.length, height = map[0].length;
394        int mask;
395        for (int x = 0; x < width; x++) {
396            for (int y = 0; y < height; y++) {
397                if(seen.contains(x, y)) 
398                {
399                    mask = 15;
400                    if(!seen.contains(x-1, y))
401                        mask ^= 1;
402                    if(!seen.contains(x+1, y))
403                        mask ^= 4;
404                    if(!seen.contains(x, y-1))
405                        mask ^= 2;
406                    if(!seen.contains(x, y+1))
407                        mask ^= 8;
408                    switch (map[x][y]) {
409                        case '─':
410                        case '━':
411                            writeInto[x][y] = symbols[5 & mask];
412                            break;
413                        case '│':
414                        case '┃':
415                            writeInto[x][y] = symbols[10 & mask];
416                            break;
417                        case '┘':
418                        case '┛':
419                            writeInto[x][y] = symbols[3 & mask];
420                            break;
421                        case '└':
422                        case '┗':
423                            writeInto[x][y] = symbols[6 & mask];
424                            break;
425                        case '┐':
426                        case '┓':
427                            writeInto[x][y] = symbols[9 & mask];
428                            break;
429                        case '┌':
430                        case '┏':
431                            writeInto[x][y] = symbols[12 & mask];
432                            break;
433                        case '┴':
434                        case '┻':
435                            writeInto[x][y] = symbols[7 & mask];
436                            break;
437                        case '┤':
438                        case '┫':
439                            writeInto[x][y] = symbols[11 & mask];
440                            break;
441                        case '┬':
442                        case '┳':
443                            writeInto[x][y] = symbols[13 & mask];
444                            break;
445                        case '├':
446                        case '┣':
447                            writeInto[x][y] = symbols[14 & mask];
448                            break;
449                        case '┼':
450                        case '╋':
451                            writeInto[x][y] = symbols[15 & mask];
452                            break;
453                        case '╴':
454                        case '╸':
455                            writeInto[x][y] = symbols[1 & mask];
456                            break;
457                        case '╵':
458                        case '╹':
459                            writeInto[x][y] = symbols[2 & mask];
460                            break;
461                        case '╶':
462                        case '╺':
463                            writeInto[x][y] = symbols[4 & mask];
464                            break;
465                        case '╷':
466                        case '╻':
467                            writeInto[x][y] = symbols[8 & mask];
468                            break;
469                    }
470                }
471            }
472        }
473        return writeInto;
474    }
