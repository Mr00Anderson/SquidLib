001package squidpony.squidgrid.mapping;
002
003import squidpony.ArrayTools;
004import squidpony.annotation.Beta;
005import squidpony.squidmath.GWTRNG;
006import squidpony.squidmath.IRNG;
007import squidpony.squidmath.IntIntOrderedMap;
008import squidpony.squidmath.IntVLA;
009
010/**

015@Beta
016public class ConnectingMapGenerator implements IDungeonGenerator {
017    
018    public int width;
019    public int height;
020    public int roomWidth;
021    public int roomHeight;
022    public char[][] dungeon;
023    public int[][] environment;
024    public IRNG rng;
025    
026    public ConnectingMapGenerator()
027    {
028        this(80, 80, new GWTRNG());
029    }
030    public ConnectingMapGenerator(int width, int height, IRNG random)
031    {
032        this(width, height, width >> 3, height >> 3, random);
033    }
034    public ConnectingMapGenerator(int width, int height, int roomWidth, int roomHeight, IRNG random)
035    {
036        this.width = Math.max(1, width);
037        this.height = Math.max(1, height);
038        this.roomWidth = Math.max(3, roomWidth);
039        this.roomHeight = Math.max(3, roomHeight);
040        dungeon = ArrayTools.fill(' ', this.width, this.height);
041        environment = new int[this.width][this.height];
042        rng = random;
043    }
044    /**
045     * Generates a dungeon or other map as a 2D char array. Any implementation may allow its own configuration and
046     * customization of how dungeons are generated, but each must provide this as a sane default. Most implementations
047     * should use the convention of '#' representing a wall and '.' representing a bare floor, but beyond that, anything
048     * could be present in the char array.
049     *
050     * @return a 2D char array representing some kind of map, probably using standard conventions for walls/floors
051     */
052    @Override
053    public char[][] generate() {
054        int gridWidth = width / roomWidth, gridHeight = height / roomHeight, gridMax = gridWidth * gridHeight;
055        if(gridWidth <= 0 || gridHeight <= 0)
056            return dungeon;
057        ArrayTools.fill(dungeon, '.');
058        ArrayTools.fill(environment, DungeonUtility.ROOM_FLOOR);
059        IntIntOrderedMap links = new IntIntOrderedMap(gridMax), surface = new IntIntOrderedMap(gridMax);
060        IntVLA choices = new IntVLA(4);
061        int dx = rng.nextSignedInt(gridWidth), dy = rng.nextSignedInt(gridHeight),
062                d = dy << 16 | dx;
063        links.put(d, 0);
064        surface.put(d, 0);
065        for (int i = 0; i < 15 && links.size() < gridMax && !surface.isEmpty(); i++) {
066            choices.clear();
067            if (dx < gridWidth - 1 && !links.containsKey(d + 1)) choices.add(1);
068            if (dy < gridHeight - 1 && !links.containsKey(d + 0x10000)) choices.add(2);
069            if (dx > 0 && !links.containsKey(d - 1)) choices.add(4);
070            if (dy > 0 && !links.containsKey(d - 0x10000)) choices.add(8);
071            if (choices.isEmpty()) {
072                surface.remove(d);
073                i--;
074                continue;
075            }
076            int choice = choices.getRandomElement(rng);
077            links.replace(d, links.get(d) | choice);
078            if (choices.size == 1)
079                surface.remove(d);
080            switch (choice) {
081                case 1:
082                    d += 1;
083                    links.put(d, 4);
084                    surface.put(d, 4);
085                    break;
086                case 2:
087                    d += 0x10000;
088                    links.put(d, 8);
089                    surface.put(d, 8);
090                    break;
091                case 4:
092                    d -= 1;
093                    links.put(d, 1);
094                    surface.put(d, 1);
095                    break;
096                default:
097                    d -= 0x10000;
098                    links.put(d, 2);
099                    surface.put(d, 2);
100                    break;
101            }
102            dx = d & 0xFFFF;
103            dy = d >>> 16;
104        }
105        while(links.size() < gridMax)
106        {
107            d = surface.randomKey(rng);
108            dx = d & 0xFFFF;
109            dy = d >>> 16;
110            for (int i = 0; i < 5 && links.size() < gridMax && !surface.isEmpty(); i++) {
111                choices.clear();
112                if (dx < gridWidth - 1 && !links.containsKey(d + 1)) choices.add(1);
113                if (dy < gridHeight - 1 && !links.containsKey(d + 0x10000)) choices.add(2);
114                if (dx > 0 && !links.containsKey(d - 1)) choices.add(4);
115                if (dy > 0 && !links.containsKey(d - 0x10000)) choices.add(8);
116                if (choices.isEmpty()) {
117                    surface.remove(d);
118                    break;
119                }
120                int choice = choices.getRandomElement(rng);
121                links.replace(d, links.get(d) | choice);
122                if (choices.size == 1)
123                    surface.remove(d);
124                switch (choice) {
125                    case 1:
126                        d += 1;
127                        links.put(d, 4);
128                        surface.put(d, 4);
129                        break;
130                    case 2:
131                        d += 0x10000;
132                        links.put(d, 8);
133                        surface.put(d, 8);
134                        break;
135                    case 4:
136                        d -= 1;
137                        links.put(d, 1);
138                        surface.put(d, 1);
139                        break;
140                    default:
141                        d -= 0x10000;
142                        links.put(d, 2);
143                        surface.put(d, 2);
144                        break;
145                }
146                dx = d & 0xFFFF;
147                dy = d >>> 16;
148            }
149        }
150        for (int i = 0; i < links.size(); i++) {
151            d = links.keyAt(i);
152            dx = d & 0xFFFF;
153            dy = d >>> 16;
154            int conn = links.getAt(i);
155            
156            dungeon[dx * roomWidth][dy * roomHeight] = '#';
157            dungeon[dx * roomWidth + roomWidth - 1][dy * roomHeight] = '#';
158            dungeon[dx * roomWidth][dy * roomHeight + roomHeight - 1] = '#';
159            dungeon[dx * roomWidth + roomWidth - 1][dy * roomHeight + roomHeight - 1] = '#';
160
161            if((conn & 1) == 0) {
162                ArrayTools.fill(dungeon, '#', dx * roomWidth + roomWidth - 1, dy * roomHeight, dx * roomWidth + roomWidth - 1, dy * roomHeight + roomHeight - 1);
163                ArrayTools.fill(environment, DungeonUtility.ROOM_WALL, dx * roomWidth + roomWidth - 1, dy * roomHeight, dx * roomWidth + roomWidth - 1, dy * roomHeight + roomHeight - 1);
164            }
165            else {
166                ArrayTools.fill(environment, DungeonUtility.CORRIDOR_FLOOR, dx * roomWidth + roomWidth - 1, dy * roomHeight + 1, dx * roomWidth + roomWidth - 1, dy * roomHeight + roomHeight - 2);
167                environment[dx * roomWidth + roomWidth - 1][dy * roomHeight] = DungeonUtility.CORRIDOR_WALL;
168                environment[dx * roomWidth + roomWidth - 1][dy * roomHeight + roomHeight - 1] = DungeonUtility.CORRIDOR_WALL;
169            }
170
171            if((conn & 2) == 0) {
172                ArrayTools.fill(dungeon, '#', dx * roomWidth, dy * roomHeight + roomHeight - 1, dx * roomWidth + roomWidth - 1, dy * roomHeight + roomHeight - 1);
173                ArrayTools.fill(environment, DungeonUtility.ROOM_WALL, dx * roomWidth, dy * roomHeight + roomHeight - 1, dx * roomWidth + roomWidth - 1, dy * roomHeight + roomHeight - 1);
174            }
175            else {
176                ArrayTools.fill(environment, DungeonUtility.CORRIDOR_FLOOR, dx * roomWidth + 1, dy * roomHeight + roomHeight - 1, dx * roomWidth + roomWidth - 2, dy * roomHeight + roomHeight - 1);
177                environment[dx * roomWidth + roomWidth - 1][dy * roomHeight] = DungeonUtility.CORRIDOR_WALL;
178                environment[dx * roomWidth][dy * roomHeight] = DungeonUtility.CORRIDOR_WALL;
179            }
180
181            if((conn & 4) == 0) {
182                ArrayTools.fill(dungeon, '#', dx * roomWidth, dy * roomHeight, dx * roomWidth, dy * roomHeight + roomHeight - 1);
183                ArrayTools.fill(environment, DungeonUtility.ROOM_WALL, dx * roomWidth, dy * roomHeight, dx * roomWidth, dy * roomHeight + roomHeight - 1);
184            }
185            else {
186                ArrayTools.fill(environment, DungeonUtility.CORRIDOR_FLOOR, dx * roomWidth, dy * roomHeight + 1, dx * roomWidth, dy * roomHeight + roomHeight - 2);
187                environment[dx * roomWidth][dy * roomHeight] = DungeonUtility.CORRIDOR_WALL;
188                environment[dx * roomWidth][dy * roomHeight + roomHeight - 1] = DungeonUtility.CORRIDOR_WALL;
189            }
190
191            if((conn & 8) == 0) {
192                ArrayTools.fill(dungeon, '#', dx * roomWidth, dy * roomHeight, dx * roomWidth + roomWidth - 1, dy * roomHeight);
193                ArrayTools.fill(environment, DungeonUtility.ROOM_WALL, dx * roomWidth, dy * roomHeight, dx * roomWidth + roomWidth - 1, dy * roomHeight);
194            }
195            else {
196                ArrayTools.fill(environment, DungeonUtility.CORRIDOR_FLOOR, dx * roomWidth + 1, dy * roomHeight, dx * roomWidth + roomWidth - 2, dy * roomHeight);
197                environment[dx * roomWidth][dy * roomHeight + roomHeight - 1] = DungeonUtility.CORRIDOR_WALL;
198                environment[dx * roomWidth + roomWidth - 1][dy * roomHeight + roomHeight - 1] = DungeonUtility.CORRIDOR_WALL;
199            }
200            
201        }
202        return dungeon;
203    }
204
205    /**
206     * Gets the most recently-produced dungeon as a 2D char array, usually produced by calling {@link #generate()} or
207     * some similar method present in a specific implementation. This normally passes a direct reference and not a copy,
208     * so you can normally modify the returned array to propagate changes back into this IDungeonGenerator.
209     *
210     * @return the most recently-produced dungeon/map as a 2D char array
211     */
212    @Override
213    public char[][] getDungeon() {
214        return dungeon;
215    }
216    /**
217     * Gets a 2D array of int constants, each representing a type of environment corresponding to a static field of
218     * DungeonUtility. This array will have the same size as the last char 2D array produced by generate(); the value
219     * of this method if called before generate() is undefined, but probably will be a 2D array of all 0 (UNTOUCHED).
220     * <ul>
221     * <li>DungeonUtility.UNTOUCHED, equal to 0, is used for any cells that aren't near a floor.</li>
222     * <li>DungeonUtility.ROOM_FLOOR, equal to 1, is used for floor cells inside wide room areas.</li>
223     * <li>DungeonUtility.ROOM_WALL, equal to 2, is used for wall cells around wide room areas.</li>
224     * <li>DungeonUtility.CAVE_FLOOR, equal to 3, is used for floor cells inside rough cave areas.</li>
225     * <li>DungeonUtility.CAVE_WALL, equal to 4, is used for wall cells around rough cave areas.</li>
226     * <li>DungeonUtility.CORRIDOR_FLOOR, equal to 5, is used for floor cells inside narrow corridor areas.</li>
227     * <li>DungeonUtility.CORRIDOR_WALL, equal to 6, is used for wall cells around narrow corridor areas.</li>
228     * </ul>
229     *
230     * @return a 2D int array where each element is an environment type constant in DungeonUtility
231     */
232    public int[][] getEnvironment() {
233        return environment;
234    }
