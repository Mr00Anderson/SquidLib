001package squidpony.squidgrid.mapping;
002
003import squidpony.ArrayTools;
004import squidpony.squidmath.*;
005
006/**
007 * A dungeon-or-other-map generator that can start with a pre-placed {@code char[][]} and connect any features in that
008 * map by placing rooms and corridors where there is blank space ({@code ' '}) in the original {@code char[][]}.
009 * <br>
010 * Created by Tommy Ettinger on 5/7/2019.
011 */
012public class ConnectingMapGenerator implements IDungeonGenerator {
013    
014    public int width;
015    public int height;
016    public char[][] dungeon;
017    public int[][] environment;
018    public IRNG rng;
019    
020    public ConnectingMapGenerator()
021    {
022        this(80, 80, new GWTRNG());
023    }
024    public ConnectingMapGenerator(int width, int height, IRNG random)
025    {
026        this.width = Math.max(1, width);
027        this.height = Math.max(1, height);
028        dungeon = ArrayTools.fill(' ', this.width, this.height);
029        environment = new int[this.width][this.height];
030        rng = random;
031    }
032    public ConnectingMapGenerator(char[][] prePlaced, IRNG random)
033    {
034        dungeon = ArrayTools.copy(prePlaced);
035        width = dungeon.length;
036        height = dungeon[0].length;
037        environment = new int[this.width][this.height];
038        rng = random;
039    }
040    /**
041     * Generates a dungeon or other map as a 2D char array. Any implementation may allow its own configuration and
042     * customization of how dungeons are generated, but each must provide this as a sane default. Most implementations
043     * should use the convention of '#' representing a wall and '.' representing a bare floor, but beyond that, anything
044     * could be present in the char array.
045     *
046     * @return a 2D char array representing some kind of map, probably using standard conventions for walls/floors
047     */
048    @Override
049    public char[][] generate() {
050        GreasedRegion empty = new GreasedRegion(dungeon, ' ').retract8way();
051        //GreasedRegion blocked = empty.copy().not();
052        GreasedRegion working = empty.copy().empty(), working2 = working.copy(), working3 = working.copy();
053        GreasedRegion rooms = working.copy(), corridors = working.copy(), doors = new GreasedRegion(dungeon, '+');
054        Coord pt;
055        int frustration = 0, targetSize = width * height * 3 >> 3;
056        while (frustration < 100 && empty.size() > targetSize)
057        {
058            pt = empty.singleRandom(rng);
059            working.insertRectangle(
060                    MathExtras.clamp(pt.x + 4 - rng.nextSignedInt(9), 0, width - 1),
061                    MathExtras.clamp(pt.y + 4 - rng.nextSignedInt(9), 0, height - 1),
062                    rng.between(6, 12), rng.between(6, 12));
063            if (empty.not().intersects(working))
064            {
065                empty.or(working);
066                rooms.or(working.retract8way());
067            }
068            else frustration++;
069            empty.not();
070            working.clear();
071        }
072        doors.or(working.remake(rooms).fringe().removeEdges().quasiRandomRegion(0.075));
073        working.remake(doors);
074        frustration = 0;
075        for(Coord door : working)
076        {
077            pt = doors.singleRandom(rng);
078            working2.insertRectangle(Math.min(door.x, pt.x), Math.min(door.y, pt.y),
079                    Math.abs(door.x - pt.x), Math.abs(door.y - pt.y)).and(empty).largestPart();
080            working3.remake(working2).expand();
081            if(working3.contains(door) && working3.contains(pt))
082                corridors.or(working2);
083            else if(++frustration >= 150)
084                break;
085        }
086        rooms.writeCharsInto(dungeon, '.');
087        rooms.writeIntsInto(environment, MixedGenerator.ROOM_FLOOR);
088        corridors.writeCharsInto(dungeon, '.');
089        corridors.writeIntsInto(environment, MixedGenerator.CORRIDOR_FLOOR);
090        doors.writeCharsInto(dungeon, '+');
091        corridors.fringe8way().writeIntsInto(environment, MixedGenerator.CORRIDOR_WALL);
092        rooms.fringe8way().writeIntsInto(environment, MixedGenerator.ROOM_WALL);
093        doors.writeIntsInto(environment, MixedGenerator.CORRIDOR_FLOOR);
094        return dungeon;
095    }
096
097    /**
098     * Gets the most recently-produced dungeon as a 2D char array, usually produced by calling {@link #generate()} or
099     * some similar method present in a specific implementation. This normally passes a direct reference and not a copy,
100     * so you can normally modify the returned array to propagate changes back into this IDungeonGenerator.
101     *
102     * @return the most recently-produced dungeon/map as a 2D char array
103     */
104    @Override
105    public char[][] getDungeon() {
106        return dungeon;
107    }
108    /**
109     * Gets a 2D array of int constants, each representing a type of environment corresponding to a static field of
110     * MixedGenerator. This array will have the same size as the last char 2D array produced by generate(); the value
111     * of this method if called before generate() is undefined, but probably will be a 2D array of all 0 (UNTOUCHED).
112     * <ul>
113     * <li>MixedGenerator.UNTOUCHED, equal to 0, is used for any cells that aren't near a floor.</li>
114     * <li>MixedGenerator.ROOM_FLOOR, equal to 1, is used for floor cells inside wide room areas.</li>
115     * <li>MixedGenerator.ROOM_WALL, equal to 2, is used for wall cells around wide room areas.</li>
116     * <li>MixedGenerator.CAVE_FLOOR, equal to 3, is used for floor cells inside rough cave areas.</li>
117     * <li>MixedGenerator.CAVE_WALL, equal to 4, is used for wall cells around rough cave areas.</li>
118     * <li>MixedGenerator.CORRIDOR_FLOOR, equal to 5, is used for floor cells inside narrow corridor areas.</li>
119     * <li>MixedGenerator.CORRIDOR_WALL, equal to 6, is used for wall cells around narrow corridor areas.</li>
120     * </ul>
121     *
122     * @return a 2D int array where each element is an environment type constant in MixedGenerator
123     */
124    public int[][] getEnvironment() {
125        return environment;
126    }
127
128}
