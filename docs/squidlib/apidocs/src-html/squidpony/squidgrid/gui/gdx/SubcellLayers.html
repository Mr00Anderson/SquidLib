020/**
021 * A subclass of SparseLayers that acts almost the same, but uses 3x3 subcells of background color for every cell that
022 * may contain a char. Methods that affect the background sometimes specify positions in subcells, which allows
023 * affecting less than one cell with some visual effect. You will usually want to use a triple-size resistance and FOV
024 * map; you can get a resistance map that respects subcells with
025 * {@link squidpony.squidgrid.mapping.DungeonUtility#generateResistances3x3(char[][])} (this looks better if it is given
026 * a "line" dungeon as produced by {@link squidpony.squidgrid.mapping.DungeonUtility#hashesToLines(char[][], boolean)}),
027 * and then you can use that resistance map with normal FOV methods, just typically at triple vision range. Much of the
028 * time, it makes sense to run both that triple-range FOV, and a normal-range FOV with a resistance map calculated in
029 * cells; you can use the normal-range FOV for gameplay and AI calculations and the triple-range for visual display. 
030 * <br>
031 * Created by Tommy Ettinger on 8/28/2018.
032 */
033public class SubcellLayers extends SparseLayers {
034    public SubcellLayers(int gridWidth, int gridHeight)
035    {
036        this(gridWidth, gridHeight, 10, 16, DefaultResources.getStretchableFont());
037    }

048    public SubcellLayers(int gridWidth, int gridHeight, float cellWidth, float cellHeight, TextCellFactory font, float xOffset, float yOffset) {
049        this.gridWidth = MathUtils.clamp(gridWidth, 1, 65535);
050        this.gridHeight = MathUtils.clamp(gridHeight, 1, 65535);
051        backgrounds = new float[this.gridWidth*3][this.gridHeight*3];
052        layers = new ArrayList<>(4);
053        if(font.initialized())
054            this.font = font;
055        else
056            this.font = font.width(cellWidth).height(cellHeight).initBySize();
057        layers.add(new SparseTextMap(gridWidth * gridHeight >> 2));
058        mapping = new IntIntMap(4);
059        mapping.put(0, 0);
060        glyphs = new ArrayList<>(16);
061        scc = DefaultResources.getSCC();
062        setBounds(xOffset, yOffset,
063                this.font.actualCellWidth * this.gridWidth, this.font.actualCellHeight * this.gridHeight);
064    }

319    /**
320     * Changes the background at position x,y to the given Color. If the color is null, then this will make the
321     * background fully transparent at the specified position.
322     *
323     * @param x     where to change the background color, x-coordinate
324     * @param y     where to change the background color, y-coordinate
325     * @param color the Color to change to; if null will be considered fully transparent
326     */
327    @Override
328    public void put(int x, int y, Color color) {
329        final float bg = (color == null) ? 0x0.0p0F : scc.filter(color).toFloatBits();
330        x *= 3;
331        y *= 3;
332        backgrounds[x  ][y  ] = bg;
333        backgrounds[x  ][y+1] = bg;
334        backgrounds[x  ][y+2] = bg;
335        backgrounds[x+1][y  ] = bg;
336        backgrounds[x+1][y+1] = bg;
337        backgrounds[x+1][y+2] = bg;
338        backgrounds[x+2][y  ] = bg;
339        backgrounds[x+2][y+1] = bg;
340        backgrounds[x+2][y+2] = bg;
341    }

499    /**
500     * Removes the foreground chars, where present, in all layers at the given x,y position.
501     * The backgrounds will be unchanged.
502     *
503     * @param x the x-coordinate of the position to remove all chars from
504     * @param y the y-coordinate of the position to remove all chars from
505     */
506    @Override
507    public void clear(int x, int y) {
508        if(x < 0 || x >= gridWidth || y < 0 || y >= gridHeight)
509            return;
510        int code = SparseTextMap.encodePosition(x, y);
511        for (int i = 0; i < layers.size(); i++) {
512            layers.get(i).remove(code);
513        }
514        x *= 3;
515        y *= 3;
516        backgrounds[x  ][y  ] = 0f;
517        backgrounds[x  ][y+1] = 0f;
518        backgrounds[x  ][y+2] = 0f;
519        backgrounds[x+1][y  ] = 0f;
520        backgrounds[x+1][y+1] = 0f;
521        backgrounds[x+1][y+2] = 0f;
522        backgrounds[x+2][y  ] = 0f;
523        backgrounds[x+2][y+1] = 0f;
524        backgrounds[x+2][y+2] = 0f;
525
526    }

528    /**
529     * Changes the background color in an area to all have the given color, as a libGDX Color (or SColor, etc.).
530     * This uses subcell measurements for x, y, width, and height, where the grid width in subcells is
531     * 3 * {@link #gridWidth} and the grid height in subcells is 3 * {@link #gridHeight}. To affect a full cell, this
532     * needs a width and height of at least 3 each; anything less will affect only some subcells.
533     *
534     * @param color  a libGDX Color to fill the area with; may be null to make the background transparent
535     * @param x      left edge's x coordinate, in subcells (3 subcells to one cell horizontally)
536     * @param y      top edge's y coordinate, in subcells (3 subcells to one cell vertically)
537     * @param width  the width of the area to change the color on, in subcells (3 subcells to one cell horizontally)
538     * @param height the height of the area to change the color on, in subcells (3 subcells to one cell vertically)
539     */
540    @Override
541    public void fillArea(Color color, int x, int y, int width, int height) {
542        fillArea(color == null ? 0f : scc.filter(color).toFloatBits(), x, y, width, height);
543    }
544    /**
545     * Changes the background color in an area to all have the given color, as a packed float.
546     * This uses subcell measurements for x, y, width, and height, where the grid width in subcells is
547     * 3 * {@link #gridWidth} and the grid height in subcells is 3 * {@link #gridHeight}. To affect a full cell, this
548     * needs a width and height of at least 3 each; anything less will affect only some subcells.
549     * 
550     * @param color a color as a packed float to fill the area with; may be 0f to make the background transparent
551     * @param x      left edge's x coordinate, in subcells (3 subcells to one cell horizontally)
552     * @param y      top edge's y coordinate, in subcells (3 subcells to one cell vertically)
553     * @param width  the width of the area to change the color on, in subcells (3 subcells to one cell horizontally)
554     * @param height the height of the area to change the color on, in subcells (3 subcells to one cell vertically)
555     */
556    public void fillArea(float color, int x, int y, int width, int height) {
557        if (x < 0) {
558            width += x;
559            x = 0;
560        }
561        if (y < 0) {
562            height += y;
563            y = 0;
564        }
565        if (width <= 0 || height <= 0)
566            return;
567        final int gw = gridWidth * 3, gh = gridHeight * 3;
568        for (int i = 0, xx = x; i < width && xx < gw; i++, xx++) {
569            for (int j = 0, yy = y; j < height && yy < gh; j++, yy++) {
570                backgrounds[xx][yy] = color;
571            }
572        }
573    }

575    /**
576     * Using the existing background color at the subcell position x,y, this performs color blending from that existing
577     * color to the given color (as a float), using the mixBy parameter to determine how much of the color parameter to
578     * use (1f will set the color in this to the parameter, while 0f for mixBy will ignore the color parameter
579     * entirely). The x and y parameters are in subcells, where 3 subcells have the same width or height as one cell (a
580     * cell holds a char, and 3x3 subcells fit in the same area).
581     * @param x the x component of the position in this panel to draw the starting color from
582     * @param y the y component of the position in this panel to draw the starting color from
583     * @param color the new color to mix with the starting color; a packed float, as made by {@link Color#toFloatBits()}
584     * @param mixBy the amount by which the new color will affect the old one, between 0 (no effect) and 1 (overwrite)
585     */
586    @Override
587    public void blend(int x, int y, float color, float mixBy)
588    {
589        backgrounds[x][y] = SColor.lerpFloatColorsBlended(backgrounds[x][y], color, mixBy);
590    }

592    /**
593     * Tints the background at position x,y (in cells) so it becomes the given encodedColor, waiting for {@code delay}
594     * (in seconds) before performing it, then after the tint is complete it returns the cell to its original color,
595     * taking duration seconds. Additionally, enqueue {@code postRunnable} for running after the created action ends.
596     * All subcells in the tinted cell will reach the same color during this animation, but the subcells can start with
597     * different colors, and they will return to those starting colors after this animation finishes.
598     * <br>
599     * This will only behave correctly if you call {@link Stage#act()} before you call {@link Stage#draw()}, but after
600     * any changes to the contents of this SparseLayers. If you change the contents, then draw, and then act, that will
601     * draw the contents without the tint this applies, then apply the tint when you call act(), then quickly overwrite
602     * the tint in the next frame. That visually appears as nothing happening other than a delay.
603     * @param delay how long to wait in seconds before starting the effect
604     * @param x the x-coordinate of the cell to tint
605     * @param y the y-coordinate of the cell to tint
606     * @param encodedColor what to transition the cell's color towards, and then transition back from, as a packed float
607     * @param duration how long the total "round-trip" transition should take in seconds
608     * @param postRunnable a Runnable to execute after the tint completes; may be null to do nothing.
609     */
610    public void tint(final float delay, final int x, final int y, final float encodedColor, float duration,
611            /* @Nullable */ Runnable postRunnable) {
612        if(x < 0 || x >= gridWidth || y < 0 || y >= gridHeight)
613            return;
614        duration = Math.max(0.015f, duration);
615        final int xx = x * 3, yy = y * 3;
616        final float
617                x0y0 = backgrounds[xx][yy], x1y0 = backgrounds[xx+1][yy], x2y0 = backgrounds[xx+2][yy],
618                x0y1 = backgrounds[xx][yy+1], x1y1 = backgrounds[xx+1][yy+1], x2y1 = backgrounds[xx+2][yy+1],
619                x0y2 = backgrounds[xx][yy+2], x1y2 = backgrounds[xx+1][yy+2], x2y2 = backgrounds[xx+2][yy+2];
620        final int nbActions = 3 + (0 < delay ? 1 : 0) + (postRunnable == null ? 0 : 1);
621        final Action[] sequence = new Action[nbActions];
622        int index = 0;
623        if (0 < delay)
624            sequence[index++] = Actions.delay(delay);
625        sequence[index++] = new TemporalAction(duration * 0.3f) {
626            @Override
627            protected void update(float percent) {
628                backgrounds[xx  ][yy  ] = SColor.lerpFloatColors(x0y0, encodedColor, percent);
629                backgrounds[xx  ][yy+1] = SColor.lerpFloatColors(x0y1, encodedColor, percent);
630                backgrounds[xx  ][yy+2] = SColor.lerpFloatColors(x0y2, encodedColor, percent);
631                backgrounds[xx+1][yy  ] = SColor.lerpFloatColors(x1y0, encodedColor, percent);
632                backgrounds[xx+1][yy+1] = SColor.lerpFloatColors(x1y1, encodedColor, percent);
633                backgrounds[xx+1][yy+2] = SColor.lerpFloatColors(x1y2, encodedColor, percent);
634                backgrounds[xx+2][yy  ] = SColor.lerpFloatColors(x2y0, encodedColor, percent);
635                backgrounds[xx+2][yy+1] = SColor.lerpFloatColors(x2y1, encodedColor, percent);
636                backgrounds[xx+2][yy+2] = SColor.lerpFloatColors(x2y2, encodedColor, percent);
637            }
638        };
639        sequence[index++] = new TemporalAction(duration * 0.7f) {
640            @Override
641            protected void update(float percent) {
642                backgrounds[xx  ][yy  ] = SColor.lerpFloatColors(encodedColor, x0y0, percent);
643                backgrounds[xx  ][yy+1] = SColor.lerpFloatColors(encodedColor, x0y1, percent);
644                backgrounds[xx  ][yy+2] = SColor.lerpFloatColors(encodedColor, x0y2, percent);
645                backgrounds[xx+1][yy  ] = SColor.lerpFloatColors(encodedColor, x1y0, percent);
646                backgrounds[xx+1][yy+1] = SColor.lerpFloatColors(encodedColor, x1y1, percent);
647                backgrounds[xx+1][yy+2] = SColor.lerpFloatColors(encodedColor, x1y2, percent);
648                backgrounds[xx+2][yy  ] = SColor.lerpFloatColors(encodedColor, x2y0, percent);
649                backgrounds[xx+2][yy+1] = SColor.lerpFloatColors(encodedColor, x2y1, percent);
650                backgrounds[xx+2][yy+2] = SColor.lerpFloatColors(encodedColor, x2y2, percent);
651            }
652        };
653        if(postRunnable != null)
654        {
655            sequence[index++] = Actions.run(postRunnable);
656        }
657        sequence[index] = Actions.delay(duration, Actions.run(new Runnable() {
658            @Override
659            public void run() {
660                backgrounds[xx  ][yy  ] = x0y0;
661                backgrounds[xx  ][yy+1] = x0y1;
662                backgrounds[xx  ][yy+2] = x0y2;
663                backgrounds[xx+1][yy  ] = x1y0;
664                backgrounds[xx+1][yy+1] = x1y1;
665                backgrounds[xx+1][yy+2] = x1y2;
666                backgrounds[xx+2][yy  ] = x2y0;
667                backgrounds[xx+2][yy+1] = x2y1;
668                backgrounds[xx+2][yy+2] = x2y2;
669            }
670        }));
671
672        addAction(Actions.sequence(sequence));
673    }
674    /**
675     * Draws the SubcellLayers and all glyphs it tracks. {@link Batch#begin()} must have already been called on the
676     * batch, and {@link Batch#end()} should be called after this returns and before the rendering code finishes for the
677     * frame.
678     * <br>
679     * This will set the shader of {@code batch} if using a distance field or MSDF font and the shader is currently not
680     * configured for such a font; it does not reset the shader to the default so that multiple Actors can all use the
681     * same shader and so specific extra glyphs or other items can be rendered after calling draw(). If you need to draw
682     * both a distance field font and full-color art, you should set the shader on the Batch to null when you want to
683     * draw full-color art, and end the Batch between drawing this object and the other art.
684     *
685     * @param batch a Batch such as a {@link FilterBatch} that must be between a begin() and end() call; usually done by Stage
686     * @param parentAlpha currently ignored
687     */
688    @Override
689    public void draw(Batch batch, float parentAlpha) {
690        float xo = getX(), yo = getY(), yOff = yo + 1f + gridHeight * font.actualCellHeight, gxo, gyo;
691        font.draw(batch, backgrounds, xo, yo, 3, 3);
692        int len = layers.size();
693        Frustum frustum = null;
694        Stage stage = getStage();
695        if(stage != null) {
696            Viewport viewport = stage.getViewport();
697            if(viewport != null)
698            {
699                Camera camera = viewport.getCamera();
700                if(camera != null)
701                {
702                    if(
703                            camera.frustum != null &&
704                                    (!camera.frustum.boundsInFrustum(xo, yOff - font.actualCellHeight - 1f, 0f, font.actualCellWidth, font.actualCellHeight, 0f) ||
705                                            !camera.frustum.boundsInFrustum(xo + font.actualCellWidth * (gridWidth-1), yo, 0f, font.actualCellWidth, font.actualCellHeight, 0f))
706                    )
707                        frustum = camera.frustum;
708                }
709            }
710        }
711        font.configureShader(batch);
712        if(frustum == null) {
713            for (int i = 0; i < len; i++) {
714                layers.get(i).draw(batch, font, xo, yOff);
715            }
716        }
717        else
718        {
719            for (int i = 0; i < len; i++) {
720                layers.get(i).draw(batch, font, frustum, xo, yOff);
721            }
722        }
723
724        int x, y;
725        for (int i = 0; i < glyphs.size(); i++) {
726            TextCellFactory.Glyph glyph = glyphs.get(i);
727            if(glyph == null)
728                continue;
729            glyph.act(Gdx.graphics.getDeltaTime());
730            if(!glyph.isVisible() ||
731                            (x = Math.round((gxo = glyph.getX() - xo) / font.actualCellWidth)) < 0 || x >= gridWidth ||
732                            (y = Math.round((gyo = glyph.getY() - yo)  / -font.actualCellHeight + gridHeight)) < 0 || y >= gridHeight ||
733                            backgrounds[x * 3 + 1][y * 3 + 1] == 0f || (frustum != null && !frustum.boundsInFrustum(gxo, gyo, 0f, font.actualCellWidth, font.actualCellHeight, 0f)))
734                continue;
735            glyph.draw(batch, 1f);
736        }
737    }
