014/**

016 * color creation.
017 *
018 * All returned SColor objects are cached so multiple requests for the same
019 * SColor will not create duplicate long term objects.

021 * @see SquidColorCenter another technique for caching colors that allows filters.
022 *
023 * @author Eben Howard - http://squidpony.com - howard@squidpony.com

034    /**
035     * Constructs a new SColorFactory with an empty cache.
036     * <br>
037     * Implementation note: SColorFactory initially used static methods and was not intended to
038     * be constructed by users, but for compatibility with Android (where static instances may
039     * linger into another run of an application), it was reworked to require an object to be
040     * constructed and for that object to contain the cache, rather than a static instance. Code
041     * written for earlier versions of SquidLib may use static methods on SColorFactory; most
042     * code written in SquidLib 3.0.0 beta 2 and newer prefers the IColorCenter implementations
043     * such as SquidColorCenter when SColor features such as color names aren't required. Since
044     * beta 4, SquidColorCenter implements most of SColorFactory's API, and also supports an
045     * IFilter for automatic changes to requested colors.
046     */
047    public SColorFactory() {
048
049        valueLookup = new IntMap<>();
050        rng = new GWTRNG();
051        colorBag = new IntMap<>();
052        palettes = new ObjectMap<>();
053        floor = 1;

057    /**
058     * Returns the SColor whose value matches the one passed in. If no SColor
059     * constant matches that value, then a cached or new SColor is returned that
060     * matches the provided value.
061     *
062     * This method constructs a list of the SColor constants the first time it
063     * is called.
064     *
065     * @param argb an int encoding 256 * 256 * 256 * alpha + 256 * 256 * red + 256 * green + blue
066     * @return the SColor with value argb
067     */
068    public SColor colorForValue(int argb) {
069        if (valueLookup.size == 0) {
070            int len = SColor.FULL_PALETTE.length;
071            SColor sc;
072            for (int i = 0; i < len; i++) {
073                sc = SColor.FULL_PALETTE[i];
074                valueLookup.put(Color.argb8888(sc), sc);
075            }
076        }
077
078        return valueLookup.containsKey(argb) ? valueLookup.get(argb) : asSColor(argb);
079    }

090    /**
091     * Utility method to blend the two colors by the amount passed in as the
092     * coefficient.
093     *
094     * @param a
095     * @param b
096     * @param coef
097     * @return
098     */
099    @SuppressWarnings("unused")
100        private int blend(int a, int b, double coef) {
101        coef = MathUtils.clamp(coef, 0, 1);
102        return (int) (a + (b - a) * coef);
103    }
104    /**
105     * Utility method to blend the two colors by the amount passed in as the
106     * coefficient.
107     *
108     * @param a
109     * @param b
110     * @param coef
111     * @return
112     */
113    private float blend(float a, float b, double coef) {
114        float cf = MathUtils.clamp((float)coef, 0, 1);
115        return (a + (b - a) * cf);
116    }

118    /**
119     * Returns an SColor that is the given distance from the first color to the
120     * second color.
121     *
122     * @param color1 The first color
123     * @param color2 The second color
124     * @param coef The percent towards the second color, as 0.0 to 1.0
125     * @return
126     */
127    public SColor blend(SColor color1, SColor color2, double coef) {
128        return asSColor(blend(color1.a, color2.a, coef),
129                blend(color1.r, color2.r, coef),
130                blend(color1.g, color2.g, coef),
131                blend(color1.b, color2.b, coef));
132    }

214    /**
215     * Returns the cached color that matches the desired rgb value.
216     *
217     * If the color is not already in the cache, it is created and added to the
218     * cache.
219     *
220     * This method does not check to see if the value is already available as a
221     * SColor constant. If such functionality is desired then please use
222     * colorForValue(int rgb) instead.
223     *
224     * @param argb
225     * @return
226     */
227    public SColor asSColor(int argb) {
228        int working = argb;
229        if (floor != 1) {//need to convert to floored values
230            int a = (argb >>> 24) & 0xff;
231            a -= a % floor;
232            int r = (argb >> 16) & 0xff;
233            r -= r % floor;
234            int g = (argb >> 8) & 0xff;
235            g -= g % floor;
236            int b = (argb) & 0xff;
237            b -= b % floor;
238
239            //put back together
240            working = ((a & 0xFF) << 24)
241                    | ((r & 0xFF) << 16)
242                    | ((g & 0xFF) << 8)
243                    | (b & 0xFF);
244        }
245
246        if (colorBag.containsKey(working)) {
247            return colorBag.get(working);
248        } else {
249            SColor color = new SColor(working);
250            colorBag.put(working, color);
251            return color;
252        }
253    }
254    /**
255     * Returns the cached color that matches the desired rgb value.
256     *
257     * If the color is not already in the cache, it is created and added to the
258     * cache.
259     *
260     * This method does not check to see if the value is already available as a
261     * SColor constant. If such functionality is desired then please use
262     * colorForValue(int rgb) instead.
263     *
264     * @param a alpha, between 0.0 and 1.0 inclusive
265     * @param r red, between 0.0 and 1.0 inclusive
266     * @param g green, between 0.0 and 1.0 inclusive
267     * @param b blue, between 0.0 and 1.0 inclusive
268     * @return an SColor with an unspecified name and the given a, r, g, and b values
269     */
270    public SColor asSColor(float a, float r, float g, float b) {
271        int aa = MathUtils.round(255 * a);
272        aa -= aa % floor;
273        int rr = MathUtils.round(255 * r);
274        rr -= rr % floor;
275        int gg = MathUtils.round(255 * g);
276        gg -= gg % floor;
277        int bb = MathUtils.round(255 * b);
278        bb -= bb % floor;
279
280        //put back together
281        final int working = ((aa & 0xFF) << 24)
282                | ((rr & 0xFF) << 16)
283                | ((gg & 0xFF) << 8)
284                | (bb & 0xFF);
285
286
287        if (colorBag.containsKey(working)) {
288            return colorBag.get(working);
289        } else {
290            SColor color = new SColor(working);
291            colorBag.put(working, color);
292            return color;
293        }
294    }

308    /**
309     * Returns an SColor with the given values, with those values clamped
310     * between 0 and 255.
311     *
312     * @param a alpha, between 0 and 255 inclusive
313     * @param r red, between 0 and 255 inclusive
314     * @param g green, between 0 and 255 inclusive
315     * @param b blue, between 0 and 255 inclusive
316     * @return an SColor with an unspecified name and the given a, r, g, and b values
317     */
318    public SColor asSColor(int a, int r, int g, int b) {
319        a = Math.max(Math.min(a, 255), 0);
320        r = Math.max(Math.min(r, 255), 0);
321        g = Math.max(Math.min(g, 255), 0);
322        b = Math.max(Math.min(b, 255), 0);
323        return asSColor((a << 24) | (r << 16) | (g << 8) | b);
324    }

403     * Returns an SColor that is the fully desaturated (greyscale) version of
404     * the provided color.
405     *
406     * @param color
407     * @return
408     */
409    public SColor desaturated(SColor color) {
410        int r = MathUtils.round(color.r * 255);
411        int g = MathUtils.round(color.g * 255);
412        int b = MathUtils.round(color.b * 255);
413
414        int average = (int) (r * 0.299 + g * 0.587 + b * 0.114);
415
416        return asSColor(average, average, average);
417    }

432    /**
433     * Returns a list of colors starting at the first color and moving to the
434     * second color. The end point colors are included in the list.
435     *
436     * @param color1 starting color
437     * @param color2 ending color
438     * @return an ArrayList of SColor going from color1 to color2; the length should be no greater than 256.
439     */
440    public ArrayList<SColor> asGradient(SColor color1, SColor color2) {
441        String name = paletteNamer(color1, color2);
442        if (palettes.containsKey(name)) {
443            return palettes.get(name);
444        }
445
446        //get the gradient
447        Queue<Coord3D> gradient = Bresenham.line3D(scolorToCoord3D(color1), scolorToCoord3D(color2));
448        ArrayList<SColor> ret = new ArrayList<>();
449        for (Coord3D coord : gradient) {
450            ret.add(coord3DToSColor(coord));
451        }
452
453        palettes.put(name, ret);
454        return ret;
455    }

479    /**
480     * Returns the SColor that is the provided percent towards the end of the
481     * palette. Bounds are checked so as long as there is at least one color in
482     * the palette, values below 0 will return the first element and values
483     * above 1 will return the last element;
484     *
485     * If there is no palette keyed to the provided name, null is returned.
486     *
487     * @param name
488     * @param percent
489     * @return
490     */
491    public SColor fromPalette(String name, float percent) {
492        ArrayList<SColor> list = palettes.get(name);
493        if (list == null) {
494            return null;
495        }
496
497        int index = Math.round(list.size() * percent);//find the index that's the given percent into the gradient
498        index = Math.min(index, list.size() - 1);
499        index = Math.max(index, 0);
500        return list.get(index);
501    }

540    /**
541     * Places the palette into the cache, along with each of the member colors.
542     *
543     * @param name
544     * @param palette
545     */
546    public void addPalette(String name, ArrayList<SColor> palette) {
547        ArrayList<SColor> temp = new ArrayList<>();
548
549        //make sure all the colors in the palette are also in the general color cache
550        for (SColor sc : palette) {
551            temp.add(asSColor(sc.a, sc.r, sc.g, sc.b));
552        }
553
554        palettes.put(name, temp);
555    }

557    /**
558     * Converts the provided color into a three dimensional coordinate point for
559     * use in the Bresenham algorithms.
560     *
561     * @param color
562     * @return
563     */
564    private Coord3D scolorToCoord3D(SColor color) {
565        return new Coord3D(MathUtils.floor(color.r * 255), MathUtils.floor(color.g * 255), MathUtils.floor(color.b * 255));
566    }
567
568    /**
569     * Converts the provided three dimensional coordinate into a color for use
570     * in the Bresenham algorithms.
571     *
572     * @param coord
573     * @return
574     */
575    private SColor coord3DToSColor(Coord3D coord) {
576        return asSColor(coord.x, coord.y, coord.z);
577    }
