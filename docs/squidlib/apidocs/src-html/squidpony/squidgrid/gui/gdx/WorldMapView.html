037    public void setWorld(WorldMapGenerator world) {
038        this.world = world;
039        if(this.width != world.width || this.height != world.height)
040        {
041            width = world.width;
042            height = world.height;
043            colorMap = new float[width][height];
044        }
045    }

047    public WorldMapView(WorldMapGenerator worldMapGenerator)
048    {
049        world = worldMapGenerator == null ? new WorldMapGenerator.LocalMap() : worldMapGenerator;
050        width = world.width;
051        height = world.height;
052        colorMap = new float[width][height];
053        this.biomeMapper = new WorldMapGenerator.DetailedBiomeMapper();
054        initialize();
055    }

138    public void initialize(float hue, float saturation, float brightness, float contrast)
139    {
140        float b, diff;
141        for (int i = 0; i < 60; i++) {
142            b = BIOME_TABLE[i];
143            diff = ((b % 1.0f) - 0.48f) * 0.27f * contrast;
144            BIOME_COLOR_TABLE[i] = b = SColor.toEditedFloat((diff >= 0)
145                    ? SColor.lightenFloat(biomeColors[(int)b], diff)
146                    : SColor.darkenFloat(biomeColors[(int)b], -diff), hue, saturation, brightness, 0f);
147            BIOME_DARK_COLOR_TABLE[i] = SColor.darkenFloat(b, 0.08f);
148        }
149        BIOME_COLOR_TABLE[60] = BIOME_DARK_COLOR_TABLE[60] = emptyColor;
150    }

152    public void generate()
153    {
154        generate(world.seedA, world.seedB, 0.9 + NumberTools.formCurvedDouble((world.seedA ^ 0x123456789ABCDL) * 0x12345689ABL ^ world.seedB) * 0.15,
155                DiverRNG.determineDouble(world.seedB * 0x12345L + 0x54321L ^ world.seedA) * 0.2 + 1.0);
156    }
157    public void generate(double landMod, double heatMod)
158    {
159        generate(world.seedA, world.seedB, landMod, heatMod);
160    }
161    
162    public void generate(int seedA, int seedB, double landMod, double heatMod) {
163        long seed = (long) seedB << 32 | (seedA & 0xFFFFFFFFL);
164        world.generate(landMod, heatMod, seed);
165        biomeMapper.makeBiomes(world);
166    }
167    public float[][] show()
168    {
169        int hc, tc, bc;
170        final int[][] heightCodeData = world.heightCodeData;
171        double[][] heightData = world.heightData;
172        int[][] heatCodeData = biomeMapper.heatCodeData;
173        int[][] biomeCodeData = biomeMapper.biomeCodeData;
174
175        for (int y = 0; y < height; y++) {
176            PER_CELL:
177            for (int x = 0; x < width; x++) {
178                hc = heightCodeData[x][y];
179                if(hc == 1000)
180                {
181                    colorMap[x][y] = emptyColor;
182                    continue;
183                }
184                tc = heatCodeData[x][y];
185                bc = biomeCodeData[x][y];
186                if(tc == 0)
187                {
188                    switch (hc)
189                    {
190                        case 0:
191                        case 1:
192                        case 2:
193                        case 3:
194                            colorMap[x][y] = SColor.lerpFloatColors(shallowColor, iceColor,
195                                    (float) ((heightData[x][y] - -1.0) / (WorldMapGenerator.sandLower - -1.0)));
196                            continue PER_CELL;
197                        case 4:
198                            colorMap[x][y] = SColor.lerpFloatColors(lightIceColor, iceColor,
199                                    (float) ((heightData[x][y] - WorldMapGenerator.sandLower) / (WorldMapGenerator.sandUpper - WorldMapGenerator.sandLower)));
200                            continue PER_CELL;
201                    }
202                }
203                switch (hc) {
204                    case 0:
205                    case 1:
206                    case 2:
207                    case 3:
208                        colorMap[x][y] = SColor.lerpFloatColors(
209                                BIOME_COLOR_TABLE[56], coastalColor,
210                                (MathExtras.clamp((float) (((heightData[x][y] + 0.06) * 8.0) / (WorldMapGenerator.sandLower + 1.0)), 0f, 1f)));
211                        break;
212                    default:
213                        colorMap[x][y] = SColor.lerpFloatColors(BIOME_COLOR_TABLE[biomeMapper.extractPartB(bc)],
214                                BIOME_DARK_COLOR_TABLE[biomeMapper.extractPartA(bc)], biomeMapper.extractMixAmount(bc));
215                }
216            }
217        }
218        
219        return colorMap;
220    }
