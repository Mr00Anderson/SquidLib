022/**
023 * A panel to display some text using libgdx directly (i.e. without using
024 * {@link SquidPanel}) as in these examples (no scrolling first, then with a
025 * scroll bar):
026 *
027 * <p>
028 * <ul>
029 * <li><img src="http://i.imgur.com/EqEXqlu.png"/></li>
030 * <li><img src="http://i.imgur.com/LYbxQZE.png"/></li>
031 * </ul>
032 * </p>
033 *
034 * <p>
035 * It supports vertical scrolling, i.e. it'll put a vertical scrollbar if
036 * there's too much text to display. This class does a lot of stuff, you
037 * typically only have to provide the textures for the scrollbars and the scroll
038 * knobs (see example below).
039 * </p>
040 *
041 * <p>
042 * A typical usage of this class is as follows:
043 *
044 * <pre>
045 * final TextPanel<Color> tp = new TextPanel<>(new GDXMarkup(), font);
046 * tp.init(screenWidth, screenHeight, text); <- first 2 params: for fullscreen
047 * final ScrollPane sp = tp.getScrollPane();
048 * sp.setScrollPaneStyle(new ScrollPaneStyle(...)); <- set textures
049 * stage.addActor(sp);
050 * stage.setScrollFocus(sp);
051 * stage.draw();
052 * </pre>
053 * </p>
054 *
055 * <p>
056 * This class shares what {@link ScrollPane} does (knobs, handling of the wheel).
057 * </p>
058 * <p>
059 * Drawing the result of {@link #getScrollPane()} will set the shader of {@code batch} if using a distance field or MSDF
060 * font and the shader is currently not configured for such a font; it does not reset the shader to the default so that
061 * multiple Actors can all use the same shader and so specific extra glyphs or other items can be rendered after calling
062 * draw(). If you need to draw both a distance field font and full-color art, you should set the shader on the Batch to
063 * null when you want to draw full-color art, and end the Batch between drawing this object and the other art.
064 * </p>
065 * @author smelC
066 *
067 * @see ScrollPane A libGDX widget for general scrolling through only the visible part of a large widget
068 * @see LinesPanel An alternative for displaying lines of text in a variable-width font
069 */
070public class TextPanel<T extends Color> {

125        /**

135        public TextPanel(/* @Nullable */IMarkup<T> markup, /* @Nullable */ TextCellFactory font) {

137                {

139                        tcf.initBySize();

143                scrollPane = new ScrollPane(textActor);

233        public void init(float width, float maxHeight, T color, String... text)
234        {
235                ArrayList<IColoredString.Impl<T>> coll = new ArrayList<>(text.length);
236                for(String t : text)
237                {
238                        coll.add(new IColoredString.Impl<>(t, color));
239                }
240                init(width, maxHeight, coll);
241        }

243        /**
244         * Draws the border. You have to call this method manually, because the
245         * border is outside the actor and hence should be drawn at the very end,
246         * otherwise it can get overwritten by UI elements.
247         *
248         * @param batch
249         */
250        public void drawBorder(Batch batch) {
251                if (borderColor != null && 0 < borderSize) {
252                        final boolean reset = batch.isDrawing();
253                        if (reset)
254                                batch.end();
255
256                        final ShapeRenderer sr = getRenderer();
257                        final Matrix4 m = batch.getTransformMatrix();
258                        sr.setTransformMatrix(m);
259                        sr.begin(ShapeType.Filled);
260                        sr.setColor(borderColor);
261//                      prepareText();
262                        final float down = font.getData().down;
263                        UIUtil.drawMarginsAround(sr, scrollPane.getX(), scrollPane.getY() + down * -0.5f + 4f, scrollPane.getWidth(),
264                                        scrollPane.getHeight() + down * -1.5f - 4f, borderSize * 2f, borderColor, borderStyle, 1f, 1f);
265                        sr.end();
266
267                        if (reset)
268                                batch.begin();
269                }
270        }

272        /**
273         * @return The text to draw, without color information present in {@link #text}.
274         */
275        public /* @Nullable */ ArrayList<String> getTypesetText() {
276                if (text == null)
277                        return null;
278                final ArrayList<String> result = new ArrayList<>();
279                for (IColoredString<T> line : text) {
280                        result.add(line.present());
281                }
282                return result;
283        }

285        /**
286         * Updates the text this will show based on the current contents of the ArrayList of IColoredString values that may
287         * be shared due to {@link #initShared(float, float, ArrayList)}, then resizes the {@link #getTextActor()} to fit
288         * the current text and lays out the {@link #getScrollPane()} to match. Called in the text actor's draw() method. 
289         */
290        protected void prepareText() {
291                if (text == null)
292                        return;
293                final BitmapFontCache cache = font.getCache();
294                //cache.clear();
295                final float w = scrollPane.getWidth();
296                float lineHeight = -font.getData().down, capHeight = font.getCapHeight();
297                int lines = 1, ci = 0;
298                StringBuilder sb = new StringBuilder(256);
299                for (int m = 0, textSize = text.size(); m < textSize; m++) {
300                        IColoredString<T> line = text.get(m);
301                        ArrayList<IColoredString.Bucket<T>> frags = line.getFragments();
302                        for (int i = 0; i < frags.size(); i++) {
303                                sb.append(frags.get(i).getText());
304                        }
305                        sb.append('\n');
306                }
307                GlyphLayout layout = cache.setText(sb, 0, 0, w, Align.left, true);                      
308                lines += layout.height / capHeight;
309                
310                ////TODO: BitmapFontCache.setColors(float, int, int) is broken in libGDX 1.9.10; find some workaround
311//              for (int m = 0, textSize = text.size(); m < textSize; m++) {
312//                      IColoredString<T> line = text.get(m);
313//                      ArrayList<IColoredString.Bucket<T>> frags = line.getFragments();
314//                      for (int i = 0; i < frags.size(); i++) {
315//                              final IColoredString.Bucket<T> b = frags.get(i);
316////                            Color c = b.getColor();
317////                            if(c != null) 
318////                                    cache.setColors(c, ci, (ci += b.length()));
319////                            else
320//                                      ci += b.length();
321//                      }
322//
323////                    if(m + 1 < textSize)
324////                    {
325//                              //cache.addText("\n", pos, (-lines) * totalTextHeight, w, Align.left, true);
326//                              //lines++;
327////                    }
328//              }
329                lineHeight *= lines;
330                if(lineHeight < 0)
331                        lineHeight = 0;
332                textActor.setHeight(/* Entire height */ lineHeight);
333                scrollPane.layout();
334
335        }

337        /**
338         * Scrolls the scroll pane this holds down by some number of rows (which may be fractional, and may be negative to
339         * scroll up). This is not a smooth scroll, and will not be animated.
340         * @param downDistance The distance in rows to scroll down, which can be negative to scroll up instead
341         */
342        public void scroll(final float downDistance)
343        {
344                prepareText();
345                scrollPane.setScrollY(scrollPane.getScrollY() + downDistance * tcf.actualCellHeight);
346        }
347
348        /**
349         * If the parameter is true, scrolls to the top of this scroll pane; otherwise scrolls to the bottom. This is not a
350         * smooth scroll, and will not be animated.
351         * @param goToTop If true, will scroll to the top edge; if false, will scroll to the bottom edge.
352         */
353        public void scrollToEdge(final boolean goToTop)
354        {
355                prepareText();
356                scrollPane.setScrollPercentY(goToTop ? 0f : 1f);
357        }

386        /**
387         * Callback done to do stuff according to whether y-scrolling is required
388         *
389         * @param required
390         *            Whether y scrolling is required.
391         */
392        protected void yScrollingCallback(boolean required) {
393                if (required) {
394                        /* Disable borders, they don't mix well with scrollbars */
395                        borderSize = 0;
396                        scrollPane.setFadeScrollBars(false);
397                        scrollPane.setForceScroll(false, true);
398                }
399        }
