078    /**
079     * Creates a new map with the specified initial capacity and load factor. This map will hold initialCapacity items before
080     * growing the backing table.
081     *
082     * @param initialCapacity If not a power of two, it is increased to the next nearest power of two.
083     */
084    public SparseTextMap(int initialCapacity, float loadFactor) {
085        if (initialCapacity < 0)
086            throw new IllegalArgumentException("initialCapacity must be >= 0: " + initialCapacity);
087        if (loadFactor <= 0f || loadFactor >= 1f)
088            throw new IllegalArgumentException("loadFactor must be > 0 and < 1: " + loadFactor);
089        initialCapacity = HashCommon.nextPowerOfTwo((int)Math.ceil(initialCapacity / loadFactor));
090        if (initialCapacity > 1 << 30)
091            throw new IllegalArgumentException("initialCapacity is too large: " + initialCapacity);
092
093        this.loadFactor = loadFactor;
094
095        threshold = (int)(initialCapacity * loadFactor);
096        mask = initialCapacity - 1;
097        shift = Long.numberOfLeadingZeros(mask);
098
099        keyTable = new int[initialCapacity];
100        charValueTable = new char[initialCapacity];
101        floatValueTable = new float[initialCapacity];
102        keys = new IntVLA(initialCapacity);
103    }

108    public SparseTextMap(SparseTextMap map) {
109        loadFactor = map.loadFactor;
110        threshold = map.threshold;
111        mask = map.mask;
112        shift = map.shift;
113        keyTable = new int[map.keyTable.length];
114        System.arraycopy(map.keyTable, 0, keyTable, 0, map.keyTable.length);
115        charValueTable = new char[map.charValueTable.length];
116        System.arraycopy(map.charValueTable, 0, charValueTable, 0, map.charValueTable.length);
117        floatValueTable = new float[map.floatValueTable.length];
118        System.arraycopy(map.floatValueTable, 0, floatValueTable, 0, map.floatValueTable.length);
119        size = map.size;
120        zeroChar = map.zeroChar;
121        zeroFloat = map.zeroFloat;
122        hasZeroValue = map.hasZeroValue;
123        keys = new IntVLA(map.keys);
124    }

165    public void draw(Batch batch, TextCellFactory textFactory, float screenOffsetX, float screenOffsetY) {
166        //textFactory.configureShader(batch);
167        final float widthInc = textFactory.actualCellWidth, heightInc = -textFactory.actualCellHeight;
168        int n;
169        for (Entry entry : entries()) {
170            n = entry.key;
171//            n ^= n >>> 16;
172//            n *= 0xA123B;
173
174//            n ^= n << 26;
175//            n ^= n >>> 15;
176//            n ^= n << 17;
177            textFactory.draw(batch, entry.charValue, entry.floatValue,
178                    (n & 0xFFFF) * widthInc + screenOffsetX, (n >>> 16) * heightInc + screenOffsetY);
179        }
180    }

339    private int fibonacci(final int item) {
340        // shift is always greater than 32, less than 64
341        return (int)(item * 0x9E3779B97F4A7C15L >>> shift);
342    }

344    private int locateKey (final int key) {
345        return locateKey(key, fibonacci(key));
346    }

348    /**
349     * Given a key and its initial placement to try in an array, this finds the actual location of the key in the array
350     * if it is present, or -1 if the key is not present. This can be overridden if a subclass needs to compare for
351     * equality differently than just by using == with int keys, but only within the same package.
352     *
353     * @param key       a K key that will be checked for equality if a similar-seeming key is found
354     * @param placement as calculated by {@link #fibonacci(int)}, almost always with {@code place(key)}
355     * @return the location in the key array of key, if found, or -1 if it was not found.
356     */
357    private int locateKey (final int key, final int placement) {
358        for (int i = placement; ; i = i + 1 & mask) {
359            // empty space is available
360            if (keyTable[i] == 0) {
361                return -1;
362            }
363            if (key == (keyTable[i])) {
364                return i;
365            }
366        }
367    }

404    public void put (int key, char charValue, float floatValue) {
405        if (key == 0) {
406            zeroChar = charValue;
407            zeroFloat = floatValue;
408            if (!hasZeroValue) {
409                hasZeroValue = true;
410                keys.add(0);
411                size++;
412            }
413            return;
414        }
415
416        int b = fibonacci(key);
417        int loc = locateKey(key, b);
418        // an identical key already exists
419        if (loc != -1) {
420            charValueTable[loc] = charValue;
421            floatValueTable[loc] = floatValue;
422            return;
423        }
424        final int[] keyTable = this.keyTable;
425        final char[] charValueTable = this.charValueTable;
426        final float[] floatValueTable = this.floatValueTable;
427        keys.add(key);
428
429        for (int i = b; ; i = (i + 1) & mask) {
430            // space is available so we insert and break
431            if (keyTable[i] == 0) {
432                keyTable[i] = key;
433                charValueTable[i] = charValue;
434                floatValueTable[i] = floatValue;
435                
436                if (++size >= threshold) {
437                    resize(keyTable.length << 1);
438                }
439                return;
440            }
441        }
442    }

522
523        int b = fibonacci(key);
524        final int[] keyTable = this.keyTable;
525        final char[] charValueTable = this.charValueTable;
526        final float[] floatValueTable = this.floatValueTable;
527
528        for (int i = b; ; i = (i + 1) & mask) {
529            // space is available so we insert and break
530            if (keyTable[i] == 0) {
531                keyTable[i] = key;
532                charValueTable[i] = charValue;
533                floatValueTable[i] = floatValue;
534 
535                if (++size >= threshold) {
536                    resize(keyTable.length << 1);
537                }
538                return;
539            }
540        }
541    }

611    public char remove(int key, char defaultValue) {
612        if (key == 0) {
613            if (!hasZeroValue) return defaultValue;
614            hasZeroValue = false;
615            size--;
616            keys.removeValue(0);
617            return zeroChar;
618        }
619
620        int loc = locateKey(key);
621        if (loc == -1) {
622            return defaultValue;
623        }
624        keys.removeValue(key);
625        final int[] keyTable = this.keyTable;
626        final float[] floatValueTable = this.floatValueTable;
627        final char[] charValueTable = this.charValueTable;
628        char oldChar = charValueTable[loc];
629        while ((key = keyTable[loc + 1 & mask]) != 0 && (loc + 1 & mask) != fibonacci(key)) {
630            keyTable[loc] = key;
631            floatValueTable[loc] = floatValueTable[loc + 1 & mask];
632            charValueTable[loc] = charValueTable[++loc & mask];
633        }
634        keyTable[loc] = 0;
635        --size;
636        return oldChar;
637    }

733     * Increases the size of the backing array to accommodate the specified number of additional items. Useful before adding many
734     * items to avoid multiple backing array resizes.
735     */
736    public void ensureCapacity(int additionalCapacity) {
737        if (additionalCapacity < 0)
738            throw new IllegalArgumentException("additionalCapacity must be >= 0: " + additionalCapacity);
739        int sizeNeeded = size + additionalCapacity;
740        keys.ensureCapacity(additionalCapacity);
741        if (sizeNeeded >= threshold)
742            resize(HashCommon.nextPowerOfTwo((int)Math.ceil(sizeNeeded / loadFactor)));
743    }

745    private void resize(int newSize) {
746        int oldCapacity = keyTable.length;
747        threshold = (int)(newSize * loadFactor);
748        mask = newSize - 1;
749        shift = Long.numberOfLeadingZeros(mask);
750
751        final int[] oldKeyTable = keyTable;
752        final float[] oldFloats = floatValueTable;
753        final char[] oldChars = charValueTable;
754
755        keyTable = new int[newSize];
756        floatValueTable = new float[newSize];
757        charValueTable = new char[newSize];
758
759        int oldSize = size;
760        size = 0;
761        hasZeroValue = false;
762        if (oldSize > 0) {
763            for (int i = 0; i < oldCapacity; i++) {
764                int key = oldKeyTable[i];
765                if (key != 0)
766                    putResize(key, oldChars[i], oldFloats[i]);
767            }
768        }
769    }

771    public int hashCode() {
772        int h = 0;
773        if (hasZeroValue) {
774            h = NumberTools.floatToIntBits(zeroFloat) ^ zeroChar;
775        }
776        int[] keyTable = this.keyTable;
777        char[] charTable = this.charValueTable;
778        float[] floatTable = this.floatValueTable;
779        for (int i = 0, n = keyTable.length; i < n; i++) {
780            int key = keyTable[i];
781            if (key != EMPTY) {
782                h += key ^ NumberTools.floatToIntBits(floatTable[i]);
783                h ^= charTable[i];
784            }
785        }
786        return h;
787    }

789    public boolean equals(Object obj) {
790        if (obj == this) return true;
791        if (!(obj instanceof SparseTextMap)) return false;
792        SparseTextMap other = (SparseTextMap) obj;
793        if (other.size != size) return false;
794        if (other.hasZeroValue != hasZeroValue) return false;
795        if (hasZeroValue && (other.zeroChar != zeroChar || other.zeroFloat != zeroFloat)) {
796            return false;
797        }
798        int[] keyTable = this.keyTable;
799        char[] charTable = this.charValueTable;
800        float[] floatTable = this.floatValueTable;
801        for (int i = 0, n = keyTable.length; i < n; i++) {
802            int key = keyTable[i];
803            if (key != EMPTY) {
804                char otherValue = other.getChar(key, '\0');
805                if (otherValue == 0 && !other.containsKey(key) 
806                        || otherValue != charTable[i]
807                        || other.getFloat(key, Float.NaN) != floatTable[i]) return false;
808            }
809        }
810        return true;
811    }

977            if (!hasNext) throw new NoSuchElementException();
978            if (!valid) throw new GdxRuntimeException("#iterator() cannot be used nested.");
979            currentIndex = nextIndex;
980            entry.key = keys.get(nextIndex);
981            if(entry.key == 0)
982            {

984                entry.floatValue = map.zeroFloat;

988                entry.charValue = map.charValueTable[loc];
989                entry.floatValue = map.floatValueTable[loc];
990            }
991            nextIndex++;
992            hasNext = nextIndex < map.size;
993            return entry;
994        }

1003        }

1006            if (currentIndex < 0)
1007                throw new IllegalStateException("next must be called before remove.");
1008            map.remove(entry.key);
1009            nextIndex--;
1010            currentIndex = -1;
1011        }
