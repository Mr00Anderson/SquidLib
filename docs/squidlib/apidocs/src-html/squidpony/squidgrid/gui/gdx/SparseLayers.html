030/**
031 * A general-purpose char display grid that supports one layer of backgrounds and arbitrarily many layers of foreground,
032 * only rendering foreground chars when something is present at a char's location. This also stores an ArrayList of
033 * {@link TextCellFactory.Glyph} items that can be added to using {@link #glyph(char, Color, int, int)} or one of
034 * several other methods; these Glyphs can have effects performed on them that are analogous to the effects on
035 * {@link SquidPanel}'s {@link AnimatedEntity} effects, such as
036 * {@link #slide(TextCellFactory.Glyph, int, int, int, int, float, Runnable)}. Unlike SquidPanel and SquidLayers, this
037 * class does not typically provide many overloads for each method, and usually limits the overloads to one that takes
038 * a packed float color and one that takes a libGDX {@link Color}, and all the other parameters are expected to be
039 * specified explicitly rather than relying on default behavior. SparseLayers tends to perform better than SquidLayers,
040 * especially when a lot of the map is unassigned/blank. It will cause less GC pressure when packed floats are used, and
041 * because {@link SColor} constants pre-calculate their packed float values, using the overloads that take a Color will
042 * perform a little better when given an SColor than a plain libGDX Color. This class makes heavy use of
043 * {@link SColor#lerpFloatColors(float, float, float)} because it should cause no GC pressure (it doesn't create
044 * temporary Color objects), and if you don't want to mutate Colors in-place (which is a bad idea for constants), then
045 * using lerpFloatColors with the packed float color API here should be very effective.
046 * <br>
047 * The names are a little different for some member fields here; there's a {@link TextCellFactory} called {@link #font},
048 * a {@code float[][]} called {@link #backgrounds} that stores packed float colors, an ArrayList of
049 * {@link SparseTextMap} that implements the sparse drawing behavior of foreground chars called {@link #layers}, and so
050 * on.
051 * <br>
052 * Created by Tommy Ettinger on 7/28/2017.
053 */

130    public SparseLayers(int gridWidth, int gridHeight, float cellWidth, float cellHeight, TextCellFactory font, float xOffset, float yOffset) {
131        this.gridWidth = MathUtils.clamp(gridWidth, 1, 65535);
132        this.gridHeight = MathUtils.clamp(gridHeight, 1, 65535);
133        backgrounds = new float[this.gridWidth][this.gridHeight];
134        layers = new ArrayList<>(4);
135        if(font.initialized())
136            this.font = font;
137        else
138            this.font = font.width(cellWidth).height(cellHeight).initBySize();
139        layers.add(new SparseTextMap(gridWidth * gridHeight >> 2));
140        mapping = new IntIntMap(4);
141        mapping.put(0, 0);
142        glyphs = new ArrayList<>(16);
143        scc = DefaultResources.getSCC();
144        setBounds(xOffset, yOffset,
145                this.font.actualCellWidth * this.gridWidth, this.font.actualCellHeight * this.gridHeight);
146    }

178     * Sets the SparseTextMap associated with the given layerNumber to the given contents. If layerNumber is too high
179     * to set an existing layer, this will add contents as a new layer on top of the others.
180     * @param layerNumber must be 0 or greater
181     * @param contents a SparseTextMap, possibly obtained with {@link #getLayer(int)}

184    {
185        if(layerNumber < 0)
186            return;
187        layerNumber = mapping.get(layerNumber, layerNumber);
188        if(layerNumber >= layers.size())
189        {
190            mapping.put(layerNumber, layers.size());
191            layers.add(contents);
192        }
193        else
194        {
195            layers.set(layerNumber, contents);
196        }

213    /**
214     * Adds a layer as a SparseTextMap to this SparseLayers and returns the one just added. The layer will generally be
215     * associated with a layer number equal to the number of layers currently present, and will be rendered over the
216     * existing layers. It might not be associated with the same number if layers were added out-of-order or with skips
217     * between associations. If you want to add a layer at some alternate layer number (which can be confusing but may
218     * be needed for some reason), there is another overload that allows you to specify the association number.
219     * If you may have added some layers out of order or skipped some numbers, you can use
220     * {@link #findLayer(SparseTextMap)} on the returned SparseTextMap to get its association number.
221     * @return a SparseTextMap that was just added; null if something went wrong
222     */
223    public SparseTextMap addLayer()
224    {
225        int association = layers.size();
226        while (mapping.containsKey(association)) {
227            ++association;
228        }
229        return addLayer(association);
230    }
231    /**
232     * Adds a layer as a SparseTextMap to this SparseLayers and returns the one just added, or returns an existing layer
233     * if one is already associated with the given number. The layer can be greater than the number of layers currently
234     * present, which will add a layer to be rendered over the existing layers, but the number that refers to that layer
235     * will not change. It is recommended that to add a layer, you only add at the value equal to
236     * {@link #getLayerCount()}, which will maintain the order and layer numbers in a sane way, and this is the behavior
237     * of the addLayer overload that does not take a parameter.
238     * @param association the number to associate the layer with; should usually be between 0 and {@link #getLayerCount()} inclusive
239     * @return a SparseTextMap that either was just added or was already associated with the given layer number; null if something went wrong
240     */
241    public SparseTextMap addLayer(int association)
242    {
243        if(association < 0)
244            return null;
245        association = mapping.get(association, association);
246        if(association >= layers.size())
247        {
248            mapping.put(association, layers.size());
249            SparseTextMap stm = new SparseTextMap(gridWidth * gridHeight >> 4);
250            layers.add(stm);
251            return stm;
252        }
253        else
254        {
255            return layers.get(association);
256        }
257    }

410    /**
411     * Places the given char 2D array, if-non-null, in the default foreground color starting at x=0, y=0, while also
412     * setting the background colors to match the given Color 2D array. If the colors argument is null, does nothing. If
413     * the chars argument is null, only affects the background colors. This will filter each Color in colors if the
414     * color center this uses has a filter.
415     * @param chars Can be {@code null}, indicating that only colors must be put.
416     * @param colors the background colors for the given chars
417     */
418    @Override
419    public void put(char[][] chars, Color[][] colors) {
420        if(chars == null)
421        {
422            if(colors != null)
423            {
424                for (int i = 0; i < colors.length; i++) {
425                    if (colors[i] == null)
426                        continue;
427                    for (int j = 0; j < colors[i].length; j++) {
428                        backgrounds[i][j] = (colors[i][j] == null) ? 0f : scc.filter(colors[i][j]).toFloatBits();
429                    }
430                }
431            }
432        }
433        else
434        {
435            SparseTextMap layer = layers.get(0);
436            if(colors == null)
437            {
438                for (int i = 0; i < chars.length; i++) {
439                    if(chars[i] == null)
440                        continue;
441                    for (int j = 0; j < chars[i].length; j++) {
442                        layer.place(i, j, chars[i][j], defaultPackedForeground);
443                    }
444                }
445            }
446            else
447            {
448                for (int i = 0; i < chars.length && i < colors.length; i++) {
449                    if(colors[i] == null || chars[i] == null)
450                        continue;
451                    for (int j = 0; j < chars[i].length && j < colors[i].length; j++) {
452                        if(colors[i][j] == null)
453                            layer.place(i, j, chars[i][j], defaultPackedForeground);
454                        else
455                            put(i, j, chars[i][j], defaultPackedForeground, scc.filter(colors[i][j]).toFloatBits());
456                    }
457                }
458            }
459        }
460    }

462     * Places the given char 2D array, if-non-null, in the default foreground color starting at x=0, y=0, while also
463     * setting the background colors to match the given 2D array of colors as packed floats. If the colors argument is
464     * null, does nothing. If the chars argument is null, only affects the background colors. This will not filter the
465     * passed colors at all.
466     * @param chars Can be {@code null}, indicating that only colors must be put.
467     * @param colors the background colors for the given chars
468     */
469    public void put(char[][] chars, float[][] colors) {
470        if(chars == null)
471        {
472            if(colors != null) {
473                for (int i = 0; i < colors.length; i++) {
474                    if (colors[i] == null)
475                        continue;
476                    System.arraycopy(colors[i], 0, backgrounds[i], 0, colors[i].length);
477                }
478            }
479        }
480        else
481        {
482            if(colors == null)
483            {
484                for (int i = 0; i < chars.length; i++) {
485                    if(chars[i] == null)
486                        continue;
487                    for (int j = 0; j < chars[i].length; j++) {
488                        put(i, j, chars[i][j], defaultPackedForeground, 0f);
489                    }
490                }
491            }
492            else
493            {
494                for (int i = 0; i < chars.length && i < colors.length; i++) {
495                    if(colors[i] == null || chars[i] == null)
496                        continue;
497                    for (int j = 0; j < chars[i].length && j < colors[i].length; j++) {
498                        put(i, j, chars[i][j], defaultPackedForeground, colors[i][j]);
499                    }
500                }

505     * Places the given char 2D array, if-non-null, with the given foreground colors in the first Color 2D array,
506     * starting at x=0, y=0, while also setting the background colors to match the second Color 2D array. If the
507     * bgColors argument is null, only affects foreground chars and colors. If the chars argument or the fgColors
508     * argument is null, only affects the background colors. Any positions where a Color in fgColors is null will not
509     * have a char placed (this can be used to restrict what is placed). This will filter each Color in the background
510     * and foreground if the color center this uses has a filter.
511     * @param chars Can be {@code null}, indicating that only colors must be put.
512     * @param fgColors the foreground Colors for the given chars
513     * @param bgColors the background Colors for the given chars
514     */
515    public void put(char[][] chars, Color[][] fgColors, Color[][] bgColors) {
516        if(chars == null || fgColors == null)
517        {
518            if(bgColors != null)
519            {
520                for (int i = 0; i < bgColors.length; i++) {
521                    if (bgColors[i] == null)
522                        continue;
523                    for (int j = 0; j < bgColors[i].length; j++) {
524                        backgrounds[i][j] = (bgColors[i][j] == null) ? 0f : scc.filter(bgColors[i][j]).toFloatBits();
525                    }
526                }
527            }
528        }
529        else
530        {
531            if(bgColors == null)
532            {
533                for (int i = 0; i < chars.length && i < fgColors.length; i++) {
534                    if(chars[i] == null || fgColors[i] == null)
535                        continue;
536                    for (int j = 0; j < chars[i].length && j < fgColors[i].length; j++) {
537                        put(i, j, chars[i][j], fgColors[i][j], null);
538                    }
539                }
540            }
541            else
542            {
543                for (int i = 0; i < chars.length && i < bgColors.length && i < fgColors.length; i++) {
544                    if(bgColors[i] == null || chars[i] == null || fgColors[i] == null)
545                        continue;
546                    for (int j = 0; j < chars[i].length && j < bgColors[i].length && j < fgColors[i].length; j++) {                         
547                        put(i, j, chars[i][j], fgColors[i][j], bgColors[i][j]);
548                    }
549                }
550            }
551        }

554     * Places the given char 2D array, if-non-null, with the given foreground colors in the first float 2D array,
555     * starting at x=0, y=0, while also setting the background colors to match the second float 2D array. If the
556     * bgColors argument is null, only affects foreground chars and colors. If the chars argument or the fgColors
557     * argument is null, only affects the background colors. Any positions where a float in fgColors is 0 will not
558     * have a char placed (this can be used to restrict what is placed). This will not filter any colors.
559     * @param chars Can be {@code null}, indicating that only colors must be put.
560     * @param fgColors the foreground colors for the given chars, as packed floats
561     * @param bgColors the background colors for the given chars, as packed floats
562     */
563    public void put(char[][] chars, float[][] fgColors, float[][] bgColors) {
564        if(chars == null || fgColors == null)
565        {
566            if(bgColors != null)
567            {
568                for (int i = 0; i < bgColors.length; i++) {
569                    if (bgColors[i] == null)
570                        continue;
571                    System.arraycopy(bgColors[i], 0, backgrounds[i], 0, bgColors[i].length);
572                }
573            }
574        }
575        else
576        {
577            if(bgColors == null)
578            {
579                for (int i = 0; i < chars.length && i < fgColors.length; i++) {
580                    if(chars[i] == null || fgColors[i] == null)
581                        continue;
582                    for (int j = 0; j < chars[i].length && j < fgColors[i].length; j++) {
583                        put(i, j, chars[i][j], fgColors[i][j], 0f);
584                    }
585                }
586            }
587            else
588            {
589                for (int i = 0; i < chars.length && i < bgColors.length && i < fgColors.length; i++) {
590                    if(bgColors[i] == null || chars[i] == null || fgColors[i] == null)
591                        continue;
592                    for (int j = 0; j < chars[i].length && j < bgColors[i].length && j < fgColors[i].length; j++) {
593                        put(i, j, chars[i][j], fgColors[i][j], bgColors[i][j]);
594                    }
595                }
596            }
597        }
598    }
