107    public WildMapView(WildMap wildMap)
108    {
109        if(wildMap == null)
110        {
111            this.wildMap = new WildMap();
112        }
113        else
114        {
115            this.wildMap = wildMap;
116        }
117        this.viewer = defaultViewer(this.wildMap.rng);
118        width = this.wildMap.width;
119        height = this.wildMap.height;
120        colorMap = new float[width][height];
121        initialize();
122    }
123    public WildMapView(WildMap wildMap, Map<String, ? extends ICellVisible> viewer)
124    {
125        if(wildMap == null) { // default to forest map, biome 21; ignore given viewer, it may be null
126            this.wildMap = new WildMap();
127            this.viewer = defaultViewer(this.wildMap.rng);
128        }
129        else
130        {
131            this.wildMap = wildMap;
132            this.viewer = viewer == null ? defaultViewer(this.wildMap.rng) : viewer;
133        }
134        width = this.wildMap.width;
135        height = this.wildMap.height;
136        colorMap = new float[width][height];
137        initialize();
138    }

162    public final float[] BIOME_COLOR_TABLE = new float[61], BIOME_DARK_COLOR_TABLE = new float[61];

164    public void initialize()
165    {
166        initialize(0f, 0f, 0f, 1f);

169    public void initialize(float hue, float saturation, float brightness, float contrast)
170    {
171        float b, diff;
172        for (int i = 0; i < 60; i++) {
173            b = BIOME_TABLE[i];
174            diff = ((b % 1.0f) - 0.48f) * 0.27f * contrast;
175            BIOME_COLOR_TABLE[i] = b = SColor.toEditedFloat((diff >= 0)
176                    ? SColor.lightenFloat(biomeColors[(int)b], diff)
177                    : SColor.darkenFloat(biomeColors[(int)b], -diff), hue, saturation, brightness, 0f);
178            BIOME_DARK_COLOR_TABLE[i] = SColor.darkenFloat(b, 0.08f);
179        }
180        BIOME_COLOR_TABLE[60] = BIOME_DARK_COLOR_TABLE[60] = emptyColor;
181    }

183    /**
184     * Initializes the colors to use for each biome (these are almost always mixed with other biome colors in practice).
185     * Each parameter may be null to use the default for an Earth-like world; otherwise it should be a libGDX
186     * {@link Color} or some subclass, like {@link SColor}. All non-null parameters should probably be fully opaque,
187     * except {@code emptyColor}, which is only used for world maps that show empty space (like a globe, as produced by
188     * {@link WorldMapGenerator.RotatingSpaceMap}).
189     * @param desertColor hot, dry, barren land; may be sandy, but many real-world deserts don't have much sand
190     * @param savannaColor hot, mostly-dry land with some parched vegetation; also called scrub or chaparral
191     * @param tropicalRainforestColor hot, extremely wet forests with dense rich vegetation
192     * @param grasslandColor prairies that are dry and usually wind-swept, but not especially hot or cold
193     * @param woodlandColor part-way between a prairie and a forest; not especially hot or cold
194     * @param seasonalForestColor forest that becomes barren in winter (deciduous trees); not especially hot or cold
195     * @param temperateRainforestColor forest that tends to be slightly warm but very wet
196     * @param borealForestColor forest that tends to be cold and very wet
197     * @param tundraColor very cold plains that still have some low-lying vegetation; also called taiga 
198     * @param iceColor cold barren land covered in permafrost; also used for rivers and lakes that are frozen
199     * @param beachColor sandy or otherwise light-colored shorelines; here, these are more common in warmer places
200     * @param rockyColor rocky or otherwise rugged shorelines; here, these are more common in colder places
201     * @param shallowColor the color of very shallow water; will be mixed with {@code deepColor} to get most ocean colors
202     * @param deepColor the color of very deep water; will be mixed with {@code shallowColor} to get most ocean colors
203     * @param emptyColor the color used for empty space off the edge of the world map; may be transparent
204     */
205    public void initialize(
206            Color desertColor,
207            Color savannaColor,
208            Color tropicalRainforestColor,
209            Color grasslandColor,
210            Color woodlandColor,
211            Color seasonalForestColor,
212            Color temperateRainforestColor,
213            Color borealForestColor,
214            Color tundraColor,
215            Color iceColor,
216            Color beachColor,
217            Color rockyColor,
218            Color shallowColor,
219            Color deepColor,
220            Color emptyColor
221    )
222    {
223        biomeColors[ 0] = desertColor == null ? WorldMapView.desertColor : desertColor.toFloatBits();
224        biomeColors[ 1] = savannaColor == null ? WorldMapView.savannaColor : savannaColor.toFloatBits();
225        biomeColors[ 2] = tropicalRainforestColor == null ? WorldMapView.tropicalRainforestColor : tropicalRainforestColor.toFloatBits();
226        biomeColors[ 3] = grasslandColor == null ? WorldMapView.grasslandColor : grasslandColor.toFloatBits();
227        biomeColors[ 4] = woodlandColor == null ? WorldMapView.woodlandColor : woodlandColor.toFloatBits();
228        biomeColors[ 5] = seasonalForestColor == null ? WorldMapView.seasonalForestColor : seasonalForestColor.toFloatBits();
229        biomeColors[ 6] = temperateRainforestColor == null ? WorldMapView.temperateRainforestColor : temperateRainforestColor.toFloatBits();
230        biomeColors[ 7] = borealForestColor == null ? WorldMapView.borealForestColor : borealForestColor.toFloatBits();
231        biomeColors[ 8] = tundraColor == null ? WorldMapView.tundraColor : tundraColor.toFloatBits();
232        biomeColors[ 9] = iceColor == null ? WorldMapView.iceColor : iceColor.toFloatBits();
233        biomeColors[10] = beachColor == null ? WorldMapView.beachColor : beachColor.toFloatBits();
234        biomeColors[11] = rockyColor == null ? WorldMapView.rockyColor : rockyColor.toFloatBits();
235        biomeColors[12] = shallowColor == null ? WorldMapView.shallowColor : shallowColor.toFloatBits();
236        biomeColors[13] = deepColor == null ? WorldMapView.deepColor : deepColor.toFloatBits();
237        biomeColors[14] = emptyColor == null ? WorldMapView.emptyColor : emptyColor.toFloatBits();
238        float b, diff;
239        for (int i = 0; i < 60; i++) {
240            b = BIOME_TABLE[i];
241            diff = ((b % 1.0f) - 0.48f) * 0.27f;
242            BIOME_COLOR_TABLE[i] = b = (diff >= 0)
243                    ? SColor.lightenFloat(biomeColors[(int)b], diff)
244                    : SColor.darkenFloat(biomeColors[(int)b], -diff);
245            BIOME_DARK_COLOR_TABLE[i] = SColor.darkenFloat(b, 0.08f);
246        }
247        BIOME_COLOR_TABLE[60] = BIOME_DARK_COLOR_TABLE[60] = biomeColors[14];
248        biomeColors[ 0] = WorldMapView.desertColor;
249        biomeColors[ 1] = WorldMapView.savannaColor;
250        biomeColors[ 2] = WorldMapView.tropicalRainforestColor;
251        biomeColors[ 3] = WorldMapView.grasslandColor;
252        biomeColors[ 4] = WorldMapView.woodlandColor;
253        biomeColors[ 5] = WorldMapView.seasonalForestColor;
254        biomeColors[ 6] = WorldMapView.temperateRainforestColor;
255        biomeColors[ 7] = WorldMapView.borealForestColor;
256        biomeColors[ 8] = WorldMapView.tundraColor;
257        biomeColors[ 9] = WorldMapView.iceColor;
258        biomeColors[10] = WorldMapView.beachColor;
259        biomeColors[11] = WorldMapView.rockyColor;
260        biomeColors[12] = WorldMapView.shallowColor;
261        biomeColors[13] = WorldMapView.deepColor;
262        biomeColors[14] = WorldMapView.emptyColor;
263    }

266    public void generate()
267    {
268        wildMap.generate();
269        float baseColor = BIOME_COLOR_TABLE[wildMap.biome & 1023];
270        int h = 1234567890, change, seed = wildMap.rng.nextInt();
271        ICellVisible icv;
272        for (int x = 0; x < width; x++) {
273            for (int y = 0; y < height; y++) {
274                change = h += Noise.IntPointHash.hashAll(x, y, 
275                        seed);
276                if((icv = viewer.get(wildMap.floorTypes.get(wildMap.floors[x][y]))) != null) 
277                    colorMap[x][y] = SColor.toEditedFloat(icv.getPackedColor(),
278                        0x1p-12f * ((h & 0xFF) - 0x9F + (change >>> 8 & 0x3F)),
279                        0x1p-11f * ((h >>> 8 & 0xFF) - 0xB0 + (change >>> 16 & 0x3F)) - 0.0625f,
280                        0x1.3p-12f * ((h >>> 16 & 0xFF) - 0x90 + (change >>> 24 & 0x3F)),
281                        0f);
282                else 
283                    colorMap[x][y] = SColor.toEditedFloat(baseColor,
284                        0x1p-12f * ((h & 0xFF) - 0x9F + (change >>> 8 & 0x3F)),
285                        0x1p-11f * ((h >>> 8 & 0xFF) - 0xB0 + (change >>> 16 & 0x3F)) - 0.0625f,
286                        0x1.3p-12f * ((h >>> 16 & 0xFF) - 0x90 + (change >>> 24 & 0x3F)),
287                        0f);
288
289            }
290        }
291    }

293    public void show(SparseLayers layers)
294    {
295        ArrayTools.insert(colorMap, layers.backgrounds, 0, 0);
296        int c;
297        ICellVisible icv;
298        for (int x = 0; x < width && x < layers.gridWidth; x++) {
299            for (int y = 0; y < height && y < layers.gridHeight; y++) {
300                if((c = wildMap.content[x][y]) >= 0 && (icv = viewer.get(wildMap.contentTypes.get(c))) != null)
301                    layers.put(x, y, icv);
302                else if((icv = viewer.get(wildMap.floorTypes.get(wildMap.floors[x][y]))) != null)
303                    layers.put(x, y, icv.getSymbol(), SColor.contrastLuma(colorMap[x][y]));
304            }
305        }
306    }
