142    /**
143     * Alters the lightnesses that affect the background colors, accepting a parameter for
144     * animation frame if rippling water and waving grass using {@link PerlinNoise} are desired. It may make sense to
145     * pass some fraction of the current time, as given by {@link System#currentTimeMillis()}, instead of a frame.
146     * Also allows additional chars to be treated like deep and shallow water regarding the ripple animation.
147     * Does not allocate a new 2D int array like {@link MapUtility#generateLightnessModifiers(char[][])} or its
148     * equivalent in DungeonUtility, and directly mutates the lightness data. You can use the {@link #lightnesses} field
149     * to access the 2D array directly, where you can make further modifications to the lighting.
150     * @param frame         a counter that typically should increase by between 10.0 and 20.0 each second; higher numbers make
151     *                      water and grass move more, and 0.013 multiplied by the current time in milliseconds works well
152     *                      as long as only the smaller digits of the time are used; this can be accomplished with
153     *                      {@code (System.currentTimeMillis() & 0xFFFFFF) * 0.013} .
154     * @param deepLiquid    a char that will be treated like deep water when animating ripples
155     * @param shallowLiquid a char that will be treated like shallow water when animating ripples
156     * @return a 2D array of lightness values from -255 to 255 but usually close to 0; can be passed to SquidLayers
157     */
158    public SquidLayers autoLight (double frame, char deepLiquid, char shallowLiquid) {
159        final char[][] map = foregroundPanel.contents;
160        final int w = getTotalWidth(), h = getTotalHeight();
161        for (int i = 0; i < w; i++) {
162            for (int j = 0; j < h; j++) {
163                switch (map[i][j]) {
164                    case '\1':
165                    case '├':
166                    case '┤':
167                    case '┴':
168                    case '┬':
169                    case '┌':
170                    case '┐':
171                    case '└':
172                    case '┘':
173                    case '│':
174                    case '─':
175                    case '┼':
176                    case '#':
177                        lightnesses[i][j] = 30;
178                        break;
179                    case '.':
180                    case ' ':
181                    case '\u0006':
182                        lightnesses[i][j] = 0;
183                        break;
184                    case ':':
185                        lightnesses[i][j] = -15;
186                        break;
187                    case '+':
188                    case '/':
189                        lightnesses[i][j] = -10;
190                        break;
191                    case ',':
192                        lightnesses[i][j] = (int) (85 * (PerlinNoise.noise(i * 1.45, j * 1.45, frame * 0.45) * 0.55 - 0.7));
193                        break;
194                    case '~':
195                        lightnesses[i][j] = (int) (100 * (PerlinNoise.noise(i * 1.45, j * 1.45, frame * 0.45) * 0.4 - 0.65));
196                        break;
197                    case '"':
198                        lightnesses[i][j] = (int) (95 * (PerlinNoise.noise(i * 1.45, j * 1.45, frame * 0.5) * 0.3 - 1.5));
199                        break;
200                    case '^':
201                        lightnesses[i][j] = 40;
202                        break;
203                    default:
204                        if (map[i][j] == deepLiquid)
205                            lightnesses[i][j] = (int) (180 * (PerlinNoise.noise(i * 4.2, j * 4.2, frame * 0.55) * 0.45 - 0.7));
206                        else if (map[i][j] == shallowLiquid)
207                            lightnesses[i][j] = (int) (110 * (PerlinNoise.noise(i * 3.1, j * 3.1, frame * 0.3) * 0.4 - 0.65));
208                        else
209                            lightnesses[i][j] = 0;
210                }
211            }
212        }
213        return this;
214    }

409    /**
410     * Create a new SquidLayers widget with the given TextCellFactory, the given number of cells for gridWidth
411     * and gridHeight, the size in pixels for each cell given by cellWidth and cellHeight, the given
412     * SquidColorCenters for background and foreground, and the given 2D char array for an area map that may be
413     * sized differently than gridWidth by gridHeight (it is usually larger than gridWidth by gridHeight, which
414     * allows camera scrolling across the map). This requires some special work with the Camera and Viewports to
415     * get working correctly; in the squidlib module's examples,
416     * <a href="https://github.com/SquidPony/SquidLib/blob/master/squidlib/src/test/java/squidpony/gdx/examples/EverythingDemo.java">EverythingDemo</a>
417     * may be a good place to see how this can be done. You can pass null for actualMap, which will simply create
418     * a char array to use internally that is exactly gridWidth by gridHeight, in cells. Consider using predefined
419     * TextCellFactory objects from {@link DefaultResources}, which will be configured by this constructor for
420     * sizing when passed here.
421     *
422     * @param gridWidth     in grid cells
423     * @param gridHeight    in grid cells
424     * @param cellWidth     in pixels
425     * @param cellHeight    in pixels
426     * @param tcf           A TextCellFactory that will be (re-)initialized here with the given cellHeight and cellWidth.
427     * @param bgColorCenter a SquidColorCenter (possibly with a filter) to use for the background
428     * @param fgColorCenter a SquidColorCenter (possibly with a filter) to use for the foreground
429     * @param actualMap     will often be a different size than gridWidth by gridHeight, which enables camera scrolling
430     */
431    public SquidLayers(int gridWidth, int gridHeight, int cellWidth, int cellHeight, TextCellFactory tcf,
432                       SquidColorCenter bgColorCenter, SquidColorCenter fgColorCenter, char[][] actualMap) {
433
434        width = gridWidth;
435        height = gridHeight;
436
437        this.cellWidth = cellWidth;
438        this.cellHeight = cellHeight;
439
440        textFactory = tcf.width(cellWidth).height(cellHeight).initBySize();
441
442        if (actualMap == null || actualMap.length <= 0) {
443            backgroundPanel = new SquidPanel(gridWidth, gridHeight, textFactory, bgColorCenter);
444            foregroundPanel = new SquidPanel(gridWidth, gridHeight, textFactory, fgColorCenter);
445            lightnesses = new int[width][height];
446        } else {
447            backgroundPanel = new SquidPanel(gridWidth, gridHeight, textFactory, bgColorCenter, 0, 0, ArrayTools.fill(' ', actualMap.length, actualMap[0].length));
448            foregroundPanel = new SquidPanel(gridWidth, gridHeight, textFactory, fgColorCenter, 0, 0, actualMap);
449            lightnesses = new int[actualMap.length][actualMap[0].length];
450        }
451        animationDuration = foregroundPanel.DEFAULT_ANIMATION_DURATION;
452
453        extraPanels = new ArrayList<>();
454
455        addActorAt(0, backgroundPanel);
456        addActorAt(2, foregroundPanel);
457
458        setSize(backgroundPanel.getWidth(), backgroundPanel.getHeight());
459    }

461    /**
462     * Add an extra layer on top of the foreground layer. Use putInto methods to specify the layer when adding a char (0
463     * is background, 1 is unused, 2 is foreground, and the first call to this method creates layer 3).
464     *
465     * @return this for chaining
466     */
467    public SquidLayers addExtraLayer() {
468        SquidPanel sp;
469        if (width != foregroundPanel.getTotalWidth() || height != foregroundPanel.getTotalHeight())
470            sp = new SquidPanel(width, height, textFactory, foregroundPanel.getColorCenter(), 0, 0,
471                    ArrayTools.fill(' ', foregroundPanel.getTotalWidth(), foregroundPanel.getTotalHeight()));
472        else
473            sp = new SquidPanel(width, height, textFactory);
474        addActor(sp);
475        extraPanels.add(sp);
476        return this;
477    }

480     * Sets the size of the text in the given layer  (but not the size of the cells) to the given width and height in
481     * pixels (which may be stretched by viewports later on, if your program uses them).
482     *
483     * @param layer the layer to affect; 0 is background, 1 is unused, 2 is foreground, 3 and higher are extra panels
484     * @param wide  the width of a glyph in pixels
485     * @param high  the height of a glyph in pixels
486     * @return this for chaining
487     */
488    public SquidLayers setTextSize(int layer, float wide, float high) {
489        SquidPanel p = backgroundPanel;
490        switch (layer) {
491            case 0:
492                break;
493            case 1:
494                break;
495            case 2:
496                p = foregroundPanel;
497                break;
498            default:
499                p = extraPanels.get(layer - 3);
500        }
501        p.setTextSize(wide, high);
502        return this;
503    }

514        textFactory.tweakHeight(high).tweakWidth(wide).initBySize();
515        setTextSize(0, wide, high);

703     * colors with another 2D Color array, and lightness variations for the background in a 2D int array (255 will make
704     * the background equal the background panel's {@link SquidPanel#getLightingColor()}, -255 will use the background

714     */
715    public SquidLayers put(int x, int y, char[][] c, Color[][] foregrounds, Color[][] backgrounds, int[][] backgroundLightness) {
716
717        foregroundPanel.put(x, y, c, foregrounds);
718        for (int i = x; i < getTotalWidth() && i - x < backgroundLightness.length; i++) {
719            for (int j = y; j < getTotalHeight() && j - y < backgroundLightness[i].length; j++) {
720                backgroundPanel.put(i, j, backgrounds[i - x][j - y],
721                        (clamp(lightnesses[i][j] + 256 + backgroundLightness[i - x][j - y], 1, 511)) * 0.001953125f);
722            }
723        }
724        return this;
725    }

831    /**
832     * A utility method that draws a 1-cell-wide black box around the text you request (as s) and replaces the contents
833     * of anything that was below or adjacent to the string's new position. Useful for message boxes.
834     *
835     * @param x in grid cells.
836     * @param y in grid cells.
837     * @param s the string to print inside the box
838     * @return this, for chaining
839     */
840    public SquidLayers putBoxedString(int x, int y, String s) {
841        if (y > 0 && y + 1 < getTotalHeight() && x > 0 && x + 1 < getTotalWidth()) {
842            for (int j = y - 1; j < y + 2 && j < getTotalHeight(); j++) {
843                for (int i = x - 1; i < s.length() + x + 2 && i < getTotalWidth(); i++) {
844                    foregroundPanel.put(i, j, ' ');
845                    lightnesses[i][j] = -200;
846
847                    backgroundPanel.put(i, j, backgroundPanel.getAt(i, j),
848                            SColor.DARK_GRAY, 0f * 0.001953125f);
849                }
850            }
851        }
852        foregroundPanel.put(x, y, s, SColor.CREAM);
853
854        return this;
855    }

1255    /**
1256     * Sets an extra panel to match the given SquidPanel (it can also be a subclass of SquidPanel, a likely use).
1257     * This tries to find the SquidPanel at the given index in the extra panels that may have been added to this with
1258     * {@link #addExtraLayer()}; if index is invalid because it is negative, this does nothing. If index is higher
1259     * than the highest index for a layer, this will simply add {@code panel} to the extra panels, possibly at an
1260     * earlier index (it will use the next available index, which could easily be 0 or 1). If index is valid, this sets
1261     * the extra panel at that index to {@code panel}, without changes.
1262     * This does not do any validation of size, but the {@link SquidPanel#getTotalWidth()} and
1263     * {@link SquidPanel#getTotalHeight()} of panel should almost always match this object's {@link #getTotalWidth()}
1264     * and {@link #getTotalHeight()}, though there are good reasons why the cell sizes or text sizes might not match.
1265     * This method is probably most useful when you want a system of {@link ImageSquidPanel} layers; as long as you have
1266     * the reference to an ImageSquidPanel outside this class, you can call its additional methods there and have it
1267     * affect {@code panel} here.
1268     *
1269     * @param panel a SquidPanel, ImageSquidPanel, or other subclass of SquidPanel that should have identical size
1270     * @param index the 0-based index into the extra panels list to try to assign panel to, or to append panel to
1271     */
1272
1273    public void setExtraPanel(SquidPanel panel, int index) {
1274        if (index < 0 || panel == null || extraPanels == null)
1275            return;
1276        if (index >= extraPanels.size()) {
1277            extraPanels.add(panel);
1278            addActor(panel);
1279        } else {
1280            removeActor(extraPanels.get(index));
1281            extraPanels.set(index, panel);
1282            addActorAt(index, panel);
1283        }
1284    }
