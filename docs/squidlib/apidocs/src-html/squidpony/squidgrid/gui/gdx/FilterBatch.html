013/**
014 * A drop-in substitute for {@link com.badlogic.gdx.graphics.g2d.SpriteBatch} that filters any colors used to tint text
015 * or images using a {@link FloatFilter}. The filter may have an effect on speed in some cases, but even moderately
016 * complex filters like {@link FloatFilters.YCbCrFilter} seem to perform perfectly well, spiking at above 1000 FPS on
017 * SparseDemo with a filter that changes parameters. This class should be preferred over SpriteBatch when using
018 * SquidLib's style of frequently changing the Batch color; as of libGDX 1.9.9, SpriteBatch is somewhat slower than
019 * FilterBatch when FilterBatch is using the default {@link FloatFilters#identityFilter}. This is because 1.9.9 changed
020 * SpriteBatch to do more work when setting a color, and FilterBatch keeps the older, faster way of color changing. It
021 * may be slower if used for Tiled map renderers in libGDX or for {@link com.badlogic.gdx.graphics.g2d.NinePatch}
022 * instances, which both need to reset a Color object frequently, but not much else in libGDX specifically needs the
023 * Color object {@link #getColor()} returns, and many things do fine with {@link #getPackedColor()}, which is faster.
024 * <br>
025 * Most FloatFilter varieties don't have much overhead, and I encourage you to try some to quickly produce graphical
026 * effects in a game. As an example, changing the warmth of a color (as well as its lightness) is relatively easy with a
027 * {@link FloatFilters.YCwCmFilter} that adds to or subtracts from Cw (chromatic warmth) and may change Y (lightness).
028 * With that same filter, if the protagonist is partially blinded, you could set the multipliers for Cw and Cm
029 * (chromatic mildness; determines whether a color is closer to mild colors like green/yellow or bold colors like
030 * red/blue) to floats between 0.1 and 0.5 or so, which takes even intense colors and makes them diluted and grayish.
031 * You could also change Y to be brighter or darker depending on whether a bright light blinded the protagonist, or some
032 * mud was thrown in their eyes, etc.
033 * <br>
034 * Created by Tommy Ettinger on 8/2/2018.
035 */

115    /** Constructs a new FilterBatch. Sets the projection matrix to an orthographic projection with y-axis point upwards, x-axis
116     * point to the right and the origin being in the bottom left corner of the screen. The projection will be pixel perfect with
117     * respect to the current screen resolution.
118     * <p>
119     * The defaultShader specifies the shader to use. Note that the names for uniforms for this default shader are different than
120     * the ones expect for shaders set with {@link #setShader(ShaderProgram)}. See {@link #createDefaultShader()}.
121     * @param size The max number of sprites in a single batch. Max of 8191.
122     * @param defaultShader The default shader to use. This is not owned by the FilterBatch and must be disposed separately. */
123    public FilterBatch (int size, ShaderProgram defaultShader, FloatFilter filter) {
124        // 32767 is max vertex index, so 32767 / 4 vertices per sprite = 8191 sprites max.
125        if (size > 8191) throw new IllegalArgumentException("Can't have more than 8191 sprites per batch: " + size);
126
127        Mesh.VertexDataType vertexDataType = (Gdx.gl30 != null) ? Mesh.VertexDataType.VertexBufferObjectWithVAO : Mesh.VertexDataType.VertexArray;
128
129        mesh = new Mesh(vertexDataType, false, size * 4, size * 6,
130                new VertexAttribute(VertexAttributes.Usage.Position, 2, ShaderProgram.POSITION_ATTRIBUTE),
131                new VertexAttribute(VertexAttributes.Usage.ColorPacked, 4, ShaderProgram.COLOR_ATTRIBUTE),
132                new VertexAttribute(VertexAttributes.Usage.TextureCoordinates, 2, ShaderProgram.TEXCOORD_ATTRIBUTE + "0"));
133
134        projectionMatrix.setToOrtho2D(0, 0, Gdx.graphics.getWidth(), Gdx.graphics.getHeight());
135
136        vertices = new float[size * SPRITE_SIZE];
137
138        int len = size * 6;
139        short[] indices = new short[len];
140        short j = 0;
141        for (int i = 0; i < len; i += 6, j += 4) {
142            indices[i] = j;
143            indices[i + 1] = (short)(j + 1);
144            indices[i + 2] = (short)(j + 2);
145            indices[i + 3] = (short)(j + 2);
146            indices[i + 4] = (short)(j + 3);
147            indices[i + 5] = j;
148        }
149        mesh.setIndices(indices);
150
151        if (defaultShader == null) {
152            shader = createDefaultShader();
153            ownsShader = true;
154        } else
155            shader = defaultShader;
156        this.filter = filter;
157    }

159    /** Returns a new instance of the default shader used by FilterBatch for GL2 when no shader is specified. */
160    static public ShaderProgram createDefaultShader () {
161        String vertexShader = "attribute vec4 " + ShaderProgram.POSITION_ATTRIBUTE + ";\n" //
162                + "attribute vec4 " + ShaderProgram.COLOR_ATTRIBUTE + ";\n" //
163                + "attribute vec2 " + ShaderProgram.TEXCOORD_ATTRIBUTE + "0;\n" //
164                + "uniform mat4 u_projTrans;\n" //
165                + "varying vec4 v_color;\n" //
166                + "varying vec2 v_texCoords;\n" //
167                + "\n" //
168                + "void main()\n" //
169                + "{\n" //
170                + "   v_color = " + ShaderProgram.COLOR_ATTRIBUTE + ";\n" //
171                + "   v_color.a = v_color.a * (255.0/254.0);\n" //
172                + "   v_texCoords = " + ShaderProgram.TEXCOORD_ATTRIBUTE + "0;\n" //
173                + "   gl_Position =  u_projTrans * " + ShaderProgram.POSITION_ATTRIBUTE + ";\n" //
174                + "}\n";
175        String fragmentShader = "#ifdef GL_ES\n" //
176                + "#define LOWP lowp\n" //
177                + "precision mediump float;\n" //
178                + "#else\n" //
179                + "#define LOWP \n" //
180                + "#endif\n" //
181                + "varying LOWP vec4 v_color;\n" //
182                + "varying vec2 v_texCoords;\n" //
183                + "uniform sampler2D u_texture;\n" //
184                + "void main()\n"//
185                + "{\n" //
186                + "  gl_FragColor = v_color * texture2D(u_texture, v_texCoords);\n" //
187                + "}";
188
189        ShaderProgram shader = new ShaderProgram(vertexShader, fragmentShader);
190        if (!shader.isCompiled()) throw new IllegalArgumentException("Error compiling shader: " + shader.getLog());
191        return shader;
192    }

194    @Override
195    public void begin () {
196        if (drawing) throw new IllegalStateException("FilterBatch.end must be called before begin.");
197        renderCalls = 0;
198
199        Gdx.gl.glDepthMask(false);
200        if (customShader != null)
201            customShader.begin();
202        else
203            shader.begin();
204        setupMatrices();
205
206        drawing = true;
207    }

209    @Override
210    public void end () {
211        if (!drawing) throw new IllegalStateException("FilterBatch.begin must be called before end.");
212        if (idx > 0) flush();
213        lastTexture = null;
214        drawing = false;
215
216        GL20 gl = Gdx.gl;
217        gl.glDepthMask(true);
218        if (isBlendingEnabled()) gl.glDisable(GL20.GL_BLEND);
219
220        if (customShader != null)
221            customShader.end();
222        else
223            shader.end();
224    }

246    @Override
247    public Color getColor () {
248        final int intBits = NumberTools.floatToIntBits(color);
249        Color color = tempColor;
250        color.r = (intBits & 0xff) / 255f;
251        color.g = ((intBits >>> 8) & 0xff) / 255f;
252        color.b = ((intBits >>> 16) & 0xff) / 255f;
253        color.a = ((intBits >>> 24) & 0xff) / 255f;
254        return color;
255    }

262    @Override
263    public void draw (Texture texture, float x, float y, float originX, float originY, float width, float height, float scaleX,
264                      float scaleY, float rotation, int srcX, int srcY, int srcWidth, int srcHeight, boolean flipX, boolean flipY) {
265        if (!drawing) throw new IllegalStateException("FilterBatch.begin must be called before draw.");
266
267        float[] vertices = this.vertices;
268
269        if (texture != lastTexture)
270            switchTexture(texture);
271        else if (idx == vertices.length) //
272            flush();
273
274        // bottom left and top right corner points relative to origin
275        final float worldOriginX = x + originX;
276        final float worldOriginY = y + originY;
277        float fx = -originX;
278        float fy = -originY;
279        float fx2 = width - originX;
280        float fy2 = height - originY;
281
282        // scale
283        if (scaleX != 1 || scaleY != 1) {
284            fx *= scaleX;
285            fy *= scaleY;
286            fx2 *= scaleX;
287            fy2 *= scaleY;
288        }
289
290        // construct corner points, start from top left and go counter clockwise
291        final float p1x = fx;
292        final float p1y = fy;
293        final float p2x = fx;
294        final float p2y = fy2;
295        final float p3x = fx2;
296        final float p3y = fy2;
297        final float p4x = fx2;
298        final float p4y = fy;
299
300        float x1;
301        float y1;
302        float x2;
303        float y2;
304        float x3;
305        float y3;
306        float x4;
307        float y4;
308
309        // rotate
310        if (rotation != 0) {
311            final float cos = MathUtils.cosDeg(rotation);
312            final float sin = MathUtils.sinDeg(rotation);
313
314            x1 = cos * p1x - sin * p1y;
315            y1 = sin * p1x + cos * p1y;
316
317            x2 = cos * p2x - sin * p2y;
318            y2 = sin * p2x + cos * p2y;
319
320            x3 = cos * p3x - sin * p3y;
321            y3 = sin * p3x + cos * p3y;
322
323            x4 = x1 + (x3 - x2);
324            y4 = y3 - (y2 - y1);
325        } else {
326            x1 = p1x;
327            y1 = p1y;
328
329            x2 = p2x;
330            y2 = p2y;
331
332            x3 = p3x;
333            y3 = p3y;
334
335            x4 = p4x;
336            y4 = p4y;
337        }
338
339        x1 += worldOriginX;
340        y1 += worldOriginY;
341        x2 += worldOriginX;
342        y2 += worldOriginY;
343        x3 += worldOriginX;
344        y3 += worldOriginY;
345        x4 += worldOriginX;
346        y4 += worldOriginY;
347
348        float u = srcX * invTexWidth;
349        float v = (srcY + srcHeight) * invTexHeight;
350        float u2 = (srcX + srcWidth) * invTexWidth;
351        float v2 = srcY * invTexHeight;
352
353        if (flipX) {
354            float tmp = u;
355            u = u2;
356            u2 = tmp;
357        }
358
359        if (flipY) {
360            float tmp = v;
361            v = v2;
362            v2 = tmp;
363        }
364
365        float color = this.color;
366        int idx = this.idx;
367        vertices[idx] = x1;
368        vertices[idx + 1] = y1;
369        vertices[idx + 2] = color;
370        vertices[idx + 3] = u;
371        vertices[idx + 4] = v;
372
373        vertices[idx + 5] = x2;
374        vertices[idx + 6] = y2;
375        vertices[idx + 7] = color;
376        vertices[idx + 8] = u;
377        vertices[idx + 9] = v2;
378
379        vertices[idx + 10] = x3;
380        vertices[idx + 11] = y3;
381        vertices[idx + 12] = color;
382        vertices[idx + 13] = u2;
383        vertices[idx + 14] = v2;
384
385        vertices[idx + 15] = x4;
386        vertices[idx + 16] = y4;
387        vertices[idx + 17] = color;
388        vertices[idx + 18] = u2;
389        vertices[idx + 19] = v;
390        this.idx = idx + 20;
391    }

591    @Override
592    public void draw (Texture texture, float[] spriteVertices, int offset, int count) {
593        if (!drawing) throw new IllegalStateException("FilterBatch.begin must be called before draw.");
594
595        int verticesLength = vertices.length;
596        int remainingVertices = verticesLength;
597        if (texture != lastTexture)
598            switchTexture(texture);
599        else {
600            remainingVertices -= idx;
601            if (remainingVertices == 0) {
602                flush();
603                remainingVertices = verticesLength;
604            }
605        }
606        int copyCount = Math.min(remainingVertices, count);
607
608        System.arraycopy(spriteVertices, offset, vertices, idx, copyCount);
609        idx += copyCount;
610        count -= copyCount;
611        while (count > 0) {
612            offset += copyCount;
613            flush();
614            copyCount = Math.min(verticesLength, count);
615            System.arraycopy(spriteVertices, offset, vertices, 0, copyCount);
616            idx += copyCount;
617            count -= copyCount;
618        }
619    }

985    @Override
986    public void flush () {
987        if (idx == 0) return;
988
989        renderCalls++;
990        totalRenderCalls++;
991        int spritesInBatch = idx / 20;
992        if (spritesInBatch > maxSpritesInBatch) maxSpritesInBatch = spritesInBatch;
993        int count = spritesInBatch * 6;
994
995        lastTexture.bind();
996        Mesh mesh = this.mesh;
997        mesh.setVertices(vertices, 0, idx);
998        mesh.getIndicesBuffer().position(0);
999        mesh.getIndicesBuffer().limit(count);
1000
1001        if (blendingDisabled) {
1002            Gdx.gl.glDisable(GL20.GL_BLEND);
1003        } else {
1004            Gdx.gl.glEnable(GL20.GL_BLEND);
1005            if (blendSrcFunc != -1) Gdx.gl.glBlendFuncSeparate(blendSrcFunc, blendDstFunc, blendSrcFuncAlpha, blendDstFuncAlpha);
1006        }
1007
1008        mesh.render(customShader != null ? customShader : shader, GL20.GL_TRIANGLES, 0, count);
1009
1010        idx = 0;
1011    }
