020    /**
021     * Wraps the functionality of {@link SColor#toEditedFloat(float, float, float, float, float)} so it can be called as
022     * a FloatFilter, adding values to hue, saturation, and value (clamping saturation and value and wrapping hue).
023     */
024    public static class HSVFilter extends FloatFilter {
025        public float hueAddend, saturationAddend, valueAddend;
026
027        public HSVFilter(float saturation, float value) {
028            this(0f, saturation, value);
029        }
030
031        public HSVFilter(float hueAdd, float saturationAdd, float valueAdd) {
032            hueAddend = hueAdd;
033            saturationAddend = saturationAdd;
034            valueAddend = valueAdd;
035        }
036
037        /**
038         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
039         *
040         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
041         * @return a packed float color, as produced by {@link Color#toFloatBits()}
042         */
043        @Override
044        public float alter(float color) {
045            return SColor.toEditedFloat(color, hueAddend, saturationAddend, valueAddend, 0f);
046        }

049    /**
050     * Changes all colors this alters to have the same hue as, and a closer saturation to, a target color given in the
051     * constructor, as well as optionally multiplying saturation of the result and/or adding lightness/value. A good
052     * example usage of this is to make a sepia-tone effect with
053     * {@code new FloatFilters.ColorizeFilter(SColor.CLOVE_BROWN, 0.6f, 0.0f)}.
054     */
055    public static class ColorizeFilter extends FloatFilter {
056        public float targetCb, targetCr, lumaAddend;
057
058        public ColorizeFilter(float color) {
059            this(color, 1f, 0f);
060        }
061
062        public ColorizeFilter(Color color) {
063            this(color.toFloatBits(), 1f, 0f);
064        }
065
066        public ColorizeFilter(Color color, float chromaMul, float lumaAdd) {
067            this(color.toFloatBits(), chromaMul, lumaAdd);
068        }
069
070        public ColorizeFilter(float color, float chromaMul, float lumaAdd) {
071            targetCb = SColor.chromaBOfFloat(color) * chromaMul;
072            targetCr = SColor.chromaROfFloat(color) * chromaMul;
073            lumaAddend = lumaAdd;
074        }
075
076        /**
077         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
078         *
079         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
080         * @return a packed float color, as produced by {@link Color#toFloatBits()}
081         */
082        @Override
083        public float alter(float color) {
084            final int bits = NumberTools.floatToIntBits(color);
085            return SColor.floatGetYCbCr((bits & 0x000000ff) * (0x1.010102p-8f  * 0.299f) +
086                    (bits & 0x0000ff00) * (0x1.010102p-16f * 0.587f) +
087                    (bits & 0x00ff0000) * (0x1.010102p-24f * 0.114f),
088                    targetCb, targetCr,
089                    ((bits & 0xfe000000) >>> 24) * 0x1.020408p-8f);
090        }
091    }

188     * A FloatFilter that makes all colors given to it grayscale, using only their luma as calculated by
189     * {@link SColor#lumaOfFloat(float)} as the lightness (it does also preserve alpha transparency).

191    public static class GrayscaleFilter extends FloatFilter {
192        /**
193         * Takes a packed float color and produces a grayscale packed float color that this FloatFilter edited.
194         * Uses the luma calculation from {@link SColor#lumaOfFloat(float)} instead of the value calculation from
195         * {@link SColor#valueOfFloat(float)}; luma tends to be more visually-accurate on modern monitors.
196         *

198         * @return a packed float color, as produced by {@link Color#toFloatBits()}

202            final int bits = NumberTools.floatToIntBits(color);
203            color = (bits & 0x000000ff) * (0x1.010102p-8f * 0.299f) +
204                    (bits & 0x0000ff00) * (0x1.010102p-16f * 0.587f) +
205                    (bits & 0x00ff0000) * (0x1.010102p-24f * 0.114f);
206            return floatGet(color, color, color, (bits >>> 24 & 0xFE) * 0.003937008f);
207        }

210    /**
211     * Like {@link HSVFilter}, but edits its input colors in YCbCr color space, and multiplies rather than adds.
212     * Y is luma, and affects how bright the color is (luma 1 is white, luma 0 is black). Cb is Chroma(blue) amd Cr is
213     * Chroma(red), two inter-related channels that determine the hue and vividness of a specific color. When Cb and Cr
214     * are both 0, the color is grayscale. When Cb is 0.5 and Cr is -0.5, the color is blue unless Y is very high or
215     * low. When Cb is -0.5 and Cr is 0.5, the color is red with the same caveats re: Y. When Cb and Cr are both -0.5,
216     * the color is green (same caveats), and when both are 0.5, the color is purple. When Y is 0.5, Cb and Cr form a
217     * graph like this:
218     * <br>
219     * <img src="https://upload.wikimedia.org/wikipedia/commons/thumb/9/91/YCbCr-CbCr_Y50.png/240px-YCbCr-CbCr_Y50.png" />
220     * <br>
221     * Valid values for Cb and Cr are from -0.5 to 0.5 at the widest part of the range (it shrinks as Y approaches 0 or
222     * 1), but there aren't really invalid values here because this filter will clamp results with higher or lower
223     * channel values than a color can have. Each of yMul, cbMul, and crMul can have any float value, but yMul should be
224     * positive (unless you want this to only produce solid black). Similarly, cbMul and crMul will not produce
225     * meaningful results if they are very large (either positive or negative); it's recommended to use values between
226     * 0.0 and 1.0 for both if you want to desaturate colors or values somewhat greater than 1.0 to oversaturate them.
227     */
228    public static class YCbCrFilter extends FloatFilter {
229        public float yMul, cbMul, crMul;
230
231        public YCbCrFilter(float yMul) {
232            this(yMul, 1f, 1f);
233        }
234
235        public YCbCrFilter(float yMul, float cbMul, float crMul) {
236            this.yMul = yMul;
237            this.cbMul = cbMul;
238            this.crMul = crMul;
239        }
240
241        /**
242         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
243         *
244         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
245         * @return a packed float color, as produced by {@link Color#toFloatBits()}
246         */
247        @Override
248        public float alter(float color) {
249            final int bits = NumberTools.floatToIntBits(color);
250            final float opacity = (bits >>> 24 & 0xFE) * 0.003937008f;
251            float luma = yMul * (
252                    (bits & 0x000000ff) * (0x1.010102p-8f * 0.299f) +
253                            (bits & 0x0000ff00) * (0x1.010102p-16f * 0.587f) +
254                            (bits & 0x00ff0000) * (0x1.010102p-24f * 0.114f));
255            final float chromaB = cbMul * (
256                    (bits & 0x000000ff) * (0x1.010102p-8f * -0.168736f) +
257                            (bits & 0x0000ff00) * (0x1.010102p-16f * -0.331264f) +
258                            (bits & 0x00ff0000) * (0x1.010102p-24f * 0.5f));
259            final float chromaR = crMul * (
260                    (bits & 0x000000ff) * (0x1.010102p-8f * 0.5f) +
261                            (bits & 0x0000ff00) * (0x1.010102p-16f * -0.418688f) +
262                            (bits & 0x00ff0000) * (0x1.010102p-24f * -0.081312f));
263            return floatGet(MathUtils.clamp(luma + chromaR * 1.402f, 0f, 1f),
264                    MathUtils.clamp(luma - chromaB * 0.344136f - chromaR * 0.714136f, 0f, 1f),
265                    MathUtils.clamp(luma + chromaB * 1.772f, 0f, 1f),
266                    opacity);
267        }

270    /**
271     * Like {@link YCbCrFilter}, but edits its input colors in YCoCg color space, or like {@link HSVFilter} except it
272     * doesn't add, it multiplies. Most of the time you should prefer {@link YCbCrFilter} as long as it isn't a
273     * performance bottleneck; this method is faster but less accurate. Y is luminance, ranging from 0 (dark) to 1
274     * (light), and affects how bright the color is, but isn't very accurate perceptually. Co is Chrominance(orange) and
275     * Cg is Chrominance(green) (both range from -0.5 to 0.5), two inter-related channels that determine the hue and
276     * vividness of a specific color. When Co and Cg are both 0, the color is grayscale. When Co is 0.5 and Cg is -0.5,
277     * the color is red unless Y is very high or low. When Co is -0.5 and Cg is 0.5, the color is cyan with the same
278     * caveats re: Y. When Co and Cg are both -0.5, the color is blue (same caveats), and when both are 0.5, the color
279     * is yellow.
280     * <br>
281     * Valid values for Co and Cg are from -0.5 to 0.5 at the widest part of the range (it shrinks as Y approaches 0 or
282     * 1), but there aren't really invalid values here because this filter will clamp results with higher or lower
283     * channel values than a color can have. Each of yMul, coMul, and cgMul can have any float value, but yMul should be
284     * positive (unless you want this to only produce solid black). Similarly, coMul and cgMul will not produce
285     * meaningful results if they are very large (either positive or negative); it's recommended to use values between
286     * 0.0 and 1.0 for both if you want to desaturate colors or values somewhat greater than 1.0 to oversaturate them.
287     */
288    public static class YCoCgFilter extends FloatFilter {
289        public float yMul, coMul, cgMul;
290
291        public YCoCgFilter(float luminanceMul, float orangeMul, float greenMul) {
292            this.yMul = luminanceMul;
293            this.coMul = orangeMul;
294            this.cgMul = greenMul;
295        }
296
297        /**
298         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
299         *
300         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
301         * @return a packed float color, as produced by {@link Color#toFloatBits()}
302         */
303        @Override
304        public float alter(float color) {
305            final int bits = NumberTools.floatToIntBits(color);
306            final float opacity = (bits >>> 24 & 0xFE) * 0.003937008f;
307            final float y = yMul * (((bits & 0x000000ff) + ((bits & 0x0000ff00) >>> 7) + ((bits & 0x00ff0000) >>> 16)) * 0x1.010102p-10f);
308            final float co = coMul * (((bits & 0x000000ff) - ((bits & 0x00ff0000) >>> 16)) * 0x1.010102p-9f);
309            final float cg = cgMul * ((((bits & 0x0000ff00) >>> 7) - (bits & 0x000000ff) - ((bits & 0x00ff0000) >>> 16)) * 0x1.010102p-10f);
310
311            final float t = y - cg;
312            return floatGet(MathUtils.clamp(t + co, 0f, 1f),
313                    MathUtils.clamp(y + cg, 0f, 1f),
314                    MathUtils.clamp(t - co, 0f, 1f),
315                    opacity);
316        }
317    }

319    /**
320     * Like {@link YCbCrFilter} or  {@link YCoCgFilter}, but edits its input colors in YCwCm color space, which is very
321     * similar to YCoCg but has chroma/chrominance components that are useful aesthetically on their own. You may often
322     * prefer {@link YCbCrFilter} because it calculates lightness (luma) more precisely, but the Cb (blue-ness) and Cr
323     * (red-ness) components are less useful for some purposes individually. Y is luminance, ranging from 0 (dark) to 1
324     * (light), and affects how bright the color is, but isn't very accurate perceptually. Cw is Chroma(warm) and
325     * Cm is Chroma(mild) (both range from -1.0 to 1.0), two inter-related channels that determine the hue and vividness
326     * of a specific color. When Cw and Cm are both 0, the color is grayscale. When Cw is 1 and Cm is -1, the color
327     * is red or like red. When Cw is -1 and Cm is 1, the color is green or like green. When Cw and Cm are both -1, the
328     * color is blue or like blue, and when both are 1, the color is roughly yellow or brown (depending on Y).
329     * <br>
330     * Valid values for Cw and Cm are from -1.0 to 1.0, but there aren't really invalid values here because this filter
331     * will clamp results with higher or lower channel values than a color can have. Each of yMul, cwMul, and cmMul can
332     * have any float value, but yMul should be positive (unless you want this to only produce solid black). Similarly,
333     * cwMul and cmMul will not produce meaningful results if they are very large (either positive or negative); it's
334     * recommended to use values between 0.0 and 1.0 for both if you want to desaturate colors or values somewhat
335     * greater than 1.0 to oversaturate them. Unlike {@link YCbCrFilter} and {@link YCoCgFilter}, you can benefit from
336     * setting cwMul independently of the other chroma component, which can be used to emphasize warm vs. cool colors if
337     * cwMul is greater than 1.0, or to de-emphasize that comparison if it is between 0.0 and 1.0. A similar option is
338     * possible for cmMul, but it isn't as clear of an artistic convention; a high cmMul will separate green-and-yellow
339     * colors further from red-purple-and-blue colors. Also unlike the other YCC filters, this allows an additive change
340     * to Y, Cw, and Cm applied after the multiplicative change but before converting to RGB and clamping. This can be
341     * used to make all colors warmer or cooler (such as for volcano or frozen scenes) by adding or subtracting from Cw,
342     * for instance. It can also lighten or darken all colors by changing luma.
343     */
344    public static class YCwCmFilter extends FloatFilter {
345        public float yMul, cwMul, cmMul, yAdd, cwAdd, cmAdd;
346
347        public YCwCmFilter(float yMul, float cwMul, float cmMul) {
348            this(yMul, cwMul, cmMul, 0f, 0f, 0f);
349        }
350        public YCwCmFilter(float yMul, float cwMul, float cmMul, float yAdd, float cwAdd, float cmAdd) {
351            this.yMul = yMul;
352            this.cwMul = cwMul;
353            this.cmMul = cmMul;
354            this.yAdd = yAdd;
355            this.cwAdd = cwAdd;
356            this.cmAdd = cmAdd;
357        }
358
359        /**
360         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
361         *
362         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
363         * @return a packed float color, as produced by {@link Color#toFloatBits()}
364         */
365        @Override
366        public float alter(float color) {
367            final int bits = NumberTools.floatToIntBits(color);
368            final float opacity = (bits >>> 24 & 0xFE) * 0.003937008f;
369            final float luma = yAdd + yMul * ((bits & 0xFF) * 0x3p-11f + (bits >>> 8 & 0xFF) * 0x1p-9f + (bits >>> 16 & 0xFF) * 0x1p-11f);
370            final float warm = cwAdd + cwMul * (((bits & 0xFF) - (bits >>> 16 & 0xff)) * 0x1.010102p-8f);
371            final float mild = 0.5f * (cmAdd + cmMul * (((bits >>> 8 & 0xff) - (bits >>> 16 & 0xff)) * 0x1.010102p-8f));
372
373            return floatGet(MathUtils.clamp(luma + warm * 0.625f - mild, 0f, 1f),
374                    MathUtils.clamp(luma + mild - warm * 0.375f, 0f, 1f),
375                    MathUtils.clamp(luma - warm * 0.375f - mild, 0f, 1f),
376                    opacity);
377        }
378    }

380    /**
381     * A FloatFilter that chains together one or more FloatFilters one after the next, passing the float output of one
382     * as input to the next until the chain has all been called.
383     */
384    public static class ChainFilter extends FloatFilter {
385        public FloatFilter[] filters;
386
387        /**
388         * Takes a vararg or array of FloatFilter objects and produces a ChainFilter that will call all of them in order
389         * on any color given to this to alter.
390         * @param filters an array or vararg of FloatFilter objects; none can be null
391         */
392        public ChainFilter(FloatFilter... filters)
393        {
394            if(filters == null || filters.length == 0)
395                this.filters = new FloatFilter[]{new IdentityFilter()};
396            this.filters = filters;
397        }
398
399        /**
400         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
401         *
402         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
403         * @return a packed float color, as produced by {@link Color#toFloatBits()}
404         */
405        @Override
406        public float alter(float color) {
407            for (int i = 0; i < filters.length; i++) {
408                color = filters[i].alter(color);
409            }
410            return color;
411        }
412    }

437         * alpha component that is less than 1, then amount is effectively multiplied by that alpha.
438         * @param target a packed float color; must not be null
439         * @param amount a float that determines how much target will affect an input color; will be clamped between 0f and 1f

462     * given colors. Uses {@link SColor#lerpFloatColorsBlended(float, float, float)} to mix a requested color with the
463     * chosen target color, and this means the alpha of the target color affects the amount of change instead of the
464     * resulting alpha. Changing the alpha of the colors this is given can be done easily with
465     * {@link SColor#translucentColor(float, float)}, and this allows you to specify varying amounts to mix by.
466     */
467    public static class MultiLerpFilter extends FloatFilter {
468        public float[] targets;
469        public float amount;
470        /**
471         * Builds a MultiLerpFilter with an array of Color objects (which will be converted to an array of packed float
472         * colors) and an amount as a float. The amount is how much the target colors will affect input colors, from 0f
473         * to 1f. If a target color has an alpha component that is less than 1, then amount is effectively multiplied by
474         * that alpha. If you want to edit the alpha without duplicating Color objects, you can use
475         * {@link SColor#translucentColor(Color, float)} to make a float array to pass to
476         *{@link #MultiLerpFilter(float, float...)}.
477         * @param amount a float that determines how much target will affect an input color; will be clamped between 0f and 1f
478         * @param targets an array of libGDX Color objects; must not be null or empty
479         */
480        public MultiLerpFilter(float amount, Color[] targets) {
481            this.targets = new float[targets.length];
482            for (int i = 0; i < targets.length; i++) {
483                this.targets[i] = targets[i].toFloatBits();
484            }
485            this.amount = MathUtils.clamp(amount, 0f, 1f);
486        }
487
488        /**
489         * Builds a MultiLerpFilter with an array of packed float colors and an amount as a float.
490         * The amount is how much the target color will affect input colors, from 0f to 1f. If the target color has an
491         * alpha component that is less than 1, then amount is effectively multiplied by that alpha.
492         * @param amount a float that determines how much target will affect an input color; will be clamped between 0f and 1f
493         * @param targets an array or vararg of packed float colors; must not be null or empty
494         */
495
496        public MultiLerpFilter(float amount, float... targets)
497        {
498            this.targets = targets;
499            this.amount = MathUtils.clamp(amount, 0f, 1f);
500        }
501
502        /**
503         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
504         *
505         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
506         * @return a packed float color, as produced by {@link Color#toFloatBits()}
507         */
508        @Override
509        public float alter(float color) {
510            int choice = 0, diff = SColor.difference2(targets[0], color);
511            for (int i = 1; i < targets.length; i++) {
512                if(diff != (diff = Math.min(SColor.difference2(targets[i], color), diff)))
513                    choice = i;
514            }
515            return lerpFloatColorsBlended(color, targets[choice], amount);
516        }
517    }

519    /**
520     * A FloatFilter that limits the colors it can return to a fixed palette, and won't return any colors that are
521     * missing from that palette (although it can always return fully-transparent). {@link PaletteReducerFilter} is also
522     * an option; it uses more memory but is faster to look up colors in larger palettes (it has a maximum size of 256
523     * colors, though, which this class doesn't).
524     */
525    public static class PaletteFilter extends FloatFilter {
526        public float[] targets;
527        /**
528         * Builds a PaletteFilter with an array of Color objects that this will choose from. The array will be converted
529         * to an array of packed float colors, and not referenced directly.
530         * @param targets an array of libGDX Color objects; must not be null or empty
531         */
532        public PaletteFilter(final Color[] targets) {
533            this.targets = new float[targets.length];
534            for (int i = 0; i < targets.length; i++) {
535                this.targets[i] = targets[i].toFloatBits();
536            }
537        }
538
539        /**
540         * Builds a PaletteFilter with an array of packed float colors that this will choose from. The array will be
541         * referenced directly, not copied, so if you change the contents of targets, it will be reflected here.
542         * @param targets an array or vararg of packed float colors; must not be null or empty
543         */
544
545        public PaletteFilter(final float... targets)
546        {
547            this.targets = targets;
548        }
549
550        /**
551         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
552         *
553         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
554         * @return a packed float color, as produced by {@link Color#toFloatBits()}
555         */
556        @Override
557        public float alter(float color) {
558            if(color >= 0f) //if color is halfway-transparent or closer to transparent...
559                return 0f; // return fully transparent
560            int choice = 0, diff = SColor.difference2(targets[0], color);
561            for (int i = 1; i < targets.length; i++) {
562                if(diff != (diff = Math.min(SColor.difference2(targets[i], color), diff)))
563                    choice = i;
564            }
565            return targets[choice];
566        }
567    }

569    /**
570     * A FloatFilter that limits the colors it can return to a fixed palette as determined by a {@link PaletteReducer},
571     * and won't return any colors that are missing from that palette (although it can always return fully-transparent).
572     * This is like {@link PaletteFilter} but trades off memory usage (it uses about 33KB to store a large-ish lookup
573     * table) to improve speed on large palettes. This can't use a palette larger than 256 colors (including transparent
574     * almost always).
575     */
576    public static class PaletteReducerFilter extends FloatFilter {
577        public PaletteReducer reducer;
578
579        /**
580         * Builds a PaletteReducerFilter that will use the 256-color (including transparent) DawnBringer Aurora palette.
581         */
582        public PaletteReducerFilter()
583        {
584            reducer = new PaletteReducer();
585        }
586        
587        /**
588         * Builds a PaletteReducerFilter with the given PaletteReducer, which will be referenced without copying. You
589         * can call {@link PaletteReducer#exact(Color[])} or other similar methods before this filter is used to set up
590         * the palette, and this is often done before the PaletteReducer is passed here.
591         * @param palette a PaletteReducer that should have the desired palette set up before this is used
592         */
593        public PaletteReducerFilter(final PaletteReducer palette) {
594            reducer = palette;
595        }
596
597        /**
598         * Builds a PaletteReducerFilter with an array or vararg of libGDX colors (at most 256 colors, and often
599         * starting with a transparent color) that this will choose from. The array will have its contents read but will
600         * not be held onto, so later changes won't affect it.
601         * @param targets an array or vararg of libGDX colors; must not be null
602         */
603
604        public PaletteReducerFilter(final Color... targets)
605        {
606            reducer = new PaletteReducer(targets);
607        }
608
609        /**
610         * Takes a packed float color and produces a potentially-different packed float color that this FloatFilter edited.
611         *
612         * @param color a packed float color, as produced by {@link Color#toFloatBits()}
613         * @return a packed float color, as produced by {@link Color#toFloatBits()}
614         */
615        @Override
616        public float alter(float color) {
617            return reducer.reduceFloat(color);
618        }
619    }
