012/**
013 * GDXMarkup implements IMarkup for libGDX Color objects, and can start blocks of markup that libGDX understands that
014 * display text in a given Color. Typically, the singleton {@link #instance} would be passed to a class that uses
015 * IMarkup of Color, and then anything else will be handled internally as Colors are given to the using class. This does
016 * extend GDX's markup to handle bold and italic options for text; this only works if you are using a {@link TextFamily}
017 * as your {@link TextCellFactory}, such as {@link DefaultResources#getSlabFamily()}, and only if you use
018 * {@link #colorString(CharSequence)} or {@link #styleString(CharSequence)} to generate a value that can be drawn later.
019 * You can use {@link #colorStringOnly(CharSequence)} if you aren't using a TextFamily; it ignores bold and italic style
020 * tags and removes them from the IColoredString it returns, but otherwise colors and case-adjusts text as tags say.
021 * <br>
022 * The notation for colors is the same as in the rest of libGDX, but if you make an IColoredString with colorString(),
023 * it doesn't need any flag to be changed on your BitmapFont (like it does for GDX markup normally). This notation looks
024 * like {@code [#FF00FF]Magenta text here[]}, which starts a tag for a hex color, uses the hex value for bright magenta,
025 * then contains the text "Magenta text here" which will be shown in bright magenta, followed by "[]" to change the
026 * color (and style, see next) back to the default white (and normal style). You can also use the names of colors, as
027 * defined in the documentation for every {@link SColor}, and some libGDX Color values as well; this looks like
028 * {@code [Inside Of A Bottle]Gray text[]} to produce the words "Gray text" with the color
029 * {@link SColor#INSIDE_OF_A_BOTTLE} (note that the docs for that SColor say what the precise name is, and case needs to
030 * match; you can also look up the {@link SColor#name} field). Another option for colors is to use hue, saturation,
031 * value, and optionally opacity as 3 or 4 floats, in that order; this looks like {@code [@ 0 0.7 0.96]} (hue,
032 * saturation, and value, with opacity implicitly 1) or {@code [@ 0 0.7 0.96 0.8]} (0.8 opacity is a bit translucent).
033 * The HSV markup option is an addition to libGDX's syntax; any number of spaces/tabs can be used between HSV components
034 * and the space between @ and the first component is optional. You can use {@code [[} to escape an opening bracket, and
035 * {@code []} to reset formatting. As an addition to GDX color markup, you can change case with {@code [!]} to make text
036 * ALL CAPS or {@code [=]} to make it lower-case (encountering a tag toggles it, so {@code [!]yelling[!] are we?} would
037 * become {@code YELLING are we?}), using {@code []} to reset all markup or {@code [,]} to reset just case and font
038 * style (see next) markup. If using a TextFamily such as {@link DefaultResources#getLeanFamily()}, you can toggle the
039 * font style as bold with {@code [*]} and as italic with {@code [/]}. If bold is on when this encounters another bold
040 * tag, it will turn bold off; the same is true for italic. These formatting styles can overlap and do not need to be
041 * nested as in HTML; this notation is valid: {@code [*]bold, [/]bold and italic, [*] just italic,[] plain}. While
042 * {@code []} resets both color, case, and style to white color, normal case, and regular style, there is also
043 * {@code [,]} to reset only style/case, or {@code [WHITE]} to reset only color (to white).
044 * <br>
045 * Created by Tommy Ettinger on 1/23/2016.
046 */

053    private static String floatToHex(float f)
054    {
055        //String s = Integer.toHexString(MathUtils.round(f * 255));
056        //if(s.length() < 2) return "0" + s;
057        //else return s;
058        int r = MathUtils.round(f * 255);
059        return String.valueOf(StringKit.hexDigits[r >>> 4 & 15]) + StringKit.hexDigits[r & 15];
060    }
061    @Override
062    public String getMarkup(Color value) {
063        return "[#" + floatToHex(value.r) + floatToHex(value.g) + floatToHex(value.b) + floatToHex(value.a) + "]";
064    }

066    @Override
067    public String closeMarkup() {
068        return "[]";
069    }

071    private final Matcher markupMatcher = Pattern.compile("({=p}[^\\[]+)|(?:\\[({=e}\\[))|(?:\\[#({=h}[0-9A-Fa-f]{6,8})\\])|(?:\\[@\\s*({=q}({=qh}[0-9]*\\.?[0-9]+)\\s+({=qs}[0-9]*\\.?[0-9]+)\\s+({=qv}[0-9]*\\.?[0-9]+)(?:\\s+({=qo}[0-9]*\\.?[0-9]+))?)\\])|(?:\\[({=b}\\*)\\])|(?:\\[({=i}/)\\])|(?:\\[({=U}!)\\])|(?:\\[({=L}=)\\])|(?:\\[({=u},)\\])|(?:\\[({=n}[^\\]]+?)\\])|(?:\\[({=r}\\]))").matcher();
072    private static final char BOLD = '\u4000', ITALIC = '\u8000', REGULAR = '\0';
073    private final StringBuilder sb = new StringBuilder(128);

075    /**
076     * Takes a CharSequence (such as a String or StringBuilder) that contains the markup this class understands, and
077     * produces an IColoredString (of Color) with the color markup tags used to mark text in colors in the resulting
078     * IColoredString, and any style markup tags used to mark sections of text as bold or italic for a TextFamily to
079     * render (normal TextCellFactory rendering may show bold/italic text as gibberish).
080     * @param markupString a String or other CharSequence containing color and/or style markup tags
081     * @return an IColoredString (of Color) with all the markup applied and removed from the text after applying
082     */
083    public IColoredString<Color> colorString(final CharSequence markupString)
084    {
085        markupMatcher.setTarget(markupString);
086        IColoredString<Color> cs = new IColoredString.Impl<>();
087        Color current = Color.WHITE;
088        char mod = REGULAR;
089        int casing = 0;
090        while (markupMatcher.find())
091        {
092            if(markupMatcher.getGroup("p", sb))
093            {
094                switch (casing) {
095                    case -1:
096                        for (int i = 0; i < sb.length(); i++) {
097                            sb.setCharAt(i, Character.toLowerCase((char) (sb.charAt(i) | mod)));
098                        }
099                        break;
100                    case 1:
101                        for (int i = 0; i < sb.length(); i++) {
102                            sb.setCharAt(i, Character.toUpperCase((char) (sb.charAt(i) | mod)));
103                        }
104                        break;
105                    default:
106                        for (int i = 0; i < sb.length(); i++) {
107                            sb.setCharAt(i, (char) (sb.charAt(i) | mod));
108                        }
109                        break;
110                }
111                cs.append(sb.toString(), current);
112            }
113            else if(markupMatcher.isCaptured("e"))
114            {
115                cs.append((char) ('[' | mod), current);
116            }
117            else if(markupMatcher.isCaptured("r"))
118            {
119                current = Color.WHITE;
120                mod = REGULAR;
121                casing = 0;
122            }
123            else if(markupMatcher.isCaptured("u"))
124            {
125                mod = REGULAR;
126                casing = 0;
127            }
128            else if(markupMatcher.getGroup("h", sb))
129            {
130                if(sb.length() == 6)
131                    current = DefaultResources.getSCC().get(StringKit.intFromHex(sb) << 8 | 0xFF);
132                else
133                    current = DefaultResources.getSCC().get(StringKit.intFromHex(sb));
134            }
135            else if(markupMatcher.getGroup("n", sb))
136            {
137                current = Colors.get(sb.toString());
138            }
139            else if(markupMatcher.isCaptured("q"))
140            {
141                current = markupMatcher.isCaptured("qo")
142                        ? DefaultResources.getSCC().getHSV(
143                        Float.parseFloat(markupMatcher.group("qh")),
144                        Float.parseFloat(markupMatcher.group("qs")),
145                        Float.parseFloat(markupMatcher.group("qv")),
146                        Float.parseFloat(markupMatcher.group("qo")))
147                        :  DefaultResources.getSCC().getHSV(
148                        Float.parseFloat(markupMatcher.group("qh")),
149                        Float.parseFloat(markupMatcher.group("qs")),
150                        Float.parseFloat(markupMatcher.group("qv")));
151            }
152            else if(markupMatcher.isCaptured("b"))
153            {
154                mod ^= BOLD;
155            }
156            else if(markupMatcher.isCaptured("i"))
157            {
158                mod ^= ITALIC;
159            }
160            else if(markupMatcher.isCaptured("U"))
161            {
162                casing = casing == 1 ? 0 : 1;
163            }
164            else if(markupMatcher.isCaptured("L"))
165            {
166                casing = casing == -1 ? 0 : -1;
167            }
168            sb.setLength(0);
169        }
170        return cs;
171    }

260    /**
261     * Similar to {@link #colorString(CharSequence)}, but leaves color tags as they are and only uses case and style
262     * tags, like {@code [*]} for bold, {@code [/]} for italic, {@code [!]} for ALL CAPS and {@code [=]} for lower-case.
263     * The StringBuilder this returns can be converted to a String or used directly for further modification, but styles
264     * will probably only render correctly if using a TextFamily. You should be aware that if {@code []} is used to
265     * reset both color and style in the given markupString, then only the style will be reset here but the {@code []}
266     * will be removed, which may affect colors if the result is later given to something that expects color markup to
267     * also have been closed. A solution for this is to use {@code [,]} to reset only styles, and to avoid using
268     * {@code []} to change color by explicitly using {@code [WHITE]} to set the text color back to pure white. This way
269     * is only necessary if you have color markup in the markupString you pass to this method.
270     * @param markupString a String containing color markup (which is left as-is) and/or case/style markup (which is used)
271     * @return a StringBuilder based on markupString that has the case/style markup applied and other markup left there
272     */
273    public StringBuilder styleString(final CharSequence markupString)
274    {
275        markupMatcher.setTarget(markupString);
276        char mod = REGULAR;
277        int casing = 0;
278        StringBuilder fsb = new StringBuilder(markupString.length());
279        while (markupMatcher.find())
280        {
281            if(markupMatcher.getGroup("p", sb))
282            {
283                switch (casing) {
284                    case -1:
285                        for (int i = 0; i < sb.length(); i++) {
286                            sb.setCharAt(i, Character.toLowerCase((char) (sb.charAt(i) | mod)));
287                        }
288                        break;
289                    case 1:
290                        for (int i = 0; i < sb.length(); i++) {
291                            sb.setCharAt(i, Character.toUpperCase((char) (sb.charAt(i) | mod)));
292                        }
293                        break;
294                    default:
295                        for (int i = 0; i < sb.length(); i++) {
296                            sb.setCharAt(i, (char) (sb.charAt(i) | mod));
297                        }
298                        break;
299                }
300                fsb.append(sb);
301            }
302            else if(markupMatcher.isCaptured("e"))
303            {
304                fsb.append((char) ('[' | mod));
305            }
306            else if(markupMatcher.isCaptured("r") || markupMatcher.isCaptured("u"))
307            {
308                mod = REGULAR;
309            }
310            else if(markupMatcher.isCaptured("h") || markupMatcher.isCaptured("n") || markupMatcher.isCaptured("q"))
311            {
312                markupMatcher.getGroup(0, fsb);
313            }
314            else if(markupMatcher.isCaptured("b"))
315            {
316                mod ^= BOLD;
317            }
318            else if(markupMatcher.isCaptured("i"))
319            {
320                mod ^= ITALIC;
321            }
322            else if(markupMatcher.isCaptured("U"))
323            {
324                casing = casing == 1 ? 0 : 1;
325            }
326            else if(markupMatcher.isCaptured("L"))
327            {
328                casing = casing == -1 ? 0 : -1;
329            }
330            sb.setLength(0);
331
332        }
333        return fsb;
334    }

336    /**
337     * Directly styles one char value, toggling its bold state if {@code bold} is true, and toggling its italic state if
338     * {@code italic} is true. When either or both of bold or italic are enabled in a char, that char will be at some
339     * unrelated section of Unicode, and the char will be hard to identify unless you call {@link #unstyleChar(char)} on
340     * it or render it with a TextFamily (which will render bold and italic chars as the correct glyph and style).
341     * @param basis the char to potentially make bold or italic; can be bold or italic already
342     * @param bold if true, the bold-ness of basis will be toggled; if false, does not change bold data
343     * @param italic if true, the italic-ness of basis will be toggled; if false, does not change italic data
344     * @return a char that will look like basis when rendered appropriately but with bold and italic settings applied
345     */
346    public char styleChar(char basis, boolean bold, boolean italic)
347    {
348        return (char) (basis ^ (bold ? '\u4000' : '\0') ^ (italic ? '\u8000' : '\0'));
349    }

351    /**
352     * If given a char that has added style information for bold/italic modes (colors aren't stored in char data), this
353     * removes the bold/italic data and makes the char what it will be rendered as in a normal font (not a special
354     * TextFamily, which renders later sections of Unicode as bold and/or italic).
355     * @param styled a char that should have any bold or italic data set to normal (non-bold, non-italic)
356     * @return a char that will not be bold or italic
357     */
358    public char unstyleChar(char styled)
359    {
360        return (char) (styled & '\u3fff');
361    }
