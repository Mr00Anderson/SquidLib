022/** PNG-8 encoder with compression. An instance can be reused to encode multiple PNGs with minimal allocation.
023 * You can configure the target palette and how this can dither colors via the {@link #palette} field, which is a
024 * {@link PaletteReducer} object that is allowed to be null and can be reused. The methods
025 * {@link PaletteReducer#exact(Color[])} or {@link PaletteReducer#analyze(Pixmap)} can be used to make the target
026 * palette match a specific set of colors or the colors in an existing image. You can use
027 * {@link PaletteReducer#setDitherStrength(float)} to reduce (or increase) dither strength; the dithering algorithm used
028 * here is a modified version of the algorithm presented in "Simple gradient-based error-diffusion method" by Xaingyu Y.
029 * Hu in the Journal of Electronic Imaging, 2016. This algorithm uses pseudo-randomly-generated noise (it is
030 * deterministic, and is seeded using the color information) to adjust Floyd-Steinberg dithering. It yields
031 * surprisingly non-random-looking dithers, but still manages to break up artificial patterns most of the time.
032 * Note that much of the time you will want to use {@link #writePrecisely(FileHandle, Pixmap, boolean)} instead of
033 * {@link #write(FileHandle, Pixmap, boolean, boolean)}, since writePrecisely will attempt to reproduce the exact colors
034 * if there are 256 colors or less in the Pixmap, and will automatically change to calling write() if there are more
035 * than 256 colors.

300    /**
301     * Attempts to write the given Pixmap exactly as a PNG-8 image to output; this attempt will only succeed if there
302     * are no more than 256 colors in the Pixmap (treating all partially transparent colors as fully transparent).
303     * If the attempt fails, this falls back to calling {@link #write(OutputStream, Pixmap, boolean, boolean)}, which
304     * can dither the image to use no more than 255 colors (plus fully transparent) based on ditherFallback and will
305     * always analyze the Pixmap to get an accurate-enough palette, using the given threshold for analysis (which is
306     * typically between 1 and 1000, and most often near 200-400). All other write() methods in this class will
307     * reduce the color depth somewhat, but as long as the color count stays at 256 or less, this will keep the
308     * non-alpha components of colors exactly.
309     * @param output an OutputStream that will not be closed
310     * @param pixmap a Pixmap to write to the given output stream
311     * @param ditherFallback if the Pixmap contains too many colors, this determines whether it will dither the output
312     * @param threshold the analysis threshold to use if there are too many colors (min 0, practical max is over 100000)
313     * @throws IOException if OutputStream things fail for any reason
314     */
315    public void writePrecisely(OutputStream output, Pixmap pixmap, boolean ditherFallback, int threshold) throws IOException {
316        IntIntMap colorToIndex = new IntIntMap(256);
317        colorToIndex.put(0, 0);
318        int color;
319        int hasTransparent = 0;
320        final int w = pixmap.getWidth(), h = pixmap.getHeight();
321        for (int y = 0; y < h; y++) {
322            int py = flipY ? (h - y - 1) : y;
323            for (int px = 0; px < w; px++) {
324                color = pixmap.getPixel(px, py);
325                if((color & 0xFE) != 0xFE) {
326                    if(hasTransparent == 0 && colorToIndex.size >= 256)                     
327                    {
328                        write(output, pixmap, true, ditherFallback, threshold);
329                        return;
330                    }
331                    hasTransparent = 1;
332                }
333                else if(!colorToIndex.containsKey(color))
334                {
335                    colorToIndex.put(color, colorToIndex.size & 255);
336                    if(colorToIndex.size == 257 && hasTransparent == 0)
337                    {
338                        colorToIndex.remove(0, 0);
339                    }
340                    if(colorToIndex.size > 256)
341                    {
342                        write(output, pixmap, true, ditherFallback, threshold);
343                        return;
344                    }
345                }
346            }
347        }
348        int[] paletteArray = new int[colorToIndex.size];
349        for(IntIntMap.Entry ent : colorToIndex)
350        {
351            paletteArray[ent.value] = ent.key;
352        }
353        DeflaterOutputStream deflaterOutput = new DeflaterOutputStream(buffer, deflater);
354        DataOutputStream dataOutput = new DataOutputStream(output);
355        dataOutput.write(SIGNATURE);
356
357        buffer.writeInt(IHDR);
358        buffer.writeInt(pixmap.getWidth());
359        buffer.writeInt(pixmap.getHeight());
360        buffer.writeByte(8); // 8 bits per component.
361        buffer.writeByte(COLOR_INDEXED);
362        buffer.writeByte(COMPRESSION_DEFLATE);
363        buffer.writeByte(FILTER_NONE);
364        buffer.writeByte(INTERLACE_NONE);
365        buffer.endChunk(dataOutput);
366
367        buffer.writeInt(PLTE);
368        for (int i = 0; i < paletteArray.length; i++) {
369            int p = paletteArray[i];
370            buffer.write(p>>>24);
371            buffer.write(p>>>16);
372            buffer.write(p>>>8);
373        }
374        buffer.endChunk(dataOutput);
375
376        if(hasTransparent == 1) {
377            buffer.writeInt(TRNS);
378            buffer.write(0);
379            buffer.endChunk(dataOutput);
380        }
381        buffer.writeInt(IDAT);
382        deflater.reset();
383
384        int lineLen = pixmap.getWidth();
385        byte[] lineOut, curLine, prevLine;
386        if (lineOutBytes == null) {
387            lineOut = (lineOutBytes = new ByteArray(lineLen)).items;
388            curLine = (curLineBytes = new ByteArray(lineLen)).items;
389            prevLine = (prevLineBytes = new ByteArray(lineLen)).items;
390        } else {
391            lineOut = lineOutBytes.ensureCapacity(lineLen);
392            curLine = curLineBytes.ensureCapacity(lineLen);
393            prevLine = prevLineBytes.ensureCapacity(lineLen);
394            for (int i = 0, n = lastLineLen; i < n; i++)
395            {
396                prevLine[i] = 0;
397            }
398        }
399
400        lastLineLen = lineLen;
401
402        for (int y = 0; y < h; y++) {
403            int py = flipY ? (h - y - 1) : y;
404            for (int px = 0; px < w; px++) {
405                color = pixmap.getPixel(px, py);
406                curLine[px] = (byte) colorToIndex.get(color, 0);
407            }
408
409            lineOut[0] = (byte)(curLine[0] - prevLine[0]);
410
411            //Paeth
412            for (int x = 1; x < lineLen; x++) {
413                int a = curLine[x - 1] & 0xff;
414                int b = prevLine[x] & 0xff;
415                int c = prevLine[x - 1] & 0xff;
416                int p = a + b - c;
417                int pa = p - a;
418                if (pa < 0) pa = -pa;
419                int pb = p - b;
420                if (pb < 0) pb = -pb;
421                int pc = p - c;
422                if (pc < 0) pc = -pc;
423                if (pa <= pb && pa <= pc)
424                    c = a;
425                else if (pb <= pc)
426                    c = b;
427                lineOut[x] = (byte)(curLine[x] - c);
428            }
429
430            deflaterOutput.write(PAETH);
431            deflaterOutput.write(lineOut, 0, lineLen);
432
433            byte[] temp = curLine;
434            curLine = prevLine;
435            prevLine = temp;
436        }
437        deflaterOutput.finish();
438        buffer.endChunk(dataOutput);
439
440        buffer.writeInt(IEND);
441        buffer.endChunk(dataOutput);
442
443        output.flush();
444
445    }

447    private void writeSolid (OutputStream output, Pixmap pixmap) throws IOException{
448        final int[] paletteArray = palette.paletteArray;
449        final byte[] paletteMapping = palette.paletteMapping;
450
451        DeflaterOutputStream deflaterOutput = new DeflaterOutputStream(buffer, deflater);
452        DataOutputStream dataOutput = new DataOutputStream(output);
453        dataOutput.write(SIGNATURE);
454
455        buffer.writeInt(IHDR);
456        buffer.writeInt(pixmap.getWidth());
457        buffer.writeInt(pixmap.getHeight());
458        buffer.writeByte(8); // 8 bits per component.
459        buffer.writeByte(COLOR_INDEXED);
460        buffer.writeByte(COMPRESSION_DEFLATE);
461        buffer.writeByte(FILTER_NONE);
462        buffer.writeByte(INTERLACE_NONE);
463        buffer.endChunk(dataOutput);
464
465        buffer.writeInt(PLTE);
466        for (int i = 0; i < paletteArray.length; i++) {
467            int p = paletteArray[i];
468            buffer.write(p>>>24);
469            buffer.write(p>>>16);
470            buffer.write(p>>>8);
471        }
472        buffer.endChunk(dataOutput);
473
474        boolean hasTransparent = false;
475        if(paletteArray[0] == 0) {
476            hasTransparent = true;
477            buffer.writeInt(TRNS);
478            buffer.write(0);
479            buffer.endChunk(dataOutput);
480        }
481        buffer.writeInt(IDAT);
482        deflater.reset();
483
484        int lineLen = pixmap.getWidth();
485        byte[] lineOut, curLine, prevLine;
486        if (lineOutBytes == null) {
487            lineOut = (lineOutBytes = new ByteArray(lineLen)).items;
488            curLine = (curLineBytes = new ByteArray(lineLen)).items;
489            prevLine = (prevLineBytes = new ByteArray(lineLen)).items;
490        } else {
491            lineOut = lineOutBytes.ensureCapacity(lineLen);
492            curLine = curLineBytes.ensureCapacity(lineLen);
493            prevLine = prevLineBytes.ensureCapacity(lineLen);
494            for (int i = 0, n = lastLineLen; i < n; i++)
495            {
496                prevLine[i] = 0;
497            }
498        }
499
500        lastLineLen = lineLen;
501
502        int color;
503        final int w = pixmap.getWidth(), h = pixmap.getHeight();
504        for (int y = 0; y < h; y++) {
505            int py = flipY ? (h - y - 1) : y;
506            for (int px = 0; px < w; px++) {
507                color = pixmap.getPixel(px, py);
508                if ((color & 0x80) == 0 && hasTransparent)
509                    curLine[px] = 0;
510                else {
511                    int rr = ((color >>> 24)       );
512                    int gg = ((color >>> 16) & 0xFF);
513                    int bb = ((color >>> 8)  & 0xFF);
514                    curLine[px] = paletteMapping[((rr << 7) & 0x7C00)
515                                    | ((gg << 2) & 0x3E0)
516                                    | ((bb >>> 3))];
517                }
518            }
519
520            lineOut[0] = (byte)(curLine[0] - prevLine[0]);
521
522            //Paeth
523            for (int x = 1; x < lineLen; x++) {
524                int a = curLine[x - 1] & 0xff;
525                int b = prevLine[x] & 0xff;
526                int c = prevLine[x - 1] & 0xff;
527                int p = a + b - c;
528                int pa = p - a;
529                if (pa < 0) pa = -pa;
530                int pb = p - b;
531                if (pb < 0) pb = -pb;
532                int pc = p - c;
533                if (pc < 0) pc = -pc;
534                if (pa <= pb && pa <= pc)
535                    c = a;
536                else if (pb <= pc)
537                    c = b;
538                lineOut[x] = (byte)(curLine[x] - c);
539            }
540
541            deflaterOutput.write(PAETH);
542            deflaterOutput.write(lineOut, 0, lineLen);
543
544            byte[] temp = curLine;
545            curLine = prevLine;
546            prevLine = temp;
547        }
548        deflaterOutput.finish();
549        buffer.endChunk(dataOutput);
550
551        buffer.writeInt(IEND);
552        buffer.endChunk(dataOutput);
553
554        output.flush();
555    }

557    private void writeDithered (OutputStream output, Pixmap pixmap) throws IOException{
558        DeflaterOutputStream deflaterOutput = new DeflaterOutputStream(buffer, deflater);
559        final int[] paletteArray = palette.paletteArray;
560        final byte[] paletteMapping = palette.paletteMapping;
561        
562        DataOutputStream dataOutput = new DataOutputStream(output);
563        dataOutput.write(SIGNATURE);
564
565        buffer.writeInt(IHDR);
566        buffer.writeInt(pixmap.getWidth());
567        buffer.writeInt(pixmap.getHeight());
568        buffer.writeByte(8); // 8 bits per component.
569        buffer.writeByte(COLOR_INDEXED);
570        buffer.writeByte(COMPRESSION_DEFLATE);
571        buffer.writeByte(FILTER_NONE);
572        buffer.writeByte(INTERLACE_NONE);
573        buffer.endChunk(dataOutput);
574
575        buffer.writeInt(PLTE);
576        for (int i = 0; i < paletteArray.length; i++) {
577            int p = paletteArray[i];
578            buffer.write(p>>>24);
579            buffer.write(p>>>16);
580            buffer.write(p>>>8);
581        }
582        buffer.endChunk(dataOutput);
583
584        boolean hasTransparent = false;
585        if(paletteArray[0] == 0) {
586            hasTransparent = true;
587            buffer.writeInt(TRNS);
588            buffer.write(0);
589            buffer.endChunk(dataOutput);
590        }
591        buffer.writeInt(IDAT);
592        deflater.reset();
593
594        final int w = pixmap.getWidth(), h = pixmap.getHeight();
595        byte[] lineOut, curLine, prevLine;
596        if (lineOutBytes == null) {
597            lineOut = (lineOutBytes = new ByteArray(w)).items;
598            curLine = (curLineBytes = new ByteArray(w)).items;
599            prevLine = (prevLineBytes = new ByteArray(w)).items;
600        } else {
601            lineOut = lineOutBytes.ensureCapacity(w);
602            curLine = curLineBytes.ensureCapacity(w);
603            prevLine = prevLineBytes.ensureCapacity(w);
604            for (int i = 0, n = lastLineLen; i < n; i++)
605            {
606                prevLine[i] = 0;
607            }
608        }
609
610        lastLineLen = w;
611
612        int color;
613        float adj, str = palette.ditherStrength;
614        for (int y = 0; y < h; y++) {
615            int py = flipY ? (h - y - 1) : y;
616            for (int px = 0; px < w; px++) {
617                color = pixmap.getPixel(px, py) & 0xF8F8F880;
618                if ((color & 0x80) == 0 && hasTransparent)
619                    curLine[px] = 0;
620                else {
621                    adj = (((px * 0xC13FA9A902A6328FL + py * 0x91E10DA5C79E7B1DL >>> 40) * 0x1p-26f - 0x1p-3f) * str);
622                    color |= (color >>> 5 & 0x07070700) | 0xFE;
623                    int rr = MathUtils.clamp((int) (((color >>> 24)       ) * (1f - adj)), 0, 0xFF);
624                    int gg = MathUtils.clamp((int) (((color >>> 16) & 0xFF) * (1f + adj)), 0, 0xFF);
625                    int bb = MathUtils.clamp((int) (((color >>> 8)  & 0xFF) * (1f + adj)), 0, 0xFF);
626                    curLine[px] = paletteMapping[((rr << 7) & 0x7C00)
627                                    | ((gg << 2) & 0x3E0)
628                                    | ((bb >>> 3))];
629                }
630            }
631
632            lineOut[0] = (byte)(curLine[0] - prevLine[0]);
633
634            //Paeth
635            for (int x = 1; x < w; x++) {
636                int a = curLine[x - 1] & 0xff;
637                int b = prevLine[x] & 0xff;
638                int c = prevLine[x - 1] & 0xff;
639                int p = a + b - c;
640                int pa = p - a;
641                if (pa < 0) pa = -pa;
642                int pb = p - b;
643                if (pb < 0) pb = -pb;
644                int pc = p - c;
645                if (pc < 0) pc = -pc;
646                if (pa <= pb && pa <= pc)
647                    c = a;
648                else if (pb <= pc)
649                    c = b;
650                lineOut[x] = (byte)(curLine[x] - c);
651            }
652
653            deflaterOutput.write(PAETH);
654            deflaterOutput.write(lineOut, 0, w);
655
656            byte[] temp = curLine;
657            curLine = prevLine;
658            prevLine = temp;
659        }
660        deflaterOutput.finish();
661        buffer.endChunk(dataOutput);
662
663        buffer.writeInt(IEND);
664        buffer.endChunk(dataOutput);
665
666        output.flush();
667    }

674    static class ChunkBuffer extends DataOutputStream {
675        final ByteArrayOutputStream buffer;
676        final CRC32 crc;
677
678        ChunkBuffer (int initialSize) {
679            this(new ByteArrayOutputStream(initialSize), new CRC32());
680        }
681
682        private ChunkBuffer (ByteArrayOutputStream buffer, CRC32 crc) {
683            super(new CheckedOutputStream(buffer, crc));
684            this.buffer = buffer;
685            this.crc = crc;
686        }
687
688        public void endChunk (DataOutputStream target) throws IOException {
689            flush();
690            target.writeInt(buffer.size() - 4);
691            buffer.writeTo(target);
692            target.writeInt((int)crc.getValue());
693            buffer.reset();
694            crc.reset();
695        }
696    }
