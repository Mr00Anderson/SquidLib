129    /**
130     * Color difference metric; returns large numbers even for smallish differences.
131     * If this returns 250 or more, the colors may be perceptibly different; 500 or more almost guarantees it.
132     *
133     * @param color1 an RGBA8888 color as an int
134     * @param color2 an RGBA8888 color as an int
135     * @return the difference between the given colors, as a positive int
136     */
137    public static int difference(final int color1, final int color2) {
138        int rmean = ((color1 >>> 24) + (color2 >>> 24));
139        int r = (color1 >>> 24) - (color2 >>> 24);
140        int g = (color1 >>> 16 & 0xFF) - (color2 >>> 16 & 0xFF) << 1;
141        int b = (color1 >>> 8 & 0xFF) - (color2 >>> 8 & 0xFF);
142//        return (((512 + rmean) * r * r) >> 8) + g * g + (((767 - rmean) * b * b) >> 8);
143        return (((1024 + rmean) * r * r) >> 9) + g * g + (((767 - rmean) * b * b) >> 8);
144    }

239     * Gets a pseudo-random float between -0.65625f and 0.65625f, determined by the upper 23 bits of seed.
240     * This currently uses a uniform distribution for its output, but earlier versions intentionally used a non-uniform
241     * one; a non-uniform distribution can sometimes work well but is very dependent on how error propagates through a
242     * dithered image, and in bad cases can produce bands of bright mistakenly-error-adjusted colors.
243     * @param seed any int, but only the most-significant 23 bits will be used
244     * @return a float between -0.65625f and 0.65625f, with fairly uniform distribution as long as seed is uniform
245     */
246    public static float randomXi(int seed)
247    {
248        return ((seed >> 9) * 0x1.5p-23f);
249//        return NumberUtils.intBitsToFloat((seed & 0x7FFFFF & ((seed >>> 11 & 0x400000)|0x3FFFFF)) | 0x3f800000) - 1.4f;

251    }

253    /**
254     * Builds the palette information this PNG8 stores from the RGBA8888 ints in {@code rgbaPalette}, up to 256 colors.
255     * Alpha is not preserved except for the first item in rgbaPalette, and only if it is {@code 0} (fully transparent
256     * black); otherwise all items are treated as opaque. If rgbaPalette is null, empty, or only has one color, then
257     * this defaults to DawnBringer's Aurora palette with 256 hand-chosen colors (including transparent).
258     *
259     * @param rgbaPalette an array of RGBA8888 ints; all will be used up to 256 items or the length of the array
260     */
261    public void exact(int[] rgbaPalette) {
262        if (rgbaPalette == null || rgbaPalette.length < 2) {
263            rgbaPalette = auroraPalette;
264        }
265        Arrays.fill(paletteArray, 0);
266        Arrays.fill(paletteMapping, (byte) 0);
267        final int plen = Math.min(256, rgbaPalette.length);
268        int color, c2;
269        int dist;
270        for (int i = 0; i < plen; i++) {
271            color = rgbaPalette[i];
272            paletteArray[i] = color;
273            paletteMapping[(color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F)] = (byte) i;
274        }
275        int rr, gg, bb;
276        for (int r = 0; r < 32; r++) {
277            rr = (r << 3 | r >>> 2);
278            for (int g = 0; g < 32; g++) {
279                gg = (g << 3 | g >>> 2);
280                for (int b = 0; b < 32; b++) {
281                    c2 = r << 10 | g << 5 | b;
282                    if (paletteMapping[c2] == 0) {
283                        bb = (b << 3 | b >>> 2);
284                        dist = 0x7FFFFFFF;
285                        for (int i = 1; i < plen; i++) {
286                            if (dist > (dist = Math.min(dist, difference(paletteArray[i], rr, gg, bb))))
287                                paletteMapping[c2] = (byte) i;
288                        }
289                    }
290                }
291            }
292        }
293    }

308    /**
309     * Builds the palette information this PNG8 stores from the Color objects in {@code colorPalette}, up to 256 colors.
310     * Alpha is not preserved except for the first item in colorPalette, and only if its r, g, b, and a values are all
311     * 0f (fully transparent black); otherwise all items are treated as opaque. If rgbaPalette is null, empty, only has
312     * one color, or limit is less than 2, then this defaults to DawnBringer's Aurora palette with 256 hand-chosen
313     * colors (including transparent).
314     *
315     * @param colorPalette an array of Color objects; all will be used up to 256 items, limit, or the length of the array
316     * @param limit        a limit on how many Color items to use from colorPalette; useful if colorPalette is from an Array
317     */
318    public void exact(Color[] colorPalette, int limit) {
319        if (colorPalette == null || colorPalette.length < 2 || limit < 2) {
320            exact(auroraPalette);
321            return;
322        }
323        Arrays.fill(paletteArray, 0);
324        Arrays.fill(paletteMapping, (byte) 0);
325        final int plen = Math.min(Math.min(256, colorPalette.length), limit);
326        int color, c2;
327        int dist;
328        for (int i = 0; i < plen; i++) {
329            color = Color.rgba8888(colorPalette[i]);
330            paletteArray[i] = color;
331            paletteMapping[(color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F)] = (byte) i;
332        }
333        int rr, gg, bb;
334        for (int r = 0; r < 32; r++) {
335            rr = (r << 3 | r >>> 2);
336            for (int g = 0; g < 32; g++) {
337                gg = (g << 3 | g >>> 2);
338                for (int b = 0; b < 32; b++) {
339                    c2 = r << 10 | g << 5 | b;
340                    if (paletteMapping[c2] == 0) {
341                        bb = (b << 3 | b >>> 2);
342                        dist = 0x7FFFFFFF;
343                        for (int i = 1; i < plen; i++) {
344                            if (dist > (dist = Math.min(dist, difference(paletteArray[i], rr, gg, bb))))
345                                paletteMapping[c2] = (byte) i;
346                        }
347                    }
348                }
349            }
350        }
351    }
352    /**
353     * Analyzes {@code pixmap} for color count and frequency, building a palette with at most 256 colors if there are
354     * too many colors to store in a PNG-8 palette. If there are 256 or less colors, this uses the exact colors
355     * (although with at most one transparent color, and no alpha for other colors); if there are more than 256 colors
356     * or any colors have 50% or less alpha, it will reserve a palette entry for transparent (even if the image has no
357     * transparency). Because calling {@link #reduce(Pixmap)} (or any of PNG8's write methods) will dither colors that
358     * aren't exact, and dithering works better when the palette can choose colors that are sufficiently different, this
359     * uses a threshold value to determine whether it should permit a less-common color into the palette, and if the
360     * second color is different enough (as measured by {@link #difference(int, int)}) by a value of at least 400, it is
361     * allowed in the palette, otherwise it is kept out for being too similar to existing colors. This doesn't return a
362     * value but instead stores the palette info in this object; a PaletteReducer can be assigned to the
363     * {@link PNG8#palette} field or can be used directly to {@link #reduce(Pixmap)} a Pixmap.
364     *
365     * @param pixmap a Pixmap to analyze, making a palette which can be used by this to {@link #reduce(Pixmap)} or by PNG8
366     */
367    public void analyze(Pixmap pixmap) {
368        analyze(pixmap, 400);
369    }

385     * aren't exact, and dithering works better when the palette can choose colors that are sufficiently different, this
386     * takes a threshold value to determine whether it should permit a less-common color into the palette, and if the
387     * second color is different enough (as measured by {@link #difference(int, int)}) by a value of at least
388     * {@code threshold}, it is allowed in the palette, otherwise it is kept out for being too similar to existing
389     * colors. The threshold is usually between 250 and 1000, and 400 is a good default. This doesn't return a value but
390     * instead stores the palette info in this object; a PaletteReducer can be assigned to the {@link PNG8#palette}
391     * field or can be used directly to {@link #reduce(Pixmap)} a Pixmap.
392     *
393     * @param pixmap    a Pixmap to analyze, making a palette which can be used by this to {@link #reduce(Pixmap)} or by PNG8
394     * @param threshold a minimum color difference as produced by {@link #difference(int, int)}; usually between 250 and 1000, 400 is a good default
395     */
396    public void analyze(Pixmap pixmap, int threshold) {
397        Arrays.fill(paletteArray, 0);
398        Arrays.fill(paletteMapping, (byte) 0);
399        int color;
400        final int width = pixmap.getWidth(), height = pixmap.getHeight();
401        IntIntMap counts = new IntIntMap(256);
402        int hasTransparent = 0;
403        int[] reds = new int[256], greens = new int[256], blues = new int[256];
404        for (int y = 0; y < height; y++) {
405            for (int x = 0; x < width; x++) {
406                color = pixmap.getPixel(x, y);
407                if ((color & 0x80) != 0) {
408                    color |= (color >>> 5 & 0x07070700) | 0xFE;
409                    counts.getAndIncrement(color, 0, 1);
410                } else {
411                    hasTransparent = 1;
412                }
413            }
414        }
415        final int cs = counts.size;
416        if (cs + hasTransparent <= 256) {
417            int i = hasTransparent;
418            for(IntIntMap.Entry e : counts) {
419                color = e.key;
420                paletteArray[i] = color;
421                color = (color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F);
422                paletteMapping[color] = (byte) i;
423                reds[i] = color >>> 10;
424                greens[i] = color >>> 5 & 31;
425                blues[i] = color & 31;
426                i++;
427            }
428        } else // reduce color count
429        {
430            ArrayList<IntIntMap.Entry> es = new ArrayList<>(cs);
431            for(IntIntMap.Entry e : counts)
432            {
433                IntIntMap.Entry e2 = new IntIntMap.Entry();
434                e2.key = e.key;
435                e2.value = e.value;
436                es.add(e2);
437            }
438            Collections.sort(es, entryComparator);
439            int i = 1, c = 0;
440            PER_BEST:
441            for (; i < 256 && c < cs;) {
442                color = es.get(c++).key;
443                for (int j = 1; j < i; j++) {
444                    if (difference(color, paletteArray[j]) < threshold)
445                        continue PER_BEST;
446                }
447                paletteArray[i] = color;
448                color = (color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F);
449                paletteMapping[color] = (byte) i;
450                reds[i] = color >>> 10;
451                greens[i] = color >>> 5 & 31;
452                blues[i] = color & 31;
453                i++;
454            }
455        }
456        int c2, dist;
457        for (int r = 0; r < 32; r++) {
458            for (int g = 0; g < 32; g++) {
459                for (int b = 0; b < 32; b++) {
460                    c2 = r << 10 | g << 5 | b;
461                    if (paletteMapping[c2] == 0) {
462                        dist = 0x7FFFFFFF;
463                        for (int i = 1; i < 256; i++) {
464                            if (dist > (dist = Math.min(dist, difference(reds[i], r, greens[i], g, blues[i], b))))
465                                paletteMapping[c2] = (byte) i;
466                        }
467                    }
468                }
469            }
470        }
471    }

484    /**
485     * Modifies the given Pixmap so it only uses colors present in this PaletteReducer, dithering when it can.
486     * The dithering algorithm here is based on
487     * <a href="http://extremelearning.com.au/unreasonable-effectiveness-of-quasirandom-sequences">one suggested by
488     * Martin Roberts</a> using a generalization of the golden ratio, using the quasi-random value per pixel to get a
489     * multiplier for all channels (instead of a threshold for choosing just black or white). This adds noise but tends
490     * to look better than most other approaches, and is an ordered dither with all the various advantages that offers.
491     * If you want to reduce the colors in a Pixmap based on what it currently contains, call
492     * {@link #analyze(Pixmap)} with {@code pixmap} as its argument, then call this method with the same
493     * Pixmap. You may instead want to use a known palette instead of one computed from a Pixmap;
494     * {@link #exact(int[])} is the tool for that job. If you can compute the PaletteReducer once and reuse it, that
495     * will save some time on repeated calls to reduce(). <a href="https://imgur.com/a/wfdYtJn">A gallery of different
496     * dithering techniques is here</a>; this method is last.
497     * @param pixmap a Pixmap that will be modified in place
498     * @return the given Pixmap, for chaining
499     */
500    public Pixmap reduce (Pixmap pixmap) {
501        boolean hasTransparent = (paletteArray[0] == 0);
502        final int lineLen = pixmap.getWidth(), h = pixmap.getHeight();
503        Pixmap.Blending blending = pixmap.getBlending();
504        pixmap.setBlending(Pixmap.Blending.None);
505        int color, used;
506        float adj;
507        byte paletteIndex;
508        for (int y = 0; y < h; y++) {
509            for (int px = 0; px < lineLen; px++) {
510                color = pixmap.getPixel(px, y) & 0xF8F8F880;
511                if ((color & 0x80) == 0 && hasTransparent)
512                    pixmap.drawPixel(px, y, 0);
513                else {
514                    adj = (((px * 0xC13FA9A902A6328FL + y * 0x91E10DA5C79E7B1DL >>> 40) * 0x1p-26f - 0x1p-3f) * ditherStrength);
515                    color |= (color >>> 5 & 0x07070700) | 0xFE;
516                    int rr = MathUtils.clamp((int) (((color >>> 24)       ) * (1f + adj)), 0, 0xFF);
517                    int gg = MathUtils.clamp((int) (((color >>> 16) & 0xFF) * (1f + adj)), 0, 0xFF);
518                    int bb = MathUtils.clamp((int) (((color >>> 8)  & 0xFF) * (1f + adj)), 0, 0xFF);
519                    paletteIndex =
520                            paletteMapping[((rr << 7) & 0x7C00)
521                                    | ((gg << 2) & 0x3E0)
522                                    | ((bb >>> 3))];
523                    used = paletteArray[paletteIndex & 0xFF];
524                    pixmap.drawPixel(px, y, used);
525                }
526            }
527        }
528        pixmap.setBlending(blending);
529        return pixmap;
530    }

794     * the Journal of Electronic Imaging, 2016. This algorithm uses pseudo-randomly-generated noise to adjust
795     * Floyd-Steinberg dithering, with input for the pseudo-random state obtained by the non-transparent color values as
796     * they are encountered. Very oddly, this tends to produce less random-seeming dither than
797     * {@link #reduceBurkes(Pixmap)}, with this method often returning regular checkerboards where Burkes may produce
798     * splotches of color. If you want to reduce the colors in a Pixmap based on what it currently contains, call
799     * {@link #analyze(Pixmap)} with {@code pixmap} as its argument, then call this method with the same
800     * Pixmap. You may instead want to use a known palette instead of one computed from a Pixmap;

804     * compute the PaletteReducer once and reuse it, that will save some time. This method is probably slower than

806     * generate two random values per non-transparent pixel. The random number "algorithm" this uses isn't very good
807     * because it doesn't have to be good, it should just be fast and avoid clear artifacts; it's similar to one of

867                    eb = curErrorBlue[px];
868                    color |= (color >>> 5 & 0x07070700) | 0xFE;
869                    int rr = MathUtils.clamp(((color >>> 24)       ) + (er), 0, 0xFF);
870                    int gg = MathUtils.clamp(((color >>> 16) & 0xFF) + (eg), 0, 0xFF);
871                    int bb = MathUtils.clamp(((color >>> 8)  & 0xFF) + (eb), 0, 0xFF);
872                    paletteIndex =
873                            paletteMapping[((rr << 7) & 0x7C00)
874                                    | ((gg << 2) & 0x3E0)
875                                    | ((bb >>> 3))];
876                    used = paletteArray[paletteIndex & 0xFF];
877                    pixmap.drawPixel(px, y, used);
878                    rdiff = (color>>>24)-    (used>>>24);
879                    gdiff = (color>>>16&255)-(used>>>16&255);
880                    bdiff = (color>>>8&255)- (used>>>8&255);
881                    state += (color + 0x41C64E6D) ^ color >>> 7;
882                    state = (state << 21 | state >>> 11);
883                    xi1 = randomXi(state);
884                    state ^= (state << 15 | state >>> 17) + 0x9E3779B9;
885                    xi2 = randomXi(state);
886
887//                    state += rdiff ^ rdiff << 9;
888//                    state = (state << 21 | state >>> 11);
889//                    xir1 = randomXi(state);
890//                    state = (state << 21 | state >>> 11);
891//                    xir2 = randomXi(state);
892//                    state += gdiff ^ gdiff << 9;
893//                    state = (state << 21 | state >>> 11);
894//                    xig1 = randomXi(state);
895//                    state = (state << 21 | state >>> 11);
896//                    xig2 = randomXi(state);
897//                    state += bdiff ^ bdiff << 9;
898//                    state = (state << 21 | state >>> 11);
899//                    xib1 = randomXi(state);
900//                    state = (state << 21 | state >>> 11);
901//                    xib2 = randomXi(state);
902                    if(px < lineLen - 1)
903                    {
904                        curErrorRed[px+1]   += rdiff * w7 * (1f + xi1);
905                        curErrorGreen[px+1] += gdiff * w7 * (1f + xi1);
906                        curErrorBlue[px+1]  += bdiff * w7 * (1f + xi1);
907                    }
908                    if(ny < h)
909                    {
910                        if(px > 0)
911                        {
912                            nextErrorRed[px-1]   += rdiff * w3 * (1f + xi2);
913                            nextErrorGreen[px-1] += gdiff * w3 * (1f + xi2);
914                            nextErrorBlue[px-1]  += bdiff * w3 * (1f + xi2);
915                        }
916                        if(px < lineLen - 1)
917                        {
918                            nextErrorRed[px+1]   += rdiff * w1 * (1f - xi2);
919                            nextErrorGreen[px+1] += gdiff * w1 * (1f - xi2);
920                            nextErrorBlue[px+1]  += bdiff * w1 * (1f - xi2);
921                        }
922                        nextErrorRed[px]   += rdiff * w5 * (1f - xi1);
923                        nextErrorGreen[px] += gdiff * w5 * (1f - xi1);
924                        nextErrorBlue[px]  += bdiff * w5 * (1f - xi1);
925                    }
926                }
927            }
928
929        }
930        pixmap.setBlending(blending);
931        return pixmap;
932    }

962     * palette this holds. Both the parameter and the returned color are RGBA8888 ints.
963     * @param color an RGBA8888 int that represents a color this should try to find a similar color for in its palette
964     * @return an RGBA8888 int representing a color from this palette, or 0 if color is mostly transparent
965     * (0 is often but not always in the palette)
966     */
967    public int reduceSingle(int color)
968    {
969        if((color & 0x80) == 0) // less visible than half-transparent
970            return 0; // transparent
971        return paletteArray[paletteMapping[
972                (color >>> 17 & 0x7C00) 
973                        | (color >>> 14 & 0x3E0)
974                        | (color >>> 11 & 0x1F)] & 0xFF];

992                        | (color >>> 6 & 0x3E0)
993                        | (color >>> 19 & 0x1F)] & 0xFF] & 0xFFFFFFFE);

997    /**
998     * Modifies {@code color} so its RGB values will match the closest color in this PaletteReducer's palette. If color
999     * has {@link Color#a} less than 0.5f, this will simply set color to be fully transparent, with rgba all 0.
1000     * @param color a libGDX Color that will be modified in-place; do not use a Color constant, use {@link Color#cpy()}
1001     *              or a temporary Color
1002     * @return color, after modifications.
1003     */
1004    public Color reduceInPlace(Color color)
1005    {
1006        if(color.a < 0.5f)
1007            return color.set(0);
1008        return color.set(paletteArray[paletteMapping[
1009                ((int) (color.r * 0x1f.8p+10) & 0x7C00)
1010                        | ((int) (color.g * 0x1f.8p+5) & 0x3E0)
1011                        | ((int) (color.r * 0x1f.8p+0))] & 0xFF]);
1012    }
