130     * If this returns 250 or more, the colors may be perceptibly different; 500 or more almost guarantees it.
131     *
132     * @param color1 an RGBA8888 color as an int
133     * @param color2 an RGBA8888 color as an int
134     * @return the difference between the given colors, as a positive int
135     */
136    public static int difference(final int color1, final int color2) {
137        int rmean = ((color1 >>> 24) + (color2 >>> 24));
138        int r = (color1 >>> 24) - (color2 >>> 24);
139        int g = (color1 >>> 16 & 0xFF) - (color2 >>> 16 & 0xFF) << 1;
140        int b = (color1 >>> 8 & 0xFF) - (color2 >>> 8 & 0xFF);
141//        return (((512 + rmean) * r * r) >> 8) + g * g + (((767 - rmean) * b * b) >> 8);
142        return (((1024 + rmean) * r * r) >> 9) + g * g + (((767 - rmean) * b * b) >> 8);
143    }

252    /**
253     * Builds the palette information this PNG8 stores from the RGBA8888 ints in {@code rgbaPalette}, up to 256 colors.
254     * Alpha is not preserved except for the first item in rgbaPalette, and only if it is {@code 0} (fully transparent
255     * black); otherwise all items are treated as opaque. If rgbaPalette is null, empty, or only has one color, then
256     * this defaults to DawnBringer's Aurora palette with 256 hand-chosen colors (including transparent).
257     *
258     * @param rgbaPalette an array of RGBA8888 ints; all will be used up to 256 items or the length of the array
259     */
260    public void exact(int[] rgbaPalette) {
261        if (rgbaPalette == null || rgbaPalette.length < 2) {
262            rgbaPalette = auroraPalette;
263        }
264        Arrays.fill(paletteArray, 0);
265        Arrays.fill(paletteMapping, (byte) 0);
266        final int plen = Math.min(256, rgbaPalette.length);
267        int color, c2;
268        int dist;
269        for (int i = 0; i < plen; i++) {
270            color = rgbaPalette[i];
271            paletteArray[i] = color;
272            paletteMapping[(color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F)] = (byte) i;
273        }
274        int rr, gg, bb;
275        for (int r = 0; r < 32; r++) {
276            rr = (r << 3 | r >>> 2);
277            for (int g = 0; g < 32; g++) {
278                gg = (g << 3 | g >>> 2);
279                for (int b = 0; b < 32; b++) {
280                    c2 = r << 10 | g << 5 | b;
281                    if (paletteMapping[c2] == 0) {
282                        bb = (b << 3 | b >>> 2);
283                        dist = 0x7FFFFFFF;
284                        for (int i = 1; i < plen; i++) {
285                            if (dist > (dist = Math.min(dist, difference(paletteArray[i], rr, gg, bb))))
286                                paletteMapping[c2] = (byte) i;
287                        }
288                    }
289                }
290            }
291        }
292    }

328            color = Color.rgba8888(colorPalette[i]);
329            paletteArray[i] = color;
330            paletteMapping[(color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F)] = (byte) i;
331        }
332        int rr, gg, bb;
333        for (int r = 0; r < 32; r++) {
334            rr = (r << 3 | r >>> 2);
335            for (int g = 0; g < 32; g++) {
336                gg = (g << 3 | g >>> 2);
337                for (int b = 0; b < 32; b++) {
338                    c2 = r << 10 | g << 5 | b;

351    /**
352     * Analyzes {@code pixmap} for color count and frequency, building a palette with at most 256 colors if there are
353     * too many colors to store in a PNG-8 palette. If there are 256 or less colors, this uses the exact colors
354     * (although with at most one transparent color, and no alpha for other colors); if there are more than 256 colors
355     * or any colors have 50% or less alpha, it will reserve a palette entry for transparent (even if the image has no
356     * transparency). Because calling {@link #reduce(Pixmap)} (or any of PNG8's write methods) will dither colors that
357     * aren't exact, and dithering works better when the palette can choose colors that are sufficiently different, this
358     * uses a threshold value to determine whether it should permit a less-common color into the palette, and if the
359     * second color is different enough (as measured by {@link #difference(int, int)}) by a value of at least 400, it is
360     * allowed in the palette, otherwise it is kept out for being too similar to existing colors. This doesn't return a
361     * value but instead stores the palette info in this object; a PaletteReducer can be assigned to the
362     * {@link PNG8#palette} field or can be used directly to {@link #reduce(Pixmap)} a Pixmap.
363     *
364     * @param pixmap a Pixmap to analyze, making a palette which can be used by this to {@link #reduce(Pixmap)} or by PNG8
365     */
366    public void analyze(Pixmap pixmap) {
367        analyze(pixmap, 400);
368    }

386     * second color is different enough (as measured by {@link #difference(int, int)}) by a value of at least
387     * {@code threshold}, it is allowed in the palette, otherwise it is kept out for being too similar to existing
388     * colors. The threshold is usually between 250 and 1000, and 400 is a good default. This doesn't return a value but
389     * instead stores the palette info in this object; a PaletteReducer can be assigned to the {@link PNG8#palette}
390     * field or can be used directly to {@link #reduce(Pixmap)} a Pixmap.
391     *
392     * @param pixmap    a Pixmap to analyze, making a palette which can be used by this to {@link #reduce(Pixmap)} or by PNG8
393     * @param threshold a minimum color difference as produced by {@link #difference(int, int)}; usually between 250 and 1000, 400 is a good default
394     */
395    public void analyze(Pixmap pixmap, int threshold) {
396        Arrays.fill(paletteArray, 0);
397        Arrays.fill(paletteMapping, (byte) 0);
398        int color;
399        final int width = pixmap.getWidth(), height = pixmap.getHeight();
400        IntIntMap counts = new IntIntMap(256);
401        int hasTransparent = 0;
402        int[] reds = new int[256], greens = new int[256], blues = new int[256];
403        for (int y = 0; y < height; y++) {
404            for (int x = 0; x < width; x++) {
405                color = pixmap.getPixel(x, y);
406                if ((color & 0x80) != 0) {
407                    color |= (color >>> 5 & 0x07070700) | 0xFE;
408                    counts.getAndIncrement(color, 0, 1);
409                } else {
410                    hasTransparent = 1;
411                }
412            }
413        }
414        final int cs = counts.size;
415        if (cs + hasTransparent <= 256) {
416            int i = hasTransparent;
417            for(IntIntMap.Entry e : counts) {
418                color = e.key;
419                paletteArray[i] = color;
420                color = (color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F);
421                paletteMapping[color] = (byte) i;
422                reds[i] = color >>> 10;
423                greens[i] = color >>> 5 & 31;
424                blues[i] = color & 31;
425                i++;
426            }
427        } else // reduce color count
428        {
429            ArrayList<IntIntMap.Entry> es = new ArrayList<>(cs);
430            for(IntIntMap.Entry e : counts)
431            {
432                IntIntMap.Entry e2 = new IntIntMap.Entry();
433                e2.key = e.key;
434                e2.value = e.value;
435                es.add(e2);
436            }
437            Collections.sort(es, entryComparator);
438            int i = 1, c = 0;
439            PER_BEST:
440            for (; i < 256 && c < cs;) {
441                color = es.get(c++).key;
442                for (int j = 1; j < i; j++) {
443                    if (difference(color, paletteArray[j]) < threshold)
444                        continue PER_BEST;
445                }
446                paletteArray[i] = color;
447                color = (color >>> 17 & 0x7C00) | (color >>> 14 & 0x3E0) | (color >>> 11 & 0x1F);
448                paletteMapping[color] = (byte) i;
449                reds[i] = color >>> 10;
450                greens[i] = color >>> 5 & 31;
451                blues[i] = color & 31;
452                i++;
453            }
454        }
455        int c2, dist;
456        for (int r = 0; r < 32; r++) {
457            for (int g = 0; g < 32; g++) {
458                for (int b = 0; b < 32; b++) {
459                    c2 = r << 10 | g << 5 | b;
460                    if (paletteMapping[c2] == 0) {
461                        dist = 0x7FFFFFFF;
462                        for (int i = 1; i < 256; i++) {
463                            if (dist > (dist = Math.min(dist, difference(reds[i], r, greens[i], g, blues[i], b))))
464                                paletteMapping[c2] = (byte) i;
465                        }
466                    }
467                }
468            }
469        }
470    }

936     * Modifies the given Pixmap so it only uses colors present in this PaletteReducer, dithering when it can using the

938     * contains, call {@link #analyze(Pixmap)} with {@code pixmap} as its argument, then call this method with the same

943     * {@link #reduceWithNoise(Pixmap)} but tends to be less splotchy. It might not be incredibly fast because of the
944     * extra calculations it has to do for dithering, but if you can compute the PaletteReducer once and reuse it, that
945     * will save some time. This method is probably about the same speed as {@link #reduceBurkes(Pixmap)}.

971            }

973        }

975        pixmap.setBlending(Pixmap.Blending.None);
976        int color, used, rdiff, gdiff, bdiff;
977        byte er, eg, eb, paletteIndex;
978        float w1 = ditherStrength * 0.125f, w3 = w1 * 3f, w5 = w1 * 5f, w7 = w1 * 7f;
979        for (int y = 0; y < h; y++) {
980            int ny = y + 1;
981            for (int i = 0; i < lineLen; i++) {

1002                            paletteMapping[((rr << 7) & 0x7C00)
1003                                    | ((gg << 2) & 0x3E0)

1009                    bdiff = (color>>>8&255)- (used>>>8&255);

1013                        curErrorGreen[px+1] += gdiff * w7;
